1//! The core union-find implementation.
2//!
3//! This implementation is inspired by "Concurrent Disjoint Set Union" by
4//! Siddhartha Jayanti and Robert Tarjan, but it includes changes and extensions
5//! to make it work with egglog's requirements. On the whole, it's a bit simpler
6//! than the implementation in the paper. A few notes:
7//!
8//! * We don't bother with union-by-rank: this isn't in use in egg or egglog. We
9//!   currently implement "union by min" which is a simple heuristic that folks
10//!   _hypothesize_ does well in practice for egraph applications. Depending on
11//!   how well this pans out, we may want to use something closer to the
12//!   randomized link by rank from the paper.
13//!
14//! * As in the paper, we use splitting for path compression, using the two-try
15//!   variant from the paper.
16use std::{cmp, sync::Arc};
17
18use super::{atomic_int::AtomicInt, buffer::Buffer};
19
20pub(crate) struct ConcurrentUnionFind<T> {
21    data: Arc<Buffer<T>>,
22}
23
24impl<T> Clone for ConcurrentUnionFind<T> {
25    fn clone(&self) -> Self {
26        Self {
27            data: self.data.clone(),
28        }
29    }
30}
31
32impl<T: AtomicInt> Default for ConcurrentUnionFind<T> {
33    fn default() -> Self {
34        Self::with_capacity(32)
35    }
36}
37
38impl<T: AtomicInt> ConcurrentUnionFind<T> {
39    pub fn deep_copy(&self) -> Self {
40        ConcurrentUnionFind {
41            data: Arc::new(Buffer::clone(&*self.data)),
42        }
43    }
44}
45
46impl<T: AtomicInt> ConcurrentUnionFind<T> {
47    /// Reset the union-find datastructure, setting each element to point to itself.
48    ///
49    /// This method blocks until all threads have finished their current operations.
50    pub fn reset(&self) {
51        self.data.re_init(T::from_usize);
52    }
53
54    /// Create a new union-find datastructure with the given capacity.
55    pub fn with_capacity(capacity: usize) -> ConcurrentUnionFind<T> {
56        let data = Arc::new(Buffer::new(capacity, T::from_usize));
57        ConcurrentUnionFind { data }
58    }
59
60    /// Get the canonical value associated with `elt`.
61    ///
62    /// Note that it may be the case that `find(x) != find(y)` even if `x` and
63    /// `y` belong to the same equivalence class, as the canonical value for the
64    /// class may have changed between the `find(x)` and `find(y)` calls. To get
65    /// a "ground truth" view on this, use the `same_set` method.
66    pub fn find(&self, elt: T::Underlying) -> T::Underlying {
67        self.data.with_access(
68            T::as_usize(elt) + 1,
69            |buf| Self::find_impl(buf, elt),
70            T::from_usize,
71        )
72    }
73
74    /// Check if `l` and `r` belong to the same equivalence class.
75    pub fn same_set(&self, l: T::Underlying, r: T::Underlying) -> bool {
76        let max_elt = cmp::max(l, r);
77        self.data.with_access(
78            T::as_usize(max_elt) + 1,
79            |buf| {
80                let mut l = Self::find_impl(buf, l);
81                let mut r = Self::find_impl(buf, r);
82                while l != r {
83                    let next = buf[T::as_usize(l)].load();
84                    if next == l {
85                        return false;
86                    }
87                    l = Self::find_impl(buf, l);
88                    r = Self::find_impl(buf, r);
89                }
90                true
91            },
92            T::from_usize,
93        )
94    }
95
96    /// Merge the equivalence classes of `l` and `r`, returning the new parent
97    /// and new child classes. If `l` and `r` are already in the same class,
98    /// then their canonical representative is returned twice.
99    pub fn merge(
100        &self,
101        l: T::Underlying,
102        r: T::Underlying,
103    ) -> (
104        T::Underlying, /* parent */
105        T::Underlying, /* child */
106    ) {
107        self.data.with_access(
108            T::as_usize(cmp::max(l, r)) + 1,
109            |buf| {
110                let mut l = l;
111                let mut r = r;
112                loop {
113                    l = Self::find_impl(buf, l);
114                    r = Self::find_impl(buf, r);
115                    if l != r {
116                        // We do "union by min": common in egraphs due to the
117                        // hypothesis that smaller ids will be
118                        // better-represented in the egraph, and hence
119                        // perturbing the maximum id is likely to result in less
120                        // work for rebuilding.
121                        let parent = cmp::min(l, r);
122                        let child = cmp::max(l, r);
123                        match buf[T::as_usize(child)].cas(child, parent) {
124                            Ok(_) => return (parent, child),
125                            Err(_) => continue,
126                        }
127                    } else {
128                        return (l, l);
129                    }
130                }
131            },
132            T::from_usize,
133        )
134    }
135
136    fn find_impl(buf: &[T], elt: T::Underlying) -> T::Underlying {
137        macro_rules! load {
138            ($x:expr) => {
139                buf[T::as_usize($x)].load()
140            };
141        }
142        let mut cur = elt;
143        let mut next = load!(cur);
144        let mut grand = load!(next);
145        while next != grand {
146            let _ = buf[T::as_usize(cur)].cas(next, grand);
147            // This is what the paper calls "two-try" splitting.
148            // next = load!(cur);
149            // grand = load!(next);
150            // let _ = buf[T::as_usize(cur)].cas(next, grand);
151            cur = next;
152            next = load!(cur);
153            grand = load!(next);
154        }
155        next
156    }
157}
