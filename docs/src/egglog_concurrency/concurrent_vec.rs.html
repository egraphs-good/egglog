17/// A simple concurrent vector type supporting push operations that do not block
18/// reads. Concurrent pushes are serialized, but reads need not wait for writes
19/// to complete, except when the vector needs to be resized.
20pub struct ConcurrentVec<T> {
21    data: ReadOptimizedLock<Vec<MaybeUninit<SyncUnsafeCell<T>>>>,
22    /// The index of the next element to be pushed. This is used to determine
23    /// how many cells have been written to successfully without grabbing any
24    /// exclusive locks.
25    head: AtomicUsize,
26    /// Used to synchronize writes.
27    write_lock: Mutex<()>,
28}
29
30impl<T> Default for ConcurrentVec<T> {
31    fn default() -> Self {
32        Self::new()
33    }
34}
35
36impl<T> Drop for ConcurrentVec<T> {
37    fn drop(&mut self) {
38        let mut writer = self.data.lock();
39        let len = self.head.load(Ordering::Acquire);
40        if mem::needs_drop::<T>() {
41            for i in 0..len {
42                // SAFETY: we own the data, have exclusive access, and know that the
43                // data is valid (you need a valid reference to the vector to
44                // increment `head`, any such call to `head` must have exited by
45                // now).
46                unsafe { writer[i].as_mut_ptr().drop_in_place() };
47            }
48        }
49    }
50}
51
52impl<T> ConcurrentVec<T> {
53    /// Create a new `AsyncVec` with the default capacity (128).
54    pub fn new() -> Self {
55        Self::with_capacity(128)
56    }
57
58    pub fn resize_with(&self, new_len: usize, mut f: impl FnMut() -> T) {
59        let head = self.head.load(Ordering::Acquire);
60        if new_len <= head {
61            return;
62        }
63        // We need to write to the vector and potentially resize it. Grab a write lock.
64        let _guard = self.write_lock.lock().unwrap();
65        let head = self.head.load(Ordering::Acquire);
66        if new_len <= head {
67            return;
68        }
69        self.push_at(f(), new_len - 1, || {
70            MaybeUninit::new(SyncUnsafeCell(UnsafeCell::new(f())))
71        });
72        self.head.store(new_len, Ordering::Release);
73    }
74
75    /// Create a new `ConcurrentVec` with the given starting capacity.
76    pub fn with_capacity(capacity: usize) -> Self {
77        let capacity = capacity.next_power_of_two();
78        Self {
79            data: ReadOptimizedLock::new(Vec::with_capacity(capacity)),
80            head: AtomicUsize::new(0),
81            write_lock: Mutex::new(()),
82        }
83    }
84    /// Push `item` onto the vector. Other calls to `push` may have to complete
85    /// in order for this item to be visible.
86    pub fn push(&self, item: T) -> usize {
87        let _guard = self.write_lock.lock().unwrap();
88        let index = self.head.load(Ordering::Acquire);
89        self.push_at(item, index, MaybeUninit::uninit);
90        self.head.store(index + 1, Ordering::Release);
91        index
92    }
93
94    fn push_at(
95        &self,
96        item: T,
97        index: usize,
98        mut init_fn: impl FnMut() -> MaybeUninit<SyncUnsafeCell<T>>,
99    ) {
100        let handle = self.data.read();
101        if let Some(slot) = handle.get(index) {
102            // SAFETY: we are tansferring ownership of `item` to the slot.
103            unsafe { ((*slot.as_ptr()).0.get()).write(item) };
104            return;
105        }
106        // `index` is out of bounds. Need to resize.
107        mem::drop(handle);
108        let mut writer = self.data.lock();
109        if index >= writer.len() {
110            writer.resize_with((index + 1).next_power_of_two(), &mut init_fn);
111        }
112        mem::drop(writer);
113        self.push_at(item, index, init_fn);
114    }
115
116    pub fn read(&self) -> impl Deref<Target = [T]> + '_ {
117        let valid_prefix = self.head.load(Ordering::Acquire);
118        let reader = self.data.read();
119        ReadHandle {
120            valid_prefix,
121            reader,
122        }
123    }
124}
125
126struct ReadHandle<'a, T> {
127    valid_prefix: usize,
128    reader: MutexReader<'a, Vec<MaybeUninit<SyncUnsafeCell<T>>>>,
129}
130
131impl<T> Deref for ReadHandle<'_, T> {
132    type Target = [T];
133
134    fn deref(&self) -> &[T] {
135        // SAFETY: all elements up to `prefix` are valid, and MaybeUninit<T> has
136        // a compatible layout with T so long as T is properly initialized.
137        //
138        // NB: transmuting an UnsafeCell<T> to <T> may not be safe long-term,
139        // even though this code passes miri.
140        unsafe {
141            mem::transmute::<&[MaybeUninit<SyncUnsafeCell<T>>], &[T]>(
142                &self.reader[0..self.valid_prefix],
143            )
144        }
145    }
146}

148struct SyncUnsafeCell<T>(UnsafeCell<T>);
149
150unsafe impl<T: Send> Send for SyncUnsafeCell<T> {}
151unsafe impl<T: Sync> Sync for SyncUnsafeCell<T> {}
