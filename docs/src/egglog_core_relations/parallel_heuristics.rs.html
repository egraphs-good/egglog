1//! Basic heuristics for whether or not to use a parallel or serial verion of an algorithm.
2//!
3//! The parallel implementations in this crate generally have a noticeable overhead when compared
4//! to the serial versions on small problem sizes.
5
6/// These are operations that work on a per-table or per-rule level where the size of the workload
7/// is hard to gauge ahead of time. In this case, we gate parallel execution based on the number of
8/// threads available and whether the total size of the database exceeds a certain threshold.
9pub(crate) fn parallelize_db_level_op(_db_size: usize) -> bool {
10    #[cfg(debug_assertions)]
11    {
12        use rand::Rng;
13        rand::rng().random_bool(0.5)
14    }
15    #[cfg(not(debug_assertions))]
16    {
17        _db_size > 10_000 && rayon::current_num_threads() > 1
18    }
19}
