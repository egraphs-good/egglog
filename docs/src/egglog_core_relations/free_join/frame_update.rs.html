1//! A data-structure for low-overhead buffering of updates in a free join
2//! execution.
3//!
4//! Free Join is a recursive algorithm that that discovers a candidate binding for a particular
5//! variable in a query and then recursively runs the rest of the join restricted for that binding
6//! holding. Once the "sub-join" finishes, the outer recursive call backtracks, adds a separate
7//! binding, and then repeats.
8//!
9//! The Free Join paper observed that this resulted in poor cache behavior because for every cell
10//! iterated over in an outer stage, we had to do several other steps on successive inner stages.
11//! Instead, we can accumulate a set of new bindings in a separate buffer and then iterate over
12//! those bindings in recursive calls. When parallelism is enabled, this data-structure allows us
13//! hand over an entire batch of recursive calls to a separate thread to process independently.
14
15use crate::numeric_id::{DenseIdMap, define_id};
16
17use crate::{Subset, Value};
18
19use super::{AtomId, Variable};
20
21define_id!(pub SubsetId, u32, "An offset into a buffer of subsets");
22
23enum UpdateCell {
24    PushBinding(Variable, Value),
25    RefineAtom(AtomId, SubsetId),
26    EndFrame,
27}
28
29pub(super) enum UpdateInstr {
30    PushBinding(Variable, Value),
31    RefineAtom(AtomId, Subset),
32    /// Marks the end of the current frame. Time to make a recursive call.
33    EndFrame,
34}
35
36/// A flat buffer of updates that is used to prepare a sequence of recursive calls to free join.
37#[derive(Default)]
38pub(super) struct FrameUpdates {
39    subsets: DenseIdMap<SubsetId, Subset>,
40    updates: Vec<UpdateCell>,
41    frames: usize,
42    last_start: usize,
43}
44
45impl FrameUpdates {
46    pub(super) fn with_capacity(capacity: usize) -> FrameUpdates {
47        FrameUpdates {
48            subsets: DenseIdMap::with_capacity(capacity),
49            updates: Vec::with_capacity(capacity * 2),
50            frames: 0,
51            last_start: 0,
52        }
53    }
54
55    /// Bind `var` to `val` in the current frame.
56    pub(super) fn push_binding(&mut self, var: Variable, val: Value) {
57        self.updates.push(UpdateCell::PushBinding(var, val));
58    }
59
60    /// Refine `atom` to consider only the given `subset` in the current frame.
61    pub(super) fn refine_atom(&mut self, atom: AtomId, subset: Subset) {
62        let subset = self.subsets.push(subset);
63        self.updates.push(UpdateCell::RefineAtom(atom, subset));
64    }
65
66    /// Roll back the updates to the last frame start. Note that repeated calls
67    /// to this method will still only roll back one frame (total).
68    pub(super) fn rollback(&mut self) {
69        self.updates.truncate(self.last_start);
70    }
71
72    /// Finish the current frame and prepare for the next one.
73    pub(super) fn finish_frame(&mut self) {
74        self.updates.push(UpdateCell::EndFrame);
75        self.last_start = self.updates.len();
76        self.frames += 1;
77    }
78
79    /// Get the number of frames that have been finished.
80    pub(super) fn frames(&self) -> usize {
81        self.frames
82    }
83
84    pub(super) fn clear(&mut self) {
85        self.subsets.clear();
86        self.updates.clear();
87    }
88
89    pub(super) fn drain(&mut self, f: impl FnMut(UpdateInstr)) {
90        let start = if matches!(self.updates.first(), Some(UpdateCell::EndFrame)) {
91            1 // Skip the first EndFrame
92        } else {
93            0
94        };
95        self.updates
96            .drain(start..)
97            .map(|cell| match cell {
98                UpdateCell::PushBinding(var, val) => UpdateInstr::PushBinding(var, val),
99                UpdateCell::RefineAtom(atom, subset) => {
100                    UpdateInstr::RefineAtom(atom, self.subsets.take(subset).unwrap())
101                }
102                UpdateCell::EndFrame => UpdateInstr::EndFrame,
103            })
104            .for_each(f);
105        self.subsets.clear();
106        self.frames = 0;
107        self.last_start = 0;
108    }
109}
