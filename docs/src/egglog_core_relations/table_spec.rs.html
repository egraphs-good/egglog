3//! Tables are a mapping from some set of keys to another set of values. Tables
4//! can also be "sorted by" a columna dn "partitioned by" another. This can help

41/// The version of a table.
42#[derive(Clone, Debug, PartialEq, Eq)]
43pub struct TableVersion {
44    /// New major generations invalidate all existing RowIds for a table.
45    pub major: Generation,
46    /// New minor generations within a major generation do not invalidate
47    /// existing RowIds, but they may indicate that `all` can return a larger
48    /// subset than before.
49    pub minor: Offset,
50    // NB: we may want to make `Offset` and `RowId` the same.
51}

53#[derive(Clone)]
54pub struct TableSpec {
55    /// The number of key columns for the table.
56    pub n_keys: usize,
57
58    /// The number of non-key (i.e. value) columns in the table.
59    ///
60    /// The total "arity" of the table is `n_keys + n_vals`.
61    pub n_vals: usize,
62
63    /// Columns that cannot be cached across generations.
64    ///
65    /// These columns should (e.g.) never have indexes built for them, as they
66    /// will go out of date too quickly.
67    pub uncacheable_columns: DenseIdMap<ColumnId, bool>,
68
69    /// Whether or not deletions are supported for this table.
70    ///
71    /// Tables where this value is false are allowed to panic on calls to
72    /// `stage_remove`.
73    pub allows_delete: bool,
74}
75
76impl TableSpec {
77    /// The total number of columns stored by the table.
78    pub fn arity(&self) -> usize {
79        self.n_keys + self.n_vals
80    }

103/// Custom functions used for tables that encode a bulk value-level rebuild of other tables.
104///
105/// The initial use-case for this trait is to support optimized implementations of rebuilding,
106/// where `Rebuilder` is implemented as a Union-find.
107///
108/// Value-level rebuilds are difficult to implement efficiently using rules as they require
109/// searching for changes to any column for a table: while it is possible to do, implementing this
110/// custom is more efficient in the case of rebuilding.
111pub trait Rebuilder: Send + Sync {
112    /// The column that contains values that should be rebuilt. If this is set, callers can use
113    /// this functionality to perform rebuilds incrementally.
114    fn hint_col(&self) -> Option<ColumnId>;
115    fn rebuild_val(&self, val: Value) -> Value;
116    /// Rebuild a contiguous slice of rows in the table.
117    fn rebuild_buf(
118        &self,
119        buf: &RowBuffer,
120        start: RowId,
121        end: RowId,
122        out: &mut TaggedRowBuffer,
123        exec_state: &mut ExecutionState,
124    );
125    /// Rebuild an arbitrary subset of the table.
126    fn rebuild_subset(
127        &self,
128        other: WrappedTableRef,
129        subset: SubsetRef,
130        out: &mut TaggedRowBuffer,
131        exec_state: &mut ExecutionState,
132    );
133    /// Rebuild a slice of values in place, returning true if any values were changed.
134    fn rebuild_slice(&self, vals: &mut [Value]) -> bool;
135}

145/// An interface for a table.
146pub trait Table: Any + Send + Sync {
147    /// A variant of clone that returns a boxed trait object; this trait object
148    /// must contain all of the data associated with the current table.
149    fn dyn_clone(&self) -> Box<dyn Table>;
150
151    /// If this table can perform a table-level rebuild, construct a [`Rebuilder`] for it.
152    fn rebuilder<'a>(&'a self, _cols: &[ColumnId]) -> Option<Box<dyn Rebuilder + 'a>> {
153        None
154    }
155
156    /// Rebuild the table according to the given [`Rebuilder`] implemented by `table`, if
157    /// there is one. Applying a rebuild can cause more mutations to be buffered, which can in turn
158    /// be flushed by a call to [`Table::merge`].
159    ///
160    /// Note that value-level rebuilds are only relevant for tables that opt into it. As a result,
161    /// tables do nothing by default.
162    ///
163    /// Returns whether any rows may be removed or inserted.
164    fn apply_rebuild(
165        &mut self,
166        _table_id: TableId,
167        _table: &WrappedTable,
168        _next_ts: Value,
169        _exec_state: &mut ExecutionState,
170    ) -> bool {
171        // Default implementation does nothing.
172        false
173    }
174
175    /// Refresh rows whose rebuildable columns mention one of `dirty_ids` by re-inserting the same
176    /// logical row with a fresh timestamp.
177    ///
178    /// This is the narrow escape hatch used when some external rebuild step
179    /// changes the semantics of an id in place, so seminaive needs a new
180    /// parent-row delta even though the row's key columns do not otherwise
181    /// change.
182    ///
183    /// One source of such ids is [`crate::ContainerRebuildSummary::dirty_ids`].
184    ///
185    /// Tables that do not maintain rebuildable id columns can use the default
186    /// no-op implementation.
187    fn refresh_rows_for_values(&mut self, _dirty_ids: &[Value], _next_ts: Value) -> bool {
188        false
189    }
190
191    /// A boilerplate method to make it easier to downcast values of `Table`.
192    ///
193    /// Implementors should be able to implement this method by returning
194    /// `self`.
195    fn as_any(&self) -> &dyn Any;
196
197    /// The schema of the table.
198    ///
199    /// These are immutable properties of the table; callers can assume they
200    /// will never change.
201    fn spec(&self) -> TableSpec;
202
203    /// Clear all table contents. If the table is nonempty, this will change the
204    /// generation of the table. This method also clears any pending data.
205    fn clear(&mut self);
206
207    // Used in queries:
208
209    /// Get a subset corresponding to all rows in the table.
210    fn all(&self) -> Subset;
211
212    /// Get the length of the table.
213    ///
214    /// This is not in general equal to the length of the `all` subset: the size
215    /// of a subset is allowed to be larger than the number of table entries in
216    /// range of the subset.
217    fn len(&self) -> usize;
218
219    /// Check if the table is empty.
220    fn is_empty(&self) -> bool {
221        self.len() == 0
222    }
223
224    /// Get the current version for the table. [`RowId`]s and [`Subset`]s are
225    /// only valid for a given major generation.
226    fn version(&self) -> TableVersion;
227
228    /// Get the subset of the table that has appeared since the last offset.
229    fn updates_since(&self, offset: Offset) -> Subset;
230
231    /// Iterate over the given subset of the table, starting at an opaque
232    /// `start` token, ending after up to `n` rows, returning the next start
233    /// token if more rows remain. Only invoke `f` on rows that match the given
234    /// constraints.
235    ///
236    /// This method is _not_ object safe, but it is used to define various
237    /// "default" implementations of object-safe methods like `scan` and
238    /// `pivot`.
239    fn scan_generic_bounded(
240        &self,
241        subset: SubsetRef,
242        start: Offset,
243        n: usize,
244        cs: &[Constraint],
245        f: impl FnMut(RowId, &[Value]),
246    ) -> Option<Offset>
247    where
248        Self: Sized;
249
250    /// Iterate over the given subset of the table.
251    ///
252    /// This is a variant of [`Table::scan_generic_bounded`] that iterates over
253    /// the entire table.
254    fn scan_generic(&self, subset: SubsetRef, mut f: impl FnMut(RowId, &[Value]))
255    where
256        Self: Sized,
257    {
258        let mut cur = Offset::new(0);
259        while let Some(next) = self.scan_generic_bounded(subset, cur, usize::MAX, &[], |id, row| {
260            f(id, row);
261        }) {
262            cur = next;
263        }
264    }
265
266    /// Filter a given subset of the table for the rows that are live
267    fn refine_live(&self, subset: Subset) -> Subset {
268        // NB: This relies on Value::stale() being strictly larger than any other value in the table.
269        self.refine_one(
270            subset,
271            &Constraint::LtConst {
272                col: ColumnId::new_const(0),
273                val: Value::stale(),
274            },
275        )
276    }
277
278    /// Filter a given subset of the table for the rows matching the single constraint.
279    ///
280    /// Implementors must provide at least one of `refine_one` or `refine`.`
281    fn refine_one(&self, subset: Subset, c: &Constraint) -> Subset {
282        self.refine(subset, std::slice::from_ref(c))
283    }
284
285    /// Filter a given subset of the table for the rows matching the given constraints.
286    ///
287    /// Implementors must provide at least one of `refine_one` or `refine`.`
288    fn refine(&self, subset: Subset, cs: &[Constraint]) -> Subset {
289        cs.iter()
290            .fold(subset, |subset, c| self.refine_one(subset, c))
291    }
292
293    /// An optional method for quickly generating a subset from a constraint.
294    /// The standard use-case here is to apply constraints based on a column
295    /// that is known to be sorted.
296    ///
297    /// These constraints are very helpful for query planning; it is a good idea
298    /// to implement them.
299    fn fast_subset(&self, _: &Constraint) -> Option<Subset> {
300        None
301    }
302
303    /// A helper routine that leverages the existing `fast_subset` method to
304    /// preprocess a set of constraints into "fast" and "slow" ones, returning
305    /// the subet of indexes that match the fast one.
306    fn split_fast_slow(
307        &self,
308        cs: &[Constraint],
309    ) -> (
310        Subset,                  /* the subset of the table matching all fast constraints */
311        Pooled<Vec<Constraint>>, /* the fast constraints */
312        Pooled<Vec<Constraint>>, /* the slow constraints */
313    ) {
314        with_pool_set(|ps| {
315            let mut fast = ps.get::<Vec<Constraint>>();
316            let mut slow = ps.get::<Vec<Constraint>>();
317            let mut subset = self.all();
318            for c in cs {
319                if let Some(sub) = self.fast_subset(c) {
320                    subset.intersect(sub.as_ref(), &ps.get_pool());
321                    fast.push(c.clone());
322                } else {
323                    slow.push(c.clone());
324                }
325            }
326            (subset, fast, slow)
327        })
328    }

330    // Used in actions:
331
332    /// Look up a single row by the given key values, if it is in the table.
333    ///
334    /// The number of values specified by `keys` should match the number of
335    /// primary keys for the table.
336    fn get_row(&self, key: &[Value]) -> Option<Row>;
337
338    /// Look up the given column of single row by the given key values, if it is
339    /// in the table.
340    ///
341    /// The number of values specified by `keys` should match the number of
342    /// primary keys for the table.
343    fn get_row_column(&self, key: &[Value], col: ColumnId) -> Option<Value> {
344        self.get_row(key).map(|row| row.vals[col.index()])
345    }
346
347    /// Merge any updates to the table, and potentially update the generation for
348    /// the table.
349    fn merge(&mut self, exec_state: &mut ExecutionState) -> TableChange;
350
351    /// Create a new buffer for staging mutations on this table. Mutations staged to a
352    /// MutationBuffer that is then dropped may not take effect until the next call to
353    /// [`Table::merge`].
354    fn new_buffer(&self) -> Box<dyn MutationBuffer>;
355}

357/// A trait specifying a buffer of pending mutations for a [`Table`].
358///
359/// Dropping an object implementing this trait should "flush" the pending
360/// mutations to the table. Calling  [`Table::merge`] on that table would then
361/// apply those mutations, making them visible for future readers.
362pub trait MutationBuffer: Any + Send + Sync {
363    /// Stage the keyed entries for insertion. Changes may not be visible until
364    /// this buffer is dropped, and after `merge` is called on the underlying
365    /// table.
366    fn stage_insert(&mut self, row: &[Value]);
367
368    /// Stage the keyed entries for removal. Changes may not be visible until
369    /// this buffer is dropped, and after `merge` is called on the underlying
370    /// table.
371    fn stage_remove(&mut self, key: &[Value]);
372
373    /// Get a fresh handle to the same table.
374    fn fresh_handle(&self) -> Box<dyn MutationBuffer>;
375}

379pub(crate) fn wrapper<T: Table>() -> Box<dyn TableWrapper> {

383impl<T: Table> TableWrapper for WrapperImpl<T> {
384    fn dyn_clone(&self) -> Box<dyn TableWrapper> {
385        Box::new(Self(PhantomData))
386    }
387    fn scan_bounded(
388        &self,
389        table: &dyn Table,
390        subset: SubsetRef,
391        start: Offset,
392        n: usize,
393        out: &mut TaggedRowBuffer,
394    ) -> Option<Offset> {
395        let table = table.as_any().downcast_ref::<T>().unwrap();
396        table.scan_generic_bounded(subset, start, n, &[], |row_id, row| {
397            out.add_row(row_id, row);
398        })
399    }
400    fn group_by_col(&self, table: &dyn Table, subset: SubsetRef, col: ColumnId) -> ColumnIndex {
401        let table = table.as_any().downcast_ref::<T>().unwrap();
402        let mut res = ColumnIndex::new();
403        table.scan_generic(subset, |row_id, row| {
404            res.add_row(&[row[col.index()]], row_id);
405        });
406        res
407    }
408    fn group_by_key(&self, table: &dyn Table, subset: SubsetRef, cols: &[ColumnId]) -> TupleIndex {
409        let table = table.as_any().downcast_ref::<T>().unwrap();
410        let mut res = TupleIndex::new(cols.len());
411        match cols {
412            [] => {}
413            [col] => table.scan_generic(subset, |row_id, row| {
414                res.add_row(&[row[col.index()]], row_id);
415            }),
416            [x, y] => table.scan_generic(subset, |row_id, row| {
417                res.add_row(&[row[x.index()], row[y.index()]], row_id);
418            }),
419            [x, y, z] => table.scan_generic(subset, |row_id, row| {
420                res.add_row(&[row[x.index()], row[y.index()], row[z.index()]], row_id);
421            }),
422            _ => {
423                let mut scratch = SmallVec::<[Value; 8]>::new();
424                table.scan_generic(subset, |row_id, row| {
425                    for col in cols {
426                        scratch.push(row[col.index()]);
427                    }
428                    res.add_row(&scratch, row_id);
429                    scratch.clear();
430                });
431            }
432        }
433        res
434    }
435    fn scan_project(
436        &self,
437        table: &dyn Table,
438        subset: SubsetRef,
439        cols: &[ColumnId],
440        start: Offset,
441        n: usize,
442        cs: &[Constraint],
443        out: &mut dyn RowSink,
444    ) -> Option<Offset> {
445        let table = table.as_any().downcast_ref::<T>().unwrap();
446        match cols {
447            [] => None,
448            [col] => table.scan_generic_bounded(subset, start, n, cs, |id, row| {
449                out.add_row(id, &[row[col.index()]]);
450            }),
451            [x, y] => table.scan_generic_bounded(subset, start, n, cs, |id, row| {
452                out.add_row(id, &[row[x.index()], row[y.index()]]);
453            }),
454            [x, y, z] => table.scan_generic_bounded(subset, start, n, cs, |id, row| {
455                out.add_row(id, &[row[x.index()], row[y.index()], row[z.index()]]);
456            }),
457            _ => {
458                let mut scratch = SmallVec::<[Value; 8]>::with_capacity(cols.len());
459                table.scan_generic_bounded(subset, start, n, cs, |id, row| {
460                    for col in cols {
461                        scratch.push(row[col.index()]);
462                    }
463                    out.add_row(id, &scratch);
464                    scratch.clear();
465                })
466            }
467        }
468    }
469
470    fn lookup_row_vectorized(
471        &self,
472        table: &dyn Table,
473        mask: &mut Mask,
474        bindings: &mut Bindings,
475        args: &[QueryEntry],
476        col: ColumnId,
477        out_var: Variable,
478    ) {
479        let table = table.as_any().downcast_ref::<T>().unwrap();
480        let mut out = with_pool_set(PoolSet::get::<Vec<Value>>);
481        for_each_binding_with_mask!(mask, args, bindings, |iter| {
482            iter.fill_vec(&mut out, Value::stale, |_, args| {
483                table.get_row_column(args.as_slice(), col)
484            })
485        });
486        bindings.insert(out_var, &out);
487    }
488
489    fn lookup_with_default_vectorized(
490        &self,
491        table: &dyn Table,
492        mask: &mut Mask,
493        bindings: &mut Bindings,
494        args: &[QueryEntry],
495        col: ColumnId,
496        default: QueryEntry,
497        out_var: Variable,
498    ) {
499        let table = table.as_any().downcast_ref::<T>().unwrap();
500        let mut out = with_pool_set(|ps| ps.get::<Vec<Value>>());
501        for_each_binding_with_mask!(mask, args, bindings, |iter| {
502            match default {
503                QueryEntry::Var(default) => iter.zip(&bindings[default]).fill_vec(
504                    &mut out,
505                    Value::stale,
506                    |_, (args, default)| {
507                        Some(
508                            table
509                                .get_row_column(args.as_slice(), col)
510                                .unwrap_or(*default),
511                        )
512                    },
513                ),
514                QueryEntry::Const(default) => iter.fill_vec(&mut out, Value::stale, |_, args| {
515                    Some(
516                        table
517                            .get_row_column(args.as_slice(), col)
518                            .unwrap_or(default),
519                    )
520                }),
521            }
522        });
523        bindings.insert(out_var, &out);
524    }

527/// A WrappedTable takes a Table and extends it with a number of helpful,
528/// object-safe methods for accessing a table.
529///
530/// It essentially acts like an extension trait: it is a separate type to allow
531/// object-safe extension methods to call methods that require `Self: Sized`.
532/// The implementations here downcast manually to the type used when
533/// constructing the WrappedTable.
534pub struct WrappedTable {
535    inner: Box<dyn Table>,
536    wrapper: Box<dyn TableWrapper>,
537}
538
539impl WrappedTable {
540    pub(crate) fn new<T: Table>(inner: T) -> Self {
541        let wrapper = wrapper::<T>();
542        let inner = Box::new(inner);
543        Self { inner, wrapper }
544    }
545
546    /// Clone the contents of the table.
547    pub fn dyn_clone(&self) -> Self {
548        WrappedTable {
549            inner: self.inner.dyn_clone(),
550            wrapper: self.wrapper.dyn_clone(),
551        }
552    }
553
554    pub(crate) fn as_ref(&self) -> WrappedTableRef<'_> {
555        WrappedTableRef {
556            inner: &*self.inner,
557            wrapper: &*self.wrapper,
558        }
559    }
560
561    /// Starting at the given [`Offset`] into `subset`, scan up to `n` rows and
562    /// write them to `out`. Return the next starting offset. If no offset is
563    /// returned then the subset has been scanned completely.
564    pub fn scan_bounded(
565        &self,
566        subset: SubsetRef,
567        start: Offset,
568        n: usize,
569        out: &mut TaggedRowBuffer,
570    ) -> Option<Offset> {
571        self.as_ref().scan_bounded(subset, start, n, out)
572    }
573
574    /// Group the contents of the given subset by the given column.
575    pub(crate) fn group_by_col(&self, subset: SubsetRef, col: ColumnId) -> ColumnIndex {
576        self.as_ref().group_by_col(subset, col)
577    }
578
579    /// A multi-column vairant of [`WrappedTable::group_by_col`].
580    pub(crate) fn group_by_key(&self, subset: SubsetRef, cols: &[ColumnId]) -> TupleIndex {
581        self.as_ref().group_by_key(subset, cols)
582    }
583
584    /// A variant fo [`WrappedTable::scan_bounded`] that projects a subset of
585    /// columns and only appends rows that match the given constraints.
586    pub fn scan_project(
587        &self,
588        subset: SubsetRef,
589        cols: &[ColumnId],
590        start: Offset,
591        n: usize,
592        cs: &[Constraint],
593        out: &mut dyn RowSink,
594    ) -> Option<Offset> {
595        self.as_ref().scan_project(subset, cols, start, n, cs, out)
596    }
597
598    /// Return the contents of the subset as a [`TaggedRowBuffer`].
599    pub fn scan(&self, subset: SubsetRef) -> TaggedRowBuffer {
600        self.as_ref().scan(subset)
601    }
602
603    /// Return the number of rows currently stored in the table.
604    pub fn len(&self) -> usize {
605        self.inner.len()
606    }
607
608    /// Check if the table is empty.
609    pub fn is_empty(&self) -> bool {
610        self.inner.is_empty()
611    }
612
613    pub(crate) fn lookup_row_vectorized(
614        &self,
615        mask: &mut Mask,
616        bindings: &mut Bindings,
617        args: &[QueryEntry],
618        col: ColumnId,
619        out_var: Variable,
620    ) {
621        self.as_ref()
622            .lookup_row_vectorized(mask, bindings, args, col, out_var)
623    }
624
625    #[allow(clippy::too_many_arguments)]
626    pub(crate) fn lookup_with_default_vectorized(
627        &self,
628        mask: &mut Mask,
629        bindings: &mut Bindings,
630        args: &[QueryEntry],
631        col: ColumnId,
632        default: QueryEntry,
633        out_var: Variable,
634    ) {
635        self.as_ref()
636            .lookup_with_default_vectorized(mask, bindings, args, col, default, out_var)
637    }
638}

670        table: &dyn Table,
671        subset: SubsetRef,
672        cols: &[ColumnId],
673        start: Offset,
674        n: usize,
675        cs: &[Constraint],
676        out: &mut dyn RowSink,
677    ) -> Option<Offset>;
678
679    fn scan(&self, table: &dyn Table, subset: SubsetRef) -> TaggedRowBuffer {
680        let arity = table.spec().arity();
681        let mut buf = TaggedRowBuffer::new(arity);
682        assert!(
683            self.scan_bounded(table, subset, Offset::new(0), usize::MAX, &mut buf)
684                .is_none()
685        );
686        buf
687    }

706        args: &[QueryEntry],

708        default: QueryEntry,

713/// An extra layer of indirection over a [`WrappedTable`] that does not require that the caller
714/// actually own the table. This is useful when a table implementation needs to construct a
715/// WrappedTable on its own.
716#[derive(Clone, Copy)]
717pub struct WrappedTableRef<'a> {
718    inner: &'a dyn Table,
719    wrapper: &'a dyn TableWrapper,
720}
721
722impl WrappedTableRef<'_> {
723    pub(crate) fn with_wrapper<T: Table, R>(
724        inner: &T,
725        f: impl for<'a> FnOnce(WrappedTableRef<'a>) -> R,
726    ) -> R {
727        let wrapper = WrapperImpl::<T>(PhantomData);
728        f(WrappedTableRef {
729            inner,
730            wrapper: &wrapper,
731        })
732    }
733
734    /// Starting at the given [`Offset`] into `subset`, scan up to `n` rows and
735    /// write them to `out`. Return the next starting offset. If no offset is
736    /// returned then the subset has been scanned completely.
737    pub fn scan_bounded(
738        &self,
739        subset: SubsetRef,
740        start: Offset,
741        n: usize,
742        out: &mut TaggedRowBuffer,
743    ) -> Option<Offset> {
744        self.wrapper.scan_bounded(self.inner, subset, start, n, out)
745    }
746
747    /// Group the contents of the given subset by the given column.
748    pub(crate) fn group_by_col(&self, subset: SubsetRef, col: ColumnId) -> ColumnIndex {
749        self.wrapper.group_by_col(self.inner, subset, col)
750    }
751
752    /// A multi-column vairant of [`WrappedTable::group_by_col`].
753    pub(crate) fn group_by_key(&self, subset: SubsetRef, cols: &[ColumnId]) -> TupleIndex {
754        self.wrapper.group_by_key(self.inner, subset, cols)
755    }
756
757    /// A variant fo [`WrappedTable::scan_bounded`] that projects a subset of
758    /// columns and only appends rows that match the given constraints.
759    pub fn scan_project(
760        &self,
761        subset: SubsetRef,
762        cols: &[ColumnId],
763        start: Offset,
764        n: usize,
765        cs: &[Constraint],
766        out: &mut dyn RowSink,
767    ) -> Option<Offset> {
768        self.wrapper
769            .scan_project(self.inner, subset, cols, start, n, cs, out)
770    }
771
772    /// Return the contents of the subset as a [`TaggedRowBuffer`].
773    pub fn scan(&self, subset: SubsetRef) -> TaggedRowBuffer {
774        self.wrapper.scan(self.inner, subset)
775    }
776
777    /// Return the number of rows currently stored in the table.
778    pub fn len(&self) -> usize {
779        self.inner.len()
780    }
781
782    pub(crate) fn lookup_row_vectorized(
783        &self,
784        mask: &mut Mask,
785        bindings: &mut Bindings,
786        args: &[QueryEntry],
787        col: ColumnId,
788        out_var: Variable,
789    ) {
790        self.wrapper
791            .lookup_row_vectorized(self.inner, mask, bindings, args, col, out_var);
792    }
793
794    #[allow(clippy::too_many_arguments)]
795    pub(crate) fn lookup_with_default_vectorized(
796        &self,
797        mask: &mut Mask,
798        bindings: &mut Bindings,
799        args: &[QueryEntry],
800        col: ColumnId,
801        default: QueryEntry,
802        out_var: Variable,
803    ) {
804        self.wrapper.lookup_with_default_vectorized(
805            self.inner, mask, bindings, args, col, default, out_var,
806        );
807    }
808}
809
810impl Deref for WrappedTableRef<'_> {
811    type Target = dyn Table;
812
813    fn deref(&self) -> &Self::Target {
814        self.inner
815    }
