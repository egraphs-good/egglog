49/// A set of rules to run against a [`Database`].
50///
51/// See [`Database::new_rule_set`] for more information.
52#[derive(Default)]
53pub struct RuleSet {
54    /// The contents of the queries (i.e. the LHS of the rules) for each rule in the set, along
55    /// with a description of the rule.
56    ///
57    /// The action here is used to map between rule descriptions and plans, which contain ActionIds. The current
58    /// accounting logic assumes that rules and actions stand in a bijection. If we relaxed that
59    /// later on, most of the core logic would still work but the accounting logic could get more
60    /// complex.
61    pub(crate) plans: IdVec<RuleId, (Plan, Arc<str> /* description */, SymbolMap)>,
62    pub(crate) actions: DenseIdMap<ActionId, ActionInfo>,
63}
64
65impl RuleSet {
66    pub fn build_cached_plan(&self, rule_id: RuleId) -> CachedPlan {
67        let (plan, desc, symbol_map) = self.plans.get(rule_id).expect("rule must exist");
68        let actions = self
69            .actions
70            .get(plan.actions())
71            .expect("action must exist")
72            .clone();
73        CachedPlan {
74            plan: plan.clone(),
75            desc: desc.clone(),
76            symbol_map: symbol_map.clone(),
77            actions,
78        }
79    }
80}

82/// Builder for a [`RuleSet`].
83///
84/// There are in general two ways to add rules to a rule set:
85///
86/// 1. Use the QueryBuilder and RuleBuilder APIs to construct a rule from scratch.
87/// 2. Use a previously cached plan and add extra constraints to it.
88///
89/// The pattern this is used by egglog is as follows: An egglog rule is first compiled to a cached
90/// plan using builder patterns at declaration time, and each time the rule is run, it is added to
91/// a ruleset using the cached plan and possibly some extra constraints (e.g., timestamp).
92///
93/// See [`Database::new_rule_set`] for more information.
94pub struct RuleSetBuilder<'outer> {
95    rule_set: RuleSet,
96    db: &'outer mut Database,
97}

116    pub fn new_rule<'a>(&'a mut self) -> QueryBuilder<'outer, 'a> {
117        let instrs = with_pool_set(PoolSet::get);
118        QueryBuilder {
119            rsb: self,
120            instrs,
121            query: Query {
122                var_info: Default::default(),
123                atoms: Default::default(),
124                // start with an invalid ActionId

126                plan_strategy: Default::default(),
127                fun_deps: Default::default(),

132    fn reprocess_constraints(
133        &self,
134        table: TableId,
135        atom: AtomId,
136        constraints: &[Constraint],
137    ) -> JoinHeader {
138        let processed = self.db.process_constraints(table, constraints);
139        if !processed.slow.is_empty() {
140            panic!(
141                "Cached plans only support constraints with a fast pushdown. \
142                 Got: {constraints:?} for table {table:?}",
143            );
144        }
145        JoinHeader {
146            atom,
147            constraints: processed.fast,
148            subset: processed.subset,
149        }
150    }
151
152    fn push_extra_constraints(
153        &self,
154        headers: &mut Vec<JoinHeader>,
155        atoms: &Arc<DenseIdMap<AtomId, Atom>>,
156        extra_constraints: &[(AtomId, Constraint)],
157    ) {
158        for (atom_id, constraint) in extra_constraints {
159            let atom_info = atoms.get(*atom_id).expect("atom must exist in plan");
160            let table = atom_info.table;
161
162            let header =
163                self.reprocess_constraints(table, *atom_id, std::slice::from_ref(constraint));
164            headers.push(header);
165        }
166    }
167
168    fn reprocess_existing_headers(
169        &self,
170        headers: &mut Vec<JoinHeader>,
171        atoms: &Arc<DenseIdMap<AtomId, Atom>>,
172        existing: &[JoinHeader],
173    ) {
174        for JoinHeader {
175            atom, constraints, ..
176        } in existing
177        {
178            let atom_info = atoms.get(*atom).expect("atom must exist in plan");
179            let table = atom_info.table;
180
181            headers.push(self.reprocess_constraints(table, *atom, constraints));
182        }
183    }
184
185    fn get_rule_with_extra_constraints(
186        &mut self,
187        cached: &CachedPlan,
188        action_id: ActionId,
189        extra_constraints: &[(AtomId, Constraint)],
190    ) -> Plan {
191        match &cached.plan {
192            Plan::SinglePlan(cached_plan) => {
193                let mut headers = vec![];
194                let stages = JoinStages {
195                    instrs: cached_plan.stages.instrs.clone(),
196                };
197
198                self.push_extra_constraints(&mut headers, &cached_plan.atoms, extra_constraints);
199
200                self.reprocess_existing_headers(
201                    &mut headers,
202                    &cached_plan.atoms,
203                    &cached_plan.header,
204                );
205
206                Plan::SinglePlan(SinglePlan {
207                    atoms: cached_plan.atoms.clone(),
208                    header: headers,
209                    stages,
210                    actions: action_id,
211                })
212            }
213            Plan::DecomposedPlan(cached_plan) => {
214                let mut blocks = Vec::with_capacity(cached_plan.stages.blocks.len());
215                let mut headers = vec![];
216                self.push_extra_constraints(&mut headers, &cached_plan.atoms, extra_constraints);
217                self.reprocess_existing_headers(
218                    &mut headers,
219                    &cached_plan.atoms,
220                    &cached_plan.header,
221                );
222
223                // Process body blocks
224                for cached_block in cached_plan.stages.blocks.iter() {
225                    let stages = JoinStages {
226                        instrs: cached_block.0.instrs.clone(),
227                    };
228
229                    blocks.push((stages, cached_block.1.clone()));
230                }
231
232                // Process result blocks
233                let result_block = JoinStages {
234                    instrs: cached_plan.result_block.instrs.clone(),
235                };
236
237                Plan::DecomposedPlan(DecomposedPlan {
238                    atoms: cached_plan.atoms.clone(),
239                    header: headers,
240                    stages: JoinStageBlocks { blocks },
241                    actions: action_id,
242                    result_block,
243                })
244            }
245        }
246    }
247
248    pub fn add_rule_from_cached_plan(
249        &mut self,
250        cached: &CachedPlan,
251        extra_constraints: &[(AtomId, Constraint)],
252    ) -> RuleId {
253        // First, patch in the new action id.
254        let action_id = self.rule_set.actions.push(cached.actions.clone());
255        let plan = self.get_rule_with_extra_constraints(cached, action_id, extra_constraints);
256        self.rule_set
257            .plans
258            .push((plan, cached.desc.clone(), cached.symbol_map.clone()))
259    }

262    pub fn build(self) -> RuleSet {
263        self.rule_set
264    }

322    /// Add the given atom to the query, with the given variables and constraints.
323    ///
324    /// NB: it is possible to constrain two non-equal variables to be equal
325    /// given this setup. Doing this will not cause any problems but
326    /// nevertheless is not recommended.
327    ///
328    /// The returned `AtomId` can be used to refer to this atom when adding constraints in
329    /// [`RuleSetBuilder::add_rule_from_cached_plan`].
330    ///
331    /// # Panics
332    /// Like most methods that take a [`TableId`], this method will panic if the
333    /// given table is not declared in the corresponding database.
334    pub fn add_atom<'b>(
335        &mut self,
336        table_id: TableId,
337        vars: &[QueryEntry],
338        cs: impl IntoIterator<Item = &'b Constraint>,
339    ) -> Result<AtomId, QueryError> {
340        let info = &self.rsb.db.tables[table_id];
341        let arity = info.spec.arity();
342        let check_constraint = |c: &Constraint| {
343            let process_col = |col: &ColumnId| -> Result<(), QueryError> {
344                if col.index() >= arity {
345                    Err(QueryError::InvalidConstraint {
346                        constraint: c.clone(),
347                        column: col.index(),
348                        table: table_id,
349                        arity,
350                    })
351                } else {
352                    Ok(())
353                }
354            };
355            match c {
356                Constraint::Eq { l_col, r_col } => {
357                    process_col(l_col)?;
358                    process_col(r_col)
359                }
360                Constraint::EqConst { col, .. }
361                | Constraint::LtConst { col, .. }
362                | Constraint::GtConst { col, .. }
363                | Constraint::LeConst { col, .. }
364                | Constraint::GeConst { col, .. } => process_col(col),
365            }
366        };
367        if arity != vars.len() {
368            return Err(QueryError::BadArity {
369                table: table_id,
370                expected: arity,
371                got: vars.len(),
372            });
373        }
374        let cs = Vec::from_iter(
375            cs.into_iter()
376                .cloned()
377                .chain(vars.iter().enumerate().filter_map(|(i, qe)| match qe {
378                    QueryEntry::Var(_) => None,
379                    QueryEntry::Const(c) => Some(Constraint::EqConst {
380                        col: ColumnId::from_usize(i),
381                        val: *c,
382                    }),
383                })),
384        );
385        cs.iter().try_fold((), |_, c| check_constraint(c))?;
386        let processed = self.rsb.db.process_constraints(table_id, &cs);
387        let mut atom = Atom {
388            table: table_id,
389            var_columns: Default::default(),
390            constraints: processed,
391        };
392        let next_atom = AtomId::from_usize(self.query.atoms.n_ids());
393        let mut subatoms = HashMap::<Variable, SubAtom>::default();
394        for (i, qe) in vars.iter().enumerate() {
395            let var = match qe {
396                QueryEntry::Var(var) => *var,
397                QueryEntry::Const(_) => {
398                    continue;
399                }
400            };
401            if var == Variable::placeholder() {
402                continue;
403            }
404            let col = ColumnId::from_usize(i);
405            if let Some(prev) = atom.var_columns.insert(var, col) {
406                atom.constraints.slow.push(Constraint::Eq {
407                    l_col: col,
408                    r_col: prev,
409                })
410            };
411            subatoms
412                .entry(var)
413                .or_insert_with(|| SubAtom::new(next_atom))
414                .vars
415                .push(col);
416        }
417        for (var, subatom) in subatoms {
418            self.query
419                .var_info
420                .get_mut(var)
421                .expect("all variables must be bound in current query")
422                .occurrences
423                .push(subatom);
424        }
425
426        // Add functional dependencies for this atom.
427        let get_var = |qe: &QueryEntry| match qe {
428            QueryEntry::Var(v) => Some(*v),
429            QueryEntry::Const(_) => None,
430        };
431        let antecedent = vars[..info.spec().n_keys]
432            .iter()
433            .filter_map(get_var)
434            .collect::<Vec<_>>();
435        let consequent = vars[info.spec().n_keys..]
436            .iter()
437            .filter_map(get_var)
438            .collect::<Vec<_>>();
439        self.query.fun_deps.add_dependency(antecedent, consequent);
440
441        Ok(self.query.atoms.push(atom))
442    }
443}

445#[derive(Debug, Error)]
446pub enum QueryError {
447    #[error("table {table:?} has {expected:?} keys but got {got:?}")]
448    KeyArityMismatch {
449        table: TableId,
450        expected: usize,
451        got: usize,
452    },
453    #[error("table {table:?} has {expected:?} columns but got {got:?}")]
454    TableArityMismatch {
455        table: TableId,
456        expected: usize,
457        got: usize,
458    },
459
460    #[error(
461        "counter used in column {column_id:?} of table {table:?}, which is declared as a base value"
462    )]
463    CounterUsedInBaseColumn {
464        table: TableId,
465        column_id: ColumnId,
466        base: BaseValueId,
467    },
468
469    #[error("attempt to compare two groups of values, one of length {l}, another of length {r}")]
470    MultiComparisonMismatch { l: usize, r: usize },
471
472    #[error("table {table:?} expected {expected:?} columns but got {got:?}")]
473    BadArity {
474        table: TableId,
475        expected: usize,
476        got: usize,
477    },
478
479    #[error("expected {expected:?} columns in schema but got {got:?}")]
480    InvalidSchema { expected: usize, got: usize },
481
482    #[error(
483        "constraint {constraint:?} on table {table:?} references column {column:?}, but the table has arity {arity:?}"
484    )]
485    InvalidConstraint {
486        constraint: Constraint,
487        column: usize,
488        table: TableId,
489        arity: usize,
490    },

500impl RuleBuilder<'_, '_> {
501    fn table_info(&self, table: TableId) -> &TableInfo {
502        self.qb.rsb.db.get_table_info(table)
503    }

505    /// Build the finished query.
506    pub fn build(self) -> RuleId {
507        self.build_with_description("")

510    fn build_symbol_map(&self) -> SymbolMap {
511        let var_info = &self.qb.query.var_info;
512        SymbolMap {
513            atoms: self
514                .qb
515                .query
516                .atoms
517                .iter()
518                .filter_map(|(id, atom)| {
519                    let name = self.table_info(atom.table).name.clone();
520                    name.map(|name| (id, name))
521                })
522                .collect(),
523            vars: var_info
524                .iter()
525                .filter_map(|(id, info)| info.name.as_ref().map(|name| (id, name.clone())))
526                .collect(),
527        }
528    }

530    pub fn build_with_description(mut self, desc: impl Into<String>) -> RuleId {
531        let var_info = &self.qb.query.var_info;
532        let symbol_map = self.build_symbol_map();
533        // Generate an id for our actions and slot them in.
534        let used_vars = SmallVec::from_iter(var_info.iter().filter_map(|(v, info)| {
535            if info.used_in_rhs && !info.defined_in_rhs {
536                Some(v)
537            } else {
538                None
539            }
540        }));
541        let action_id = self.qb.rsb.rule_set.actions.push(ActionInfo {
542            instrs: Arc::new(self.qb.instrs),
543            used_vars,
544        });
545        self.qb.query.action = action_id;
546        // Plan the query
547        let plan = self.qb.rsb.db.plan_query(self.qb.query);
548        let desc: String = desc.into();
549        // Add it to the ruleset.
550        self.qb
551            .rsb
552            .rule_set
553            .plans
554            .push((plan, desc.into(), symbol_map))
555    }

565    /// Return a variable containing the result of looking up the specified
566    /// column from the row corresponding to given keys in the given
567    /// table.
568    ///
569    /// If the key does not currently have a mapping in the table, the values
570    /// specified by `default_vals` will be inserted.
571    pub fn lookup_or_insert(
572        &mut self,
573        table: TableId,
574        args: &[QueryEntry],
575        default_vals: &[WriteVal],
576        dst_col: ColumnId,
577    ) -> Result<Variable, QueryError> {
578        let table_info = self.table_info(table);
579        self.validate_keys(table, table_info, args)?;
580        self.validate_vals(table, table_info, default_vals.iter())?;
581        let res = self.qb.new_var();
582        self.qb.instrs.push(Instr::LookupOrInsertDefault {
583            table,
584            args: args.to_vec(),
585            default: default_vals.to_vec(),
586            dst_col,
587            dst_var: res,
588        });
589        self.qb.mark_used(args);
590        self.qb
591            .mark_used(default_vals.iter().filter_map(|x| match x {
592                WriteVal::QueryEntry(qe) => Some(qe),
593                WriteVal::IncCounter(_) | WriteVal::CurrentVal(_) => None,
594            }));
595        self.qb.mark_defined(&res.into());
596        Ok(res)
597    }

702        func: ExternalFunctionId,
703        args: &[QueryEntry],
704    ) -> Result<Variable, QueryError> {
705        let res = self.qb.new_var();
706        self.qb.instrs.push(Instr::External {
707            func,
708            args: args.to_vec(),
709            dst: res,
710        });
711        self.qb.mark_used(args);
712        self.qb.mark_defined(&res.into());
713        Ok(res)
714    }
715
716    /// Look up the given key in the given table. If the lookup fails, then call the given external
717    /// function with the given arguments. Bind the result to the returned variable. If the
718    /// external function returns None (and the lookup fails) then the execution of the rule halts.
719    pub fn lookup_with_fallback(
720        &mut self,
721        table: TableId,
722        key: &[QueryEntry],
723        dst_col: ColumnId,
724        func: ExternalFunctionId,
725        func_args: &[QueryEntry],
726    ) -> Result<Variable, QueryError> {
727        let table_info = self.table_info(table);
728        self.validate_keys(table, table_info, key)?;
729        let res = self.qb.new_var();
730        self.qb.instrs.push(Instr::LookupWithFallback {
731            table,
732            table_key: key.to_vec(),
733            func,
734            func_args: func_args.to_vec(),
735            dst_var: res,
736            dst_col,
737        });
738        self.qb.mark_used(key);
739        self.qb.mark_used(func_args);
740        self.qb.mark_defined(&res.into());
741        Ok(res)
742    }
743
744    pub fn call_external_with_fallback(
745        &mut self,
746        f1: ExternalFunctionId,
747        args1: &[QueryEntry],
748        f2: ExternalFunctionId,
749        args2: &[QueryEntry],
750    ) -> Result<Variable, QueryError> {
751        let res = self.qb.new_var();
752        self.qb.instrs.push(Instr::ExternalWithFallback {
753            f1,
754            args1: args1.to_vec(),
755            f2,
756            args2: args2.to_vec(),
757            dst: res,
758        });
759        self.qb.mark_used(args1);
760        self.qb.mark_used(args2);
761        self.qb.mark_defined(&res.into());
762        Ok(res)
763    }
764
765    /// Continue execution iff the two arguments are equal.
766    pub fn assert_eq(&mut self, l: QueryEntry, r: QueryEntry) {
767        self.qb.instrs.push(Instr::AssertEq(l, r));
768        self.qb.mark_used(&[l, r]);
769    }
770
771    /// Continue execution iff the two arguments are not equal.
772    pub fn assert_ne(&mut self, l: QueryEntry, r: QueryEntry) -> Result<(), QueryError> {
773        self.qb.instrs.push(Instr::AssertNe(l, r));
774        self.qb.mark_used(&[l, r]);
775        Ok(())
776    }
777
778    /// Continue execution iff there is some `i` such that `l[i] != r[i]`.
779    ///
780    /// This is useful when doing egglog-style rebuilding.
781    pub fn assert_any_ne(&mut self, l: &[QueryEntry], r: &[QueryEntry]) -> Result<(), QueryError> {
782        if l.len() != r.len() {
783            return Err(QueryError::MultiComparisonMismatch {
784                l: l.len(),
785                r: r.len(),
786            });
787        }
788
789        let mut ops = Vec::with_capacity(l.len() + r.len());
790        ops.extend_from_slice(l);
791        ops.extend_from_slice(r);
792        self.qb.instrs.push(Instr::AssertAnyNe {
793            ops,
794            divider: l.len(),
795        });
796        self.qb.mark_used(l);
797        self.qb.mark_used(r);
798        Ok(())
799    }
800
801    fn validate_row(
802        &self,
803        table: TableId,
804        info: &TableInfo,
805        vals: &[QueryEntry],
806    ) -> Result<(), QueryError> {
807        if vals.len() != info.spec.arity() {
808            Err(QueryError::TableArityMismatch {
809                table,
810                expected: info.spec.arity(),
811                got: vals.len(),
812            })
813        } else {
814            Ok(())
815        }
816    }
817
818    fn validate_keys(
819        &self,
820        table: TableId,
821        info: &TableInfo,
822        keys: &[QueryEntry],
823    ) -> Result<(), QueryError> {
824        if keys.len() != info.spec.n_keys {
825            Err(QueryError::KeyArityMismatch {
826                table,
827                expected: info.spec.n_keys,
828                got: keys.len(),
829            })
830        } else {
831            Ok(())
832        }
833    }
834
835    fn validate_vals<'b>(
836        &self,
837        table: TableId,
838        info: &TableInfo,
839        vals: impl Iterator<Item = &'b WriteVal>,
840    ) -> Result<(), QueryError> {
841        for (i, _) in vals.enumerate() {
842            let col = i + info.spec.n_keys;
843            if col >= info.spec.arity() {
844                return Err(QueryError::TableArityMismatch {
845                    table,
846                    expected: info.spec.arity(),
847                    got: col,
848                });
849            }
850        }
851        Ok(())
852    }
853}

931/// A functional dependency inferencer.
932///
933/// A functional dependency (x, y, ...) -> (u, v, ...) means that if we know
934/// the values of x, y, ..., then we can determine u, v, ...
935///
936/// This data structure can compute the closure of a set of variables under
937/// a set of functional dependencies.
938#[derive(Clone, Default)]
939pub(crate) struct FunDeps {
940    /// List of functional dependencies (antecedent -> consequent)
941    dependencies: Vec<(Vec<Variable>, Vec<Variable>)>,
942}
943
944impl FunDeps {
945    /// Add a functional dependency: antecedent -> consequent.
946    pub fn add_dependency(&mut self, antecedent: Vec<Variable>, consequent: Vec<Variable>) {
947        // Don't add trivial dependencies.
948        if !antecedent.is_empty() {
949            self.dependencies.push((antecedent, consequent));
950        }
951    }
952
953    /// Returns all variables that can be determined from the input variables
954    /// using the functional dependencies.
955    pub fn closure(
956        &self,
957        variables: impl IntoIterator<Item = Variable>,
958    ) -> DenseIdMap<Variable, ()> {
959        let mut result: DenseIdMap<Variable, ()> =
960            DenseIdMap::from_iter(variables.into_iter().map(|v| (v, ())));
961        let mut changed = true;
962
963        while changed {
964            changed = false;
965            for (antecedent, consequent) in &self.dependencies {
966                // If all variables in the antecedent are in the result,
967                // add all variables in the consequent.
968                if antecedent.iter().all(|v| result.contains_key(*v)) {
969                    for v in consequent {
970                        if !result.contains_key(*v) {
971                            result.insert(*v, ());
972                            changed = true;
973                        }
974                    }
975                }
976            }
977        }
978
979        result
980    }
981}
982
983impl std::fmt::Debug for FunDeps {
984    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
985        use std::fmt::Write;
986
987        let mut deps = String::new();
988
989        for (i, (ant, cons)) in self.dependencies.iter().enumerate() {
990            if i > 0 {
991                deps.push_str("; ");
992            }
993
994            deps.push('{');
995            for (j, v) in ant.iter().enumerate() {
996                if j > 0 {
997                    deps.push_str(", ");
998                }
999                write!(&mut deps, "{v:?}")?;
1000            }
1001            deps.push('}');
1002
1003            deps.push_str(" -> ");
1004
1005            deps.push('{');
1006            for (j, v) in cons.iter().enumerate() {
1007                if j > 0 {
1008                    deps.push_str(", ");
1009                }
1010                write!(&mut deps, "{v:?}")?;
1011            }
1012            deps.push('}');
1013        }
1014
1015        write!(f, "FunDeps {{ {deps} }}")
1016    }
1017}
