1//! Basic utilities around sharding a hashbrown `HashTable`.
2
3use crate::numeric_id::NumericId;
4use hashbrown::HashTable;
5
6use crate::common::{ShardData, ShardId};
7
8#[derive(Clone)]
9pub(crate) struct ShardedHashTable<T> {
10    shard_data: ShardData,
11    shards: Vec<HashTable<T>>,
12}
13
14impl<T> Default for ShardedHashTable<T> {
15    fn default() -> Self {
16        let cur_threads = rayon::current_num_threads();
17        if cur_threads == 1 {
18            Self::with_shards(1)
19        } else {
20            Self::with_shards(cur_threads * 2)
21        }
22    }
23}
24
25impl<T> ShardedHashTable<T> {
26    pub(crate) fn clear(&mut self) {
27        self.shards.iter_mut().for_each(|s| s.clear());
28    }
29    pub(crate) fn with_shards(shards: usize) -> Self {
30        let shard_data = ShardData::new(shards);
31        let shards = (0..shard_data.n_shards())
32            .map(|_| HashTable::new())
33            .collect::<Vec<_>>();
34        Self { shard_data, shards }
35    }
36
37    /// Extract a [`ShardData`] allowing users to compute shard information for
38    /// this table.
39    pub(crate) fn shard_data(&self) -> ShardData {
40        self.shard_data
41    }
42
43    pub(crate) fn get_shard(&self, shard_id: ShardId) -> &HashTable<T> {
44        &self.shards[shard_id.index()]
45    }
46
47    pub(crate) fn mut_shards(&mut self) -> &mut [HashTable<T>] {
48        &mut self.shards
49    }
50}
