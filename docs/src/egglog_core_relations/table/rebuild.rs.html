20// Helper macro used for adjusting sort before inserting to a mutation buffer.
21macro_rules! insert_row {
22    ($this: expr, $mutation_buf: expr, $row:expr, $next_ts:expr) => {{
23        let row = $row;
24        let this = &*$this;
25        let next_ts = $next_ts;
26        if let Some(sort_by) = this.sort_by {
27            row[sort_by.index()] = next_ts;
28        }
29        $mutation_buf.stage_insert(row);
30    }};
31}

33impl SortedWritesTable {
34    fn refresh_rebuild_index(&mut self) {
35        let mut index = mem::replace(
36            &mut self.rebuild_index,
37            Index::new(vec![], ColumnIndex::new()),
38        );
39        WrappedTableRef::with_wrapper(self, |wrapped| {
40            index.refresh(wrapped);
41        });
42        self.rebuild_index = index;
43    }
44
45    pub(super) fn do_rebuild(
46        &mut self,
47        table_id: TableId,
48        table: &WrappedTable,
49        next_ts: Value,
50        exec_state: &mut ExecutionState,
51    ) -> bool {
52        if self.to_rebuild.is_empty() {
53            return false;
54        }
55        let Some(rebuilder) = table.rebuilder(&self.to_rebuild) else {
56            return false;
57        };
58        // First, decide whether to do an incremental or full rebuild.
59        if let Some(hint_col) = rebuilder.hint_col() {
60            // Incremental rebuilds are possible if we can scan the subset of the columns that are
61            // relevant.
62            let to_scan = self.subset_tracker.recent_updates(table_id, table);
63            if incremental_rebuild(
64                to_scan.size(),
65                self.data.next_row().index(),
66                parallelize_rebuild(to_scan.size()),
67            ) {
68                self.rebuild_incremental(table, &*rebuilder, hint_col, to_scan, next_ts, exec_state)
69            } else {
70                self.rebuild_nonincremental(&*rebuilder, next_ts, exec_state)
71            }
72        } else {
73            self.rebuild_nonincremental(&*rebuilder, next_ts, exec_state)
74        }
75    }

77    pub(super) fn refresh_rows_for_values(&mut self, dirty_ids: &[Value], next_ts: Value) -> bool {
78        if dirty_ids.is_empty() || self.to_rebuild.is_empty() {
79            return false;
80        }
81        // Reuse the rebuild index to find rows whose rebuildable columns mention
82        // one of the same-id dirty container ids.
83        self.refresh_rebuild_index();
84
85        let mut candidate_rows = HashSet::<RowId>::default();
86        for value in dirty_ids {
87            let Some(subset) = self.rebuild_index.get_subset(value) else {
88                continue;
89            };
90            subset.offsets(|row_id| {
91                candidate_rows.insert(row_id);
92            });
93        }
94
95        if candidate_rows.is_empty() {
96            return false;
97        }
98
99        let mut changed = false;
100        let mut mutation_buf = self.new_buffer();
101        let mut refreshed_row = Vec::<Value>::with_capacity(self.n_columns);
102        for row_id in candidate_rows {
103            let Some(current_row) = self.data.get_row(row_id) else {
104                continue;
105            };
106            // Preserve the logical row and only advance its sort/timestamp
107            // column, so seminaive treats this as a fresh parent-row delta.
108            mutation_buf.stage_remove(&current_row[0..self.n_keys]);
109            refreshed_row.clear();
110            refreshed_row.extend_from_slice(current_row);
111            if let Some(sort_by) = self.sort_by {
112                refreshed_row[sort_by.index()] = next_ts;
113            }
114            mutation_buf.stage_insert(&refreshed_row);
115            changed = true;
116        }
117        changed
118    }

120    fn rebuild_incremental(
121        &mut self,
122        table: &WrappedTable,
123        rebuilder: &dyn Rebuilder,
124        search_col: ColumnId,
125        to_scan: Subset,
126        next_ts: Value,
127        exec_state: &mut ExecutionState,
128    ) -> bool {
129        self.refresh_rebuild_index();
130        let mut buf = TaggedRowBuffer::new(1);
131        table.scan_project(
132            to_scan.as_ref(),
133            &[search_col],
134            Offset::new(0),
135            usize::MAX,
136            &[],
137            &mut buf,
138        );
139
140        if parallelize_rebuild(to_scan.size()) {
141            WrappedTableRef::with_wrapper(self, |wrapped| {
142                buf.par_iter()
143                    .fold(
144                        || (self.new_buffer(), exec_state.clone(), false),
145                        |(mut mutation_buf, mut exec_state, mut changed), (_, row)| {
146                            let Some(subset) = self.rebuild_index.get_subset(&row[0]) else {
147                                return (mutation_buf, exec_state, changed);
148                            };
149                            let mut scanned = TaggedRowBuffer::new(self.n_columns);
150                            rebuilder.rebuild_subset(
151                                wrapped,
152                                subset,
153                                &mut scanned,
154                                &mut exec_state,
155                            );
156                            for (row_id, row) in scanned.non_stale_mut() {
157                                let to_remove =
158                                    self.data.get_row(row_id).map(|x| &x[0..self.n_keys]);
159                                if let Some(key) = to_remove {
160                                    mutation_buf.stage_remove(key);
161                                }
162                                changed = true;
163                                insert_row!(self, mutation_buf, row, next_ts);
164                            }
165                            (mutation_buf, exec_state, changed)
166                        },
167                    )
168                    .map(|(_, _, changed)| changed)
169                    .max()
170                    .unwrap_or(false)
171            })
172        } else {
173            let mut scratch = TaggedRowBuffer::new(self.n_columns);
174            let mut changed = false;
175            for (_, id) in buf.iter() {
176                let Some(subset) = self.rebuild_index.get_subset(&id[0]) else {
177                    continue;
178                };
179                WrappedTableRef::with_wrapper(self, |wrapped| {
180                    rebuilder.rebuild_subset(wrapped, subset, &mut scratch, exec_state);
181                });
182                changed |= subset.size() > 0;
183            }
184            if !scratch.is_empty() {
185                let mut write_buf = self.new_buffer();
186                for (row_id, row) in scratch.non_stale_mut() {
187                    if let Some(to_remove) = self.data.get_row(row_id).map(|x| &x[0..self.n_keys]) {
188                        write_buf.stage_remove(to_remove);
189                    }
190                    insert_row!(self, write_buf, row, next_ts);
191                }
192            }
193            changed
194        }
195    }

197    fn rebuild_nonincremental(
198        &mut self,
199        rebuilder: &dyn Rebuilder,
200        next_ts: Value,
201        exec_state: &mut ExecutionState,
202    ) -> bool {
203        const STEP_SIZE: usize = 2048;
204        if parallelize_rebuild(self.data.next_row().index()) {
205            (0..self.data.next_row().index())
206                .into_par_iter()
207                .step_by(STEP_SIZE)
208                .fold(
209                    || {
210                        (
211                            self.new_buffer(),
212                            TaggedRowBuffer::new(self.n_columns),
213                            exec_state.clone(),
214                            false,
215                        )
216                    },
217                    |(mut mutation_buf, mut buf, mut exec_state, mut changed), start| {
218                        rebuilder.rebuild_buf(
219                            &self.data.data,
220                            RowId::from_usize(start),
221                            RowId::from_usize(cmp::min(
222                                start + STEP_SIZE,
223                                self.data.next_row().index(),
224                            )),
225                            &mut buf,
226                            &mut exec_state,
227                        );
228                        for (row_id, row) in buf.non_stale_mut() {
229                            let to_remove = self.data.get_row(row_id).map(|x| &x[0..self.n_keys]);
230                            changed = true;
231                            if let Some(key) = to_remove {
232                                mutation_buf.stage_remove(key);
233                            }
234                            insert_row!(self, mutation_buf, row, next_ts);
235                        }
236                        buf.clear();
237                        (mutation_buf, buf, exec_state, changed)
238                    },
239                )
240                .map(|(_, _, _, changed)| changed)
241                .max()
242                .unwrap_or(false)
243        } else {
244            let mut buf = TaggedRowBuffer::new(self.n_columns);
245            let mut changed = false;
246
247            let max_row = self.data.next_row().index();
248            for start in (0..max_row).step_by(STEP_SIZE) {
249                rebuilder.rebuild_buf(
250                    &self.data.data,
251                    RowId::from_usize(start),
252                    RowId::from_usize(cmp::min(start + STEP_SIZE, max_row)),
253                    &mut buf,
254                    exec_state,
255                );
256            }
257            if !buf.is_empty() {
258                let mut write_buf = self.new_buffer();
259                for (row_id, row) in buf.non_stale_mut() {
260                    if let Some(to_remove) = self.data.get_row(row_id).map(|x| &x[0..self.n_keys]) {
261                        write_buf.stage_remove(to_remove);
262                    }
263                    insert_row!(self, write_buf, row, next_ts);
264                    changed = true;
265                }
266            }
267            changed
268        }
269    }
270}
271
272fn incremental_rebuild(uf_size: usize, table_size: usize, parallel: bool) -> bool {
273    if parallel {
274        table_size > 10_000 && uf_size * 8192 <= table_size
275    } else {
276        table_size > 10000 && uf_size * 8 <= table_size
277    }
