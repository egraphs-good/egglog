1use super::*;
2
3/// 64-bit floating point numbers supporting these primitives:
4/// - Arithmetic: `+`, `-`, `*`, `/`, `%`, `^`, `neg`, `abs`
5/// - Comparisons: `<`, `>`, `<=`, `>=`
6/// - Other: `min`, `max`, `to-i64`, `to-string`
7#[derive(Debug)]
8pub struct F64Sort;
9
10impl BaseSort for F64Sort {
11    type Base = F;
12
13    fn name(&self) -> &str {
14        "f64"
15    }
16
17    #[rustfmt::skip]
18    // We need the closure for division and mod operations, as they can panic.
19    // cf https://github.com/rust-lang/rust-clippy/issues/9422
20    #[allow(clippy::unnecessary_lazy_evaluations)]
21    fn register_primitives(&self, eg: &mut EGraph) {
22        add_literal_prim!(eg, "+" = |a: F, b: F| -> F { a + b });
23        add_literal_prim!(eg, "-" = |a: F, b: F| -> F { a - b });
24        add_literal_prim!(eg, "*" = |a: F, b: F| -> F { a * b });
25        add_literal_prim!(eg, "/" = |a: F, b: F| -?> F { (*b != 0.0).then(|| a / b) });
26        add_literal_prim!(eg, "%" = |a: F, b: F| -?> F { (*b != 0.0).then(|| a % b) });
27        add_literal_prim!(eg, "^" = |a: F, b: F| -> F { F::from(OrderedFloat(a.powf(**b))) });
28        add_literal_prim!(eg, "neg" = |a: F| -> F { -a });
29
30        add_literal_prim!(eg, "<" = |a: F, b: F| -?> () { (a < b).then(|| ()) });
31        add_literal_prim!(eg, ">" = |a: F, b: F| -?> () { (a > b).then(|| ()) });
32        add_literal_prim!(eg, "<=" = |a: F, b: F| -?> () { (a <= b).then(|| ()) });
33        add_literal_prim!(eg, ">=" = |a: F, b: F| -?> () { (a >= b).then(|| ()) });
34
35        add_literal_prim!(eg, "min" = |a: F, b: F| -> F { a.min(b) });
36        add_literal_prim!(eg, "max" = |a: F, b: F| -> F { a.max(b) });
37        add_literal_prim!(eg, "abs" = |a: F| -> F { F::from(a.abs()) });
38
39        // `to-f64` should be in `i64.rs`, but `F64Sort` wouldn't exist yet
40        add_literal_prim!(eg, "to-f64" = |a: i64| -> F { F::from(OrderedFloat(a as f64)) });
41        add_literal_prim!(eg, "to-i64" = |a: F| -> i64 { a.0.0 as i64 });
42        // Use debug instead of to_string so that decimal place is always printed
43        add_literal_prim!(eg, "to-string" = |a: F| -> S { S::new(format!("{:?}", a.0.0)) });
44    }
45
46    fn reconstruct_termdag(
47        &self,
48        base_values: &BaseValues,
49        value: Value,
50        termdag: &mut TermDag,
51    ) -> TermId {
52        let f = base_values.unwrap::<F>(value);
53
54        termdag.lit(Literal::Float(f.0))
55    }
56}
