5#[derive(Clone, Debug, PartialEq, Eq, Hash)]
6pub struct MultiSetContainer {
7    pub do_rebuild: bool,
8    pub data: MultiSet<Value>,
9}
10
11impl ContainerValue for MultiSetContainer {
12    fn rebuild_contents(&mut self, rebuilder: &dyn Rebuilder) -> bool {
13        // If the contents are an eq-sort then we want to rebuild
14        if self.do_rebuild {
15            let mut xs: Vec<_> = self.data.iter().copied().collect();
16            let changed = rebuilder.rebuild_slice(&mut xs);
17            self.data = xs.into_iter().collect();
18            changed
19        // if the contents are just a primitive then don't need to do anything.
20        } else {
21            false
22        }
23    }
24    fn iter(&self) -> impl Iterator<Item = Value> + '_ {
25        self.data.iter().copied()
26    }
27}

127    fn register_primitives(&self, eg: &mut EGraph) {
128        let arc = self.clone().to_arcsort();
129
130        add_primitive!(eg, "multiset-of" = {self.clone(): MultiSetSort} [xs: # (self.element())] -> @MultiSetContainer (arc) { MultiSetContainer {
131            do_rebuild: self.ctx.is_eq_container_sort(),
132            data: xs.collect()
133        } });
134
135        add_primitive!(eg, "multiset-single" = {self.clone(): MultiSetSort} |x: # (self.element()), i: i64| -?> @MultiSetContainer (arc) {
136            i.try_into().ok().map(|i|
137                MultiSetContainer {
138                do_rebuild: self.ctx.is_eq_container_sort(),
139                data: std::iter::repeat_n(x, i).collect()
140            })
141        });
142        add_primitive!(eg, "multiset-pick" = |xs: @MultiSetContainer (arc)| -> # (self.element()) { *xs.data.pick().expect("Cannot pick from an empty multiset") });
143        add_primitive!(eg, "multiset-insert" = |mut xs: @MultiSetContainer (arc), x: # (self.element())| -> @MultiSetContainer (arc) { MultiSetContainer { data: xs.data.insert( x) , ..xs } });
144        add_primitive!(eg, "multiset-remove" = |mut xs: @MultiSetContainer (arc), x: # (self.element())| -?> @MultiSetContainer (arc) { Some(MultiSetContainer { data: xs.data.remove(&x)?, ..xs } )});
145        add_primitive!(eg, "multiset-remove-swapped" = |x: # (self.element()), mut xs: @MultiSetContainer (arc)| -?> @MultiSetContainer (arc) { Some(MultiSetContainer { data: xs.data.remove(&x)?, ..xs }) });
146        add_primitive!(eg, "multiset-subtract" = |mut xs: @MultiSetContainer (arc), other: @MultiSetContainer (arc)| -?> @MultiSetContainer (arc) { Some(MultiSetContainer { data: xs.data.subtract(&other.data)?, ..xs }) });
147        add_primitive!(eg, "multiset-subtract-swapped" = |other: @MultiSetContainer (arc), mut xs: @MultiSetContainer (arc)| -?> @MultiSetContainer (arc) { Some(MultiSetContainer { data: xs.data.subtract(&other.data)?, ..xs }) });
148        add_primitive!(eg, "multiset-length"       = |xs: @MultiSetContainer (arc)| -> i64 { xs.data.len() as i64 });
149        add_primitive!(eg, "multiset-contains"     = |xs: @MultiSetContainer (arc), x: # (self.element())| -?> () { ( xs.data.contains(&x)).then_some(()) });
150        add_primitive!(eg, "multiset-not-contains" = |xs: @MultiSetContainer (arc), x: # (self.element())| -?> () { (!xs.data.contains(&x)).then_some(()) });
151        add_primitive!(eg, "multiset-contains-swapped" = |x: # (self.element()), xs: @MultiSetContainer (arc)| -?> () { (xs.data.contains(&x)).then_some(()) });
152        add_primitive!(eg, "multiset-not-contains-swapped" = |x: # (self.element()), xs: @MultiSetContainer (arc)| -?> () { (!xs.data.contains(&x)).then_some(()) });
153        add_primitive!(eg, "multiset-intersection" = |xs: @MultiSetContainer (arc), ys: @MultiSetContainer (arc)| -> @MultiSetContainer (arc) { MultiSetContainer { data: xs.data.intersection(ys.data), ..xs } });
154        add_primitive!(eg, "multiset-sum" = |xs: @MultiSetContainer (arc), ys: @MultiSetContainer (arc)| -> @MultiSetContainer (arc) { MultiSetContainer { data: xs.data.sum(ys.data), ..xs } });
155        // Set counts to one
156        add_primitive!(eg, "multiset-reset-counts" = |mut xs: @MultiSetContainer (arc)| -> @MultiSetContainer (arc) { {
157            let mut new_data = MultiSet::<Value>::new();
158            for (v, _) in xs.data.iter_counts() {
159                new_data.insert_multiple_mut(v, 1);
160            }
161            MultiSetContainer { data: new_data, ..xs }
162        }});
163        add_primitive!(eg, "multiset-pick-max" = |xs: @MultiSetContainer (arc)| -?>  # (self.element()) {
164            Some(xs.data.iter_counts().max_by_key(|(_, c)| *c)?.0)
165        });
166        add_primitive!(eg, "multiset-count" = |xs: @MultiSetContainer (arc), x: # (self.element())| -> i64 {
167            xs.data.iter_counts().find(|(v, _)| *v == x).map(|(_, c)| c as i64).unwrap_or(0)
168        });
169
170        // Add multiset-sum-multisets if the inner arcsort is also a multiset
171        for other_multiset_sort in eg.type_info.get_arcsorts_by(|f| {
172            f.name() == self.element.name()
173            // We can't query directly by arcsort type since it's wrapped in a ContainerSort which is not public
174                && f.value_type() == Some(TypeId::of::<MultiSetContainer>())
175        }) {
176            eg.add_primitive(SumMultisets {
177                name: "multiset-sum-multisets".into(),
178                multiset: other_multiset_sort.clone(),
179                multiset_of_multisets: arc.clone(),
180            });
181        }
182        let all_ms_sorts = eg
183            .type_info
184            .get_arcsorts_by(|f| f.value_type() == Some(TypeId::of::<MultiSetContainer>()));
185        for fn_sort in eg.type_info.get_sorts::<FunctionSort>() {
186            for ms_sort in &all_ms_sorts {
187                try_registering_multiset_map(eg, fn_sort.clone(), ms_sort.clone(), arc.clone());
188                if ms_sort.name() != arc.name() {
189                    try_registering_multiset_map(eg, fn_sort.clone(), arc.clone(), ms_sort.clone());
190                }
191            }
192            try_registering_multiset_non_map_primitives(eg, fn_sort.clone(), arc.clone());
193        }
194        if self.element.is_eq_sort() {
195            eg.add_primitive(UnionValues {
196                name: "multiset-union-values".into(),
197                multiset: arc.clone(),
198                action: eg.new_union_action(),
199                element: self.element.clone(),
200            });
201        }
202    }

220 * Register a multiset map primitive if the function matches the input and output multiset.
221 */
222pub(crate) fn try_registering_multiset_map(
223    eg: &mut EGraph,
224    fn_: Arc<FunctionSort>,
225    input_ms: ArcSort,
226    output_ms: ArcSort,
227) {
228    if fn_.inputs().len() != 1
229        || fn_.inputs()[0].name() != input_ms.inner_sorts()[0].name()
230        || fn_.output().name() != output_ms.inner_sorts()[0].name()
231    {
232        return;
233    }
234    eg.add_primitive(Map {
235        name: "unstable-multiset-map".into(),
236        multiset: input_ms,
237        output_multiset: output_ms,
238        fn_: fn_.clone(),
239    });
240}

242pub(crate) fn register_multiset_primitives_for_function(eg: &mut EGraph, fn_: Arc<FunctionSort>) {
243    let all_ms_sorts = eg
244        .type_info
245        .get_arcsorts_by(|f| f.value_type() == Some(TypeId::of::<MultiSetContainer>()));
246    for input_ms in &all_ms_sorts {
247        for output_ms in &all_ms_sorts {
248            try_registering_multiset_map(eg, fn_.clone(), input_ms.clone(), output_ms.clone());
249        }
250    }
251    for ms_sort in &all_ms_sorts {
252        try_registering_multiset_non_map_primitives(eg, fn_.clone(), ms_sort.clone());
253    }
254}

256fn try_registering_multiset_non_map_primitives(
257    eg: &mut EGraph,
258    fn_: Arc<FunctionSort>,
259    multiset: ArcSort,
260) {
261    let element = multiset.inner_sorts()[0].clone();
262    let element_name = element.name();
263
264    if fn_.inputs().len() == 1
265        && fn_.inputs()[0].name() == element_name
266        && fn_.output().name() == "Unit"
267    {
268        eg.add_primitive(Filter {
269            name: "unstable-multiset-filter".into(),
270            multiset: multiset.clone(),
271            fn_: fn_.clone(),
272            skip_empty: true,
273        });
274        eg.add_primitive(Filter {
275            name: "unstable-multiset-filter-not".into(),
276            multiset: multiset.clone(),
277            fn_: fn_.clone(),
278            skip_empty: false,
279        });
280    }
281
282    if fn_.inputs().len() == 2
283        && fn_.inputs()[0].name() == element_name
284        && fn_.inputs()[1].name() == element_name
285        && fn_.output().name() == element_name
286    {
287        eg.add_primitive(Reduce {
288            name: "unstable-multiset-reduce".into(),
289            multiset: multiset.clone(),
290            fn_: fn_.clone(),
291            element: element.clone(),
292        });
293    }
294
295    if fn_.inputs().len() == 2
296        && fn_.inputs()[0].name() == multiset.name()
297        && fn_.inputs()[1].name() == element_name
298        && fn_.output().name() == "i64"
299    {
300        let unit = eg.type_info.get_sort_by_name("Unit").unwrap().clone();
301        eg.add_primitive(FillIndex {
302            name: "unstable-multiset-fill-index".into(),
303            multiset: multiset.clone(),
304            unit: unit.clone(),
305            fn_: fn_.clone(),
306        });
307        eg.add_primitive(ClearIndex {
308            name: "unstable-multiset-clear-index".into(),
309            multiset: multiset.clone(),
310            unit,
311            fn_: fn_.clone(),
312        });
313    }
314
315    if fn_.inputs().len() == 1
316        && fn_.inputs()[0].name() == element_name
317        && fn_.output().name() == multiset.name()
318    {
319        eg.add_primitive(FlatMap {
320            name: "unstable-multiset-flat-map".into(),
321            multiset,
322            fn_: fn_.clone(),
323        });
324    }
325}

327#[derive(Clone)]
328struct Map {
329    name: String,
330    multiset: ArcSort,
331    fn_: Arc<FunctionSort>,
332    output_multiset: ArcSort,
333}
334
335impl Primitive for Map {
336    fn name(&self) -> &str {
337        &self.name
338    }
339
340    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
341        SimpleTypeConstraint::new(
342            self.name(),
343            vec![
344                self.fn_.clone(),
345                self.multiset.clone(),
346                self.output_multiset.clone(),
347            ],
348            span.clone(),
349        )
350        .into_box()
351    }
352
353    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
354        let fc = exec_state
355            .container_values()
356            .get_val::<FunctionContainer>(args[0])
357            .unwrap()
358            .clone();
359        let multiset = exec_state
360            .container_values()
361            .get_val::<MultiSetContainer>(args[1])
362            .unwrap()
363            .clone();
364        let mut new_data = MultiSet::<Value>::new();
365        // Filter out any elements which do not have the function defined for them
366        for (v, c) in multiset.data.iter_counts() {
367            let mapped = fc.apply(exec_state, &[v]);
368            if let Some(mapped_v) = mapped {
369                new_data.insert_multiple_mut(mapped_v, c);
370            }
371        }
372        let multiset = MultiSetContainer {
373            data: new_data,
374            ..multiset
375        };
376        Some(
377            exec_state
378                .clone()
379                .container_values()
380                .register_val(multiset, exec_state),
381        )
382    }
383}

385// (unstable-multiset-fill-index ms: MultiSet[X] index_fn: [MultiSet[X], X] -> i64) -> Unit
386// will set the index function for all elements in the multiset
387#[derive(Clone)]
388struct FillIndex {
389    name: String,
390    multiset: ArcSort,
391    unit: ArcSort,
392    fn_: Arc<FunctionSort>,
393}
394
395impl Primitive for FillIndex {
396    fn name(&self) -> &str {
397        &self.name
398    }
399
400    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
401        SimpleTypeConstraint::new(
402            self.name(),
403            vec![self.multiset.clone(), self.fn_.clone(), self.unit.clone()],
404            span.clone(),
405        )
406        .into_box()
407    }
408
409    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
410        let fc = exec_state
411            .container_values()
412            .get_val::<FunctionContainer>(args[1])
413            .unwrap()
414            .clone();
415        let multiset = exec_state
416            .container_values()
417            .get_val::<MultiSetContainer>(args[0])
418            .unwrap()
419            .clone();
420        let ResolvedFunctionId::Lookup(action) = fc.0 else {
421            panic!(
422                "Primitive functions cannot be used with unstable-multiset-fill-index, since they cannot be set"
423            );
424        };
425        for (v, c) in multiset.data.iter_counts() {
426            let mut row = vec![args[0], v];
427            // If we have already filled this multiset once, skip since it should still be accurate with the right
428            // merge function
429            if action.lookup(exec_state, &row).is_some() {
430                break;
431            }
432            row.push(exec_state.base_values().get::<i64>(c.try_into().unwrap()));
433            action.insert(exec_state, row.into_iter());
434        }
435        Some(exec_state.base_values().get::<()>(()))
436    }
437}

486// (unstable-multiset-flat-map (MultiSet[X], [X] -> MultiSet[X]) -> MultiSet[X])
487// will map the function over all elements in the multiset and flatten the result. Any element in the multiset
488// which does not have the function defined for it will be kept as-is.
489#[derive(Clone)]
490struct FlatMap {
491    name: String,
492    multiset: ArcSort,
493    fn_: Arc<FunctionSort>,
494}
495
496impl Primitive for FlatMap {
497    fn name(&self) -> &str {
498        &self.name
499    }
500
501    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
502        SimpleTypeConstraint::new(
503            self.name(),
504            vec![
505                self.fn_.clone(),
506                self.multiset.clone(),
507                self.multiset.clone(),
508            ],
509            span.clone(),
510        )
511        .into_box()
512    }
513
514    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
515        let fc = exec_state
516            .container_values()
517            .get_val::<FunctionContainer>(args[0])
518            .unwrap()
519            .clone();
520        let multiset = exec_state
521            .container_values()
522            .get_val::<MultiSetContainer>(args[1])
523            .unwrap()
524            .clone();
525        let mut new_data = MultiSet::<Value>::new();
526        for (v, c) in multiset.data.iter_counts() {
527            let mapped = fc.apply(exec_state, &[v]);
528            if let Some(mapped_ms) = mapped {
529                let mapped_ms = exec_state
530                    .container_values()
531                    .get_val::<MultiSetContainer>(mapped_ms)
532                    .unwrap();
533                for (mv, mc) in mapped_ms.data.iter_counts() {
534                    new_data.insert_multiple_mut(mv, c * mc);
535                }
536            } else {
537                new_data.insert_multiple_mut(v, c);
538            }
539        }
540        let new_container = MultiSetContainer {
541            data: new_data,
542            ..multiset
543        };
544        Some(
545            exec_state
546                .clone()
547                .container_values()
548                .register_val(new_container, exec_state),
549        )
550    }
551}

614// (multiset-sum-multisets (MultiSet[MultiSet[X]]) -> MultiSet[X])
615// will sum all multisets in the outer multiset into a single multiset
616
617#[derive(Clone)]
618struct SumMultisets {
619    name: String,
620    multiset: ArcSort,
621    multiset_of_multisets: ArcSort,
622}
623
624impl Primitive for SumMultisets {
625    fn name(&self) -> &str {
626        &self.name
627    }
628
629    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
630        SimpleTypeConstraint::new(
631            self.name(),
632            vec![self.multiset_of_multisets.clone(), self.multiset.clone()],
633            span.clone(),
634        )
635        .into_box()
636    }
637
638    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
639        let mut data = MultiSet::<Value>::new();
640        let ms_of_ms = exec_state
641            .container_values()
642            .get_val::<MultiSetContainer>(args[0])
643            .unwrap()
644            .clone();
645        for (ms_value, counts) in ms_of_ms.data.iter_counts() {
646            let ms = exec_state
647                .container_values()
648                .get_val::<MultiSetContainer>(ms_value)
649                .unwrap();
650            for (v, c) in ms.data.iter_counts() {
651                data.insert_multiple_mut(v, c * counts);
652            }
653        }
654        let multiset = MultiSetContainer {
655            data,
656            do_rebuild: self.multiset.is_eq_container_sort(),
657        };
658        Some(
659            exec_state
660                .clone()
661                .container_values()
662                .register_val(multiset, exec_state),
663        )
664    }
665}

667// (unstable-multiset-reduce ([X, X] -> X, X, MultiSet[X]) -> X
668// will reduce the multiset using the provided binary function and initial value
669// Won't use the initial value if the multiset is non-empty.
670#[derive(Clone)]
671struct Reduce {
672    name: String,
673    multiset: ArcSort,
674    fn_: Arc<FunctionSort>,
675    element: ArcSort,
676}
677
678impl Primitive for Reduce {
679    fn name(&self) -> &str {
680        &self.name
681    }
682
683    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
684        SimpleTypeConstraint::new(
685            self.name(),
686            vec![
687                self.fn_.clone(),
688                self.element.clone(),
689                self.multiset.clone(),
690                self.element.clone(),
691            ],
692            span.clone(),
693        )
694        .into_box()
695    }
696
697    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
698        let fc = exec_state
699            .container_values()
700            .get_val::<FunctionContainer>(args[0])
701            .unwrap()
702            .clone();
703        let initial = args[1];
704        let multiset = exec_state
705            .container_values()
706            .get_val::<MultiSetContainer>(args[2])
707            .unwrap()
708            .clone();
709        let mut values = multiset.data.iter().cloned().collect::<Vec<_>>();
710        let mut acc = if values.is_empty() {
711            initial
712        } else {
713            let first = values[0];
714            values.remove(0);
715            first
716        };
717        for v in values {
718            acc = fc.apply(exec_state, &[acc, v])?;
719        }
720        Some(acc)
721    }
722}

724// (multiset-union-values MultiSet[A]) -> A
725// where A: Eq
726// Unions all values in the multiset together using the union action defined for the inner type.
727#[derive(Clone)]
728struct UnionValues {
729    name: String,
730    multiset: ArcSort,
731    element: ArcSort,
732    action: UnionAction,
733}
734
735impl Primitive for UnionValues {
736    fn name(&self) -> &str {
737        &self.name
738    }
739
740    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
741        SimpleTypeConstraint::new(
742            self.name(),
743            vec![self.multiset.clone(), self.element.clone()],
744            span.clone(),
745        )
746        .into_box()
747    }
748
749    fn apply(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
750        let values = exec_state
751            .container_values()
752            .get_val::<MultiSetContainer>(args[0])?
753            .clone()
754            .data;
755        let values: Vec<_> = values.iter_counts().map(|(v, _c)| v).collect();
756        if values.is_empty() {
757            return None;
758        }
759        let first = values[0];
760        for v in values.into_iter().skip(1) {
761            self.action.union(exec_state, first, v);
762        }
763        Some(first)
764    }
765}

767// Place multiset in its own module to keep implementation details private from sort
768mod inner {
769    use std::collections::BTreeMap;
770    use std::hash::Hash;
771    /// Immutable multiset implementation, which is threadsafe and hash stable, regardless of insertion order.
772    ///
773    /// All methods that return a new multiset take ownership of the old multiset.
774    #[derive(Debug, Default, Hash, Eq, PartialEq, Clone)]
775    pub struct MultiSet<T: Clone + Hash + Ord>(
776        /// All values should be > 0
777        BTreeMap<T, usize>,
778        /// cached length
779        usize,
780    );
781
782    impl<T: Clone + Hash + Ord> MultiSet<T> {
783        /// Create a new empty multiset.
784        pub fn new() -> Self {
785            MultiSet(BTreeMap::new(), 0)
786        }
787
788        /// Check if the multiset contains a key.
789        pub fn contains(&self, value: &T) -> bool {
790            self.0.contains_key(value)
791        }
792
793        /// Return the total number of elements in the multiset.
794        pub fn len(&self) -> usize {
795            self.1
796        }
797
798        /// Return an iterator over all elements in the multiset.
799        pub fn iter(&self) -> impl Iterator<Item = &T> {
800            self.0.iter().flat_map(|(k, v)| std::iter::repeat_n(k, *v))
801        }
802
803        /// Return an iterator over values and counts
804        pub fn iter_counts(&self) -> impl Iterator<Item = (T, usize)> {
805            self.0.iter().map(|(k, v)| (k.clone(), *v))
806        }
807
808        /// Return an arbitrary element from the multiset.
809        pub fn pick(&self) -> Option<&T> {
810            self.0.keys().next()
811        }
812
813        /// Insert a value into the multiset, taking ownership of it and returning a new multiset.
814        pub fn insert(mut self, value: T) -> MultiSet<T> {
815            self.insert_multiple_mut(value, 1);
816            self
817        }
818
819        /// Remove a value from the multiset, taking ownership of it and returning a new multiset.
820        pub fn remove(mut self, value: &T) -> Option<MultiSet<T>> {
821            if let Some(v) = self.0.get(value) {
822                self.1 -= 1;
823                if *v == 1 {
824                    self.0.remove(value);
825                } else {
826                    self.0.insert(value.clone(), v - 1);
827                }
828                Some(self)
829            } else {
830                None
831            }
832        }
833
834        /// Subtract the counts of another multiset from this multiset, taking ownership of both and returning a new multiset.
835        pub fn subtract(mut self, other: &MultiSet<T>) -> Option<MultiSet<T>> {
836            for (k, v) in other.0.iter() {
837                if let Some(self_v) = self.0.get_mut(k) {
838                    if *self_v < *v {
839                        return None;
840                    }
841                    *self_v -= *v;
842                    self.1 -= *v;
843                    if *self_v == 0 {
844                        self.0.remove(k);
845                    }
846                } else {
847                    return None;
848                }
849            }
850            Some(self)
851        }
852
853        pub fn insert_multiple_mut(&mut self, value: T, n: usize) {
854            self.1 += n;
855            if let Some(v) = self.0.get(&value) {
856                self.0.insert(value, v + n);
857            } else {
858                self.0.insert(value, n);
859            }
860        }
861
862        /// Compute the sum of two multisets.
863        pub fn sum(mut self, MultiSet(other_map, other_count): Self) -> Self {
864            let target_count = self.1 + other_count;
865            for (k, v) in other_map {
866                self.insert_multiple_mut(k, v);
867            }
868            assert_eq!(self.1, target_count);
869            self
870        }
871
872        /// Compute the intersection of two multisets.
873        /// The count of each element in the result is the minimum of its counts in the two multisets.
874        pub fn intersection(self, MultiSet(other_map, _): Self) -> Self {
875            let mut new_map = BTreeMap::new();
876            for (k, v) in self.0.into_iter() {
877                if let Some(other_v) = other_map.get(&k) {
878                    let new_v = std::cmp::min(v, *other_v);
879                    new_map.insert(k, new_v);
880                }
881            }
882            let new_count = new_map.values().sum();
883            MultiSet(new_map, new_count)
884        }
885    }
886
887    impl<T: Clone + Hash + Ord> FromIterator<T> for MultiSet<T> {
888        fn from_iter<I: IntoIterator<Item = T>>(iter: I) -> Self {
889            let mut multiset = MultiSet::new();
890            for value in iter {
891                multiset.insert_multiple_mut(value, 1);
892            }
893            multiset
894        }
895    }
896}
