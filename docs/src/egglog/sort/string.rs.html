14    fn register_primitives(&self, eg: &mut EGraph) {
15        let string_concat_validator = |termdag: &mut TermDag, args: &[TermId]| -> Option<TermId> {
16            let mut concatenated = String::new();
17            for &arg in args {
18                let Term::Lit(Literal::String(s)) = termdag.get(arg) else {
19                    return None;
20                };
21                concatenated.push_str(s.as_str());
22            }
23            let result_lit = Literal::String(concatenated);
24            let result_term = termdag.lit(result_lit);
25            Some(result_term)
26        };
27        add_primitive_with_validator!(eg, "+" = [xs: S] -> S {{
28            let mut y = String::new();
29            xs.for_each(|x| y.push_str(x.as_str()));
30            y.into()
31        }}, string_concat_validator);
32
33        add_literal_prim!(eg, "replace" = |a: S, b: S, c: S| -> S {
34            a.as_str().replace(b.as_str(), c.as_str()).into()
35        });
36    }
