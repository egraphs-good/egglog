1//! Row encoding for the [`crate::EGraph`] surface API.
2//!
3//! Four traits map Rust values to and from egglog row data:
4//!
5//! - [`IntoValue`] / [`FromValue`] — one Rust value ↔ one egglog
6//!   [`Value`]. These are **open** traits: users implementing custom
7//!   sorts (or just custom Rust wrapper types) can implement them to
8//!   make their types flow through the row-encoding APIs.
9//! - [`IntoValues`] / [`FromValues`] — a whole row of Rust values
10//!   ↔ a `&[Value]`. These are **sealed**: row shape is fixed to
11//!   tuples up to arity 8, plus the [`RawValues`] / `Vec<Value>` /
12//!   `()` escape hatches.
13//!
14//! The API is **type-unsafe at the column level**: every column flows
15//! as a bare [`Value`]. The Rust trait machinery here only enforces
16//! arity (you can't pass a 3-tuple to a 2-column table) and subtype
17//! (you can't `set` on a constructor or `add` on a function). Per-
18//! column sort matching (e.g., "this column wants an `i64`, you
19//! passed a `String`") is the caller's responsibility — see the
20//! parked dynamic-typing work for what real safety would cost.
21
22use crate::core_relations::{BaseValues, Value};

26// ---------------------------------------------------------------------
27// ApiError — runtime check failures from the `Read` / `Write` trait
28// methods and from [`crate::EGraph::update`] callers.
29// ---------------------------------------------------------------------
30
31/// Runtime errors from the Rust API surface.
32///
33/// These signal a misuse the API can detect at runtime — wrong table
34/// subtype, wrong arity, mismatched column sorts, etc. They are *not*
35/// egglog typecheck errors and *not* backend / e-graph failures; for
36/// those, see [`enum@crate::Error`].
37#[derive(Debug, Error)]
38pub enum ApiError {
39    #[error("no table named `{name}` is registered")]
40    MissingTable { name: String },
41
42    #[error("table `{name}` is a {actual}; this method is only valid for {expected} tables")]
43    WrongSubtype {
44        name: String,
45        expected: &'static str,
46        actual: &'static str,
47    },
48
49    #[error("table `{table}`: expected {expected} input columns, got {got}")]
50    WrongArity {
51        table: String,
52        expected: usize,
53        got: usize,
54    },
55}

61// ---------------------------------------------------------------------
62// Per-column conversion: IntoValue + FromValue (open)
63// ---------------------------------------------------------------------
64
65/// Convert a Rust value into an egglog [`Value`]. Open trait: impl
66/// for your own types if you want them to flow through `set` / `add`
67/// / `lookup` keys as column inputs.
68pub trait IntoValue {
69    fn into_value(self, bv: &BaseValues) -> Value;
70}
71
72/// Convert an egglog [`Value`] back into a Rust value. Open trait:
73/// impl for your own types if you want them to flow out of
74/// `function_entries` / `query` rows.
75pub trait FromValue: Sized {
76    fn from_value(value: Value, bv: &BaseValues) -> Self;
77}
78
79// ---------------------------------------------------------------------
80// Row conversion: IntoValues + FromValues (sealed)
81// ---------------------------------------------------------------------
82
83/// Convert a Rust value into a row of egglog [`Value`]s.
84///
85/// Sealed; impls cover bare [`IntoValue`] (single-column row),
86/// tuples up to arity 8 of [`IntoValue`] values, and [`RawValues`]
87/// for variadic / pre-converted rows.
88pub trait IntoValues: sealed::Sealed {
89    fn into_values(self, bv: &BaseValues) -> impl Iterator<Item = Value>;
90}
91
92/// Convert a row of egglog [`Value`]s back into a Rust value.
93///
94/// Sealed; impls cover `()` (discard), `Vec<Value>` (raw escape
95/// hatch), and tuples up to arity 8 of [`FromValue`] types.
96pub trait FromValues: Sized + sealed::Sealed {
97    fn from_values(values: &[Value], bv: &BaseValues) -> Self;
98}

100// ---------------------------------------------------------------------
101// Built-in column impls — symmetric for IntoValue / FromValue
102// ---------------------------------------------------------------------
103
104macro_rules! impl_value_for_base {
105    ( $( $ty:ty ),+ $(,)? ) => {
106        $(
107            impl IntoValue for $ty {
108                fn into_value(self, bv: &BaseValues) -> Value {
109                    bv.get::<$ty>(self)
110                }
111            }
112            impl FromValue for $ty {
113                fn from_value(value: Value, bv: &BaseValues) -> Self {
114                    bv.unwrap::<$ty>(value)
115                }
116            }
117        )+
118    };
119}
120
121impl_value_for_base!(i64, bool, (), sort::F, sort::S, sort::Z, sort::Q);
122
123// `String` is sugar — egglog's String sort uses `sort::S` (`Boxed<String>`).
124impl IntoValue for String {
125    fn into_value(self, bv: &BaseValues) -> Value {
126        bv.get::<sort::S>(self.into())
127    }
128}
129impl FromValue for String {
130    fn from_value(value: Value, bv: &BaseValues) -> Self {
131        bv.unwrap::<sort::S>(value).0
132    }
133}
134
135// `&str` is one-directional input sugar.
136impl IntoValue for &str {
137    fn into_value(self, bv: &BaseValues) -> Value {
138        bv.get::<sort::S>(self.to_string().into())
139    }
140}
141
142// `f64` is sugar for `sort::F`.
143impl IntoValue for f64 {
144    fn into_value(self, bv: &BaseValues) -> Value {
145        use ordered_float::OrderedFloat;
146        bv.get::<sort::F>(OrderedFloat(self).into())
147    }
148}
149impl FromValue for f64 {
150    fn from_value(value: Value, bv: &BaseValues) -> Self {
151        bv.unwrap::<sort::F>(value).0.0
152    }
153}
154
155// Bare `Value` passes through unchanged.
156impl IntoValue for Value {
157    fn into_value(self, _bv: &BaseValues) -> Value {
158        self
159    }
160}
161impl FromValue for Value {
162    fn from_value(value: Value, _bv: &BaseValues) -> Self {
163        value
164    }
165}

167// ---------------------------------------------------------------------
168// Variadic / pre-converted rows: RawValues
169// ---------------------------------------------------------------------
170
171/// Wrap a `Vec<Value>` as an [`IntoValues`]. Useful for zero-arg
172/// constructors (`RawValues(vec![])`) and for rows whose column
173/// count isn't known until runtime.
174#[derive(Clone, Debug)]
175pub struct RawValues(pub Vec<Value>);
176
177impl sealed::Sealed for RawValues {}
178impl IntoValues for RawValues {
179    fn into_values(self, _bv: &BaseValues) -> impl Iterator<Item = Value> {
180        self.0.into_iter()
181    }
182}
183
184impl sealed::Sealed for Vec<Value> {}
185impl FromValues for Vec<Value> {
186    fn from_values(values: &[Value], _bv: &BaseValues) -> Self {
187        values.to_vec()
188    }
189}
190
191// `()` as a FromValues discards whatever it was handed (useful for
192// "did this match" queries). `()` as an IntoValues is a single Unit
193// column, handled by the blanket `impl<A: IntoValue> IntoValues for A`
194// below. For a zero-arg row, use `RawValues(vec![])`.
195impl FromValues for () {
196    fn from_values(_values: &[Value], _bv: &BaseValues) -> Self {}
197}
198
199// ---------------------------------------------------------------------
200// Single-column blanket impl: any T: IntoValue is a 1-column IntoValues.
201//
202// Note: this does NOT cover FromValues — that would conflict with the
203// `Vec<Value>` impl above. Single-column outputs use the (A,) tuple form.
204// ---------------------------------------------------------------------
205
206impl<A: IntoValue> sealed::Sealed for A {}
207impl<A: IntoValue> IntoValues for A {
208    fn into_values(self, bv: &BaseValues) -> impl Iterator<Item = Value> {
209        std::iter::once(self.into_value(bv))
210    }
211}

213// ---------------------------------------------------------------------
214// Tuple impls — IntoValues / FromValues for tuples of arity 1..=8
215// ---------------------------------------------------------------------
216
217macro_rules! impl_values_for_tuple {
218    ( $( ($($name:ident),+) ),+ $(,)? ) => {
219        $(
220            #[allow(non_snake_case)]
221            impl<$($name),+> sealed::Sealed for ($($name,)+) {}
222
223            #[allow(non_snake_case)]
224            impl<$($name: IntoValue),+> IntoValues for ($($name,)+) {
225                fn into_values(self, bv: &BaseValues) -> impl Iterator<Item = Value> {
226                    let ($($name,)+) = self;
227                    [ $( $name.into_value(bv) ),+ ].into_iter()
228                }
229            }
230
231            #[allow(non_snake_case)]
232            impl<$($name: FromValue),+> FromValues for ($($name,)+) {
233                fn from_values(values: &[Value], bv: &BaseValues) -> Self {
234                    let mut iter = values.iter().copied();
235                    let tuple = ( $( $name::from_value(iter.next().expect("FromValues: row shorter than tuple arity"), bv), )+ );
236                    assert!(
237                        iter.next().is_none(),
238                        "FromValues: row longer than tuple arity",
239                    );
240                    tuple
241                }
242            }
243        )+
244    };
245}
