19impl PartialEq for FuncType {
20    fn eq(&self, other: &Self) -> bool {
21        if self.name == other.name
22            && self.subtype == other.subtype
23            && self.output.name() == other.output.name()
24        {
25            if self.input.len() != other.input.len() {
26                return false;
27            }
28            for (a, b) in self.input.iter().zip(other.input.iter()) {
29                if a.name() != b.name() {
30                    return false;
31                }
32            }
33            true
34        } else {
35            false
36        }
37    }
38}

43    fn hash<H: Hasher>(&self, state: &mut H) {

45        self.subtype.hash(state);
46        self.output.name().hash(state);
47        for inp in &self.input {

49        }

64impl PrimitiveWithId {
65    /// Takes the full signature of a primitive (both input and output types).
66    /// Returns whether the primitive is compatible with this signature.
67    pub fn accept(&self, tys: &[Arc<dyn Sort>], typeinfo: &TypeInfo) -> bool {
68        let mut constraints = vec![];
69        let lits: Vec<_> = (0..tys.len())
70            .map(|i| AtomTerm::Literal(Span::Panic, Literal::Int(i as i64)))
71            .collect();
72        for (lit, ty) in lits.iter().zip(tys.iter()) {
73            constraints.push(constraint::assign(lit.clone(), ty.clone()))
74        }
75        constraints.extend(
76            self.primitive
77                .get_type_constraints(&Span::Panic)
78                .get(&lits, typeinfo),
79        );
80        let problem = Problem {
81            constraints,
82            range: HashSet::default(),
83        };
84        problem.solve(|sort| sort.name()).is_ok()
85    }
86}

118    /// Declare a sort. This corresponds to the `sort` keyword in egglog.
119    /// It can either declares a new [`EqSort`] if `presort_and_args` is not provided,
120    /// or an instantiation of a presort (e.g., containers like `Vec`).
121    pub fn declare_sort(
122        &mut self,
123        name: impl Into<String>,
124        presort_and_args: &Option<(String, Vec<Expr>)>,
125        span: Span,
126    ) -> Result<(), TypeError> {
127        let name = name.into();
128        if self.type_info.func_types.contains_key(&name) {
129            return Err(TypeError::FunctionAlreadyBound(name, span));
130        }
131
132        let sort = match presort_and_args {
133            None => Arc::new(EqSort { name }),
134            Some((presort, args)) => {
135                if let Some(mksort) = self.type_info.mksorts.get(presort) {
136                    mksort(&mut self.type_info, name, args)?
137                } else {
138                    return Err(TypeError::PresortNotFound(presort.clone(), span));
139                }
140            }
141        };
142
143        self.add_arcsort(sort, span)
144    }

146    /// Add a user-defined sort to the e-graph.
147    pub fn add_arcsort(&mut self, sort: ArcSort, span: Span) -> Result<(), TypeError> {
148        sort.register_type(&mut self.backend);
149
150        let name = sort.name();
151        match self.type_info.sorts.entry(name.to_owned()) {
152            HEntry::Occupied(_) => Err(TypeError::SortAlreadyBound(name.to_owned(), span)),
153            HEntry::Vacant(e) => {
154                e.insert(sort.clone());
155                sort.register_primitives(self);
156                Ok(())
157            }
158        }
159    }

169    /// Add a user-defined primitive with an optional validator
170    pub fn add_primitive_with_validator<T>(&mut self, x: T, validator: Option<PrimitiveValidator>)
171    where
172        T: Clone + Primitive + Send + Sync + 'static,
173    {
174        // We need to use a wrapper because of the orphan rule.
175        // If we just try to implement `ExternalFunction` directly on
176        // all `PrimitiveLike`s then it would be possible for a
177        // downstream crate to create a conflict.
178        #[derive(Clone)]
179        struct Wrapper<T>(T);
180        impl<T: Clone + Primitive + Send + Sync> ExternalFunction for Wrapper<T> {
181            fn invoke(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
182                self.0.apply(exec_state, args)
183            }
184        }
185
186        let primitive = Arc::new(x.clone());
187        let id = self.backend.register_external_func(Box::new(Wrapper(x)));
188        self.type_info
189            .primitives
190            .entry(primitive.name().to_owned())
191            .or_default()
192            .push(PrimitiveWithId {
193                primitive,
194                id,
195                validator,
196            });
197    }

210    fn typecheck_command(&mut self, command: &NCommand) -> Result<ResolvedNCommand, TypeError> {
211        let symbol_gen = &mut self.parser.symbol_gen;
212
213        let command: ResolvedNCommand = match command {
214            NCommand::Function(fdecl) => {
215                let resolved = self.type_info.typecheck_function(symbol_gen, fdecl)?;
216                // If this is a let binding, add it to global_sorts
217                // This preserves bahavior for lets after desugaring
218                if resolved.internal_let {
219                    let output_sort = self.type_info.sorts.get(&fdecl.schema.output).unwrap();
220                    self.type_info
221                        .global_sorts
222                        .insert(fdecl.name.clone(), output_sort.clone());
223                }
224                ResolvedNCommand::Function(resolved)
225            }
226            NCommand::NormRule { rule } => ResolvedNCommand::NormRule {
227                rule: self.type_info.typecheck_rule(symbol_gen, rule)?,
228            },
229            NCommand::Sort {
230                span,
231                name,
232                presort_and_args,
233                uf,
234                unionable,
235            } => {
236                // Note this is bad since typechecking should be pure and idempotent
237                // Otherwise typechecking the same program twice will fail
238                self.declare_sort(name.clone(), presort_and_args, span.clone())?;
239                // Mark as non-unionable if the sort declaration says so
240                if !unionable {
241                    self.type_info.non_unionable_sorts.insert(name.clone());
242                }
243                ResolvedNCommand::Sort {
244                    span: span.clone(),
245                    name: name.clone(),
246                    presort_and_args: presort_and_args.clone(),
247                    uf: uf.clone(),
248                    unionable: *unionable,
249                }
250            }
251            NCommand::CoreAction(Action::Let(span, var, expr)) => {
252                let expr = self
253                    .type_info
254                    .typecheck_expr(symbol_gen, expr, &Default::default())?;
255                let output_type = expr.output_type();
256                self.ensure_global_name_prefix(span, var)?;
257                self.type_info
258                    .global_sorts
259                    .insert(var.clone(), output_type.clone());
260                let var = ResolvedVar {
261                    name: var.clone(),
262                    sort: output_type,
263                    // not a global reference, but a global binding
264                    is_global_ref: false,
265                };
266                ResolvedNCommand::CoreAction(ResolvedAction::Let(span.clone(), var, expr))
267            }
268            NCommand::CoreAction(action) => ResolvedNCommand::CoreAction(
269                self.type_info
270                    .typecheck_action(symbol_gen, action, &Default::default())?,
271            ),
272            NCommand::Extract(span, expr, variants) => {
273                let res_expr =
274                    self.type_info
275                        .typecheck_expr(symbol_gen, expr, &Default::default())?;
276
277                let res_variants =
278                    self.type_info
279                        .typecheck_expr(symbol_gen, variants, &Default::default())?;
280                if res_variants.output_type().name() != I64Sort.name() {
281                    return Err(TypeError::Mismatch {
282                        expr: variants.clone(),
283                        expected: I64Sort.to_arcsort(),
284                        actual: res_variants.output_type(),
285                    });
286                }
287
288                ResolvedNCommand::Extract(span.clone(), res_expr, res_variants)
289            }
290            NCommand::Check(span, facts) => ResolvedNCommand::Check(
291                span.clone(),
292                self.type_info.typecheck_facts(symbol_gen, facts)?,
293            ),
294            NCommand::Fail(span, cmd) => {
295                ResolvedNCommand::Fail(span.clone(), Box::new(self.typecheck_command(cmd)?))
296            }
297            NCommand::RunSchedule(schedule) => ResolvedNCommand::RunSchedule(
298                self.type_info.typecheck_schedule(symbol_gen, schedule)?,
299            ),
300            NCommand::Pop(span, n) => ResolvedNCommand::Pop(span.clone(), *n),
301            NCommand::Push(n) => ResolvedNCommand::Push(*n),
302            NCommand::AddRuleset(span, ruleset) => {
303                ResolvedNCommand::AddRuleset(span.clone(), ruleset.clone())
304            }
305            NCommand::UnstableCombinedRuleset(span, name, sub_rulesets) => {
306                ResolvedNCommand::UnstableCombinedRuleset(
307                    span.clone(),
308                    name.clone(),
309                    sub_rulesets.clone(),
310                )
311            }
312            NCommand::PrintOverallStatistics(span, file) => {
313                ResolvedNCommand::PrintOverallStatistics(span.clone(), file.clone())
314            }
315            NCommand::PrintFunction(span, table, size, file, mode) => {
316                ResolvedNCommand::PrintFunction(
317                    span.clone(),
318                    table.clone(),
319                    *size,
320                    file.clone(),
321                    *mode,
322                )
323            }
324            NCommand::PrintSize(span, n) => {
325                // Should probably also resolve the function symbol here
326                ResolvedNCommand::PrintSize(span.clone(), n.clone())
327            }
328            NCommand::ProveExists(span, constructor) => {
329                let func_type = self
330                    .type_info
331                    .get_func_type(constructor)
332                    .ok_or_else(|| TypeError::UnboundFunction(constructor.clone(), span.clone()))?;
333                if func_type.subtype != FunctionSubtype::Constructor {
334                    return Err(TypeError::ProveExistsRequiresConstructor(
335                        constructor.clone(),
336                        span.clone(),
337                    ));
338                }
339                ResolvedNCommand::ProveExists(span.clone(), ResolvedCall::Func(func_type.clone()))
340            }
341            NCommand::Output { span, file, exprs } => {
342                let exprs = exprs
343                    .iter()
344                    .map(|expr| {
345                        self.type_info
346                            .typecheck_expr(symbol_gen, expr, &Default::default())
347                    })
348                    .collect::<Result<Vec<_>, _>>()?;
349                ResolvedNCommand::Output {
350                    span: span.clone(),
351                    file: file.clone(),
352                    exprs,
353                }
354            }
355            NCommand::Input { span, name, file } => ResolvedNCommand::Input {
356                span: span.clone(),
357                name: name.clone(),
358                file: file.clone(),
359            },
360            NCommand::UserDefined(span, name, exprs) => {
361                ResolvedNCommand::UserDefined(span.clone(), name.clone(), exprs.clone())
362            }
363        };
364        if let ResolvedNCommand::NormRule { rule } = &command {
365            self.warn_for_prefixed_non_globals_in_rule(rule)?;
366        }
367        Ok(command)
368    }

370    fn warn_for_prefixed_non_globals_in_var(
371        &mut self,
372        span: &Span,
373        var: &ResolvedVar,
374    ) -> Result<(), TypeError> {
375        if var.is_global_ref {
376            return Ok(());
377        }
378        if var.name.starts_with(crate::GLOBAL_NAME_PREFIX) {
379            self.warn_prefixed_non_globals(span, &var.name)?;
380        }
381        Ok(())
382    }
383
384    fn warn_for_prefixed_non_globals_in_rule(
385        &mut self,
386        rule: &ResolvedRule,
387    ) -> Result<(), TypeError> {
388        let mut res: Result<(), TypeError> = Ok(());
389
390        for fact in &rule.body {
391            fact.visit_vars(&mut |span, var| {
392                if res.is_ok() {
393                    res = self.warn_for_prefixed_non_globals_in_var(span, var);
394                }
395            });
396        }
397
398        rule.head.visit_vars(&mut |span, var| {
399            if res.is_ok() {
400                res = self.warn_for_prefixed_non_globals_in_var(span, var);
401            }
402        });
403        res
404    }

409    pub fn add_presort<S: Presort>(&mut self, span: Span) -> Result<(), TypeError> {
410        let name = S::presort_name();
411        match self.mksorts.entry(name.to_owned()) {
412            HEntry::Occupied(_) => Err(TypeError::SortAlreadyBound(name.to_owned(), span)),
413            HEntry::Vacant(e) => {
414                e.insert(S::make_sort);
415                self.reserved_primitives.extend(S::reserved_primitives());
416                Ok(())
417            }
418        }

421    /// Returns all sorts that satisfy the type and predicate.
422    pub fn get_sorts_by<S: Sort>(&self, pred: impl Fn(&Arc<S>) -> bool) -> Vec<Arc<S>> {
423        let mut results = Vec::new();
424        for sort in self.sorts.values() {
425            let sort = sort.clone().as_arc_any();
426            if let Ok(sort) = Arc::downcast(sort) {
427                if pred(&sort) {
428                    results.push(sort);
429                }
430            }
431        }
432        results
433    }

481    fn function_to_functype(&self, func: &FunctionDecl) -> Result<FuncType, TypeError> {
482        let input = func
483            .schema
484            .input
485            .iter()
486            .map(|name| {
487                if let Some(sort) = self.sorts.get(name) {
488                    Ok(sort.clone())
489                } else {
490                    Err(TypeError::UndefinedSort(name.clone(), func.span.clone()))
491                }
492            })
493            .collect::<Result<Vec<_>, _>>()?;
494        let output = if let Some(sort) = self.sorts.get(&func.schema.output) {
495            Ok(sort.clone())
496        } else {
497            Err(TypeError::UndefinedSort(
498                func.schema.output.clone(),
499                func.span.clone(),
500            ))
501        }?;
502
503        Ok(FuncType {
504            name: func.name.clone(),
505            subtype: func.subtype,
506            input,
507            output: output.clone(),
508        })

511    fn typecheck_function(
512        &mut self,
513        symbol_gen: &mut SymbolGen,
514        fdecl: &FunctionDecl,
515    ) -> Result<ResolvedFunctionDecl, TypeError> {
516        if self.sorts.contains_key(&fdecl.name) {
517            return Err(TypeError::SortAlreadyBound(
518                fdecl.name.clone(),
519                fdecl.span.clone(),
520            ));
521        }
522        if self.is_primitive(&fdecl.name) {
523            return Err(TypeError::PrimitiveAlreadyBound(
524                fdecl.name.clone(),
525                fdecl.span.clone(),
526            ));
527        }
528        // View tables (with term_constructor) must have at least one input (the e-class)
529        if fdecl.term_constructor.is_some() && fdecl.schema.input.is_empty() {
530            return Err(TypeError::TermConstructorNoInputs(
531                fdecl.name.clone(),
532                fdecl.span.clone(),
533            ));
534        }
535        let ftype = self.function_to_functype(fdecl)?;
536        if self.func_types.insert(fdecl.name.clone(), ftype).is_some() {
537            return Err(TypeError::FunctionAlreadyBound(
538                fdecl.name.clone(),
539                fdecl.span.clone(),
540            ));
541        }
542        let mut bound_vars = IndexMap::default();
543        let output_type = self.sorts.get(&fdecl.schema.output).unwrap();
544        if fdecl.subtype == FunctionSubtype::Constructor && !output_type.is_eq_sort() {
545            return Err(TypeError::ConstructorOutputNotSort(
546                fdecl.name.clone(),
547                fdecl.span.clone(),
548            ));
549        }
550        bound_vars.insert("old", (fdecl.span.clone(), output_type.clone()));
551        bound_vars.insert("new", (fdecl.span.clone(), output_type.clone()));
552
553        Ok(ResolvedFunctionDecl {
554            name: fdecl.name.clone(),
555            subtype: fdecl.subtype,
556            schema: fdecl.schema.clone(),
557            resolved_schema: ResolvedCall::Func(self.func_types.get(&fdecl.name).unwrap().clone()),
558            merge: match &fdecl.merge {
559                Some(merge) => Some(self.typecheck_expr(symbol_gen, merge, &bound_vars)?),
560                None => None,
561            },
562            cost: fdecl.cost,
563            unextractable: fdecl.unextractable,
564            internal_hidden: fdecl.internal_hidden,
565            internal_let: fdecl.internal_let,
566            span: fdecl.span.clone(),
567            term_constructor: fdecl.term_constructor.clone(),
568        })
569    }

571    fn typecheck_schedule(
572        &self,
573        symbol_gen: &mut SymbolGen,
574        schedule: &Schedule,
575    ) -> Result<ResolvedSchedule, TypeError> {
576        let schedule = match schedule {
577            Schedule::Repeat(span, times, schedule) => ResolvedSchedule::Repeat(
578                span.clone(),
579                *times,
580                Box::new(self.typecheck_schedule(symbol_gen, schedule)?),
581            ),
582            Schedule::Sequence(span, schedules) => {
583                let schedules = schedules
584                    .iter()
585                    .map(|schedule| self.typecheck_schedule(symbol_gen, schedule))
586                    .collect::<Result<Vec<_>, _>>()?;
587                ResolvedSchedule::Sequence(span.clone(), schedules)
588            }
589            Schedule::Saturate(span, schedule) => ResolvedSchedule::Saturate(
590                span.clone(),
591                Box::new(self.typecheck_schedule(symbol_gen, schedule)?),
592            ),
593            Schedule::Run(span, RunConfig { ruleset, until }) => {
594                let until = until
595                    .as_ref()
596                    .map(|facts| self.typecheck_facts(symbol_gen, facts))
597                    .transpose()?;
598                ResolvedSchedule::Run(
599                    span.clone(),
600                    ResolvedRunConfig {
601                        ruleset: ruleset.clone(),
602                        until,
603                    },
604                )
605            }
606        };
607
608        Result::Ok(schedule)
609    }

611    fn typecheck_rule(
612        &self,
613        symbol_gen: &mut SymbolGen,
614        rule: &Rule,
615    ) -> Result<ResolvedRule, TypeError> {
616        let Rule {
617            span,
618            head,
619            body,
620            name,
621            ruleset,
622        } = rule;
623        let mut constraints = vec![];
624
625        let (query, mapped_query) = Facts(body.clone()).to_query(self, symbol_gen);
626        constraints.extend(query.get_constraints(self)?);
627
628        let mut binding = query.get_vars();
629        // We lower to core actions with `union_to_set_optimization`
630        // later in the pipeline. For typechecking we do not need it.
631        let mut ctx = CoreActionContext::new(self, &mut binding, symbol_gen, false);
632        let (actions, mapped_action) = head.to_core_actions(&mut ctx)?;
633
634        let mut problem = Problem::default();
635        problem.add_rule(
636            &CoreRule {
637                span: span.clone(),
638                body: query,
639                head: actions,
640            },
641            self,
642            symbol_gen,
643        )?;
644
645        let assignment = problem
646            .solve(|sort: &ArcSort| sort.name())
647            .map_err(|e| e.to_type_error())?;
648
649        let body: Vec<ResolvedFact> = assignment.annotate_facts(&mapped_query, self);
650        let actions: ResolvedActions = assignment.annotate_actions(&mapped_action, self)?;
651
652        self.check_lookup_actions(&actions)?;
653
654        Ok(ResolvedRule {
655            span: span.clone(),
656            body,
657            head: actions,
658            name: name.clone(),
659            ruleset: ruleset.clone(),
660        })
661    }

663    fn check_lookup_expr(&self, expr: &ResolvedExpr) -> Result<(), TypeError> {
664        if let Some(span) = self.expr_has_function_lookup(expr) {
665            return Err(TypeError::LookupInRuleDisallowed(
666                "function".to_string(),
667                span,
668            ));
669        }
670        Ok(())
671    }
672
673    fn check_lookup_actions(&self, actions: &ResolvedActions) -> Result<(), TypeError> {
674        for action in actions.iter() {
675            match action {
676                GenericAction::Let(_, _, rhs) => self.check_lookup_expr(rhs)?,
677                GenericAction::Set(_, _, args, rhs) => {
678                    for arg in args.iter() {
679                        self.check_lookup_expr(arg)?;
680                    }
681                    self.check_lookup_expr(rhs)?;
682                }
683                GenericAction::Union(_, lhs, rhs) => {
684                    self.check_lookup_expr(lhs)?;
685                    self.check_lookup_expr(rhs)?;
686                }
687                GenericAction::Change(_, _, _, args) => {
688                    for arg in args.iter() {
689                        self.check_lookup_expr(arg)?;
690                    }
691                }
692                GenericAction::Panic(..) => {}
693                GenericAction::Expr(_, expr) => self.check_lookup_expr(expr)?,
694            }
695        }
696        Ok(())
697    }
698
699    pub fn typecheck_facts(
700        &self,
701        symbol_gen: &mut SymbolGen,
702        facts: &[Fact],
703    ) -> Result<Vec<ResolvedFact>, TypeError> {
704        let (query, mapped_facts) = Facts(facts.to_vec()).to_query(self, symbol_gen);
705        let mut problem = Problem::default();
706        problem.add_query(&query, self)?;
707        let assignment = problem
708            .solve(|sort: &ArcSort| sort.name())
709            .map_err(|e| e.to_type_error())?;
710        let annotated_facts = assignment.annotate_facts(&mapped_facts, self);
711        Ok(annotated_facts)
712    }
713
714    fn typecheck_actions(
715        &self,
716        symbol_gen: &mut SymbolGen,
717        actions: &Actions,
718        binding: &IndexMap<&str, (Span, ArcSort)>,
719    ) -> Result<ResolvedActions, TypeError> {
720        let mut binding_set: IndexSet<String> =
721            binding.keys().copied().map(str::to_string).collect();
722        // We lower to core actions with `union_to_set_optimization`
723        // later in the pipeline. For typechecking we do not need it.
724        let mut ctx = CoreActionContext::new(self, &mut binding_set, symbol_gen, false);
725        let (actions, mapped_action) = actions.to_core_actions(&mut ctx)?;
726        let mut problem = Problem::default();
727
728        // add actions to problem
729        problem.add_actions(&actions, self, symbol_gen)?;
730
731        // add bindings from the context
732        for (var, (span, sort)) in binding {
733            problem.assign_local_var_type(var, span.clone(), sort.clone())?;
734        }
735
736        let assignment = problem
737            .solve(|sort: &ArcSort| sort.name())
738            .map_err(|e| e.to_type_error())?;
739
740        let annotated_actions = assignment.annotate_actions(&mapped_action, self)?;
741        Ok(annotated_actions)
742    }
743
744    fn typecheck_expr(
745        &self,
746        symbol_gen: &mut SymbolGen,
747        expr: &Expr,
748        binding: &IndexMap<&str, (Span, ArcSort)>,
749    ) -> Result<ResolvedExpr, TypeError> {
750        let action = Action::Expr(expr.span(), expr.clone());
751        let typechecked_action = self.typecheck_action(symbol_gen, &action, binding)?;
752        match typechecked_action {
753            ResolvedAction::Expr(_, expr) => Ok(expr),
754            _ => unreachable!(),
755        }
756    }
757
758    fn typecheck_action(
759        &self,
760        symbol_gen: &mut SymbolGen,
761        action: &Action,
762        binding: &IndexMap<&str, (Span, ArcSort)>,
763    ) -> Result<ResolvedAction, TypeError> {
764        self.typecheck_actions(symbol_gen, &Actions::singleton(action.clone()), binding)
765            .map(|v| {
766                assert_eq!(v.len(), 1);
767                v.0.into_iter().next().unwrap()
768            })
769    }

808    /// Check if an expression contains non-global function lookups (FunctionSubtype::Custom calls).
809    /// Global function calls are allowed since they get desugared to constructors.
810    /// Returns Some(span) if a lookup is found, None otherwise.
811    pub fn expr_has_function_lookup(&self, expr: &ResolvedExpr) -> Option<Span> {
812        use ast::GenericExpr;
813
814        expr.find(&mut |e| {
815            if let GenericExpr::Call(span, ResolvedCall::Func(func_type), _) = e {
816                if func_type.subtype == FunctionSubtype::Custom {
817                    // Skip global functions - they get desugared to constructors
818                    if !self.is_global(&func_type.name) {
819                        return Some(span.clone());
820                    }
821                }
822            }
823            None
824        })
825    }

828#[derive(Debug, Clone, Error)]
829pub enum TypeError {
830    #[error("{}\nArity mismatch, expected {expected} args: {expr}", .expr.span())]
831    Arity { expr: Expr, expected: usize },
832    #[error(
833        "{}\n Expect expression {expr} to have type {}, but get type {}",
834        .expr.span(), .expected.name(), .actual.name(),
835    )]
836    Mismatch {
837        expr: Expr,
838        expected: ArcSort,
839        actual: ArcSort,
840    },
841    #[error("{1}\nUnbound symbol {0}")]
842    Unbound(String, Span),
843    #[error("{1}\nVariable {0} is ungrounded")]
844    Ungrounded(String, Span),
845    #[error("{1}\nUndefined sort {0}")]
846    UndefinedSort(String, Span),
847    #[error("{2}\nSort {0} definition is disallowed: {1}")]
848    DisallowedSort(String, String, Span),
849    #[error("{1}\nUnbound function {0}")]
850    UnboundFunction(String, Span),
851    #[error("{1}\nprove-exists requires constructor function, but {0} is not a constructor")]
852    ProveExistsRequiresConstructor(String, Span),
853    #[error("{1}\nFunction already bound {0}")]
854    FunctionAlreadyBound(String, Span),
855    #[error("{1}\nSort {0} already declared.")]
856    SortAlreadyBound(String, Span),
857    #[error("{1}\nPrimitive {0} already declared.")]
858    PrimitiveAlreadyBound(String, Span),
859    #[error("Function type mismatch: expected {} => {}, actual {} => {}", .1.iter().map(|s| s.name().to_string()).collect::<Vec<_>>().join(", "), .0.name(), .3.iter().map(|s| s.name().to_string()).collect::<Vec<_>>().join(", "), .2.name())]
860    FunctionTypeMismatch(ArcSort, Vec<ArcSort>, ArcSort, Vec<ArcSort>),
861    #[error("{1}\nPresort {0} not found.")]
862    PresortNotFound(String, Span),
863    #[error("{}\nFailed to infer a type for: {}", .0.span(), .0)]
864    InferenceFailure(Expr),
865    #[error("{1}\nVariable {0} was already defined")]
866    AlreadyDefined(String, Span),
867    #[error("{1}\nThe output type of constructor function {0} must be sort")]
868    ConstructorOutputNotSort(String, Span),
869    #[error("{1}\nValue lookup of non-constructor function {0} in rule is disallowed.")]
870    LookupInRuleDisallowed(String, Span),
871    #[error("{1}\nCannot set constructor {0}. Use `union` instead or declare {0} as a function.")]
872    SetConstructorDisallowed(String, Span),
873    #[error("All alternative definitions considered failed\n{}", .0.iter().map(|e| format!("  {e}\n")).collect::<Vec<_>>().join(""))]
874    AllAlternativeFailed(Vec<TypeError>),
875    #[error("{}\nCannot union values of sort {}", .1, .0.name())]
876    NonEqsortUnion(ArcSort, Span),
877    #[error("{}\nCannot union values of sort {} because it is marked as non-unionable (e.g. from a relation)", .1, .0.name())]
878    NonUnionableSort(ArcSort, Span),
879    #[error(
880        "{1}\nView table {0} with :term-constructor must have at least one input (the e-class)."
881    )]
882    TermConstructorNoInputs(String, Span),
883    #[error(
884        "{span}\nNon-global variable `{name}` must not start with `{}`.",
885        crate::GLOBAL_NAME_PREFIX
886    )]
887    NonGlobalPrefixed { name: String, span: Span },
888    #[error(
889        "{span}\nGlobal `{name}` must start with `{}`.",
890        crate::GLOBAL_NAME_PREFIX
891    )]
892    GlobalMissingPrefix { name: String, span: Span },
893}
