20impl PrimitiveWithId {
21    /// Takes the full signature of a primitive (both input and output types).
22    /// Returns whether the primitive is compatible with this signature.
23    pub fn accept(&self, tys: &[Arc<dyn Sort>], typeinfo: &TypeInfo) -> bool {
24        let mut constraints = vec![];
25        let lits: Vec<_> = (0..tys.len())
26            .map(|i| AtomTerm::Literal(Span::Panic, Literal::Int(i as i64)))
27            .collect();
28        for (lit, ty) in lits.iter().zip(tys.iter()) {
29            constraints.push(constraint::assign(lit.clone(), ty.clone()))
30        }
31        constraints.extend(
32            self.0
33                .get_type_constraints(&Span::Panic)
34                .get(&lits, typeinfo),
35        );
36        let problem = Problem {
37            constraints,
38            range: HashSet::default(),
39        };
40        problem.solve(|sort| sort.name()).is_ok()
41    }
42}

67    /// Also look at [`prelude::add_base_sort`] for a convenience method for adding user-defined sorts

72    /// Declare a sort. This corresponds to the `sort` keyword in egglog.
73    /// It can either declares a new [`EqSort`] if `presort_and_args` is not provided,
74    /// or an instantiation of a presort (e.g., containers like `Vec`).
75    pub fn declare_sort(
76        &mut self,
77        name: impl Into<String>,
78        presort_and_args: &Option<(String, Vec<Expr>)>,
79        span: Span,
80    ) -> Result<(), TypeError> {
81        let name = name.into();
82        if self.type_info.func_types.contains_key(&name) {
83            return Err(TypeError::FunctionAlreadyBound(name, span));
84        }
85
86        let sort = match presort_and_args {
87            None => Arc::new(EqSort { name }),
88            Some((presort, args)) => {
89                if let Some(mksort) = self.type_info.mksorts.get(presort) {
90                    mksort(&mut self.type_info, name, args)?
91                } else {
92                    return Err(TypeError::PresortNotFound(presort.clone(), span));
93                }
94            }
95        };
96
97        self.add_arcsort(sort, span)
98    }

100    /// Add a user-defined sort to the e-graph.
101    pub fn add_arcsort(&mut self, sort: ArcSort, span: Span) -> Result<(), TypeError> {
102        sort.register_type(&mut self.backend);
103
104        let name = sort.name();
105        match self.type_info.sorts.entry(name.to_owned()) {
106            HEntry::Occupied(_) => Err(TypeError::SortAlreadyBound(name.to_owned(), span)),
107            HEntry::Vacant(e) => {
108                e.insert(sort.clone());
109                sort.register_primitives(self);
110                Ok(())
111            }
112        }
113    }

115    /// Add a user-defined primitive
116    pub fn add_primitive<T>(&mut self, x: T)
117    where
118        T: Clone + Primitive + Send + Sync + 'static,
119    {
120        // We need to use a wrapper because of the orphan rule.
121        // If we just try to implement `ExternalFunction` directly on
122        // all `PrimitiveLike`s then it would be possible for a
123        // downstream crate to create a conflict.
124        #[derive(Clone)]
125        struct Wrapper<T>(T);
126        impl<T: Clone + Primitive + Send + Sync> ExternalFunction for Wrapper<T> {
127            fn invoke(&self, exec_state: &mut ExecutionState, args: &[Value]) -> Option<Value> {
128                self.0.apply(exec_state, args)
129            }
130        }
131
132        let prim = Arc::new(x.clone());
133        let ext = self.backend.register_external_func(Wrapper(x));
134        self.type_info
135            .primitives
136            .entry(prim.name().to_owned())
137            .or_default()
138            .push(PrimitiveWithId(prim, ext));
139    }

152    fn typecheck_command(&mut self, command: &NCommand) -> Result<ResolvedNCommand, TypeError> {
153        let symbol_gen = &mut self.parser.symbol_gen;
154
155        let command: ResolvedNCommand = match command {
156            NCommand::Function(fdecl) => {
157                ResolvedNCommand::Function(self.type_info.typecheck_function(symbol_gen, fdecl)?)
158            }
159            NCommand::NormRule { rule } => ResolvedNCommand::NormRule {
160                rule: self.type_info.typecheck_rule(symbol_gen, rule)?,
161            },
162            NCommand::Sort(span, sort, presort_and_args) => {
163                // Note this is bad since typechecking should be pure and idempotent
164                // Otherwise typechecking the same program twice will fail
165                self.declare_sort(sort.clone(), presort_and_args, span.clone())?;
166                ResolvedNCommand::Sort(span.clone(), sort.clone(), presort_and_args.clone())
167            }
168            NCommand::CoreAction(Action::Let(span, var, expr)) => {
169                let expr = self
170                    .type_info
171                    .typecheck_expr(symbol_gen, expr, &Default::default())?;
172                let output_type = expr.output_type();
173                self.type_info
174                    .global_sorts
175                    .insert(var.clone(), output_type.clone());
176                let var = ResolvedVar {
177                    name: var.clone(),
178                    sort: output_type,
179                    // not a global reference, but a global binding
180                    is_global_ref: false,
181                };
182                ResolvedNCommand::CoreAction(ResolvedAction::Let(span.clone(), var, expr))
183            }
184            NCommand::CoreAction(action) => ResolvedNCommand::CoreAction(
185                self.type_info
186                    .typecheck_action(symbol_gen, action, &Default::default())?,
187            ),
188            NCommand::Extract(span, expr, variants) => {
189                let res_expr =
190                    self.type_info
191                        .typecheck_expr(symbol_gen, expr, &Default::default())?;
192
193                let res_variants =
194                    self.type_info
195                        .typecheck_expr(symbol_gen, variants, &Default::default())?;
196                if res_variants.output_type().name() != I64Sort.name() {
197                    return Err(TypeError::Mismatch {
198                        expr: variants.clone(),
199                        expected: I64Sort.to_arcsort(),
200                        actual: res_variants.output_type(),
201                    });
202                }
203
204                ResolvedNCommand::Extract(span.clone(), res_expr, res_variants)
205            }
206            NCommand::Check(span, facts) => ResolvedNCommand::Check(
207                span.clone(),
208                self.type_info.typecheck_facts(symbol_gen, facts)?,
209            ),
210            NCommand::Fail(span, cmd) => {
211                ResolvedNCommand::Fail(span.clone(), Box::new(self.typecheck_command(cmd)?))
212            }
213            NCommand::RunSchedule(schedule) => ResolvedNCommand::RunSchedule(
214                self.type_info.typecheck_schedule(symbol_gen, schedule)?,
215            ),
216            NCommand::Pop(span, n) => ResolvedNCommand::Pop(span.clone(), *n),
217            NCommand::Push(n) => ResolvedNCommand::Push(*n),
218            NCommand::AddRuleset(span, ruleset) => {
219                ResolvedNCommand::AddRuleset(span.clone(), ruleset.clone())
220            }
221            NCommand::UnstableCombinedRuleset(span, name, sub_rulesets) => {
222                ResolvedNCommand::UnstableCombinedRuleset(
223                    span.clone(),
224                    name.clone(),
225                    sub_rulesets.clone(),
226                )
227            }
228            NCommand::PrintOverallStatistics(span, file) => {
229                ResolvedNCommand::PrintOverallStatistics(span.clone(), file.clone())
230            }
231            NCommand::PrintFunction(span, table, size, file, mode) => {
232                ResolvedNCommand::PrintFunction(
233                    span.clone(),
234                    table.clone(),
235                    *size,
236                    file.clone(),
237                    *mode,
238                )
239            }
240            NCommand::PrintSize(span, n) => {
241                // Should probably also resolve the function symbol here
242                ResolvedNCommand::PrintSize(span.clone(), n.clone())
243            }
244            NCommand::Output { span, file, exprs } => {
245                let exprs = exprs
246                    .iter()
247                    .map(|expr| {
248                        self.type_info
249                            .typecheck_expr(symbol_gen, expr, &Default::default())
250                    })
251                    .collect::<Result<Vec<_>, _>>()?;
252                ResolvedNCommand::Output {
253                    span: span.clone(),
254                    file: file.clone(),
255                    exprs,
256                }
257            }
258            NCommand::Input { span, name, file } => ResolvedNCommand::Input {
259                span: span.clone(),
260                name: name.clone(),
261                file: file.clone(),
262            },
263            NCommand::UserDefined(span, name, exprs) => {
264                ResolvedNCommand::UserDefined(span.clone(), name.clone(), exprs.clone())
265            }
266        };
267        Ok(command)
268    }

274        let name = S::presort_name();
275        match self.mksorts.entry(name.to_owned()) {
276            HEntry::Occupied(_) => Err(TypeError::SortAlreadyBound(name.to_owned(), span)),
277            HEntry::Vacant(e) => {
278                e.insert(S::make_sort);
279                self.reserved_primitives.extend(S::reserved_primitives());
280                Ok(())
281            }

286    pub fn get_sorts_by<S: Sort>(&self, pred: impl Fn(&Arc<S>) -> bool) -> Vec<Arc<S>> {
287        let mut results = Vec::new();
288        for sort in self.sorts.values() {
289            let sort = sort.clone().as_arc_any();
290            if let Ok(sort) = Arc::downcast(sort) {
291                if pred(&sort) {
292                    results.push(sort);
293                }
294            }
295        }
296        results

338    fn function_to_functype(&self, func: &FunctionDecl) -> Result<FuncType, TypeError> {
339        let input = func
340            .schema
341            .input
342            .iter()
343            .map(|name| {
344                if let Some(sort) = self.sorts.get(name) {
345                    Ok(sort.clone())
346                } else {
347                    Err(TypeError::UndefinedSort(name.clone(), func.span.clone()))
348                }
349            })
350            .collect::<Result<Vec<_>, _>>()?;
351        let output = if let Some(sort) = self.sorts.get(&func.schema.output) {
352            Ok(sort.clone())
353        } else {
354            Err(TypeError::UndefinedSort(
355                func.schema.output.clone(),
356                func.span.clone(),
357            ))
358        }?;
359
360        Ok(FuncType {
361            name: func.name.clone(),
362            subtype: func.subtype,
363            input,
364            output: output.clone(),
365        })
366    }

368    fn typecheck_function(
369        &mut self,
370        symbol_gen: &mut SymbolGen,
371        fdecl: &FunctionDecl,
372    ) -> Result<ResolvedFunctionDecl, TypeError> {
373        if self.sorts.contains_key(&fdecl.name) {
374            return Err(TypeError::SortAlreadyBound(
375                fdecl.name.clone(),
376                fdecl.span.clone(),
377            ));
378        }
379        if self.is_primitive(&fdecl.name) {
380            return Err(TypeError::PrimitiveAlreadyBound(
381                fdecl.name.clone(),
382                fdecl.span.clone(),
383            ));
384        }
385        let ftype = self.function_to_functype(fdecl)?;
386        if self.func_types.insert(fdecl.name.clone(), ftype).is_some() {
387            return Err(TypeError::FunctionAlreadyBound(
388                fdecl.name.clone(),
389                fdecl.span.clone(),
390            ));
391        }
392        let mut bound_vars = IndexMap::default();
393        let output_type = self.sorts.get(&fdecl.schema.output).unwrap();
394        if fdecl.subtype == FunctionSubtype::Constructor && !output_type.is_eq_sort() {
395            return Err(TypeError::ConstructorOutputNotSort(
396                fdecl.name.clone(),
397                fdecl.span.clone(),
398            ));
399        }
400        bound_vars.insert("old", (fdecl.span.clone(), output_type.clone()));
401        bound_vars.insert("new", (fdecl.span.clone(), output_type.clone()));
402
403        Ok(ResolvedFunctionDecl {
404            name: fdecl.name.clone(),
405            subtype: fdecl.subtype,
406            schema: fdecl.schema.clone(),
407            merge: match &fdecl.merge {
408                Some(merge) => Some(self.typecheck_expr(symbol_gen, merge, &bound_vars)?),
409                None => None,
410            },
411            cost: fdecl.cost,
412            unextractable: fdecl.unextractable,
413            let_binding: fdecl.let_binding,
414            span: fdecl.span.clone(),
415        })
416    }

418    fn typecheck_schedule(
419        &self,
420        symbol_gen: &mut SymbolGen,
421        schedule: &Schedule,
422    ) -> Result<ResolvedSchedule, TypeError> {
423        let schedule = match schedule {
424            Schedule::Repeat(span, times, schedule) => ResolvedSchedule::Repeat(
425                span.clone(),
426                *times,
427                Box::new(self.typecheck_schedule(symbol_gen, schedule)?),
428            ),
429            Schedule::Sequence(span, schedules) => {
430                let schedules = schedules
431                    .iter()
432                    .map(|schedule| self.typecheck_schedule(symbol_gen, schedule))
433                    .collect::<Result<Vec<_>, _>>()?;
434                ResolvedSchedule::Sequence(span.clone(), schedules)
435            }
436            Schedule::Saturate(span, schedule) => ResolvedSchedule::Saturate(
437                span.clone(),
438                Box::new(self.typecheck_schedule(symbol_gen, schedule)?),
439            ),
440            Schedule::Run(span, RunConfig { ruleset, until }) => {
441                let until = until
442                    .as_ref()
443                    .map(|facts| self.typecheck_facts(symbol_gen, facts))
444                    .transpose()?;
445                ResolvedSchedule::Run(
446                    span.clone(),
447                    ResolvedRunConfig {
448                        ruleset: ruleset.clone(),
449                        until,
450                    },
451                )
452            }
453        };
454
455        Result::Ok(schedule)
456    }

458    fn typecheck_rule(
459        &self,
460        symbol_gen: &mut SymbolGen,
461        rule: &Rule,
462    ) -> Result<ResolvedRule, TypeError> {
463        let Rule {
464            span,
465            head,
466            body,
467            name,
468            ruleset,
469        } = rule;
470        let mut constraints = vec![];
471
472        let (query, mapped_query) = Facts(body.clone()).to_query(self, symbol_gen);
473        constraints.extend(query.get_constraints(self)?);
474
475        let mut binding = query.get_vars();
476        let (actions, mapped_action) = head.to_core_actions(self, &mut binding, symbol_gen)?;
477
478        let mut problem = Problem::default();
479        problem.add_rule(
480            &CoreRule {
481                span: span.clone(),
482                body: query,
483                head: actions,
484            },
485            self,
486            symbol_gen,
487        )?;
488
489        let assignment = problem
490            .solve(|sort: &ArcSort| sort.name())
491            .map_err(|e| e.to_type_error())?;
492
493        let body: Vec<ResolvedFact> = assignment.annotate_facts(&mapped_query, self);
494        let actions: ResolvedActions = assignment.annotate_actions(&mapped_action, self)?;
495
496        Self::check_lookup_actions(&actions)?;
497
498        Ok(ResolvedRule {
499            span: span.clone(),
500            body,
501            head: actions,
502            name: name.clone(),
503            ruleset: ruleset.clone(),
504        })
505    }

507    fn check_lookup_expr(expr: &GenericExpr<ResolvedCall, ResolvedVar>) -> Result<(), TypeError> {
508        match expr {
509            GenericExpr::Call(span, head, args) => {
510                match head {
511                    ResolvedCall::Func(t) => {
512                        // Only allowed to lookup constructor or relation
513                        if t.subtype != FunctionSubtype::Constructor
514                            && t.subtype != FunctionSubtype::Relation
515                        {
516                            Err(TypeError::LookupInRuleDisallowed(
517                                head.to_string(),
518                                span.clone(),
519                            ))
520                        } else {
521                            Ok(())
522                        }
523                    }
524                    ResolvedCall::Primitive(_) => Ok(()),
525                }?;
526                for arg in args.iter() {
527                    Self::check_lookup_expr(arg)?
528                }
529                Ok(())
530            }
531            _ => Ok(()),
532        }
533    }
534
535    fn check_lookup_actions(actions: &ResolvedActions) -> Result<(), TypeError> {
536        for action in actions.iter() {
537            match action {
538                GenericAction::Let(_, _, rhs) => Self::check_lookup_expr(rhs),
539                GenericAction::Set(_, _, args, rhs) => {
540                    for arg in args.iter() {
541                        Self::check_lookup_expr(arg)?
542                    }
543                    Self::check_lookup_expr(rhs)
544                }
545                GenericAction::Union(_, lhs, rhs) => {
546                    Self::check_lookup_expr(lhs)?;
547                    Self::check_lookup_expr(rhs)
548                }
549                GenericAction::Change(_, _, _, args) => {
550                    for arg in args.iter() {
551                        Self::check_lookup_expr(arg)?
552                    }
553                    Ok(())
554                }
555                GenericAction::Panic(..) => Ok(()),
556                GenericAction::Expr(_, expr) => Self::check_lookup_expr(expr),
557            }?
558        }
559        Ok(())
560    }

562    fn typecheck_facts(
563        &self,
564        symbol_gen: &mut SymbolGen,
565        facts: &[Fact],
566    ) -> Result<Vec<ResolvedFact>, TypeError> {
567        let (query, mapped_facts) = Facts(facts.to_vec()).to_query(self, symbol_gen);
568        let mut problem = Problem::default();
569        problem.add_query(&query, self)?;
570        let assignment = problem
571            .solve(|sort: &ArcSort| sort.name())
572            .map_err(|e| e.to_type_error())?;
573        let annotated_facts = assignment.annotate_facts(&mapped_facts, self);
574        Ok(annotated_facts)
575    }

577    fn typecheck_actions(
578        &self,
579        symbol_gen: &mut SymbolGen,
580        actions: &Actions,
581        binding: &IndexMap<&str, (Span, ArcSort)>,
582    ) -> Result<ResolvedActions, TypeError> {
583        let mut binding_set: IndexSet<String> =
584            binding.keys().copied().map(str::to_string).collect();
585        let (actions, mapped_action) =
586            actions.to_core_actions(self, &mut binding_set, symbol_gen)?;
587        let mut problem = Problem::default();
588
589        // add actions to problem
590        problem.add_actions(&actions, self, symbol_gen)?;
591
592        // add bindings from the context
593        for (var, (span, sort)) in binding {
594            problem.assign_local_var_type(var, span.clone(), sort.clone())?;
595        }
596
597        let assignment = problem
598            .solve(|sort: &ArcSort| sort.name())
599            .map_err(|e| e.to_type_error())?;
600
601        let annotated_actions = assignment.annotate_actions(&mapped_action, self)?;
602        Ok(annotated_actions)
603    }

605    fn typecheck_expr(
606        &self,
607        symbol_gen: &mut SymbolGen,
608        expr: &Expr,
609        binding: &IndexMap<&str, (Span, ArcSort)>,
610    ) -> Result<ResolvedExpr, TypeError> {
611        let action = Action::Expr(expr.span(), expr.clone());
612        let typechecked_action = self.typecheck_action(symbol_gen, &action, binding)?;
613        match typechecked_action {
614            ResolvedAction::Expr(_, expr) => Ok(expr),
615            _ => unreachable!(),
616        }
617    }

663#[derive(Debug, Clone, Error)]
664pub enum TypeError {
665    #[error("{}\nArity mismatch, expected {expected} args: {expr}", .expr.span())]
666    Arity { expr: Expr, expected: usize },
667    #[error(
668        "{}\n Expect expression {expr} to have type {}, but get type {}",
