1//! User-facing execution state wrappers.
2//!
3//! Four wrappers around `core_relations::ExecutionState` expose different
4//! subsets of the database API based on the context in which a primitive runs:
5//!
6//! | Wrapper       | DB reads | DB writes | Used for                                 |
7//! |---------------|----------|-----------|------------------------------------------|
8//! | `PureState`   | no       | no        | rule LHS (seminaive on)                  |
9//! | `WriteState`  | no       | yes       | rule RHS (seminaive on)                  |
10//! | `ReadState`   | yes      | no        | top-level query-shaped commands          |
11//! | `FullState`   | yes      | yes       | top-level action-shaped commands, `eval` |
12//!
13//! Methods come from sealed capability traits implemented on each
14//! wrapper:
15//!
16//! - [`Core`] — base values, counters, container interning, conversion
17//!   sugar. Implemented for all four wrappers.
18//! - [`Read`] — name-indexed table lookup (`state.lookup("name", &[…])`).
19//!   Implemented for [`ReadState`] and [`FullState`].
20//! - [`Write`] — name-indexed writes (`set`/`add`/`remove`/
21//!   `subsume`/`union`/`panic`). Implemented for [`WriteState`] and
22//!   [`FullState`].
23//!
24//! Privileged seams (`call_external_func`, raw `&mut ExecutionState`)
25//! used by the `FunctionContainer` higher-order dispatch live on the
26//! crate-private [`Internal`] trait. User code cannot reach them.
27//!
28//! [`PurePrim`]: crate::PurePrim
29//! [`WritePrim`]: crate::WritePrim
30//! [`ReadPrim`]: crate::ReadPrim
31//! [`FullPrim`]: crate::FullPrim
32

54/// The four contexts a primitive may run in, named after the
55/// capability profile they grant. Each variant maps 1:1 to one of the
56/// state wrappers below: `Pure` ↔ [`PureState`], `Write` ↔ [`WriteState`],
57/// `Read` ↔ [`ReadState`], `Full` ↔ [`FullState`]. The egglog
58/// typechecker filters primitive definitions by whether they carry a
59/// runtime id for the surrounding `Context` at each call site.
60#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash, enum_map::Enum)]
61pub enum Context {
62    /// No DB reads, no DB writes. The body (LHS) of a rule running
63    /// under seminaive evaluation: a body read of live state means
64    /// the rule won't re-fire when the read row's contents change
65    /// in a later iteration; a body write makes no semantic sense.
66    Pure,
67    /// DB writes allowed, DB reads forbidden. The head (RHS) of a
68    /// rule running under seminaive evaluation: same re-firing
69    /// concern as `Pure` rules out reads, but staged writes are
70    /// fine.
71    Write,
72    /// DB reads allowed, DB writes forbidden. Top-level query-shaped
73    /// commands (`check`, condition evaluation) and the body of a
74    /// `:naive` rule. Reads are safe because there is no seminaive
75    /// epoch to violate.
76    Read,
77    /// DB reads and writes both allowed. Top-level action-shaped
78    /// commands (`eval`, `let`, action-mode `run-schedule`) and the
79    /// head of a `:naive` rule.
80    Full,
81}
82
83impl Context {
84    pub const ALL: [Context; 4] = [Context::Pure, Context::Write, Context::Read, Context::Full];
85}

87// =====================================================================
88// Sealed traits.
89//
90// These traits are `pub(crate)`: external users cannot bring them
91// into scope, so they cannot call the methods defined here even on
92// values they have. They give the public capability traits (`Core`,
93// `Write`) a way to reach the underlying `ExecutionState` from
94// default methods, and carry the privileged seams used by the
95// `FunctionContainer` higher-order dispatch.
96// =====================================================================
97
98/// Crate-private accessor + privileged-dispatch trait. Required
99/// methods are the accessors that every wrapper supplies; default
100/// methods are the privileged seams used by the `FunctionContainer`
101/// higher-order dispatch.
102pub(crate) trait Internal<'a, 'db: 'a>: 'a {
103    fn es(&self) -> &ExecutionState<'db>;
104    fn es_mut(&mut self) -> &mut ExecutionState<'db>;
105    /// The call-site [`Context`] this primitive was invoked from.
106    /// Stamped onto the wrapper at construction time by the
107    /// `ExternalFunction` wrapper closure; read by
108    /// [`Core::apply_function`] to route higher-order dispatch.
109    fn ctx(&self) -> Context;
110
111    fn call_external_func(&mut self, id: ExternalFunctionId, args: &[Value]) -> Option<Value> {
112        self.es_mut().call_external_func(id, args)
113    }
114    fn raw_exec_state(&mut self) -> &mut ExecutionState<'db> {
115        self.es_mut()
116    }
117    fn apply_resolved_function(&mut self, _func: &FuncType, _args: &[Value]) -> Option<Value> {
118        None
119    }
120
121    fn apply_table_function(
122        &mut self,
123        subtype: FunctionSubtype,
124        action: &TableAction,
125        args: &[Value],
126    ) -> Option<Value> {
127        match (subtype, self.ctx()) {
128            (FunctionSubtype::Constructor, Context::Write | Context::Full) => {
129                action.lookup_or_insert(self.es_mut(), args)
130            }
131            (FunctionSubtype::Custom, Context::Read | Context::Full) => {
132                action.lookup(self.es(), args)
133            }
134            _ => None,
135        }
136    }
137}

140/// wrapper that has a registry (`ReadState`, `WriteState`, `FullState`)

143pub(crate) trait RegistrySealed<'a, 'db: 'a>: Internal<'a, 'db> {

145}

151/// Core methods available on every state wrapper: base values,
152/// counters, container interning, value/base/container conversion sugar.
153/// Always seminaive-safe.
154#[allow(private_bounds)]
155pub trait Core<'a, 'db: 'a>: Internal<'a, 'db> {
156    /// Base-value pool (interned primitives like `i64`, `String`, …).
157    fn base_values(&self) -> &'a BaseValues {
158        self.es().base_values()
159    }
160    /// Signal that rule execution should stop after this firing.
161    fn trigger_early_stop(&self) {
162        self.es().trigger_early_stop()
163    }
164    /// Has someone called `trigger_early_stop`?
165    fn should_stop(&self) -> bool {
166        self.es().should_stop()
167    }
168    /// Container values for this EGraph.
169    fn container_values(&self) -> &'a ContainerValues {
170        self.es().container_values()
171    }
172
173    /// Register a container value, returning its interned `Value`.
174    fn register_container<C: ContainerValue>(&mut self, container: C) -> Value {
175        // `container_values()` returns `&'a ContainerValues` — a reference
176        // tied to the inner ExecutionState's lifetime, not to `&self` —
177        // so it doesn't conflict with the subsequent `&mut` reborrow.
178        // Avoiding the clone of `ExecutionState` here matters: this is
179        // hot-path code (every container intern goes through it) and
180        // the clone copies a non-trivial amount of state.
181        let cv = self.container_values();
182        let es = self.es_mut();
183        cv.register_val(container, es)
184    }
185
186    /// Convert an egglog [`Value`] to a Rust base type, assuming that the
187    /// value belongs to `T`.
188    fn value_to_base<T: BaseValue>(&self, x: Value) -> T {
189        self.es().base_values().unwrap::<T>(x)
190    }
191
192    /// Convert a Rust base type to an egglog [`Value`].
193    fn base_to_value<T: BaseValue>(&self, x: T) -> Value {
194        self.es().base_values().get::<T>(x)
195    }
196
197    /// Look up the Rust container behind an egglog [`Value`], if any.
198    fn value_to_container<T: ContainerValue>(
199        &self,
200        x: Value,
201    ) -> Option<impl Deref<Target = T> + 'a> {
202        self.es().container_values().get_val::<T>(x)
203    }
204
205    /// Intern a Rust container into the e-graph and return its
206    /// [`Value`]. Sugar over `self.register_container(x)`.
207    fn container_to_value<T: ContainerValue>(&mut self, x: T) -> Value {
208        self.register_container(x)
209    }
210
211    /// Dispatch a wrapped `unstable-fn` value. This is the public entry
212    /// point for higher-order primitive bodies: the call-site
213    /// [`Context`] is stamped onto the state by the registration
214    /// wrapper (see [`EGraph::add_pure_primitive`] and the matching
215    /// `add_read_primitive` / `add_write_primitive` /
216    /// `add_full_primitive`), so the caller can't supply a wrong
217    /// context — there is no `ctx` parameter to lie about.
218    ///
219    /// [`EGraph::add_pure_primitive`]: crate::EGraph::add_pure_primitive
220    fn apply_function(
221        &mut self,
222        fc: &crate::sort::FunctionContainer,
223        args: &[Value],
224    ) -> Option<Value> {
225        let ctx = self.ctx();
226        let mut pure = PureState::wrap(self.raw_exec_state(), ctx);
227        fc.apply(&mut pure, args)
228    }
229
230    /// Dispatch an already type-specialized primitive in the current
231    /// call-site context.
232    ///
233    /// This is a trusted evaluator hook, not an authorization boundary:
234    /// callers must only pass primitives that were resolved for this same
235    /// call-site context and whose surrounding expression has already been
236    /// checked to require no more capability than this state wrapper provides.
237    /// For example, a primitive body evaluator should typecheck the body under
238    /// the runtime context it will register for, infer the body's required
239    /// context, and register the primitive using the matching state wrapper.
240    fn apply_primitive(
241        &mut self,
242        primitive: &crate::core::SpecializedPrimitive,
243        args: &[Value],
244    ) -> Option<Value> {
245        let id = primitive.external_id(self.ctx());
246        self.es_mut().call_external_func(id, args)
247    }
248
249    /// Evaluate an already typechecked expression in this primitive call
250    /// context, using `bindings` for local variables.
251    ///
252    /// Primitive calls dispatch through the current call-site context.
253    /// Table-backed function calls follow the same capability split as
254    /// `unstable-app`: custom functions require a read-capable state,
255    /// and constructors require a write-capable state that can mint on miss.
256    ///
257    /// This method assumes `expr` came from a trusted preparation pipeline that
258    /// typechecked it for this exact runtime context and rejected expressions
259    /// whose required capabilities exceed the receiver's state wrapper. It
260    /// should not be used to evaluate arbitrary resolved expressions from a
261    /// wider context inside a less-capable wrapper.
262    fn eval_resolved_expr(
263        &mut self,
264        expr: &ResolvedExpr,
265        bindings: &[(&str, Value)],
266    ) -> Option<Value> {
267        match expr {
268            ResolvedExpr::Lit(_, literal) => Some(match literal {
269                Literal::Int(x) => self.base_to_value(*x),
270                Literal::Float(x) => self.base_to_value(F::from(*x)),
271                Literal::String(x) => self.base_to_value(S::new(x.clone())),
272                Literal::Bool(x) => self.base_to_value(*x),
273                Literal::Unit => self.base_to_value(()),
274            }),
275            ResolvedExpr::Var(_, resolved_var) => {
276                assert!(
277                    !resolved_var.is_global_ref,
278                    "global variable {:?} reached direct expression evaluation before remove_globals",
279                    resolved_var.name
280                );
281                bindings
282                    .iter()
283                    .find_map(|(name, value)| (*name == resolved_var.name).then_some(*value))
284            }
285            ResolvedExpr::Call(_, resolved_call, children) => {
286                let mut values = Vec::with_capacity(children.len());
287                for child in children {
288                    values.push(self.eval_resolved_expr(child, bindings)?);
289                }
290                match resolved_call {
291                    ResolvedCall::Primitive(primitive) => self.apply_primitive(primitive, &values),
292                    ResolvedCall::Func(func) => self.apply_resolved_function(func, &values),
293                }
294            }
295        }
296    }
297}

299/// Read-side methods — name-indexed table lookup and iteration.
300/// Implemented for [`ReadState`] and [`FullState`]; *not* for
301/// [`PureState`] or [`WriteState`] (a `Write` context body must not
302/// depend on live DB state).
303///
304/// The single-entry methods (`lookup`, `eclass_of`, `contains`)
305/// return `None` if absent — never insert. The iteration /
306/// introspection methods (`function_entries`, `constructor_enodes`,
307/// `table_size`, `table_sizes`) walk the current contents of the
308/// database.
309///
310/// Detectable misuse (wrong table subtype, wrong arity) is reported
311/// as [`crate::ApiError`] via the method's `Result`. Per-column sort
312/// matching is **not** checked at this layer.
313#[allow(private_bounds)]
314pub trait Read<'a, 'db: 'a>: Core<'a, 'db> + RegistrySealed<'a, 'db> {
315    /// Look up a function's output value at the given key. Returns
316    /// `Ok(None)` if the row is absent. The returned `Value` is raw —
317    /// extract a Rust type via [`Core::value_to_base`] for base sorts
318    /// or [`Core::value_to_container`] for containers.
319    ///
320    /// **Only valid for `function` tables.** Constructors error;
321    /// use [`Read::eclass_of`] for those.
322    fn lookup<K: IntoValues>(&self, name: &str, key: K) -> Result<Option<Value>, Error> {
323        let action = lookup_action(self.registry(), name)?;
324        check_subtype(name, &action, TableKind::Function, "function")?;
325        let key_values: ValueRow = key.into_values(self.base_values()).collect();
326        check_arity(name, &action, key_values.len())?;
327        Ok(action.lookup(self.es(), &key_values))
328    }
329
330    /// Look up a constructor's eclass at the given inputs, without
331    /// minting a fresh one on miss. Returns `Ok(None)` if absent.
332    ///
333    /// **Only valid for constructor tables.** Functions error;
334    /// use [`Read::lookup`] for those.
335    fn eclass_of<K: IntoValues>(&self, name: &str, inputs: K) -> Result<Option<Value>, Error> {
336        let action = lookup_action(self.registry(), name)?;
337        check_subtype(name, &action, TableKind::Constructor, "constructor")?;
338        let key_values: ValueRow = inputs.into_values(self.base_values()).collect();
339        check_arity(name, &action, key_values.len())?;
340        Ok(action.lookup(self.es(), &key_values))
341    }
342
343    /// True iff a row with the given key exists in the table. Works
344    /// for any subtype — never mints.
345    fn contains<K: IntoValues>(&self, name: &str, key: K) -> Result<bool, Error> {
346        let action = lookup_action(self.registry(), name)?;
347        let key_values: ValueRow = key.into_values(self.base_values()).collect();
348        check_arity(name, &action, key_values.len())?;
349        Ok(action.lookup(self.es(), &key_values).is_some())
350    }
351
352    /// Return the current row count for the named table, or `None` if no table
353    /// with that name is registered.
354    fn table_size(&self, name: &str) -> Option<usize> {
355        self.registry()
356            .lookup_table(name)
357            .map(|action| action.row_count(self.es()))
358    }
359
360    /// Snapshot the registered table names and their current row counts.
361    fn table_sizes(&self) -> Vec<(&str, usize)> {
362        self.registry().table_sizes(self.es())
363    }
364
365    /// Call `f` on each [`Enode`] of a constructor / relation table.
366    /// Errors with `WrongSubtype` if `name` is a function. To stop
367    /// early, use [`Read::constructor_enodes_while`].
368    fn constructor_enodes(&self, name: &str, mut f: impl FnMut(Enode<'_>)) -> Result<(), Error> {
369        self.constructor_enodes_while(name, |enode| {
370            f(enode);
371            true
372        })
373    }
374
375    /// Like [`Read::constructor_enodes`], but stops as soon as `f`
376    /// returns `false`.
377    fn constructor_enodes_while(
378        &self,
379        name: &str,
380        mut f: impl FnMut(Enode<'_>) -> bool,
381    ) -> Result<(), Error> {
382        let action = lookup_action(self.registry(), name)?;
383        check_subtype(name, &action, TableKind::Constructor, "constructor")?;
384        action.for_each_while(self.es(), |row| {
385            let (eclass, children) = row
386                .vals
387                .split_last()
388                .expect("constructor row has at least an eclass column");
389            f(Enode {
390                children,
391                eclass: *eclass,
392                subsumed: row.subsumed,
393            })
394        });
395        Ok(())
396    }
397
398    /// Call `f` on each [`FunctionEntry`] of a function table. Errors
399    /// with `WrongSubtype` if `name` is a constructor. To stop early,
400    /// use [`Read::function_entries_while`].
401    fn function_entries(
402        &self,
403        name: &str,
404        mut f: impl FnMut(FunctionEntry<'_>),
405    ) -> Result<(), Error> {
406        self.function_entries_while(name, |entry| {
407            f(entry);
408            true
409        })
410    }
411
412    /// Like [`Read::function_entries`], but stops as soon as `f`
413    /// returns `false`.
414    fn function_entries_while(
415        &self,
416        name: &str,
417        mut f: impl FnMut(FunctionEntry<'_>) -> bool,
418    ) -> Result<(), Error> {
419        let action = lookup_action(self.registry(), name)?;
420        check_subtype(name, &action, TableKind::Function, "function")?;
421        action.for_each_while(self.es(), |row| {
422            let (output, inputs) = row
423                .vals
424                .split_last()
425                .expect("function row has at least an output column");
426            f(FunctionEntry {
427                inputs,
428                output: *output,
429                subsumed: row.subsumed,
430            })
431        });
432        Ok(())
433    }
434}

436/// One enode from [`Read::constructor_enodes`]. Columns are raw
437/// [`Value`]s; convert with [`Core::value_to_base`] / [`Core::value_to_container`].
438#[derive(Clone, Copy, Debug)]
439pub struct Enode<'a> {
440    /// The constructor's input columns.
441    pub children: &'a [Value],
442    /// The eclass id this enode belongs to.
443    pub eclass: Value,
444    /// Whether this enode has been subsumed.
445    pub subsumed: bool,
446}

448/// One entry from [`Read::function_entries`]. Columns are raw
449/// [`Value`]s; convert with [`Core::value_to_base`] / [`Core::value_to_container`].
450#[derive(Clone, Copy, Debug)]
451pub struct FunctionEntry<'a> {
452    /// The function's key (input) columns.
453    pub inputs: &'a [Value],
454    /// The value the function maps the key to.

456    /// Whether this entry has been subsumed.
457    pub subsumed: bool,
458}

460/// Action-side write methods — name-indexed inserts/removes/subsumes
461/// plus union and panic. Implemented for [`WriteState`] and
462/// [`FullState`]; *not* for [`PureState`] or [`ReadState`].
463///
464/// Detectable misuse (wrong table subtype, wrong arity) is reported
465/// as [`crate::ApiError`] via the method's `Result`. Per-column sort
466/// matching is **not** checked at this layer.
467#[allow(private_bounds)]
468pub trait Write<'a, 'db: 'a>: Core<'a, 'db> + RegistrySealed<'a, 'db> {
469    /// Set a function table's value at the given key — mirrors the
470    /// egglog `(set (f k) v)` action.
471    ///
472    /// **Only valid for `function` tables.** Constructors error;
473    /// use [`Write::add`] for those.
474    fn set<K: IntoValues, V: IntoValue>(
475        &mut self,
476        name: &str,
477        key: K,
478        value: V,
479    ) -> Result<(), Error> {
480        let action = lookup_action(self.registry(), name)?;
481        check_subtype(name, &action, TableKind::Function, "function")?;
482        let bv = self.base_values();
483        let mut row: ValueRow = key.into_values(bv).collect();
484        check_arity(name, &action, row.len())?;
485        row.push(value.into_value(bv));
486        action.insert(self.es_mut(), row.into_iter());
487        Ok(())
488    }
489
490    /// Mint or look up an eclass for a constructor — mirrors the
491    /// egglog `(Cons k1 k2 ...)` expression form. Pass inputs only;
492    /// the output eclass is minted (or returned if a row with these
493    /// inputs already exists).
494    ///
495    /// **Only valid for constructor tables.** Functions error;
496    /// use [`Write::set`] for those.
497    fn add<R: IntoValues>(&mut self, name: &str, inputs: R) -> Result<Value, Error> {
498        let action = lookup_action(self.registry(), name)?;
499        check_subtype(name, &action, TableKind::Constructor, "constructor")?;
500        let key: ValueRow = inputs.into_values(self.base_values()).collect();
501        check_arity(name, &action, key.len())?;
502        let value = action
503            .lookup_or_insert(self.es_mut(), &key)
504            .expect("constructor lookup_or_insert returned None");
505        Ok(value)
506    }
507
508    /// Remove a row from the named table. Works for any subtype.
509    fn remove<K: IntoValues>(&mut self, name: &str, key: K) -> Result<(), Error> {
510        let action = lookup_action(self.registry(), name)?;
511        let key_values: ValueRow = key.into_values(self.base_values()).collect();
512        check_arity(name, &action, key_values.len())?;
513        action.remove(self.es_mut(), &key_values);
514        Ok(())
515    }
516
517    /// Subsume a row in the named table.
518    fn subsume<K: IntoValues>(&mut self, name: &str, key: K) -> Result<(), Error> {
519        let action = lookup_action(self.registry(), name)?;
520        let key_values: ValueRow = key.into_values(self.base_values()).collect();
521        check_arity(name, &action, key_values.len())?;
522        action.subsume(self.es_mut(), key_values.into_iter());
523        Ok(())
524    }
525
526    /// Union two values in the e-graph's union-find. The caller is
527    /// responsible for ensuring both values belong to the same
528    /// eq-sort.
529    fn union(&mut self, x: Value, y: Value) -> Result<(), Error> {
530        let action = *self.registry().union_action();
531        action.union(self.es_mut(), x, y);
532        Ok(())
533    }
534
535    /// Trigger a panic from a primitive. Always returns `None` so the
536    /// caller can propagate with `?`.
537    fn panic(&mut self) -> Option<()> {
538        let panic_id = self.registry().default_panic_id();
539        self.es_mut().call_external_func(panic_id, &[]);
540        None
541    }
542}

544fn lookup_action(registry: &ActionRegistry, name: &str) -> Result<TableAction, Error> {
545    registry.lookup_table(name).cloned().ok_or_else(|| {
546        ApiError::MissingTable {
547            name: name.to_string(),
548        }
549        .into()
550    })
551}
552
553fn check_subtype(
554    name: &str,
555    action: &TableAction,
556    expected: TableKind,
557    expected_label: &'static str,
558) -> Result<(), Error> {
559    if action.kind() == expected {
560        return Ok(());
561    }
562    let actual_label = match action.kind() {
563        TableKind::Function => "function",
564        TableKind::Constructor => "constructor",
565    };
566    Err(ApiError::WrongSubtype {
567        name: name.to_string(),
568        expected: expected_label,
569        actual: actual_label,
570    }
571    .into())
572}
573
574fn check_arity(table: &str, action: &TableAction, got: usize) -> Result<(), Error> {
575    let expected = action.input_arity();
576    if got != expected {
577        return Err(ApiError::WrongArity {
578            table: table.to_string(),
579            expected,
580            got,
581        }
582        .into());
583    }
584    Ok(())
585}

587fn apply_registered_function<'a, 'db: 'a>(
588    state: &mut impl RegistrySealed<'a, 'db>,
589    func: &FuncType,
590    args: &[Value],
591) -> Option<Value> {
592    let action = lookup_action(state.registry(), &func.name).ok()?;
593    state.apply_table_function(func.subtype, &action, args)
594}

596// =====================================================================
597// The four wrapper types — plain structs, methods come from traits.
598// =====================================================================
599
600/// Wrapper for [`Context::Pure`]. Implements [`Core`] only.
601///
602/// ```compile_fail
603/// // Pure context cannot write: `Write` is not implemented.
604/// use egglog::Write;
605/// fn _no_writes<'a, 'db>(state: &mut egglog::PureState<'a, 'db>) {
606///     state.set("foo", (1_i64,), 2_i64);
607/// }
608/// ```
609///
610/// ```compile_fail
611/// // Pure context cannot reach raw `ExecutionState`.
612/// fn _no_raw<'a, 'db>(state: &mut egglog::PureState<'a, 'db>) {
613///     state.raw_exec_state();
614/// }
615/// ```
616pub struct PureState<'a, 'db> {
617    pub(crate) inner: &'a mut ExecutionState<'db>,
618    /// The call-site [`Context`] the wrapping primitive was invoked
619    /// from. Stamped by the wrapper closure at invocation time and
620    /// read by [`PureState::apply_function`]; user code cannot
621    /// observe or modify it directly.
622    pub(crate) ctx: Context,
623}
624
625/// Wrapper for [`Context::Read`]. Implements [`Core`] + [`Read`].
626pub struct ReadState<'a, 'db> {
627    pub(crate) inner: &'a mut ExecutionState<'db>,
628    pub(crate) registry: &'a ActionRegistry,
629    pub(crate) ctx: Context,
630}
631
632/// Wrapper for [`Context::Write`]. Implements [`Core`] + [`Write`].
633///
634/// ```compile_fail
635/// // Write context cannot reach raw `ExecutionState`.
636/// fn _no_raw<'a, 'db>(state: &mut egglog::WriteState<'a, 'db>) {
637///     state.raw_exec_state();
638/// }
639/// ```
640pub struct WriteState<'a, 'db> {
641    pub(crate) inner: &'a mut ExecutionState<'db>,
642    pub(crate) registry: &'a ActionRegistry,
643    pub(crate) ctx: Context,
644}
645
646/// Wrapper for [`Context::Full`]. Implements [`Core`] + [`Read`] + [`Write`].
647///
648/// ```compile_fail
649/// // Even `FullState` cannot reach the raw `ExecutionState`.
650/// fn _no_raw<'a, 'db>(state: &mut egglog::FullState<'a, 'db>) {
651///     state.raw_exec_state();
652/// }
653/// ```
654pub struct FullState<'a, 'db> {
655    pub(crate) inner: &'a mut ExecutionState<'db>,
656    pub(crate) registry: &'a ActionRegistry,
657    pub(crate) ctx: Context,
658}
659
660impl<'a, 'db: 'a> PureState<'a, 'db> {

662        Self { inner: es, ctx }
663    }
664    pub const fn valid_contexts() -> &'static [Context] {
665        &Context::ALL
666    }
667}
