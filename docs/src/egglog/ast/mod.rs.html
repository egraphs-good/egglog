34/// A [`NCommand`] is a desugared [`Command`], where syntactic sugars
35/// like [`Command::Datatype`] and [`Command::Rewrite`]
36/// are eliminated.
37/// Most of the heavy lifting in egglog is done over [`NCommand`]s.
38///
39/// [`GenericNCommand`] is a generalization of [`NCommand`], like how [`GenericCommand`]
40/// is a generalization of [`Command`], allowing annotations over `Head` and `Leaf`.
41///
42/// TODO: The name "NCommand" used to denote normalized command, but this
43/// meaning is obsolete. A future PR should rename this type to something
44/// like "DCommand".
45#[derive(Debug, Clone, Eq, PartialEq, Hash)]
46pub enum GenericNCommand<Head, Leaf>
47where
48    Head: Clone + Display,
49    Leaf: Clone + PartialEq + Eq + Display + Hash,
50{
51    Sort(
52        Span,
53        String,
54        Option<(String, Vec<GenericExpr<String, String>>)>,
55    ),
56    Function(GenericFunctionDecl<Head, Leaf>),
57    AddRuleset(Span, String),
58    UnstableCombinedRuleset(Span, String, Vec<String>),
59    NormRule {
60        rule: GenericRule<Head, Leaf>,
61    },
62    CoreAction(GenericAction<Head, Leaf>),
63    Extract(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
64    RunSchedule(GenericSchedule<Head, Leaf>),
65    PrintOverallStatistics(Span, Option<String>),
66    Check(Span, Vec<GenericFact<Head, Leaf>>),
67    PrintFunction(
68        Span,
69        String,
70        Option<usize>,
71        Option<String>,
72        PrintFunctionMode,
73    ),
74    PrintSize(Span, Option<String>),
75    Output {
76        span: Span,
77        file: String,
78        exprs: Vec<GenericExpr<Head, Leaf>>,
79    },
80    Push(usize),
81    Pop(Span, usize),
82    Fail(Span, Box<GenericNCommand<Head, Leaf>>),
83    Input {
84        span: Span,
85        name: String,
86        file: String,
87    },
88    UserDefined(Span, String, Vec<Expr>),
89}

96    pub fn to_command(&self) -> GenericCommand<Head, Leaf> {
97        match self {
98            GenericNCommand::Sort(span, name, params) => {
99                GenericCommand::Sort(span.clone(), name.clone(), params.clone())
100            }
101            GenericNCommand::Function(f) => match f.subtype {
102                FunctionSubtype::Constructor => GenericCommand::Constructor {
103                    span: f.span.clone(),
104                    name: f.name.clone(),
105                    schema: f.schema.clone(),
106                    cost: f.cost,
107                    unextractable: f.unextractable,
108                },
109                FunctionSubtype::Relation => GenericCommand::Relation {
110                    span: f.span.clone(),
111                    name: f.name.clone(),
112                    inputs: f.schema.input.clone(),
113                },
114                FunctionSubtype::Custom => GenericCommand::Function {
115                    span: f.span.clone(),
116                    schema: f.schema.clone(),
117                    name: f.name.clone(),
118                    merge: f.merge.clone(),
119                },
120            },
121            GenericNCommand::AddRuleset(span, name) => {
122                GenericCommand::AddRuleset(span.clone(), name.clone())
123            }
124            GenericNCommand::UnstableCombinedRuleset(span, name, others) => {
125                GenericCommand::UnstableCombinedRuleset(span.clone(), name.clone(), others.clone())
126            }
127            GenericNCommand::NormRule { rule } => GenericCommand::Rule { rule: rule.clone() },
128            GenericNCommand::RunSchedule(schedule) => GenericCommand::RunSchedule(schedule.clone()),
129            GenericNCommand::PrintOverallStatistics(span, file) => {
130                GenericCommand::PrintOverallStatistics(span.clone(), file.clone())
131            }
132            GenericNCommand::CoreAction(action) => GenericCommand::Action(action.clone()),
133            GenericNCommand::Extract(span, expr, variants) => {
134                GenericCommand::Extract(span.clone(), expr.clone(), variants.clone())
135            }
136            GenericNCommand::Check(span, facts) => {
137                GenericCommand::Check(span.clone(), facts.clone())
138            }
139            GenericNCommand::PrintFunction(span, name, n, file, mode) => {
140                GenericCommand::PrintFunction(span.clone(), name.clone(), *n, file.clone(), *mode)
141            }
142            GenericNCommand::PrintSize(span, name) => {
143                GenericCommand::PrintSize(span.clone(), name.clone())
144            }
145            GenericNCommand::Output { span, file, exprs } => GenericCommand::Output {
146                span: span.clone(),
147                file: file.to_string(),
148                exprs: exprs.clone(),
149            },
150            GenericNCommand::Push(n) => GenericCommand::Push(*n),
151            GenericNCommand::Pop(span, n) => GenericCommand::Pop(span.clone(), *n),
152            GenericNCommand::Fail(span, cmd) => {
153                GenericCommand::Fail(span.clone(), Box::new(cmd.to_command()))
154            }
155            GenericNCommand::Input { span, name, file } => GenericCommand::Input {
156                span: span.clone(),
157                name: name.clone(),
158                file: file.clone(),
159            },
160            GenericNCommand::UserDefined(span, name, exprs) => {
161                GenericCommand::UserDefined(span.clone(), name.clone(), exprs.clone())
162            }
163        }
164    }
165
166    pub fn visit_exprs(
167        self,
168        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
169    ) -> Self {
170        match self {
171            GenericNCommand::Sort(span, name, params) => GenericNCommand::Sort(span, name, params),
172            GenericNCommand::Function(func) => GenericNCommand::Function(func.visit_exprs(f)),
173            GenericNCommand::AddRuleset(span, name) => GenericNCommand::AddRuleset(span, name),
174            GenericNCommand::UnstableCombinedRuleset(span, name, rulesets) => {
175                GenericNCommand::UnstableCombinedRuleset(span, name, rulesets)
176            }
177            GenericNCommand::NormRule { rule } => GenericNCommand::NormRule {
178                rule: rule.visit_exprs(f),
179            },
180            GenericNCommand::RunSchedule(schedule) => {
181                GenericNCommand::RunSchedule(schedule.visit_exprs(f))
182            }
183            GenericNCommand::PrintOverallStatistics(span, file) => {
184                GenericNCommand::PrintOverallStatistics(span, file)
185            }
186            GenericNCommand::CoreAction(action) => {
187                GenericNCommand::CoreAction(action.visit_exprs(f))
188            }
189            GenericNCommand::Extract(span, expr, variants) => {
190                GenericNCommand::Extract(span, expr.visit_exprs(f), variants.visit_exprs(f))
191            }
192            GenericNCommand::Check(span, facts) => GenericNCommand::Check(
193                span,
194                facts.into_iter().map(|fact| fact.visit_exprs(f)).collect(),
195            ),
196            GenericNCommand::PrintFunction(span, name, n, file, mode) => {
197                GenericNCommand::PrintFunction(span, name, n, file, mode)
198            }
199            GenericNCommand::PrintSize(span, name) => GenericNCommand::PrintSize(span, name),
200            GenericNCommand::Output { span, file, exprs } => GenericNCommand::Output {
201                span,
202                file,
203                exprs: exprs.into_iter().map(f).collect(),
204            },
205            GenericNCommand::Push(n) => GenericNCommand::Push(n),
206            GenericNCommand::Pop(span, n) => GenericNCommand::Pop(span, n),
207            GenericNCommand::Fail(span, cmd) => {
208                GenericNCommand::Fail(span, Box::new(cmd.visit_exprs(f)))
209            }
210            GenericNCommand::Input { span, name, file } => {
211                GenericNCommand::Input { span, name, file }
212            }
213            GenericNCommand::UserDefined(span, name, exprs) => {
214                // We can't map `f` over UserDefined because UserDefined always assumes plain `Expr`s
215                GenericNCommand::UserDefined(span, name, exprs)
216            }
217        }
218    }
219}

224#[derive(Debug, Clone, PartialEq, Eq, Hash)]
225pub enum GenericSchedule<Head, Leaf> {
226    Saturate(Span, Box<GenericSchedule<Head, Leaf>>),
227    Repeat(Span, usize, Box<GenericSchedule<Head, Leaf>>),
228    Run(Span, GenericRunConfig<Head, Leaf>),
229    Sequence(Span, Vec<GenericSchedule<Head, Leaf>>),
230}
231
232impl<Head, Leaf> GenericSchedule<Head, Leaf>
233where
234    Head: Clone + Display,
235    Leaf: Clone + PartialEq + Eq + Display + Hash,
236{
237    fn visit_exprs(
238        self,
239        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
240    ) -> Self {
241        match self {
242            GenericSchedule::Saturate(span, sched) => {
243                GenericSchedule::Saturate(span, Box::new(sched.visit_exprs(f)))
244            }
245            GenericSchedule::Repeat(span, size, sched) => {
246                GenericSchedule::Repeat(span, size, Box::new(sched.visit_exprs(f)))
247            }
248            GenericSchedule::Run(span, config) => GenericSchedule::Run(span, config.visit_exprs(f)),
249            GenericSchedule::Sequence(span, scheds) => GenericSchedule::Sequence(
250                span,
251                scheds.into_iter().map(|s| s.visit_exprs(f)).collect(),
252            ),
253        }
254    }
255}

257impl<Head: Display, Leaf: Display> Display for GenericSchedule<Head, Leaf> {
258    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
259        match self {
260            GenericSchedule::Saturate(_ann, sched) => write!(f, "(saturate {sched})"),
261            GenericSchedule::Repeat(_ann, size, sched) => write!(f, "(repeat {size} {sched})"),
262            GenericSchedule::Run(_ann, config) => write!(f, "{config}"),
263            GenericSchedule::Sequence(_ann, scheds) => {
264                write!(f, "(seq {})", ListDisplay(scheds, " "))
265            }
266        }
267    }

298/// A [`Command`] is the top-level construct in egglog.
299/// It includes defining rules, declaring functions,
300/// adding to tables, and running rules (via a [`Schedule`]).
301#[derive(Debug, Clone)]
302pub enum GenericCommand<Head, Leaf>
303where
304    Head: Clone + Display,
305    Leaf: Clone + PartialEq + Eq + Display + Hash,
306{
307    /// Create a new user-defined sort, which can then
308    /// be used in new [`Command::Function`] declarations.
309    /// The [`Command::Datatype`] command desugars directly to this command, with one [`Command::Function`]
310    /// per constructor.
311    /// The main use of this command (as opposed to using [`Command::Datatype`]) is for forward-declaring a sort for mutually-recursive datatypes.
312    ///
313    /// It can also be used to create
314    /// a container sort.
315    /// For example, here's how to make a sort for vectors
316    /// of some user-defined sort `Math`:
317    /// ```text
318    /// (sort MathVec (Vec Math))
319    /// ```
320    ///
321    /// Now `MathVec` can be used as an input or output sort.
322    Sort(Span, String, Option<(String, Vec<Expr>)>),
323
324    /// Egglog supports three types of functions
325    ///
326    /// A constructor models an egg-style user-defined datatype
327    /// It can only be defined through the `datatype`/`datatype*` command
328    /// or the `constructor` command
329    ///
330    /// A relation models a datalog-style mathematical relation
331    /// It can only be defined through the `relation` command
332    ///
333    /// A custom function is a dictionary
334    /// It can only be defined through the `function` command
335    ///
336    /// The `datatype` command declares a user-defined datatype.
337    /// Datatypes can be unioned with [`Action::Union`] either
338    /// at the top level or in the actions of a rule.
339    /// This makes them equal in the implicit, global equality relation.
340    ///
341    /// Example:
342    /// ```text
343    /// (datatype Math
344    ///   (Num i64)
345    ///   (Var String)
346    ///   (Add Math Math)
347    ///   (Mul Math Math))
348    /// ```
349    ///
350    /// defines a simple `Math` datatype with variants for numbers, named variables, addition and multiplication.
351    ///
352    /// Datatypes desugar directly to a [`Command::Sort`] and a [`Command::Constructor`] for each constructor.
353    /// The code above becomes:
354    /// ```text
355    /// (sort Math)
356    /// (constructor Num (i64) Math)
357    /// (constructor Var (String) Math)
358    /// (constructor Add (Math Math) Math)
359    /// (constructor Mul (Math Math) Math)
360    ///
361    /// Datatypes are also known as algebraic data types, tagged unions and sum types.
362    Datatype {
363        span: Span,
364        name: String,
365        variants: Vec<Variant>,
366    },
367    Datatypes {
368        span: Span,
369        datatypes: Vec<(Span, String, Subdatatypes)>,
370    },
371
372    /// The `constructor` command defines a new constructor for a user-defined datatype
373    /// Example:
374    /// ```text
375    /// (constructor Add (i64 i64) Math)
376    /// ```
377    ///
378    Constructor {
379        span: Span,
380        name: String,
381        schema: Schema,
382        cost: Option<DefaultCost>,
383        unextractable: bool,
384    },
385
386    /// The `relation` command declares a named relation
387    /// Example:
388    /// ```text
389    /// (relation path (i64 i64))
390    /// (relation edge (i64 i64))
391    /// ```
392    Relation {
393        span: Span,
394        name: String,
395        inputs: Vec<String>,
396    },
397
398    /// The `function` command declare an egglog custom function, which is a database table with a
399    /// a functional dependency (also called a primary key) on its inputs to one output.
400    ///
401    /// ```text
402    /// (function <name:Ident> <schema:Schema> <cost:Cost>
403    ///        (:on_merge <List<Action>>)?
404    ///        (:merge <Expr>)?)
405    ///```
406    /// A function can have a `cost` for extraction.
407    ///
408    /// Finally, it can have a `merge` and `on_merge`, which are triggered when
409    /// the function dependency is violated.
410    /// In this case, the merge expression determines which of the two outputs
411    /// for the same input is used.
412    /// The `on_merge` actions are run after the merge expression is evaluated.
413    ///
414    /// Note that the `:merge` expression must be monotonic
415    /// for the behavior of the egglog program to be consistent and defined.
416    /// In other words, the merge function must define a lattice on the output of the function.
417    /// If values are merged in different orders, they should still result in the same output.
418    /// If the merge expression is not monotonic, the behavior can vary as
419    /// actions may be applied more than once with different results.
420    ///
421    /// ```text
422    /// (function LowerBound (Math) i64 :merge (max old new))
423    /// ```
424    ///
425    /// Specifically, a custom function can also have an EqSort output type:
426    ///
427    /// ```text
428    /// (function Add (i64 i64) Math)
429    /// ```
430    ///
431    /// All functions can be `set`
432    /// with [`Action::Set`].
433    ///
434    /// Output of a function, if being the EqSort type, can be unioned with [`Action::Union`]
435    /// with another datatype of the same `sort`.
436    ///
437    Function {
438        span: Span,
439        name: String,
440        schema: Schema,
441        merge: Option<GenericExpr<Head, Leaf>>,
442    },
443
444    /// Using the `ruleset` command, defines a new
445    /// ruleset that can be added to in [`Command::Rule`]s.
446    /// Rulesets are used to group rules together
447    /// so that they can be run together in a [`Schedule`].
448    ///
449    /// Example:
450    /// Ruleset allows users to define a ruleset- a set of rules
451    ///
452    /// ```text
453    /// (ruleset myrules)
454    /// (rule ((edge x y))
455    ///       ((path x y))
456    ///       :ruleset myrules)
457    /// (run myrules 2)
458    /// ```
459    AddRuleset(Span, String),
460    /// Using the `combined-ruleset` command, construct another ruleset
461    /// which runs all the rules in the given rulesets.
462    /// This is useful for running multiple rulesets together.
463    /// The combined ruleset also inherits any rules added to the individual rulesets
464    /// after the combined ruleset is declared.
465    ///
466    /// Example:
467    /// ```text
468    /// (ruleset myrules1)
469    /// (rule ((edge x y))
470    ///       ((path x y))
471    ///      :ruleset myrules1)
472    /// (ruleset myrules2)
473    /// (rule ((path x y) (edge y z))
474    ///       ((path x z))
475    ///       :ruleset myrules2)
476    /// (combined-ruleset myrules-combined myrules1 myrules2)
477    /// ```
478    UnstableCombinedRuleset(Span, String, Vec<String>),
479    /// ```text
480    /// (rule <body:List<Fact>> <head:List<Action>>)
481    /// ```
482    ///
483    /// defines an egglog rule.
484    /// The rule matches a list of facts with respect to
485    /// the global database, and runs the list of actions
486    /// for each match.
487    /// The matches are done *modulo equality*, meaning
488    /// equal datatypes in the database are considered
489    /// equal.
490    ///
491    /// Example:
492    /// ```text
493    /// (rule ((edge x y))
494    ///       ((path x y)))
495    ///
496    /// (rule ((path x y) (edge y z))
497    ///       ((path x z)))
498    /// ```
499    Rule { rule: GenericRule<Head, Leaf> },
500    /// `rewrite` is syntactic sugar for a specific form of `rule`
501    /// which simply unions the left and right hand sides.
502    ///
503    /// Example:
504    /// ```text
505    /// (rewrite (Add a b)
506    ///          (Add b a))
507    /// ```
508    ///
509    /// Desugars to:
510    /// ```text
511    /// (rule ((= lhs (Add a b)))
512    ///       ((union lhs (Add b a))))
513    /// ```
514    ///
515    /// Additionally, additional facts can be specified
516    /// using a `:when` clause.
517    /// For example, the same rule can be run only
518    /// when `a` is zero:
519    ///
520    /// ```text
521    /// (rewrite (Add a b)
522    ///          (Add b a)
523    ///          :when ((= a (Num 0)))
524    /// ```
525    ///
526    /// Add the `:subsume` flag to cause the left hand side to be subsumed after matching, which means it can
527    /// no longer be matched in a rule, but can still be checked against (See [`Change`] for more details.)
528    ///
529    /// ```text
530    /// (rewrite (Mul a 2) (bitshift-left a 1) :subsume)
531    /// ```
532    ///
533    /// Desugars to:
534    /// ```text
535    /// (rule ((= lhs (Mul a 2)))
536    ///       ((union lhs (bitshift-left a 1))
537    ///        (subsume (Mul a 2))))
538    /// ```
539    Rewrite(String, GenericRewrite<Head, Leaf>, Subsume),
540    /// Similar to [`Command::Rewrite`], but
541    /// generates two rules, one for each direction.
542    ///
543    /// Example:
544    /// ```text
545    /// (bi-rewrite (Mul (Var x) (Num 0))
546    ///             (Var x))
547    /// ```
548    ///
549    /// Becomes:
550    /// ```text
551    /// (rule ((= lhs (Mul (Var x) (Num 0))))
552    ///       ((union lhs (Var x))))
553    /// (rule ((= lhs (Var x)))
554    ///       ((union lhs (Mul (Var x) (Num 0)))))
555    /// ```
556    BiRewrite(String, GenericRewrite<Head, Leaf>),
557    /// Perform an [`Action`] on the global database
558    /// (see documentation for [`Action`] for more details).
559    /// Example:
560    /// ```text
561    /// (let xplusone (Add (Var "x") (Num 1)))
562    /// ```
563    Action(GenericAction<Head, Leaf>),
564    /// `extract` a datatype from the egraph, choosing
565    /// the smallest representative.
566    /// By default, each constructor costs 1 to extract
567    /// (common subexpressions are not shared in the cost
568    /// model).
569    Extract(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
570    /// Runs a [`Schedule`], which specifies
571    /// rulesets and the number of times to run them.
572    ///
573    /// Example:
574    /// ```text
575    /// (run-schedule
576    ///     (saturate my-ruleset-1)
577    ///     (run my-ruleset-2 4))
578    /// ```
579    ///
580    /// Runs `my-ruleset-1` until saturation,
581    /// then runs `my-ruleset-2` four times.
582    ///
583    /// See [`Schedule`] for more details.
584    RunSchedule(GenericSchedule<Head, Leaf>),
585    /// Print runtime statistics about rules
586    /// and rulesets so far.
587    PrintOverallStatistics(Span, Option<String>),
588    /// The `check` command checks that the given facts
589    /// match at least once in the current database.
590    /// The list of facts is matched in the same way a [`Command::Rule`] is matched.
591    ///
592    /// Example:
593    ///
594    /// ```text
595    /// (check (= (+ 1 2) 3))
596    /// (check (<= 0 3) (>= 3 0))
597    /// (fail (check (= 1 2)))
598    /// ```
599    ///
600    /// prints
601    ///
602    /// ```text
603    /// [INFO ] Checked.
604    /// [INFO ] Checked.
605    /// [ERROR] Check failed
606    /// [INFO ] Command failed as expected.
607    /// ```
608    Check(Span, Vec<GenericFact<Head, Leaf>>),
609    /// Print out rows of a given function, extracting each of the elements of the function.
610    /// Example:
611    ///
612    /// ```text
613    /// (print-function Add 20)
614    /// ```
615    /// prints the first 20 rows of the `Add` function.
616    ///
617    /// ```text
618    /// (print-function Add)
619    /// ```
620    /// prints all rows of the `Add` function.
621    ///
622    /// ```text
623    /// (print-function Add :file "add.csv")
624    /// ```
625    /// prints all rows of the `Add` function to a CSV file.
626    PrintFunction(
627        Span,
628        String,
629        Option<usize>,
630        Option<String>,
631        PrintFunctionMode,
632    ),
633    /// Print out the number of rows in a function or all functions.
634    PrintSize(Span, Option<String>),
635    /// Input a CSV file directly into a function.
636    Input {
637        span: Span,
638        name: String,
639        file: String,
640    },
641    /// Extract and output a set of expressions to a file.
642    Output {
643        span: Span,
644        file: String,
645        exprs: Vec<GenericExpr<Head, Leaf>>,
646    },
647    /// `push` the current egraph `n` times so that it is saved.
648    /// Later, the current database and rules can be restored using `pop`.
649    Push(usize),
650    /// `pop` the current egraph, restoring the previous one.
651    /// The argument specifies how many egraphs to pop.
652    Pop(Span, usize),
653    /// Assert that a command fails with an error.
654    Fail(Span, Box<GenericCommand<Head, Leaf>>),
655    /// Include another egglog file directly as text and run it.
656    Include(Span, String),
657    /// User-defined command.
658    UserDefined(Span, String, Vec<Expr>),
659}

662where
663    Head: Clone + Display,
664    Leaf: Clone + PartialEq + Eq + Display + Hash,
665{
666    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
667        match self {
668            GenericCommand::Rewrite(name, rewrite, subsume) => {
669                rewrite.fmt_with_ruleset(f, name, false, *subsume)
670            }
671            GenericCommand::BiRewrite(name, rewrite) => {
672                rewrite.fmt_with_ruleset(f, name, true, false)
673            }
674            GenericCommand::Datatype {
675                span: _,
676                name,
677                variants,
678            } => write!(f, "(datatype {name} {})", ListDisplay(variants, " ")),
679            GenericCommand::Action(a) => write!(f, "{a}"),
680            GenericCommand::Extract(_span, expr, variants) => {
681                write!(f, "(extract {expr} {variants})")
682            }
683            GenericCommand::Sort(_span, name, None) => write!(f, "(sort {name})"),
684            GenericCommand::Sort(_span, name, Some((name2, args))) => {
685                write!(f, "(sort {name} ({name2} {}))", ListDisplay(args, " "))
686            }
687            GenericCommand::Function {
688                span: _,
689                name,
690                schema,
691                merge,
692            } => {
693                write!(f, "(function {name} {schema}")?;
694                if let Some(merge) = &merge {
695                    write!(f, " :merge {merge}")?;
696                } else {
697                    write!(f, " :no-merge")?;
698                }
699                write!(f, ")")
700            }
701            GenericCommand::Constructor {
702                span: _,
703                name,
704                schema,
705                cost,
706                unextractable,
707            } => {
708                write!(f, "(constructor {name} {schema}")?;
709                if let Some(cost) = cost {
710                    write!(f, " :cost {cost}")?;
711                }
712                if *unextractable {
713                    write!(f, " :unextractable")?;
714                }
715                write!(f, ")")
716            }
717            GenericCommand::Relation {
718                span: _,
719                name,
720                inputs,
721            } => {
722                write!(f, "(relation {name} ({}))", ListDisplay(inputs, " "))
723            }
724            GenericCommand::AddRuleset(_span, name) => write!(f, "(ruleset {name})"),
725            GenericCommand::UnstableCombinedRuleset(_span, name, others) => {
726                write!(
727                    f,
728                    "(unstable-combined-ruleset {name} {})",
729                    ListDisplay(others, " ")
730                )
731            }
732            GenericCommand::Rule { rule } => rule.fmt(f),
733            GenericCommand::RunSchedule(sched) => write!(f, "(run-schedule {sched})"),
734            GenericCommand::PrintOverallStatistics(_span, file) => match file {
735                Some(file) => write!(f, "(print-stats :file {file})"),
736                None => write!(f, "(print-stats)"),
737            },
738            GenericCommand::Check(_ann, facts) => {
739                write!(f, "(check {})", ListDisplay(facts, "\n"))
740            }
741            GenericCommand::Push(n) => write!(f, "(push {n})"),
742            GenericCommand::Pop(_span, n) => write!(f, "(pop {n})"),
743            GenericCommand::PrintFunction(_span, name, n, file, mode) => {
744                write!(f, "(print-function {name}")?;
745                if let Some(n) = n {
746                    write!(f, " {n}")?;
747                }
748                if let Some(file) = file {
749                    write!(f, " :file {file:?}")?;
750                }
751                match mode {
752                    PrintFunctionMode::Default => {}
753                    PrintFunctionMode::CSV => write!(f, " :mode csv")?,
754                }
755                write!(f, ")")
756            }
757            GenericCommand::PrintSize(_span, name) => {
758                write!(f, "(print-size {})", ListDisplay(name, " "))
759            }
760            GenericCommand::Input {
761                span: _,
762                name,
763                file,
764            } => write!(f, "(input {name} {file:?})"),
765            GenericCommand::Output {
766                span: _,
767                file,
768                exprs,
769            } => write!(f, "(output {file:?} {})", ListDisplay(exprs, " ")),
770            GenericCommand::Fail(_span, cmd) => write!(f, "(fail {cmd})"),
771            GenericCommand::Include(_span, file) => write!(f, "(include {file:?})"),
772            GenericCommand::Datatypes { span: _, datatypes } => {
773                let datatypes: Vec<_> = datatypes
774                    .iter()
775                    .map(|(_, name, variants)| match variants {
776                        Subdatatypes::Variants(variants) => {
777                            format!("({name} {})", ListDisplay(variants, " "))
778                        }
779                        Subdatatypes::NewSort(head, args) => {
780                            format!("(sort {name} ({head} {}))", ListDisplay(args, " "))
781                        }
782                    })
783                    .collect();
784                write!(f, "(datatype* {})", ListDisplay(datatypes, " "))
785            }
786            GenericCommand::UserDefined(_span, name, exprs) => {
787                write!(f, "({name} {})", ListDisplay(exprs, " "))
788            }
789        }
790    }
791}

1009pub type Fact = GenericFact<String, String>;
1010pub(crate) type ResolvedFact = GenericFact<ResolvedCall, ResolvedVar>;
1011pub(crate) type MappedFact<Head, Leaf> = GenericFact<CorrespondingVar<Head, Leaf>, Leaf>;
1012
1013pub struct Facts<Head, Leaf>(pub Vec<GenericFact<Head, Leaf>>);
1014
1015impl<Head, Leaf> Facts<Head, Leaf>
1016where
1017    Head: Clone + Display,
1018    Leaf: Clone + PartialEq + Eq + Display + Hash,
1019{
1020    /// Flattens a list of facts into a Query.
1021    /// For typechecking, we need the correspondence between the original ast
1022    /// and the flattened one, so that we can annotate the original with types.
1023    /// That's why this function produces a corresponding list of facts, annotated with
1024    /// the variable names in the flattened Query.
1025    /// (Typechecking preserves the original AST this way,
1026    /// and allows terms and proof instrumentation to do the same).
1027    pub(crate) fn to_query(
1028        &self,
1029        typeinfo: &TypeInfo,
1030        fresh_gen: &mut impl FreshGen<Head, Leaf>,
1031    ) -> (Query<HeadOrEq<Head>, Leaf>, Vec<MappedFact<Head, Leaf>>) {
1032        let mut atoms = vec![];
1033        let mut new_body = vec![];
1034
1035        for fact in self.0.iter() {
1036            match fact {
1037                GenericFact::Eq(span, e1, e2) => {
1038                    let mut to_equate = vec![];
1039                    let mut process = |expr: &GenericExpr<Head, Leaf>| {
1040                        let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1041                        atoms.extend(child_atoms);
1042                        to_equate.push(expr.get_corresponding_var_or_lit(typeinfo));
1043                        expr
1044                    };
1045                    let e1 = process(e1);
1046                    let e2 = process(e2);
1047                    atoms.push(GenericAtom {
1048                        span: span.clone(),
1049                        head: HeadOrEq::Eq,
1050                        args: to_equate,
1051                    });
1052                    new_body.push(GenericFact::Eq(span.clone(), e1, e2));
1053                }
1054                GenericFact::Fact(expr) => {
1055                    let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1056                    atoms.extend(child_atoms);
1057                    new_body.push(GenericFact::Fact(expr));
1058                }
1059            }
1060        }
1061        (Query { atoms }, new_body)
1062    }
1063}

1107#[derive(Clone, Debug)]
1108pub struct GenericRewrite<Head, Leaf> {
1109    pub span: Span,
1110    pub lhs: GenericExpr<Head, Leaf>,
1111    pub rhs: GenericExpr<Head, Leaf>,
1112    pub conditions: Vec<GenericFact<Head, Leaf>>,
1113}
1114
1115impl<Head: Display, Leaf: Display> GenericRewrite<Head, Leaf> {
1116    /// Converts the rewrite into an s-expression.
1117    pub fn fmt_with_ruleset(
1118        &self,
1119        f: &mut Formatter,
1120        ruleset: &str,
1121        is_bidirectional: bool,
1122        subsume: bool,
1123    ) -> std::fmt::Result {
1124        let direction = if is_bidirectional {
1125            "birewrite"
1126        } else {
1127            "rewrite"
1128        };
1129        write!(f, "({direction} {} {}", self.lhs, self.rhs)?;
1130        if subsume {
1131            write!(f, " :subsume")?;
1132        }
1133        if !self.conditions.is_empty() {
1134            write!(f, " :when ({})", ListDisplay(&self.conditions, " "))?;
1135        }
1136        if !ruleset.is_empty() {
1137            write!(f, " :ruleset {ruleset}")?;
1138        }
1139        write!(f, ")")
1140    }
1141}

1143pub(crate) trait MappedExprExt<Head, Leaf>
1144where
1145    Head: Clone + Display,
1146    Leaf: Clone + PartialEq + Eq + Display + Hash,
1147{
1148    fn get_corresponding_var_or_lit(&self, typeinfo: &TypeInfo) -> GenericAtomTerm<Leaf>;

1150
1151impl<Head, Leaf> MappedExprExt<Head, Leaf> for MappedExpr<Head, Leaf>
1152where
1153    Head: Clone + Display,
1154    Leaf: Clone + PartialEq + Eq + Display + Hash,
1155{
1156    fn get_corresponding_var_or_lit(&self, typeinfo: &TypeInfo) -> GenericAtomTerm<Leaf> {
1157        // Note: need typeinfo to resolve whether a symbol is a global or not
1158        // This is error-prone and the complexities can be avoided by treating globals
1159        // as nullary functions.
1160        match self {
1161            GenericExpr::Var(span, v) => {
1162                if typeinfo.is_global(&v.to_string()) {
1163                    GenericAtomTerm::Global(span.clone(), v.clone())
1164                } else {
1165                    GenericAtomTerm::Var(span.clone(), v.clone())
1166                }
1167            }
1168            GenericExpr::Lit(span, lit) => GenericAtomTerm::Literal(span.clone(), lit.clone()),
1169            GenericExpr::Call(span, head, _) => GenericAtomTerm::Var(span.clone(), head.to.clone()),
1170        }
1171    }
