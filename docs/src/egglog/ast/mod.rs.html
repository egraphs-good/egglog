27/// A [`NCommand`] is a desugared [`Command`], where syntactic sugars
28/// like [`Command::Datatype`] and [`Command::Rewrite`]
29/// are eliminated.
30/// Most of the heavy lifting in egglog is done over [`NCommand`]s.
31///
32/// [`GenericNCommand`] is a generalization of [`NCommand`], like how [`GenericCommand`]
33/// is a generalization of [`Command`], allowing annotations over `Head` and `Leaf`.
34///
35/// TODO: The name "NCommand" used to denote normalized command, but this
36/// meaning is obsolete. A future PR should rename this type to something
37/// like "DCommand".
38#[derive(Debug, Clone, Eq, PartialEq, Hash)]
39pub enum GenericNCommand<Head, Leaf>
40where
41    Head: Clone + Display,
42    Leaf: Clone + PartialEq + Eq + Display + Hash,
43{
44    SetOption {
45        name: String,
46        value: GenericExpr<Head, Leaf>,
47    },
48    Sort(
49        Span,
50        String,
51        Option<(String, Vec<GenericExpr<String, String>>)>,
52    ),
53    Function(GenericFunctionDecl<Head, Leaf>),
54    AddRuleset(Span, String),
55    UnstableCombinedRuleset(Span, String, Vec<String>),
56    NormRule {
57        name: String,
58        ruleset: String,
59        rule: GenericRule<Head, Leaf>,
60    },
61    CoreAction(GenericAction<Head, Leaf>),
62    Extract(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
63    RunSchedule(GenericSchedule<Head, Leaf>),
64    PrintOverallStatistics,
65    Check(Span, Vec<GenericFact<Head, Leaf>>),
66    PrintTable(Span, String, usize),
67    PrintSize(Span, Option<String>),
68    Output {
69        span: Span,
70        file: String,
71        exprs: Vec<GenericExpr<Head, Leaf>>,
72    },
73    Push(usize),
74    Pop(Span, usize),
75    Fail(Span, Box<GenericNCommand<Head, Leaf>>),
76    Input {
77        span: Span,
78        name: String,
79        file: String,
80    },
81    UserDefined(Span, String, Vec<Expr>),
82}

89    pub fn to_command(&self) -> GenericCommand<Head, Leaf> {
90        match self {
91            GenericNCommand::SetOption { name, value } => GenericCommand::SetOption {
92                name: name.clone(),
93                value: value.clone(),
94            },
95            GenericNCommand::Sort(span, name, params) => {
96                GenericCommand::Sort(span.clone(), name.clone(), params.clone())
97            }
98            GenericNCommand::Function(f) => match f.subtype {
99                FunctionSubtype::Constructor => GenericCommand::Constructor {
100                    span: f.span.clone(),
101                    name: f.name.clone(),
102                    schema: f.schema.clone(),
103                    cost: f.cost,
104                    unextractable: f.unextractable,
105                },
106                FunctionSubtype::Relation => GenericCommand::Relation {
107                    span: f.span.clone(),
108                    name: f.name.clone(),
109                    inputs: f.schema.input.clone(),
110                },
111                FunctionSubtype::Custom => GenericCommand::Function {
112                    span: f.span.clone(),
113                    schema: f.schema.clone(),
114                    name: f.name.clone(),
115                    merge: f.merge.clone(),
116                },
117            },
118            GenericNCommand::AddRuleset(span, name) => {
119                GenericCommand::AddRuleset(span.clone(), name.clone())
120            }
121            GenericNCommand::UnstableCombinedRuleset(span, name, others) => {
122                GenericCommand::UnstableCombinedRuleset(span.clone(), name.clone(), others.clone())
123            }
124            GenericNCommand::NormRule {
125                name,
126                ruleset,
127                rule,
128            } => GenericCommand::Rule {
129                name: name.clone(),
130                ruleset: ruleset.clone(),
131                rule: rule.clone(),
132            },
133            GenericNCommand::RunSchedule(schedule) => GenericCommand::RunSchedule(schedule.clone()),
134            GenericNCommand::PrintOverallStatistics => GenericCommand::PrintOverallStatistics,
135            GenericNCommand::CoreAction(action) => GenericCommand::Action(action.clone()),
136            GenericNCommand::Extract(span, expr, variants) => {
137                GenericCommand::Extract(span.clone(), expr.clone(), variants.clone())
138            }
139            GenericNCommand::Check(span, facts) => {
140                GenericCommand::Check(span.clone(), facts.clone())
141            }
142            GenericNCommand::PrintTable(span, name, n) => {
143                GenericCommand::PrintFunction(span.clone(), name.clone(), *n)
144            }
145            GenericNCommand::PrintSize(span, name) => {
146                GenericCommand::PrintSize(span.clone(), name.clone())
147            }
148            GenericNCommand::Output { span, file, exprs } => GenericCommand::Output {
149                span: span.clone(),
150                file: file.to_string(),
151                exprs: exprs.clone(),
152            },
153            GenericNCommand::Push(n) => GenericCommand::Push(*n),
154            GenericNCommand::Pop(span, n) => GenericCommand::Pop(span.clone(), *n),
155            GenericNCommand::Fail(span, cmd) => {
156                GenericCommand::Fail(span.clone(), Box::new(cmd.to_command()))
157            }
158            GenericNCommand::Input { span, name, file } => GenericCommand::Input {
159                span: span.clone(),
160                name: name.clone(),
161                file: file.clone(),
162            },
163            GenericNCommand::UserDefined(span, name, exprs) => {
164                GenericCommand::UserDefined(span.clone(), name.clone(), exprs.clone())
165            }
166        }
167    }
168
169    pub fn visit_exprs(
170        self,
171        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
172    ) -> Self {
173        match self {
174            GenericNCommand::SetOption { name, value } => GenericNCommand::SetOption {
175                name,
176                value: f(value.clone()),
177            },
178            GenericNCommand::Sort(span, name, params) => GenericNCommand::Sort(span, name, params),
179            GenericNCommand::Function(func) => GenericNCommand::Function(func.visit_exprs(f)),
180            GenericNCommand::AddRuleset(span, name) => GenericNCommand::AddRuleset(span, name),
181            GenericNCommand::UnstableCombinedRuleset(span, name, rulesets) => {
182                GenericNCommand::UnstableCombinedRuleset(span, name, rulesets)
183            }
184            GenericNCommand::NormRule {
185                name,
186                ruleset,
187                rule,
188            } => GenericNCommand::NormRule {
189                name,
190                ruleset,
191                rule: rule.visit_exprs(f),
192            },
193            GenericNCommand::RunSchedule(schedule) => {
194                GenericNCommand::RunSchedule(schedule.visit_exprs(f))
195            }
196            GenericNCommand::PrintOverallStatistics => GenericNCommand::PrintOverallStatistics,
197            GenericNCommand::CoreAction(action) => {
198                GenericNCommand::CoreAction(action.visit_exprs(f))
199            }
200            GenericNCommand::Extract(span, expr, variants) => {
201                GenericNCommand::Extract(span, expr.visit_exprs(f), variants.visit_exprs(f))
202            }
203            GenericNCommand::Check(span, facts) => GenericNCommand::Check(
204                span,
205                facts.into_iter().map(|fact| fact.visit_exprs(f)).collect(),
206            ),
207            GenericNCommand::PrintTable(span, name, n) => {
208                GenericNCommand::PrintTable(span, name, n)
209            }
210            GenericNCommand::PrintSize(span, name) => GenericNCommand::PrintSize(span, name),
211            GenericNCommand::Output { span, file, exprs } => GenericNCommand::Output {
212                span,
213                file,
214                exprs: exprs.into_iter().map(f).collect(),
215            },
216            GenericNCommand::Push(n) => GenericNCommand::Push(n),
217            GenericNCommand::Pop(span, n) => GenericNCommand::Pop(span, n),
218            GenericNCommand::Fail(span, cmd) => {
219                GenericNCommand::Fail(span, Box::new(cmd.visit_exprs(f)))
220            }
221            GenericNCommand::Input { span, name, file } => {
222                GenericNCommand::Input { span, name, file }
223            }
224            GenericNCommand::UserDefined(span, name, exprs) => {
225                // We can't map `f` over UserDefined because UserDefined always assumes plain `Expr`s
226                GenericNCommand::UserDefined(span, name, exprs)
227            }
228        }
229    }
230}

236pub enum GenericSchedule<Head, Leaf> {
237    Saturate(Span, Box<GenericSchedule<Head, Leaf>>),
238    Repeat(Span, usize, Box<GenericSchedule<Head, Leaf>>),
239    Run(Span, GenericRunConfig<Head, Leaf>),
240    Sequence(Span, Vec<GenericSchedule<Head, Leaf>>),

243impl<Head, Leaf> GenericSchedule<Head, Leaf>
244where
245    Head: Clone + Display,
246    Leaf: Clone + PartialEq + Eq + Display + Hash,
247{
248    fn visit_exprs(
249        self,
250        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
251    ) -> Self {
252        match self {
253            GenericSchedule::Saturate(span, sched) => {
254                GenericSchedule::Saturate(span, Box::new(sched.visit_exprs(f)))
255            }
256            GenericSchedule::Repeat(span, size, sched) => {
257                GenericSchedule::Repeat(span, size, Box::new(sched.visit_exprs(f)))
258            }
259            GenericSchedule::Run(span, config) => GenericSchedule::Run(span, config.visit_exprs(f)),
260            GenericSchedule::Sequence(span, scheds) => GenericSchedule::Sequence(
261                span,
262                scheds.into_iter().map(|s| s.visit_exprs(f)).collect(),
263            ),
264        }
265    }
266}

274            GenericSchedule::Sequence(_ann, scheds) => {
275                write!(f, "(seq {})", ListDisplay(scheds, " "))

277        }

291/// A [`Command`] is the top-level construct in egglog.
292/// It includes defining rules, declaring functions,
293/// adding to tables, and running rules (via a [`Schedule`]).
294#[derive(Debug, Clone)]
295pub enum GenericCommand<Head, Leaf>
296where
297    Head: Clone + Display,
298    Leaf: Clone + PartialEq + Eq + Display + Hash,
299{
300    /// Egglog supports several *experimental* options
301    /// that can be set using the `set-option` command.
302    ///
303    /// Options supported include:
304    /// - "interactive_mode" (default: false): when enabled, egglog prints "(done)" after each command, allowing an external
305    ///   tool to know when each command has finished running.
306    SetOption {
307        name: String,
308        value: GenericExpr<Head, Leaf>,
309    },
310
311    /// Create a new user-defined sort, which can then
312    /// be used in new [`Command::Function`] declarations.
313    /// The [`Command::Datatype`] command desugars directly to this command, with one [`Command::Function`]
314    /// per constructor.
315    /// The main use of this command (as opposed to using [`Command::Datatype`]) is for forward-declaring a sort for mutually-recursive datatypes.
316    ///
317    /// It can also be used to create
318    /// a container sort.
319    /// For example, here's how to make a sort for vectors
320    /// of some user-defined sort `Math`:
321    /// ```text
322    /// (sort MathVec (Vec Math))
323    /// ```
324    ///
325    /// Now `MathVec` can be used as an input or output sort.
326    Sort(Span, String, Option<(String, Vec<Expr>)>),
327
328    /// Egglog supports three types of functions
329    ///
330    /// A constructor models an egg-style user-defined datatype
331    /// It can only be defined through the `datatype`/`datatype*` command
332    /// or the `constructor` command
333    ///
334    /// A relation models a datalog-style mathematical relation
335    /// It can only be defined through the `relation` command
336    ///
337    /// A custom function is a dictionary
338    /// It can only be defined through the `function` command
339    ///
340    /// The `datatype` command declares a user-defined datatype.
341    /// Datatypes can be unioned with [`Action::Union`] either
342    /// at the top level or in the actions of a rule.
343    /// This makes them equal in the implicit, global equality relation.
344    ///
345    /// Example:
346    /// ```text
347    /// (datatype Math
348    ///   (Num i64)
349    ///   (Var String)
350    ///   (Add Math Math)
351    ///   (Mul Math Math))
352    /// ```
353    ///
354    /// defines a simple `Math` datatype with variants for numbers, named variables, addition and multiplication.
355    ///
356    /// Datatypes desugar directly to a [`Command::Sort`] and a [`Command::Constructor`] for each constructor.
357    /// The code above becomes:
358    /// ```text
359    /// (sort Math)
360    /// (constructor Num (i64) Math)
361    /// (constructor Var (String) Math)
362    /// (constructor Add (Math Math) Math)
363    /// (constructor Mul (Math Math) Math)
364    ///
365    /// Datatypes are also known as algebraic data types, tagged unions and sum types.
366    Datatype {
367        span: Span,
368        name: String,
369        variants: Vec<Variant>,
370    },
371    Datatypes {
372        span: Span,
373        datatypes: Vec<(Span, String, Subdatatypes)>,
374    },
375
376    /// The `constructor` command defines a new constructor for a user-defined datatype
377    /// Example:
378    /// ```text
379    /// (constructor Add (i64 i64) Math)
380    /// ```
381    ///
382    Constructor {
383        span: Span,
384        name: String,
385        schema: Schema,
386        cost: Option<usize>,
387        unextractable: bool,
388    },
389
390    /// The `relation` command declares a named relation
391    /// Example:
392    /// ```text
393    /// (relation path (i64 i64))
394    /// (relation edge (i64 i64))
395    /// ```
396    Relation {
397        span: Span,
398        name: String,
399        inputs: Vec<String>,
400    },
401
402    /// The `function` command declare an egglog custom function, which is a database table with a
403    /// a functional dependency (also called a primary key) on its inputs to one output.
404    ///
405    /// ```text
406    /// (function <name:Ident> <schema:Schema> <cost:Cost>
407    ///        (:on_merge <List<Action>>)?
408    ///        (:merge <Expr>)?)
409    ///```
410    /// A function can have a `cost` for extraction.
411    ///
412    /// Finally, it can have a `merge` and `on_merge`, which are triggered when
413    /// the function dependency is violated.
414    /// In this case, the merge expression determines which of the two outputs
415    /// for the same input is used.
416    /// The `on_merge` actions are run after the merge expression is evaluated.
417    ///
418    /// Note that the `:merge` expression must be monotonic
419    /// for the behavior of the egglog program to be consistent and defined.
420    /// In other words, the merge function must define a lattice on the output of the function.
421    /// If values are merged in different orders, they should still result in the same output.
422    /// If the merge expression is not monotonic, the behavior can vary as
423    /// actions may be applied more than once with different results.
424    ///
425    /// ```text
426    /// (function LowerBound (Math) i64 :merge (max old new))
427    /// ```
428    ///
429    /// Specifically, a custom function can also have an EqSort output type:
430    ///
431    /// ```text
432    /// (function Add (i64 i64) Math)
433    /// ```
434    ///
435    /// All functions can be `set`
436    /// with [`Action::Set`].
437    ///
438    /// Output of a function, if being the EqSort type, can be unioned with [`Action::Union`]
439    /// with another datatype of the same `sort`.
440    ///
441    Function {
442        span: Span,
443        name: String,
444        schema: Schema,
445        merge: Option<GenericExpr<Head, Leaf>>,
446    },
447
448    /// Using the `ruleset` command, defines a new
449    /// ruleset that can be added to in [`Command::Rule`]s.
450    /// Rulesets are used to group rules together
451    /// so that they can be run together in a [`Schedule`].
452    ///
453    /// Example:
454    /// Ruleset allows users to define a ruleset- a set of rules
455    ///
456    /// ```text
457    /// (ruleset myrules)
458    /// (rule ((edge x y))
459    ///       ((path x y))
460    ///       :ruleset myrules)
461    /// (run myrules 2)
462    /// ```
463    AddRuleset(Span, String),
464    /// Using the `combined-ruleset` command, construct another ruleset
465    /// which runs all the rules in the given rulesets.
466    /// This is useful for running multiple rulesets together.
467    /// The combined ruleset also inherits any rules added to the individual rulesets
468    /// after the combined ruleset is declared.
469    ///
470    /// Example:
471    /// ```text
472    /// (ruleset myrules1)
473    /// (rule ((edge x y))
474    ///       ((path x y))
475    ///      :ruleset myrules1)
476    /// (ruleset myrules2)
477    /// (rule ((path x y) (edge y z))
478    ///       ((path x z))
479    ///       :ruleset myrules2)
480    /// (combined-ruleset myrules-combined myrules1 myrules2)
481    UnstableCombinedRuleset(Span, String, Vec<String>),
482    /// ```text
483    /// (rule <body:List<Fact>> <head:List<Action>>)
484    /// ```
485    ///
486    /// defines an egglog rule.
487    /// The rule matches a list of facts with respect to
488    /// the global database, and runs the list of actions
489    /// for each match.
490    /// The matches are done *modulo equality*, meaning
491    /// equal datatypes in the database are considered
492    /// equal.
493    ///
494    /// Example:
495    /// ```text
496    /// (rule ((edge x y))
497    ///       ((path x y)))
498    ///
499    /// (rule ((path x y) (edge y z))
500    ///       ((path x z)))
501    /// ```
502    Rule {
503        name: String,
504        ruleset: String,
505        rule: GenericRule<Head, Leaf>,
506    },
507    /// `rewrite` is syntactic sugar for a specific form of `rule`
508    /// which simply unions the left and right hand sides.
509    ///
510    /// Example:
511    /// ```text
512    /// (rewrite (Add a b)
513    ///          (Add b a))
514    /// ```
515    ///
516    /// Desugars to:
517    /// ```text
518    /// (rule ((= lhs (Add a b)))
519    ///       ((union lhs (Add b a))))
520    /// ```
521    ///
522    /// Additionally, additional facts can be specified
523    /// using a `:when` clause.
524    /// For example, the same rule can be run only
525    /// when `a` is zero:
526    ///
527    /// ```text
528    /// (rewrite (Add a b)
529    ///          (Add b a)
530    ///          :when ((= a (Num 0)))
531    /// ```
532    ///
533    /// Add the `:subsume` flag to cause the left hand side to be subsumed after matching, which means it can
534    /// no longer be matched in a rule, but can still be checked against (See [`Change`] for more details.)
535    ///
536    /// ```text
537    /// (rewrite (Mul a 2) (bitshift-left a 1) :subsume)
538    /// ```
539    ///
540    /// Desugars to:
541    /// ```text
542    /// (rule ((= lhs (Mul a 2)))
543    ///       ((union lhs (bitshift-left a 1))
544    ///        (subsume (Mul a 2))))
545    /// ```
546    Rewrite(String, GenericRewrite<Head, Leaf>, Subsume),
547    /// Similar to [`Command::Rewrite`], but
548    /// generates two rules, one for each direction.
549    ///
550    /// Example:
551    /// ```text
552    /// (bi-rewrite (Mul (Var x) (Num 0))
553    ///             (Var x))
554    /// ```
555    ///
556    /// Becomes:
557    /// ```text
558    /// (rule ((= lhs (Mul (Var x) (Num 0))))
559    ///       ((union lhs (Var x))))
560    /// (rule ((= lhs (Var x)))
561    ///       ((union lhs (Mul (Var x) (Num 0)))))
562    /// ```
563    BiRewrite(String, GenericRewrite<Head, Leaf>),
564    /// Perform an [`Action`] on the global database
565    /// (see documentation for [`Action`] for more details).
566    /// Example:
567    /// ```text
568    /// (let xplusone (Add (Var "x") (Num 1)))
569    /// ```
570    Action(GenericAction<Head, Leaf>),
571    /// `extract` a datatype from the egraph, choosing
572    /// the smallest representative.
573    /// By default, each constructor costs 1 to extract
574    /// (common subexpressions are not shared in the cost
575    /// model).
576    Extract(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
577    /// Runs a [`Schedule`], which specifies
578    /// rulesets and the number of times to run them.
579    ///
580    /// Example:
581    /// ```text
582    /// (run-schedule
583    ///     (saturate my-ruleset-1)
584    ///     (run my-ruleset-2 4))
585    /// ```
586    ///
587    /// Runs `my-ruleset-1` until saturation,
588    /// then runs `my-ruleset-2` four times.
589    ///
590    /// See [`Schedule`] for more details.
591    RunSchedule(GenericSchedule<Head, Leaf>),
592    /// Print runtime statistics about rules
593    /// and rulesets so far.
594    PrintOverallStatistics,
595    /// The `check` command checks that the given facts
596    /// match at least once in the current database.
597    /// The list of facts is matched in the same way a [`Command::Rule`] is matched.
598    ///
599    /// Example:
600    ///
601    /// ```text
602    /// (check (= (+ 1 2) 3))
603    /// (check (<= 0 3) (>= 3 0))
604    /// (fail (check (= 1 2)))
605    /// ```
606    ///
607    /// prints
608    ///
609    /// ```text
610    /// [INFO ] Checked.
611    /// [INFO ] Checked.
612    /// [ERROR] Check failed
613    /// [INFO ] Command failed as expected.
614    /// ```
615    Check(Span, Vec<GenericFact<Head, Leaf>>),
616    /// Print out rows a given function, extracting each of the elements of the function.
617    /// Example:
618    /// ```text
619    /// (print-function Add 20)
620    /// ```
621    /// prints the first 20 rows of the `Add` function.
622    ///
623    PrintFunction(Span, String, usize),
624    /// Print out the number of rows in a function or all functions.
625    PrintSize(Span, Option<String>),
626    /// Input a CSV file directly into a function.
627    Input {
628        span: Span,
629        name: String,
630        file: String,
631    },
632    /// Extract and output a set of expressions to a file.
633    Output {
634        span: Span,
635        file: String,
636        exprs: Vec<GenericExpr<Head, Leaf>>,
637    },
638    /// `push` the current egraph `n` times so that it is saved.
639    /// Later, the current database and rules can be restored using `pop`.
640    Push(usize),
641    /// `pop` the current egraph, restoring the previous one.
642    /// The argument specifies how many egraphs to pop.
643    Pop(Span, usize),
644    /// Assert that a command fails with an error.
645    Fail(Span, Box<GenericCommand<Head, Leaf>>),
646    /// Include another egglog file directly as text and run it.
647    Include(Span, String),
648    /// User-defined command.
649    UserDefined(Span, String, Vec<Expr>),
650}

652impl<Head, Leaf> Display for GenericCommand<Head, Leaf>
653where
654    Head: Clone + Display,
655    Leaf: Clone + PartialEq + Eq + Display + Hash,
656{
657    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
658        match self {
659            GenericCommand::SetOption { name, value } => write!(f, "(set-option {name} {value})"),
660            GenericCommand::Rewrite(name, rewrite, subsume) => {
661                rewrite.fmt_with_ruleset(f, name, false, *subsume)
662            }
663            GenericCommand::BiRewrite(name, rewrite) => {
664                rewrite.fmt_with_ruleset(f, name, true, false)
665            }
666            GenericCommand::Datatype {
667                span: _,
668                name,
669                variants,
670            } => write!(f, "(datatype {name} {})", ListDisplay(variants, " ")),
671            GenericCommand::Action(a) => write!(f, "{a}"),
672            GenericCommand::Extract(_span, expr, variants) => {
673                write!(f, "(extract {expr} {variants})")
674            }
675            GenericCommand::Sort(_span, name, None) => write!(f, "(sort {name})"),
676            GenericCommand::Sort(_span, name, Some((name2, args))) => {
677                write!(f, "(sort {name} ({name2} {}))", ListDisplay(args, " "))
678            }
679            GenericCommand::Function {
680                span: _,
681                name,
682                schema,
683                merge,
684            } => {
685                write!(f, "(function {name} {schema}")?;
686                if let Some(merge) = &merge {
687                    write!(f, " :merge {merge}")?;
688                } else {
689                    write!(f, " :no-merge")?;
690                }
691                write!(f, ")")
692            }
693            GenericCommand::Constructor {
694                span: _,
695                name,
696                schema,
697                cost,
698                unextractable,
699            } => {
700                write!(f, "(constructor {name} {schema}")?;
701                if let Some(cost) = cost {
702                    write!(f, " :cost {cost}")?;
703                }
704                if *unextractable {
705                    write!(f, " :unextractable")?;
706                }
707                write!(f, ")")
708            }
709            GenericCommand::Relation {
710                span: _,
711                name,
712                inputs,
713            } => {
714                write!(f, "(relation {name} ({}))", ListDisplay(inputs, " "))
715            }
716            GenericCommand::AddRuleset(_span, name) => write!(f, "(ruleset {name})"),
717            GenericCommand::UnstableCombinedRuleset(_span, name, others) => {
718                write!(
719                    f,
720                    "(unstable-combined-ruleset {name} {})",
721                    ListDisplay(others, " ")
722                )
723            }
724            GenericCommand::Rule {
725                ruleset,
726                name,
727                rule,
728            } => rule.fmt_with_ruleset(f, ruleset, name),
729            GenericCommand::RunSchedule(sched) => write!(f, "(run-schedule {sched})"),
730            GenericCommand::PrintOverallStatistics => write!(f, "(print-stats)"),
731            GenericCommand::Check(_ann, facts) => {
732                write!(f, "(check {})", ListDisplay(facts, "\n"))
733            }
734            GenericCommand::Push(n) => write!(f, "(push {n})"),
735            GenericCommand::Pop(_span, n) => write!(f, "(pop {n})"),
736            GenericCommand::PrintFunction(_span, name, n) => {
737                write!(f, "(print-function {name} {n})")
738            }
739            GenericCommand::PrintSize(_span, name) => {
740                write!(f, "(print-size {})", ListDisplay(name, " "))
741            }
742            GenericCommand::Input {
743                span: _,
744                name,
745                file,
746            } => write!(f, "(input {name} {file:?})"),
747            GenericCommand::Output {
748                span: _,
749                file,
750                exprs,
751            } => write!(f, "(output {file:?} {})", ListDisplay(exprs, " ")),
752            GenericCommand::Fail(_span, cmd) => write!(f, "(fail {cmd})"),
753            GenericCommand::Include(_span, file) => write!(f, "(include {file:?})"),
754            GenericCommand::Datatypes { span: _, datatypes } => {
755                let datatypes: Vec<_> = datatypes
756                    .iter()
757                    .map(|(_, name, variants)| match variants {
758                        Subdatatypes::Variants(variants) => {
759                            format!("({name} {})", ListDisplay(variants, " "))
760                        }
761                        Subdatatypes::NewSort(head, args) => {
762                            format!("(sort {name} ({head} {}))", ListDisplay(args, " "))
763                        }
764                    })
765                    .collect();
766                write!(f, "(datatype* {})", ListDisplay(datatypes, " "))
767            }
768            GenericCommand::UserDefined(_span, name, exprs) => {
769                write!(f, "({name} {})", ListDisplay(exprs, " "))
770            }
771        }
772    }
773}

991/// Facts are the left-hand side of a [`Command::Rule`].
992/// They represent a part of a database query.
993/// Facts can be expressions or equality constraints between expressions.
994///
995/// Note that primitives such as  `!=` are partial.
996/// When two things are equal, it returns nothing and the query does not match.
997/// For example, the following egglog code runs:
998/// ```text
999/// (fail (check (!= 1 1)))
1000/// ```
1001#[derive(Clone, Debug, PartialEq, Eq, Hash)]
1002pub enum GenericFact<Head, Leaf> {
1003    Eq(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
1004    Fact(GenericExpr<Head, Leaf>),
1005}
1006
1007pub struct Facts<Head, Leaf>(pub Vec<GenericFact<Head, Leaf>>);
1008
1009impl<Head, Leaf> Facts<Head, Leaf>
1010where
1011    Head: Clone + Display,
1012    Leaf: Clone + PartialEq + Eq + Display + Hash,
1013{
1014    /// Flattens a list of facts into a Query.
1015    /// For typechecking, we need the correspondence between the original ast
1016    /// and the flattened one, so that we can annotate the original with types.
1017    /// That's why this function produces a corresponding list of facts, annotated with
1018    /// the variable names in the flattened Query.
1019    /// (Typechecking preserves the original AST this way,
1020    /// and allows terms and proof instrumentation to do the same).
1021    pub(crate) fn to_query(
1022        &self,
1023        typeinfo: &TypeInfo,
1024        fresh_gen: &mut impl FreshGen<Head, Leaf>,
1025    ) -> (Query<HeadOrEq<Head>, Leaf>, Vec<MappedFact<Head, Leaf>>) {
1026        let mut atoms = vec![];
1027        let mut new_body = vec![];
1028
1029        for fact in self.0.iter() {
1030            match fact {
1031                GenericFact::Eq(span, e1, e2) => {
1032                    let mut to_equate = vec![];
1033                    let mut process = |expr: &GenericExpr<Head, Leaf>| {
1034                        let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1035                        atoms.extend(child_atoms);
1036                        to_equate.push(expr.get_corresponding_var_or_lit(typeinfo));
1037                        expr
1038                    };
1039                    let e1 = process(e1);
1040                    let e2 = process(e2);
1041                    atoms.push(GenericAtom {
1042                        span: span.clone(),
1043                        head: HeadOrEq::Eq,
1044                        args: to_equate,
1045                    });
1046                    new_body.push(GenericFact::Eq(span.clone(), e1, e2));
1047                }
1048                GenericFact::Fact(expr) => {
1049                    let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1050                    atoms.extend(child_atoms);
1051                    new_body.push(GenericFact::Fact(expr));
1052                }
1053            }
1054        }
1055        (Query { atoms }, new_body)
1056    }
1057}

1060    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
1061        match self {
1062            GenericFact::Eq(_, e1, e2) => write!(f, "(= {e1} {e2})"),
1063            GenericFact::Fact(expr) => write!(f, "{expr}"),
1064        }

1188    /// Example:
1189    /// ```text
1190    /// (datatype Math (Num i64))
1191    /// (union (Num 1) (Num 2)); Define that Num 1 and Num 2 are equivalent
1192    /// (extract (Num 1)); Extracts Num 1
1193    /// (extract (Num 2)); Extracts Num 1
1194    /// ```
1195    Union(Span, GenericExpr<Head, Leaf>, GenericExpr<Head, Leaf>),
1196    Panic(Span, String),
1197    Expr(Span, GenericExpr<Head, Leaf>),
1198    // If(Expr, Action, Action),
1199}

1266impl<Head, Leaf> GenericAction<Head, Leaf>
1267where
1268    Head: Clone + Display,
1269    Leaf: Clone + Eq + Display + Hash,
1270{
1271    // Applys `f` to all expressions in the action.
1272    pub fn map_exprs(
1273        &self,
1274        f: &mut impl FnMut(&GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
1275    ) -> Self {
1276        match self {
1277            GenericAction::Let(span, lhs, rhs) => {
1278                GenericAction::Let(span.clone(), lhs.clone(), f(rhs))
1279            }
1280            GenericAction::Set(span, lhs, args, rhs) => {
1281                let right = f(rhs);
1282                GenericAction::Set(
1283                    span.clone(),
1284                    lhs.clone(),
1285                    args.iter().map(f).collect(),
1286                    right,
1287                )
1288            }
1289            GenericAction::Change(span, change, lhs, args) => GenericAction::Change(
1290                span.clone(),
1291                *change,
1292                lhs.clone(),
1293                args.iter().map(f).collect(),
1294            ),
1295            GenericAction::Union(span, lhs, rhs) => {
1296                GenericAction::Union(span.clone(), f(lhs), f(rhs))
1297            }
1298            GenericAction::Panic(span, msg) => GenericAction::Panic(span.clone(), msg.clone()),
1299            GenericAction::Expr(span, e) => GenericAction::Expr(span.clone(), f(e)),
1300        }
1301    }
1302
1303    /// Applys `f` to all sub-expressions (including `self`)
1304    /// bottom-up, collecting the results.
1305    pub fn visit_exprs(
1306        self,
1307        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
1308    ) -> Self {
1309        match self {
1310            GenericAction::Let(span, lhs, rhs) => {
1311                GenericAction::Let(span, lhs.clone(), rhs.visit_exprs(f))
1312            }
1313            // TODO should we refactor `Set` so that we can map over Expr::Call(lhs, args)?
1314            // This seems more natural to oflatt
1315            // Currently, visit_exprs does not apply f to the first argument of Set.
1316            GenericAction::Set(span, lhs, args, rhs) => {
1317                let args = args.into_iter().map(|e| e.visit_exprs(f)).collect();
1318                GenericAction::Set(span, lhs.clone(), args, rhs.visit_exprs(f))
1319            }
1320            GenericAction::Change(span, change, lhs, args) => {
1321                let args = args.into_iter().map(|e| e.visit_exprs(f)).collect();
1322                GenericAction::Change(span, change, lhs.clone(), args)
1323            }
1324            GenericAction::Union(span, lhs, rhs) => {
1325                GenericAction::Union(span, lhs.visit_exprs(f), rhs.visit_exprs(f))
1326            }
1327            GenericAction::Panic(span, msg) => GenericAction::Panic(span, msg.clone()),
1328            GenericAction::Expr(span, e) => GenericAction::Expr(span, e.visit_exprs(f)),
1329        }
1330    }
1331
1332    pub fn subst(&self, subst: &mut impl FnMut(&Span, &Leaf) -> GenericExpr<Head, Leaf>) -> Self {
1333        self.map_exprs(&mut |e| e.subst_leaf(subst))
1334    }
1335
1336    pub fn map_def_use(self, fvar: &mut impl FnMut(Leaf, bool) -> Leaf) -> Self {
1337        macro_rules! fvar_expr {
1338            () => {
1339                |span, s: _| GenericExpr::Var(span.clone(), fvar(s.clone(), false))
1340            };
1341        }
1342        match self {
1343            GenericAction::Let(span, lhs, rhs) => {
1344                let lhs = fvar(lhs, true);
1345                let rhs = rhs.subst_leaf(&mut fvar_expr!());
1346                GenericAction::Let(span, lhs, rhs)
1347            }
1348            GenericAction::Set(span, lhs, args, rhs) => {
1349                let args = args
1350                    .into_iter()
1351                    .map(|e| e.subst_leaf(&mut fvar_expr!()))
1352                    .collect();
1353                let rhs = rhs.subst_leaf(&mut fvar_expr!());
1354                GenericAction::Set(span, lhs.clone(), args, rhs)
1355            }
1356            GenericAction::Change(span, change, lhs, args) => {
1357                let args = args
1358                    .into_iter()
1359                    .map(|e| e.subst_leaf(&mut fvar_expr!()))
1360                    .collect();
1361                GenericAction::Change(span, change, lhs.clone(), args)
1362            }
1363            GenericAction::Union(span, lhs, rhs) => {
1364                let lhs = lhs.subst_leaf(&mut fvar_expr!());
1365                let rhs = rhs.subst_leaf(&mut fvar_expr!());
1366                GenericAction::Union(span, lhs, rhs)
1367            }
1368            GenericAction::Panic(span, msg) => GenericAction::Panic(span, msg.clone()),
1369            GenericAction::Expr(span, e) => {
1370                GenericAction::Expr(span, e.subst_leaf(&mut fvar_expr!()))
1371            }
1372        }
1373    }
1374}

1390impl<Head, Leaf> GenericRule<Head, Leaf>
1391where
1392    Head: Clone + Display,
1393    Leaf: Clone + PartialEq + Eq + Display + Hash,
1394{
1395    pub(crate) fn visit_exprs(
1396        self,
1397        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
1398    ) -> Self {
1399        Self {
1400            span: self.span,
1401            head: self.head.visit_exprs(f),
1402            body: self
1403                .body
1404                .into_iter()
1405                .map(|bexpr| bexpr.visit_exprs(f))
1406                .collect(),
1407        }
1408    }
1409}
1410
1411impl<Head, Leaf> GenericRule<Head, Leaf>
1412where
1413    Head: Clone + Display,
1414    Leaf: Clone + PartialEq + Eq + Display + Hash,
1415{
1416    pub(crate) fn fmt_with_ruleset(
1417        &self,
1418        f: &mut Formatter,
1419        ruleset: &str,
1420        name: &str,
1421    ) -> std::fmt::Result {
1422        let indent = " ".repeat(7);
1423        write!(f, "(rule (")?;
1424        for (i, fact) in self.body.iter().enumerate() {
1425            if i > 0 {
1426                write!(f, "{}", indent)?;
1427            }
1428
1429            if i != self.body.len() - 1 {
1430                writeln!(f, "{}", fact)?;
1431            } else {
1432                write!(f, "{}", fact)?;
1433            }
1434        }
1435        write!(f, ")\n      (")?;
1436        for (i, action) in self.head.0.iter().enumerate() {
1437            if i > 0 {
1438                write!(f, "{}", indent)?;
1439            }
1440            if i != self.head.0.len() - 1 {
1441                writeln!(f, "{}", action)?;
1442            } else {
1443                write!(f, "{}", action)?;
1444            }
1445        }
1446        let ruleset = if !ruleset.is_empty() {
1447            format!(":ruleset {}", ruleset)
1448        } else {
1449            "".into()
1450        };
1451        let name = if !name.is_empty() {
1452            format!(":name \"{}\"", name)
1453        } else {
1454            "".into()
1455        };
1456        write!(f, ")\n{} {} {})", indent, ruleset, name)
1457    }
1458}

1462#[derive(Clone, Debug)]
1463pub struct GenericRewrite<Head, Leaf> {
1464    pub span: Span,
1465    pub lhs: GenericExpr<Head, Leaf>,
1466    pub rhs: GenericExpr<Head, Leaf>,
1467    pub conditions: Vec<GenericFact<Head, Leaf>>,
1468}
1469
1470impl<Head: Display, Leaf: Display> GenericRewrite<Head, Leaf> {
1471    /// Converts the rewrite into an s-expression.
1472    pub fn fmt_with_ruleset(
1473        &self,
1474        f: &mut Formatter,
1475        ruleset: &str,
1476        is_bidirectional: bool,
1477        subsume: bool,
1478    ) -> std::fmt::Result {
1479        let direction = if is_bidirectional {
1480            "birewrite"
1481        } else {
1482            "rewrite"
1483        };
1484        write!(f, "({direction} {} {}", self.lhs, self.rhs)?;
1485        if subsume {
1486            write!(f, " :subsume")?;
1487        }
1488        if !self.conditions.is_empty() {
1489            write!(f, " :when ({})", ListDisplay(&self.conditions, " "))?;
1490        }
1491        if !ruleset.is_empty() {
1492            write!(f, " :ruleset {ruleset}")?;
1493        }
1494        write!(f, ")")
1495    }
1496}

1503    pub(crate) fn get_corresponding_var_or_lit(
1504        &self,
1505        typeinfo: &TypeInfo,
1506    ) -> GenericAtomTerm<Leaf> {
1507        // Note: need typeinfo to resolve whether a symbol is a global or not
1508        // This is error-prone and the complexities can be avoided by treating globals
1509        // as nullary functions.
1510        match self {
1511            GenericExpr::Var(span, v) => {
1512                if typeinfo.is_global(&v.to_string()) {
1513                    GenericAtomTerm::Global(span.clone(), v.clone())
1514                } else {
1515                    GenericAtomTerm::Var(span.clone(), v.clone())
1516                }
1517            }
1518            GenericExpr::Lit(span, lit) => GenericAtomTerm::Literal(span.clone(), lit.clone()),
1519            GenericExpr::Call(span, head, _) => GenericAtomTerm::Var(span.clone(), head.to.clone()),
1520        }
1521    }
