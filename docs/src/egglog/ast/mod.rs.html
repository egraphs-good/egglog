144                GenericCommand::UnstableCombinedRuleset(span.clone(), name.clone(), others.clone())

189    pub fn visit_queries(
190        self,
191        f: &mut impl FnMut(Vec<GenericFact<Head, Leaf>>) -> Vec<GenericFact<Head, Leaf>>,
192    ) -> Self {
193        match self {
194            GenericNCommand::Check(span, query) => GenericNCommand::Check(span, f(query)),

207            | GenericNCommand::AddRuleset(..)
208            | GenericNCommand::UnstableCombinedRuleset(..)
209            | GenericNCommand::CoreAction(..)
210            | GenericNCommand::Extract(..)
211            | GenericNCommand::PrintOverallStatistics(..)
212            | GenericNCommand::PrintFunction(..)
213            | GenericNCommand::PrintSize(..)
214            | GenericNCommand::Output { .. }
215            | GenericNCommand::Push(..)
216            | GenericNCommand::Pop(..)
217            | GenericNCommand::Input { .. }
218            | GenericNCommand::UserDefined(..)

224    pub fn visit_exprs(
225        self,
226        f: &mut impl FnMut(GenericExpr<Head, Leaf>) -> GenericExpr<Head, Leaf>,
227    ) -> Self {

229            GenericNCommand::Sort {
230                span,

237                name,
238                presort_and_args,
239                uf,
240                unionable,
241            },

243            GenericNCommand::AddRuleset(span, name) => GenericNCommand::AddRuleset(span, name),
244            GenericNCommand::UnstableCombinedRuleset(span, name, rulesets) => {
245                GenericNCommand::UnstableCombinedRuleset(span, name, rulesets)
246            }
247            GenericNCommand::NormRule { rule } => GenericNCommand::NormRule {
248                rule: rule.visit_exprs(f),
249            },
250            GenericNCommand::RunSchedule(schedule) => {
251                GenericNCommand::RunSchedule(schedule.visit_exprs(f))
252            }
253            GenericNCommand::PrintOverallStatistics(span, file) => {
254                GenericNCommand::PrintOverallStatistics(span, file)
255            }
256            GenericNCommand::CoreAction(action) => {
257                GenericNCommand::CoreAction(action.visit_exprs(f))
258            }
259            GenericNCommand::Extract(span, expr, variants) => {
260                GenericNCommand::Extract(span, expr.visit_exprs(f), variants.visit_exprs(f))
261            }
262            GenericNCommand::Check(span, facts) => GenericNCommand::Check(
263                span,
264                facts.into_iter().map(|fact| fact.visit_exprs(f)).collect(),
265            ),
266            GenericNCommand::PrintFunction(span, name, n, file, mode) => {
267                GenericNCommand::PrintFunction(span, name, n, file, mode)
268            }
269            GenericNCommand::ProveExists(span, constructor) => {
270                GenericNCommand::ProveExists(span, constructor)
271            }
272            GenericNCommand::PrintSize(span, name) => GenericNCommand::PrintSize(span, name),
273            GenericNCommand::Output { span, file, exprs } => GenericNCommand::Output {
274                span,
275                file,
276                exprs: exprs.into_iter().map(f).collect(),
277            },
278            GenericNCommand::Push(n) => GenericNCommand::Push(n),
279            GenericNCommand::Pop(span, n) => GenericNCommand::Pop(span, n),
280            GenericNCommand::Fail(span, cmd) => {
281                GenericNCommand::Fail(span, Box::new(cmd.visit_exprs(f)))
282            }
283            GenericNCommand::Input { span, name, file } => {
284                GenericNCommand::Input { span, name, file }
285            }
286            GenericNCommand::UserDefined(span, name, exprs) => {
287                // We can't map `f` over UserDefined because UserDefined always assumes plain `Expr`s
288                GenericNCommand::UserDefined(span, name, exprs)
289            }
290        }
291    }

322    pub fn visit_queries(
323        self,

325    ) -> Self {
326        match self {
327            GenericSchedule::Saturate(span, generic_schedule) => {

329            }
330            GenericSchedule::Repeat(span, iters, generic_schedule) => {

333            GenericSchedule::Run(span, run_config) => GenericSchedule::Run(

337                    until: run_config.until.map(f),
338                },
339            ),
340            GenericSchedule::Sequence(span, generic_schedules) => GenericSchedule::Sequence(

343                    .into_iter()

350    /// Recursively flattens nested `Sequence` nodes into a single level.
351    /// For example, `(seq (seq a b) c)` becomes `(seq a b c)`.
352    /// Also unwraps single-element sequences into their inner schedule.
353    fn flatten_sequences(self) -> Self {
354        match self {
355            GenericSchedule::Saturate(span, sched) => {
356                GenericSchedule::Saturate(span, Box::new(sched.flatten_sequences()))
357            }
358            GenericSchedule::Repeat(span, size, sched) => {
359                GenericSchedule::Repeat(span, size, Box::new(sched.flatten_sequences()))
360            }
361            GenericSchedule::Run(span, config) => GenericSchedule::Run(span, config),
362            GenericSchedule::Sequence(span, scheds) => {
363                let mut flattened = Vec::new();
364                for sched in scheds.into_iter().map(Self::flatten_sequences) {
365                    match sched {
366                        GenericSchedule::Sequence(_, nested) => flattened.extend(nested),
367                        other => flattened.push(other),
368                    }
369                }
370
371                match flattened.len() {
372                    0 => GenericSchedule::Sequence(span, flattened),
373                    1 => flattened.into_iter().next().unwrap(),
374                    _ => GenericSchedule::Sequence(span, flattened),
375                }
376            }
377        }
378    }

507/// A [`Command`] is the top-level construct in egglog.
508/// It includes defining rules, declaring functions,
509/// adding to tables, and running rules (via a [`Schedule`]).
510///
511/// # Binding naming convention
512/// Bindings introduced by commands fall into two categories:
513/// - **Global bindings** must start with [`$`](crate::GLOBAL_NAME_PREFIX).
514/// - **Non-global bindings** must *not* start with [`$`](crate::GLOBAL_NAME_PREFIX).
515///
516/// When `--strict-mode` is enabled, violating these conventions is a type error;
517/// otherwise, egglog emits a single warning per program.
518#[derive(Debug, Clone)]
519pub enum GenericCommand<Head, Leaf>
520where
521    Head: Clone + Display,
522    Leaf: Clone + PartialEq + Eq + Display + Hash,
523{
524    /// Create a new user-defined sort, which can then
525    /// be used in new [`Command::Function`] declarations.
526    /// The [`Command::Datatype`] command desugars directly to this command, with one [`Command::Function`]
527    /// per constructor.
528    /// The main use of this command (as opposed to using [`Command::Datatype`]) is for forward-declaring a sort for mutually-recursive datatypes.
529    ///
530    /// It can also be used to create
531    /// a container sort.
532    /// For example, here's how to make a sort for vectors
533    /// of some user-defined sort `Math`:
534    /// ```text
535    /// (sort MathVec (Vec Math))
536    /// ```
537    ///
538    /// Now `MathVec` can be used as an input or output sort.
539    Sort {
540        span: Span,
541        name: String,
542        presort_and_args: Option<(String, Vec<Expr>)>,
543        /// The name of the union-find function for this sort.
544        /// Used in term encoding to canonicalize values during extraction.
545        uf: Option<String>,
546        /// Whether values of this sort can be unioned.
547        /// Defaults to true for user-defined sorts.
548        /// Set to false for relations and term tables that should not allow union.
549        unionable: bool,
550    },
551
552    /// Egglog supports three types of functions
553    ///
554    /// A constructor models an egg-style user-defined datatype
555    /// It can only be defined through the `datatype`/`datatype*` command
556    /// or the `constructor` command
557    ///
558    /// A relation models a datalog-style mathematical relation
559    /// It can only be defined through the `relation` command
560    ///
561    /// A custom function is a dictionary
562    /// It can only be defined through the `function` command
563    ///
564    /// The `datatype` command declares a user-defined datatype.
565    /// Datatypes can be unioned with [`Action::Union`] either
566    /// at the top level or in the actions of a rule.
567    /// This makes them equal in the implicit, global equality relation.
568    ///
569    /// Example:
570    /// ```text
571    /// (datatype Math
572    ///   (Num i64)
573    ///   (Var String)
574    ///   (Add Math Math)
575    ///   (Mul Math Math))
576    /// ```
577    ///
578    /// defines a simple `Math` datatype with variants for numbers, named variables, addition and multiplication.
579    ///
580    /// Datatypes desugar directly to a [`Command::Sort`] and a [`Command::Constructor`] for each constructor.
581    /// The code above becomes:
582    /// ```text
583    /// (sort Math)
584    /// (constructor Num (i64) Math)
585    /// (constructor Var (String) Math)
586    /// (constructor Add (Math Math) Math)
587    /// (constructor Mul (Math Math) Math)
588    ///
589    /// Datatypes are also known as algebraic data types, tagged unions and sum types.
590    Datatype {
591        span: Span,
592        name: String,
593        variants: Vec<Variant>,
594    },
595    Datatypes {
596        span: Span,
597        datatypes: Vec<(Span, String, Subdatatypes)>,
598    },
599
600    /// The `constructor` command defines a new constructor for a user-defined datatype
601    /// Example:
602    /// ```text
603    /// (constructor Add (i64 i64) Math)
604    /// ```
605    ///
606    /// A constructor can be `:unextractable`, in which case extraction skip this constructor entirely.
607    /// ```text
608    /// (constructor UnextractableNode (i64) Math :unextractable)
609    /// ```
610    ///
611    /// A constructor can also have a `cost` for extraction, which is used in the cost model for extraction. It defaults to 1 if not specified.
612    /// ```text
613    /// (constructor ExpensiveNode (i64) Math :cost 10)
614    /// ```
615    Constructor {
616        span: Span,
617        name: String,
618        schema: Schema,
619        cost: Option<DefaultCost>,
620        unextractable: bool,
621        /// Internal-hidden constructors are excluded from print-size output and extraction.
622        /// Used for internal tables generated by proof production.
623        hidden: bool,
624        /// Internal-let constructors are let bindings, excluded from print-size output.
625        /// Used for global let bindings that are converted to constructors.
626        let_binding: bool,
627        /// For view tables in proof encoding: the constructor to use for building
628        /// terms from the first n-1 children during extraction.
629        term_constructor: Option<String>,
630    },
631
632    /// The `relation` command declares a named relation
633    /// Example:
634    /// ```text
635    /// (relation path (i64 i64))
636    /// (relation edge (i64 i64))
637    /// ```
638    Relation {
639        span: Span,
640        name: String,
641        inputs: Vec<String>,
642    },
643
644    /// The `function` command declare an egglog custom function, which is a database table with a
645    /// a functional dependency (also called a primary key) on its inputs to one output.
646    ///
647    /// ```text
648    /// (function <name:Ident> <schema:Schema> <cost:Cost>
649    ///        (:on_merge <List<Action>>)?
650    ///        (:merge <Expr>)?)
651    ///```
652    /// A function can have a `cost` for extraction.
653    ///
654    /// Finally, it can have a `merge` and `on_merge`, which are triggered when
655    /// the function dependency is violated.
656    /// In this case, the merge expression determines which of the two outputs
657    /// for the same input is used.
658    /// The `on_merge` actions are run after the merge expression is evaluated.
659    ///
660    /// Note that the `:merge` expression must be monotonic
661    /// for the behavior of the egglog program to be consistent and defined.
662    /// In other words, the merge function must define a lattice on the output of the function.
663    /// If values are merged in different orders, they should still result in the same output.
664    /// If the merge expression is not monotonic, the behavior can vary as
665    /// actions may be applied more than once with different results.
666    ///
667    /// ```text
668    /// (function LowerBound (Math) i64 :merge (max old new))
669    /// ```
670    ///
671    /// Specifically, a custom function can also have an EqSort output type:
672    ///
673    /// ```text
674    /// (function Add (i64 i64) Math)
675    /// ```
676    ///
677    /// All functions can be `set`
678    /// with [`Action::Set`].
679    ///
680    /// Output of a function, if being the EqSort type, can be unioned with [`Action::Union`]
681    /// with another datatype of the same `sort`.
682    ///
683    Function {
684        span: Span,
685        name: String,
686        schema: Schema,
687        merge: Option<GenericExpr<Head, Leaf>>,
688        hidden: bool,
689        let_binding: bool,
690    },

693    /// ruleset that can be added to in [`Command::Rule`]s.
694    /// Rulesets are used to group rules together
695    /// so that they can be run together in a [`Schedule`].
696    ///
697    /// Example:
698    /// Ruleset allows users to define a ruleset- a set of rules
699    ///
700    /// ```text
701    /// (ruleset myrules)

712    /// after the combined ruleset is declared.

733    /// the global database, and runs the list of actions

736    /// equal datatypes in the database are considered

738    ///

742    ///       ((path x y)))

750    /// `rewrite` is syntactic sugar for a specific form of `rule`

913impl<Head, Leaf> Display for GenericCommand<Head, Leaf>
914where
915    Head: Clone + Display,
916    Leaf: Clone + PartialEq + Eq + Display + Hash,
917{
918    fn fmt(&self, f: &mut Formatter) -> std::fmt::Result {
919        match self {
920            GenericCommand::Rewrite(name, rewrite, subsume) => {
921                rewrite.fmt_with_ruleset(f, name, false, *subsume)
922            }
923            GenericCommand::BiRewrite(name, rewrite) => {
924                rewrite.fmt_with_ruleset(f, name, true, false)
925            }
926            GenericCommand::Datatype {
927                span: _,
928                name,
929                variants,
930            } => {
931                write!(f, "(datatype {name} {})", ListDisplay(variants, " "))
932            }
933            GenericCommand::Action(a) => write!(f, "{a}"),
934            GenericCommand::Extract(_span, expr, variants) => {
935                write!(f, "(extract {expr} {variants})")
936            }
937            GenericCommand::Sort {
938                name,
939                presort_and_args: None,
940                ..
941            } => {
942                write!(f, "(sort {name})")
943            }
944            GenericCommand::Sort {
945                name,
946                presort_and_args: Some((name2, args)),
947                ..
948            } => {
949                write!(f, "(sort {name} ({name2} {}))", ListDisplay(args, " "))
950            }
951            GenericCommand::Function {
952                span: _,
953                name,
954                schema,
955                merge,
956                hidden,
957                let_binding,
958            } => {
959                write!(f, "(function {name} {schema}")?;
960                if let Some(merge) = &merge {
961                    write!(f, " :merge {merge}")?;
962                } else {
963                    write!(f, " :no-merge")?;
964                }
965                if *hidden {
966                    write!(f, " :internal-hidden")?;
967                }
968                if *let_binding {
969                    write!(f, " :internal-let")?;
970                }
971                write!(f, ")")
972            }
973            GenericCommand::Constructor {
974                span: _,
975                name,
976                schema,
977                cost,
978                unextractable,
979                hidden,
980                let_binding,
981                term_constructor,
982            } => {
983                write!(f, "(constructor {name} {schema}")?;
984                if let Some(cost) = cost {
985                    write!(f, " :cost {cost}")?;
986                }
987                if *unextractable {
988                    write!(f, " :unextractable")?;
989                }
990                if *hidden {
991                    write!(f, " :internal-hidden")?;
992                }
993                if *let_binding {
994                    write!(f, " :internal-let")?;
995                }
996                if let Some(tc) = term_constructor {
997                    write!(f, " :term-constructor {tc}")?;
998                }
999                write!(f, ")")
1000            }
1001            GenericCommand::Relation {
1002                span: _,
1003                name,
1004                inputs,
1005            } => {
1006                write!(f, "(relation {name} ({}))", ListDisplay(inputs, " "))
1007            }
1008            GenericCommand::AddRuleset(_span, name) => {
1009                write!(f, "(ruleset {name})")
1010            }
1011            GenericCommand::UnstableCombinedRuleset(_span, name, others) => {
1012                write!(
1013                    f,
1014                    "(unstable-combined-ruleset {name} {})",
1015                    ListDisplay(others, " ")
1016                )
1017            }
1018            GenericCommand::Rule { rule } => rule.fmt(f),
1019            GenericCommand::RunSchedule(sched) => write!(f, "(run-schedule {sched})"),
1020            GenericCommand::PrintOverallStatistics(_span, file) => match file {
1021                Some(file) => write!(f, "(print-stats :file {file})"),
1022                None => write!(f, "(print-stats)"),
1023            },
1024            GenericCommand::Check(_ann, facts) => {
1025                write!(f, "(check {})", ListDisplay(facts, "\n"))
1026            }
1027            GenericCommand::Prove(_span, facts) => {
1028                if facts.is_empty() {
1029                    write!(f, "(prove)")
1030                } else {
1031                    write!(f, "(prove {})", ListDisplay(facts, " "))
1032                }
1033            }
1034            GenericCommand::ProveExists(_span, constructor) => {
1035                write!(f, "(prove-exists {constructor})")
1036            }
1037            GenericCommand::Push(n) => write!(f, "(push {n})"),
1038            GenericCommand::Pop(_span, n) => write!(f, "(pop {n})"),
1039            GenericCommand::PrintFunction(_span, name, n, file, mode) => {
1040                write!(f, "(print-function {name}")?;
1041                if let Some(n) = n {
1042                    write!(f, " {n}")?;
1043                }
1044                if let Some(file) = file {
1045                    write!(f, " :file {file:?}")?;
1046                }
1047                match mode {
1048                    PrintFunctionMode::Default => {}
1049                    PrintFunctionMode::CSV => write!(f, " :mode csv")?,
1050                }
1051                write!(f, ")")
1052            }
1053            GenericCommand::PrintSize(_span, name) => {
1054                write!(f, "(print-size {})", ListDisplay(name, " "))
1055            }
1056            GenericCommand::Input {
1057                span: _,
1058                name,
1059                file,
1060            } => {
1061                write!(f, "(input {name} {file:?})")
1062            }
1063            GenericCommand::Output {
1064                span: _,
1065                file,
1066                exprs,
1067            } => write!(f, "(output {file:?} {})", ListDisplay(exprs, " ")),
1068            GenericCommand::Fail(_span, cmd) => write!(f, "(fail {cmd})"),
1069            GenericCommand::Include(_span, file) => write!(f, "(include {file:?})"),
1070            GenericCommand::Datatypes { span: _, datatypes } => {
1071                let datatypes: Vec<_> = datatypes
1072                    .iter()
1073                    .map(|(_, name, variants)| match variants {
1074                        Subdatatypes::Variants(variants) => {
1075                            format!("({name} {})", ListDisplay(variants, " "))
1076                        }
1077                        Subdatatypes::NewSort(head, args) => {
1078                            format!("(sort {name} ({head} {}))", ListDisplay(args, " "))
1079                        }
1080                    })
1081                    .collect();
1082                write!(f, "(datatype* {})", ListDisplay(datatypes, " "))
1083            }
1084            GenericCommand::UserDefined(_span, name, exprs) => {
1085                write!(f, "({name} {})", ListDisplay(exprs, " "))
1086            }
1087        }
1088    }
1089}

1352impl<Head, Leaf> Facts<Head, Leaf>
1353where
1354    Head: Clone + Display,
1355    Leaf: Clone + PartialEq + Eq + Display + Hash,
1356{
1357    /// Flattens a list of facts into a Query.
1358    /// For typechecking, we need the correspondence between the original ast
1359    /// and the flattened one, so that we can annotate the original with types.
1360    /// That's why this function produces a corresponding list of facts, annotated with
1361    /// the variable names in the flattened Query.
1362    /// (Typechecking preserves the original AST this way,
1363    /// and allows terms and proof instrumentation to do the same).
1364    pub(crate) fn to_query(
1365        &self,
1366        typeinfo: &TypeInfo,
1367        fresh_gen: &mut impl FreshGen<Head, Leaf>,
1368    ) -> (Query<HeadOrEq<Head>, Leaf>, Vec<MappedFact<Head, Leaf>>) {
1369        let mut atoms = vec![];
1370        let mut new_body = vec![];
1371
1372        for fact in self.0.iter() {
1373            match fact {
1374                GenericFact::Eq(span, e1, e2) => {
1375                    let mut to_equate = vec![];
1376                    let mut process = |expr: &GenericExpr<Head, Leaf>| {
1377                        let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1378                        atoms.extend(child_atoms);
1379                        to_equate.push(expr.get_corresponding_var_or_lit(typeinfo));
1380                        expr
1381                    };
1382                    let e1 = process(e1);
1383                    let e2 = process(e2);
1384                    atoms.push(GenericAtom {
1385                        span: span.clone(),
1386                        head: HeadOrEq::Eq,
1387                        args: to_equate,
1388                    });
1389                    new_body.push(GenericFact::Eq(span.clone(), e1, e2));
1390                }
1391                GenericFact::Fact(expr) => {
1392                    let (child_atoms, expr) = expr.to_query(typeinfo, fresh_gen);
1393                    atoms.extend(child_atoms);
1394                    new_body.push(GenericFact::Fact(expr));
1395                }
1396            }
1397        }
1398        (Query { atoms }, new_body)
1399    }
1400}
