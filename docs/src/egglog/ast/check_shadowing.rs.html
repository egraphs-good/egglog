3#[derive(Clone, Debug, Default)]
4pub(crate) struct Names {
5    seen: HashMap<String, Span>,
6    global_aliases: HashMap<String, (String, Span)>,
7}
8
9impl Names {
10    fn check(&mut self, name: String, new: Span) -> Result<(), Error> {
11        if let Some(old) = self.seen.get(&name) {
12            Err(Error::Shadowing(name, old.clone(), new))
13        } else {
14            self.seen.insert(name, new);
15            Ok(())
16        }
17    }
18
19    fn track_global_alias(&mut self, name: &str, span: &Span) {
20        if let Some(stripped) = name.strip_prefix(GLOBAL_NAME_PREFIX) {
21            self.global_aliases
22                .insert(stripped.to_owned(), (name.to_owned(), span.clone()));
23        }
24    }
25
26    fn check_pattern_name(&mut self, name: &str, span: &Span) -> Result<(), Error> {
27        let canonical = name
28            .strip_prefix(GLOBAL_NAME_PREFIX)
29            .unwrap_or(name)
30            .to_owned();
31        if let Some((global_name, global_span)) = self.global_aliases.get(&canonical) {
32            return Err(Error::Shadowing(
33                format!(
34                    "pattern variable `{}` conflicts with global `{}`",
35                    name, global_name
36                ),
37                global_span.clone(),
38                span.clone(),
39            ));
40        }
41        self.check(name.to_owned(), span.clone())
42    }
43
44    /// WARNING: this function does not handle `push` and `pop`.
45    /// Because `Names` is contained on the `EGraph`, this will
46    /// work correctly when executed from `process_command`, but
47    /// a unit test that called this function multiple times without
48    /// changing the `EGraph` will be wrong.
49    pub(crate) fn check_shadowing(&mut self, command: &ResolvedNCommand) -> Result<(), Error> {
50        match command {
51            ResolvedNCommand::Sort { span, name, .. } => self.check(name.clone(), span.clone()),
52            ResolvedNCommand::Function(decl) => {
53                self.check(decl.name.clone(), decl.span.clone())?;
54                if decl.internal_let {
55                    self.track_global_alias(&decl.name, &decl.span);
56                }
57                Ok(())
58            }
59            ResolvedNCommand::AddRuleset(span, name) => self.check(name.clone(), span.clone()),
60            ResolvedNCommand::UnstableCombinedRuleset(span, name, _args) => {
61                self.check(name.clone(), span.clone())
62            }
63            ResolvedNCommand::NormRule { rule, .. } => {
64                let mut inner = self.clone();
65                inner.check_shadowing_query(&rule.body)?;
66                for action in rule.head.iter() {
67                    inner.check_shadowing_action(action)?;
68                }
69                Ok(())
70            }
71            ResolvedNCommand::CoreAction(action) => self.check_shadowing_action(action),
72            ResolvedNCommand::Check(_span, query) => {
73                let mut inner = self.clone();
74                inner.check_shadowing_query(query)
75            }
76            ResolvedNCommand::Fail(_span, command) => {
77                let mut inner = self.clone();
78                inner.check_shadowing(command)
79            }
80            ResolvedNCommand::Extract(..) => Ok(()),
81            ResolvedNCommand::RunSchedule(..) => Ok(()),
82            ResolvedNCommand::PrintOverallStatistics(..) => Ok(()),
83            ResolvedNCommand::ProveExists(..) => Ok(()),
84            ResolvedNCommand::PrintFunction(..) => Ok(()),
85            ResolvedNCommand::PrintSize(..) => Ok(()),
86            ResolvedNCommand::Input { .. } => Ok(()),
87            ResolvedNCommand::Output { .. } => Ok(()),
88            ResolvedNCommand::Push(..) => Ok(()),
89            ResolvedNCommand::Pop(..) => Ok(()),
90            ResolvedNCommand::UserDefined(..) => Ok(()),
91        }
92    }
93
94    fn check_shadowing_query(&mut self, query: &[ResolvedFact]) -> Result<(), Error> {
95        // we want to allow names in queries to shadow each other, so we first collect
96        // all of the variable names, and then we check each of those names once
97        fn collect_expr_names(expr: &ResolvedExpr, out: &mut HashMap<String, Span>) {
98            match expr {
99                ResolvedExpr::Lit(..) => {}
100                ResolvedExpr::Var(span, name) => {
101                    out.entry(name.name.clone()).or_insert_with(|| span.clone());
102                }
103                ResolvedExpr::Call(_span, _func, args) => {
104                    args.iter().for_each(|e| collect_expr_names(e, out));
105                }
106            }
107        }
108
109        let mut collected = HashMap::default();
110
111        for fact in query {
112            match fact {
113                ResolvedFact::Eq(_span, e1, e2) => {
114                    collect_expr_names(e1, &mut collected);
115                    collect_expr_names(e2, &mut collected);
116                }
117                ResolvedFact::Fact(e) => collect_expr_names(e, &mut collected),
118            }
119        }
120
121        for (name, span) in collected {
122            self.check_pattern_name(&name, &span)?;
123        }
124
125        Ok(())
126    }
127
128    fn check_shadowing_action(&mut self, action: &ResolvedAction) -> Result<(), Error> {
129        if let ResolvedAction::Let(span, name, _args) = action {
130            self.check_pattern_name(&name.name, span)
131        } else {
132            Ok(())
133        }
134    }
