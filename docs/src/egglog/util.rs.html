16pub(crate) struct ListDisplay<'a, TS>(pub TS, pub &'a str);
17
18impl<TS> Display for ListDisplay<'_, TS>
19where
20    TS: Clone + IntoIterator,
21    TS::Item: Display,
22{
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        let mut did_something = false;
25        for item in self.0.clone().into_iter() {
26            if did_something {
27                f.write_str(self.1)?;
28            }
29            Display::fmt(&item, f)?;
30            did_something = true;
31        }
32        Ok(())
33    }
34}

36/// Generates fresh symbols for internal use during typechecking and flattening.
37/// These are guaranteed not to collide with the
38/// user's symbols because they use $.
39#[derive(Debug, Clone, PartialEq, Eq)]
40pub struct SymbolGen {
41    gen: usize,
42    reserved_string: String,
43}
44
45impl SymbolGen {
46    pub fn new(reserved_string: String) -> Self {
47        Self {
48            gen: 0,
49            reserved_string,
50        }
51    }
52
53    pub fn has_been_used(&self) -> bool {
54        self.gen > 0
55    }
56}

58/// This trait lets us statically dispatch between `fresh` methods for generic structs.
59pub trait FreshGen<Head: ?Sized, Leaf> {
60    fn fresh(&mut self, name_hint: &Head) -> Leaf;
61}
62
63impl FreshGen<str, String> for SymbolGen {
64    fn fresh(&mut self, name_hint: &str) -> String {
65        let s = format!("{}{}{}", self.reserved_string, name_hint, self.gen);
66        self.gen += 1;
67        s
68    }
69}
70
71impl FreshGen<String, String> for SymbolGen {
72    fn fresh(&mut self, name_hint: &String) -> String {
73        self.fresh(name_hint.as_str())
74    }
75}
76
77impl FreshGen<ResolvedCall, ResolvedVar> for SymbolGen {
78    fn fresh(&mut self, name_hint: &ResolvedCall) -> ResolvedVar {
79        let name = format!("{}{}{}", self.reserved_string, name_hint, self.gen);
80        self.gen += 1;
81        let sort = match name_hint {
82            ResolvedCall::Func(f) => f.output.clone(),
83            ResolvedCall::Primitive(SpecializedPrimitive { output, .. }) => output.clone(),
84        };
85        ResolvedVar {
86            name,
87            sort,
88            // fresh variables are never global references, since globals
89            // are desugared away by `remove_globals`
90            is_global_ref: false,
91        }
92    }
93}
