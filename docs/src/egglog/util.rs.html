13/// Generates fresh symbols for internal use during typechecking and flattening.
14/// These are guaranteed not to collide with the
15/// user's symbols because they use $.
16#[derive(Debug, Clone, PartialEq, Eq)]
17pub struct SymbolGen {
18    count: usize,
19    reserved_string: String,
20}
21
22impl SymbolGen {
23    pub fn new(reserved_string: String) -> Self {
24        Self {
25            count: 0,
26            reserved_string,
27        }
28    }
29
30    pub fn has_been_used(&self) -> bool {
31        self.count > 0
32    }
33}
34
35/// This trait lets us statically dispatch between `fresh` methods for generic structs.
36pub trait FreshGen<Head: ?Sized, Leaf> {
37    fn fresh(&mut self, name_hint: &Head) -> Leaf;
38}
39
40impl FreshGen<str, String> for SymbolGen {
41    fn fresh(&mut self, name_hint: &str) -> String {
42        let s = format!("{}{}{}", self.reserved_string, name_hint, self.count);
43        self.count += 1;
44        s
45    }
46}
47
48impl FreshGen<String, String> for SymbolGen {
49    fn fresh(&mut self, name_hint: &String) -> String {
50        self.fresh(name_hint.as_str())
51    }
52}
53
54impl FreshGen<ResolvedCall, ResolvedVar> for SymbolGen {
55    fn fresh(&mut self, name_hint: &ResolvedCall) -> ResolvedVar {
56        let name = format!("{}{}{}", self.reserved_string, name_hint, self.count);
57        self.count += 1;
58        let sort = match name_hint {
59            ResolvedCall::Func(f) => f.output.clone(),
60            ResolvedCall::Primitive(SpecializedPrimitive { output, .. }) => output.clone(),
61        };
62        ResolvedVar {
63            name,
64            sort,
65            // fresh variables are never global references, since globals
66            // are desugared away by `remove_globals`
67            is_global_ref: false,
68        }
69    }
