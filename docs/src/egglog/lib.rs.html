119/// Output from a command.
120#[derive(Clone, Debug)]
121#[allow(clippy::large_enum_variant)]
122pub enum CommandOutput {
123    /// The size of a function
124    PrintFunctionSize(usize),
125    /// The name of all functions and their sizes
126    PrintAllFunctionsSize(Vec<(String, usize)>),
127    /// The best function found after extracting
128    ExtractBest(TermDag, DefaultCost, TermId),
129    /// The variants of a function found after extracting
130    ExtractVariants(TermDag, Vec<TermId>),
131    /// A high-level proof witnessing constructor existence
132    ProveExists {
133        proof_store: ProofStore,
134        proof_id: ProofId,
135    },
136    /// The report from all runs
137    OverallStatistics(RunReport),
138    /// A printed function and all its values
139    PrintFunction(Function, TermDag, Vec<(TermId, TermId)>, PrintFunctionMode),
140    /// The report from a single run
141    RunSchedule(RunReport),
142    /// A user defined output
143    UserDefined(Arc<dyn UserDefinedCommandOutput>),
144}

146impl std::fmt::Display for CommandOutput {
147    /// Format the command output for display, ending with a newline.
148    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
149        match self {
150            CommandOutput::PrintFunctionSize(size) => writeln!(f, "{size}"),
151            CommandOutput::PrintAllFunctionsSize(names_and_sizes) => {
152                write!(f, "(")?;
153                for (i, (name, size)) in names_and_sizes.iter().enumerate() {
154                    // indent except for the first line
155                    if i > 0 {
156                        write!(f, " ")?;
157                    }
158                    // write the pair of funciton symbol and size
159                    write!(f, "({name} {size})")?;
160                    // add a newline except at the end
161                    if i < names_and_sizes.len() - 1 {
162                        writeln!(f)?;
163                    }
164                }
165                writeln!(f, ")")
166            }
167            CommandOutput::ExtractBest(termdag, _cost, term) => {
168                writeln!(f, "{}", termdag.to_string(*term))
169            }
170            CommandOutput::ExtractVariants(termdag, terms) => {
171                writeln!(f, "(")?;
172                for expr in terms {
173                    writeln!(f, "   {}", termdag.to_string(*expr))?;
174                }
175                writeln!(f, ")")
176            }
177            CommandOutput::ProveExists {
178                proof_store,
179                proof_id,
180            } => writeln!(f, "{}", proof_store.proof_to_string(*proof_id)),
181            CommandOutput::OverallStatistics(run_report) => {
182                write!(f, "Overall statistics:\n{run_report}")
183            }
184            CommandOutput::PrintFunction(function, termdag, terms_and_outputs, mode) => {
185                let out_is_unit = function.schema.output.name() == UnitSort.name();
186                if *mode == PrintFunctionMode::CSV {
187                    let mut wtr = Writer::from_writer(vec![]);
188                    for (term_id, output) in terms_and_outputs {
189                        let term = termdag.get(*term_id);
190                        match term {
191                            Term::App(name, children) => {
192                                let mut values = vec![name.clone()];
193                                for child_id in children {
194                                    values.push(termdag.to_string(*child_id));
195                                }
196
197                                if !out_is_unit {
198                                    values.push(termdag.to_string(*output));
199                                }
200                                wtr.write_record(&values).map_err(|_| std::fmt::Error)?;
201                            }
202                            _ => panic!("Expect function_to_dag to return a list of apps."),
203                        }
204                    }
205                    let csv_bytes = wtr.into_inner().map_err(|_| std::fmt::Error)?;
206                    f.write_str(&String::from_utf8(csv_bytes).map_err(|_| std::fmt::Error)?)
207                } else {
208                    writeln!(f, "(")?;
209                    for (term, output) in terms_and_outputs.iter() {
210                        write!(f, "   {}", termdag.to_string(*term))?;
211                        if !out_is_unit {
212                            write!(f, " -> {}", termdag.to_string(*output))?;
213                        }
214                        writeln!(f)?;
215                    }
216                    writeln!(f, ")")
217                }
218            }
219            CommandOutput::RunSchedule(_report) => Ok(()),
220            CommandOutput::UserDefined(output) => {
221                write!(f, "{}", *output)
222            }
223        }
224    }

227/// The main interface for an e-graph in egglog.
228///
229/// An [`EGraph`] maintains a collection of equivalence classes of terms and provides
230/// operations for adding facts, running rules, and extracting optimal terms.
231///
232/// # Examples
233///
234/// ```
235/// use egglog::*;
236///
237/// let mut egraph = EGraph::default();
238/// egraph.parse_and_run_program(None, "(datatype Math (Num i64) (Add Math Math))").unwrap();
239/// ```
240#[derive(Clone)]
241pub struct EGraph {
242    backend: egglog_bridge::EGraph,
243    pub parser: Parser,
244    names: check_shadowing::Names,
245    /// pushed_egraph forms a linked list of pushed egraphs.
246    /// Pop reverts the egraph to the last pushed egraph.
247    pushed_egraph: Option<Box<Self>>,
248    functions: IndexMap<String, Function>,
249    rulesets: IndexMap<String, Ruleset>,
250    pub fact_directory: Option<PathBuf>,
251    pub seminaive: bool,
252    type_info: TypeInfo,
253    /// The run report unioned over all runs so far.
254    overall_run_report: RunReport,
255    schedulers: DenseIdMap<SchedulerId, SchedulerRecord>,
256    commands: IndexMap<String, Arc<dyn UserDefinedCommand>>,
257    strict_mode: bool,
258    warned_about_global_prefix: bool,
259    /// Registry for command-level macros
260    command_macros: CommandMacroRegistry,
261    proof_state: EncodingState,
262    /// In proof mode, this is the program before proof instrumentation and the version we use for proof checking.
263    proof_check_program: Vec<ResolvedNCommand>,
264}

336impl Default for EGraph {
337    fn default() -> Self {
338        let mut parser = Parser::default();
339        let proof_state = EncodingState::new(&mut parser.symbol_gen);
340        let mut eg = Self {
341            backend: Default::default(),
342            parser,
343            names: Default::default(),
344            pushed_egraph: Default::default(),
345            functions: Default::default(),
346            rulesets: Default::default(),
347            fact_directory: None,
348            seminaive: true,
349            overall_run_report: Default::default(),
350            type_info: Default::default(),
351            schedulers: Default::default(),
352            commands: Default::default(),
353            strict_mode: false,
354            warned_about_global_prefix: false,
355            command_macros: Default::default(),
356            proof_state,
357            proof_check_program: vec![],
358        };
359        add_base_sort(&mut eg, UnitSort, span!()).unwrap();
360        add_base_sort(&mut eg, StringSort, span!()).unwrap();
361        add_base_sort(&mut eg, BoolSort, span!()).unwrap();
362        add_base_sort(&mut eg, I64Sort, span!()).unwrap();
363        add_base_sort(&mut eg, F64Sort, span!()).unwrap();
364        add_base_sort(&mut eg, BigIntSort, span!()).unwrap();
365        add_base_sort(&mut eg, BigRatSort, span!()).unwrap();
366        eg.type_info.add_presort::<MapSort>(span!()).unwrap();
367        eg.type_info.add_presort::<SetSort>(span!()).unwrap();
368        eg.type_info.add_presort::<VecSort>(span!()).unwrap();
369        eg.type_info.add_presort::<FunctionSort>(span!()).unwrap();
370        eg.type_info.add_presort::<MultiSetSort>(span!()).unwrap();
371
372        // Add != with a validator that computes inequality result
373        let neq_validator = |termdag: &mut TermDag, args: &[TermId]| -> Option<TermId> {
374            if args.len() == 2 && args[0] != args[1] {
375                // Return unit literal for successful inequality
376                Some(termdag.lit(Literal::Unit))
377            } else {
378                None
379            }
380        };
381        add_primitive_with_validator!(
382            &mut eg,
383            "!=" = |a: #, b: #| -?> () {
384                (a != b).then_some(())
385            },
386            neq_validator
387        );
388
389        add_primitive_with_validator!(
390            &mut eg,
391            "bool-!=" = |a: #, b: #| -> bool {
392                (a != b)
393            },
394            |termdag: &mut TermDag, args: &[TermId]| -> Option<TermId> {
395                if args.len() == 2 {
396                    Some(termdag.lit(Literal::Bool(args[0] != args[1])))
397                } else {
398                    None
399                }
400            }
401        );
402
403        add_primitive!(&mut eg, "value-eq" = |a: #, b: #| -?> () {
404            (a == b).then_some(())
405        });
406        add_primitive!(&mut eg, "ordering-min" = |a: #, b: #| -> # {
407            if a < b { a } else { b }
408        });
409        add_primitive!(&mut eg, "ordering-max" = |a: #, b: #| -> # {
410            if a > b { a } else { b }
411        });
412
413        eg.rulesets
414            .insert("".into(), Ruleset::Rules(Default::default()));
415
416        eg
417    }

482    /// Set the number of threads used for parallel operations.
483    ///
484    /// This is a helper that simply configures the global rayon thread pool. It can only be called
485    /// once per process; subsequent calls will be ignored.
486    ///
487    /// # Panics
488    ///
489    /// Panics on wasm if `num_threads > 1`.
490    pub fn set_num_threads(num_threads: usize) {
491        #[cfg(target_family = "wasm")]
492        if num_threads > 1 {
493            panic!("cannot use more than 1 thread on wasm");
494        }
495        #[cfg(not(target_family = "wasm"))]
496        {
497            // This will fail silently if the global pool has already been configured.
498            let err = rayon::ThreadPoolBuilder::new()
499                .num_threads(num_threads)

503                log::info!("Initialize global thread pool with  {num_threads} threads");
504            } else {
505                log::warn!(
506                    "Failed to initialize global thread pool with {num_threads} threads. This may be because the thread pool was already initialized with a different number of threads. Error: {err:?}"

508            }
509        }

567    fn ensure_global_name_prefix(&mut self, span: &Span, name: &str) -> Result<(), TypeError> {
568        if name.starts_with(GLOBAL_NAME_PREFIX) {
569            return Ok(());
570        }
571        if self.strict_mode {
572            Err(TypeError::GlobalMissingPrefix {
573                name: name.to_owned(),
574                span: span.clone(),
575            })
576        } else {
577            self.warn_missing_global_prefix(span, name)?;
578            Ok(())
579        }

583        &mut self,

588            return Err(TypeError::GlobalMissingPrefix {

624    /// Push a snapshot of the e-graph into the stack.
625    ///
626    /// See [`EGraph::pop`].
627    pub fn push(&mut self) {
628        let prev_prev: Option<Box<Self>> = self.pushed_egraph.take();
629        let mut prev = self.clone();
630        prev.pushed_egraph = prev_prev;
631        self.pushed_egraph = Some(Box::new(prev));

634    /// Pop the current egraph off the stack, replacing
635    /// it with the previously pushed egraph.
636    /// It preserves the run report and messages from the popped
637    /// egraph.
638    pub fn pop(&mut self) -> Result<(), Error> {
639        match self.pushed_egraph.take() {
640            Some(mut e) => {
641                // Preserve the overall report from the popped egraph
642                std::mem::swap(&mut self.overall_run_report, &mut e.overall_run_report);
643                // Preserve the symbol generator so that fresh symbols
644                // generated after pop don't collide with ones generated before pop.
645                std::mem::swap(&mut self.parser.symbol_gen, &mut e.parser.symbol_gen);
646                *self = *e;
647                Ok(())
648            }
649            None => Err(Error::Pop(span!())),
650        }
651    }

653    fn translate_expr_to_mergefn(
654        &self,
655        expr: &ResolvedExpr,
656    ) -> Result<egglog_bridge::MergeFn, Error> {
657        match expr {
658            GenericExpr::Lit(_, literal) => {
659                let val = literal_to_value(&self.backend, literal);
660                Ok(egglog_bridge::MergeFn::Const(val))
661            }
662            GenericExpr::Var(span, resolved_var) => match resolved_var.name.as_str() {
663                "old" => Ok(egglog_bridge::MergeFn::Old),
664                "new" => Ok(egglog_bridge::MergeFn::New),
665                // NB: type-checking should already catch unbound variables here.
666                _ => Err(TypeError::Unbound(resolved_var.name.clone(), span.clone()).into()),
667            },
668            GenericExpr::Call(_, ResolvedCall::Func(f), args) => {
669                let translated_args = args
670                    .iter()
671                    .map(|arg| self.translate_expr_to_mergefn(arg))
672                    .collect::<Result<Vec<_>, _>>()?;
673                Ok(egglog_bridge::MergeFn::Function(
674                    self.functions[&f.name].backend_id,
675                    translated_args,
676                ))
677            }
678            GenericExpr::Call(_, ResolvedCall::Primitive(p), args) => {
679                let translated_args = args
680                    .iter()
681                    .map(|arg| self.translate_expr_to_mergefn(arg))
682                    .collect::<Result<Vec<_>, _>>()?;
683                Ok(egglog_bridge::MergeFn::Primitive(
684                    p.external_id(),
685                    translated_args,
686                ))
687            }
688        }
689    }

691    fn declare_function(&mut self, decl: &ResolvedFunctionDecl) -> Result<(), Error> {
692        let get_sort = |name: &String| match self.type_info.get_sort_by_name(name) {
693            Some(sort) => Ok(sort.clone()),
694            None => Err(Error::TypeError(TypeError::UndefinedSort(
695                name.to_owned(),
696                decl.span.clone(),
697            ))),
698        };
699
700        let input = decl
701            .schema
702            .input
703            .iter()
704            .map(get_sort)
705            .collect::<Result<Vec<_>, _>>()?;
706        let output = get_sort(&decl.schema.output)?;
707
708        let can_subsume = match decl.subtype {
709            FunctionSubtype::Constructor => true,
710            FunctionSubtype::Custom => false,
711        };
712
713        use egglog_bridge::{DefaultVal, MergeFn};
714        let backend_id = self.backend.add_table(egglog_bridge::FunctionConfig {
715            schema: input
716                .iter()
717                .chain([&output])
718                .map(|sort| sort.column_ty(&self.backend))
719                .collect(),
720            default: match decl.subtype {
721                FunctionSubtype::Constructor => DefaultVal::FreshId,
722                FunctionSubtype::Custom => DefaultVal::Fail,
723            },
724            merge: match decl.subtype {
725                FunctionSubtype::Constructor => MergeFn::UnionId,
726                FunctionSubtype::Custom => match &decl.merge {
727                    None => MergeFn::AssertEq,
728                    Some(expr) => self.translate_expr_to_mergefn(expr)?,
729                },
730            },
731            name: decl.name.to_string(),
732            can_subsume,
733        });
734
735        let function = Function {
736            decl: decl.clone(),
737            schema: ResolvedSchema { input, output },
738            can_subsume,
739            backend_id,
740        };
741
742        let old = self.functions.insert(decl.name.clone(), function);
743        if old.is_some() {
744            panic!(
745                "Typechecking should have caught function already bound: {}",
746                decl.name
747            );
748        }
749
750        Ok(())
751    }

816    /// Print the size of a function. If no function name is provided,
817    /// print the size of all non-hidden functions as an s-expression list of
818    /// `(name size)` pairs, e.g. `((name size) ...)`.
819    pub fn print_size(&self, sym: Option<&str>) -> Result<CommandOutput, Error> {
820        if let Some(sym) = sym {
821            // In proof mode, we have view tables instead of term tables.
822            // So we do a linear scan to find the view table first, falling back on the normal table otherwise.
823            // (We don't check the proof mode flag so that this still works after desugaring)
824            let f = self
825                .functions
826                .values()
827                .find(|f| f.decl.term_constructor.as_deref() == Some(sym))
828                .or_else(|| self.functions.get(sym))
829                .ok_or(TypeError::UnboundFunction(sym.to_owned(), span!()))?;
830            // Skip hidden and let_binding functions
831            if f.decl.internal_hidden || f.decl.internal_let {
832                return Err(TypeError::UnboundFunction(sym.to_owned(), span!()).into());
833            }
834            let size = self.backend.table_size(f.backend_id);
835            log::info!("Function {sym} has size {size}");
836            Ok(CommandOutput::PrintFunctionSize(size))
837        } else {
838            // Print size of all non-hidden, non-let_binding functions
839            // For view tables, use the term_constructor name instead
840            let mut lens = self
841                .functions
842                .iter()
843                .filter(|(_, f)| !f.decl.internal_hidden && !f.decl.internal_let)
844                .map(|(sym, f)| {
845                    let name = f
846                        .decl
847                        .term_constructor
848                        .clone()
849                        .unwrap_or_else(|| sym.clone());
850                    (name, self.backend.table_size(f.backend_id))
851                })
852                .collect::<Vec<_>>();
853
854            // Function name's alphabetical order
855            lens.sort_by_key(|(name, _)| name.clone());
856            if log_enabled!(Level::Info) {
857                for (sym, len) in &lens {
858                    log::info!("Function {sym} has size {len}");
859                }
860            }
861            Ok(CommandOutput::PrintAllFunctionsSize(lens))
862        }
863    }

865    // returns whether the egraph was updated
866    fn run_schedule(&mut self, sched: &ResolvedSchedule) -> Result<RunReport, Error> {
867        match sched {
868            ResolvedSchedule::Run(span, config) => self.run_rules(span, config),
869            ResolvedSchedule::Repeat(_span, limit, sched) => {
870                let mut report = RunReport::default();
871                for _i in 0..*limit {
872                    let rec = self.run_schedule(sched)?;
873                    let updated = rec.updated;
874                    report.union(rec);
875                    if !updated {
876                        break;
877                    }
878                }
879                Ok(report)
880            }
881            ResolvedSchedule::Saturate(_span, sched) => {
882                let mut report = RunReport::default();
883                loop {
884                    let rec = self.run_schedule(sched)?;
885                    let updated = rec.updated;
886                    report.union(rec);
887                    if !updated {
888                        break;
889                    }
890                }
891                Ok(report)
892            }
893            ResolvedSchedule::Sequence(_span, scheds) => {
894                let mut report = RunReport::default();
895                for sched in scheds {
896                    report.union(self.run_schedule(sched)?);
897                }
898                Ok(report)
899            }
900        }
901    }

903    fn run_rules(&mut self, span: &Span, config: &ResolvedRunConfig) -> Result<RunReport, Error> {
904        log::debug!("Running ruleset: {}", config.ruleset);
905        let mut report: RunReport = Default::default();
906
907        let GenericRunConfig { ruleset, until } = config;
908
909        if let Some(facts) = until
910            && self.check_facts(span, facts).is_ok()
911        {
912            log::info!(
913                "Breaking early because of facts:\n {}!",
914                ListDisplay(facts, "\n")
915            );
916            return Ok(report);
917        }
918
919        let subreport = self.step_rules(ruleset)?;
920        report.union(subreport);
921
922        if log_enabled!(Level::Debug) {
923            log::debug!("database size: {}", self.num_tuples());
924        }
925
926        Ok(report)
927    }

929    /// Runs a ruleset for an iteration.
930    ///
931    /// This applies every match it finds (under semi-naive).
932    /// See [`EGraph::step_rules_with_scheduler`] for more fine-grained control.
933    ///
934    /// This will return an error if an egglog primitive returns None in an action.
935    pub fn step_rules(&mut self, ruleset: &str) -> Result<RunReport, Error> {
936        fn collect_rule_ids(
937            ruleset: &str,
938            rulesets: &IndexMap<String, Ruleset>,
939            ids: &mut Vec<egglog_bridge::RuleId>,
940        ) {
941            match &rulesets[ruleset] {
942                Ruleset::Rules(rules) => {
943                    for (_, id) in rules.values() {
944                        ids.push(*id);
945                    }
946                }
947                Ruleset::Combined(sub_rulesets) => {
948                    for sub_ruleset in sub_rulesets {
949                        collect_rule_ids(sub_ruleset, rulesets, ids);
950                    }
951                }
952            }
953        }
954
955        let mut rule_ids = Vec::new();
956        collect_rule_ids(ruleset, &self.rulesets, &mut rule_ids);
957
958        let iteration_report = self
959            .backend
960            .run_rules(&rule_ids)
961            .map_err(|e| Error::BackendError(e.to_string()))?;
962
963        Ok(RunReport::singleton(ruleset, iteration_report))
964    }

966    fn add_rule(&mut self, rule: ast::ResolvedRule) -> Result<String, Error> {
967        // Disable union_to_set optimization in proof or term encoding mode, since
968        // it expects only `union` on constructors (not set).
969        let core_rule = rule.to_canonicalized_core_rule(
970            &self.type_info,
971            &mut self.parser.symbol_gen,
972            self.proof_state.original_typechecking.is_none(),
973        )?;
974        let (query, actions) = (&core_rule.body, &core_rule.head);
975
976        let rule_id = {
977            let mut translator = BackendRule::new(
978                self.backend.new_rule(&rule.name, self.seminaive),
979                &self.functions,
980                &self.type_info,
981            );
982            translator.query(query, false);
983            translator.actions(actions)?;
984            translator.build()
985        };
986
987        if let Some(rules) = self.rulesets.get_mut(&rule.ruleset) {
988            match rules {
989                Ruleset::Rules(rules) => {
990                    match rules.entry(rule.name.clone()) {
991                        indexmap::map::Entry::Occupied(_) => {
992                            let name = rule.name;
993                            panic!("Rule '{name}' was already present")
994                        }
995                        indexmap::map::Entry::Vacant(e) => e.insert((core_rule, rule_id)),
996                    };
997                    Ok(rule.name)
998                }
999                Ruleset::Combined(_) => Err(Error::CombinedRulesetError(rule.ruleset, rule.span)),
1000            }
1001        } else {
1002            Err(Error::NoSuchRuleset(rule.ruleset, rule.span))
1003        }
1004    }

1078        let unit_val = self.backend.base_values().get(());

1088            })));

1206    fn run_command(&mut self, command: ResolvedNCommand) -> Result<Option<CommandOutput>, Error> {
1207        match command {
1208            // Sorts are already declared during typechecking
1209            ResolvedNCommand::Sort {
1210                name,
1211                uf,
1212                proof_func,
1213                ..
1214            } => {
1215                // If the sort has a :internal-uf field, store the mapping for extraction
1216                if let Some(uf_name) = uf {
1217                    self.proof_state.uf_parent.insert(name.clone(), uf_name);
1218                }
1219                // If the sort has a :internal-proof-func field, store the mapping for proof lookup.
1220                // This annotation is set by proof instrumentation and consumed here.
1221                if let Some(proof_func_name) = proof_func {
1222                    self.proof_state
1223                        .proof_func_parent
1224                        .insert(name.clone(), proof_func_name);
1225                }
1226                log::info!("Declared sort {name}.")
1227            }
1228            ResolvedNCommand::Function(fdecl) => {
1229                self.declare_function(&fdecl)?;
1230                log::info!("Declared {} {}.", fdecl.subtype, fdecl.name)
1231            }
1232            ResolvedNCommand::AddRuleset(_span, name) => {
1233                self.add_ruleset(name.clone());
1234                log::info!("Declared ruleset {name}.");
1235            }
1236            ResolvedNCommand::UnstableCombinedRuleset(_span, name, others) => {
1237                self.add_combined_ruleset(name.clone(), others);
1238                log::info!("Declared ruleset {name}.");
1239            }
1240            ResolvedNCommand::NormRule { rule } => {
1241                let name = rule.name.clone();
1242                self.add_rule(rule)?;
1243                log::info!("Declared rule {name}.")
1244            }
1245            ResolvedNCommand::RunSchedule(sched) => {
1246                let report = self.run_schedule(&sched)?;
1247                log::info!("Ran schedule {sched}.");
1248                log::info!("Report: {report}");
1249                self.overall_run_report.union(report.clone());
1250                return Ok(Some(CommandOutput::RunSchedule(report)));
1251            }
1252            ResolvedNCommand::PrintOverallStatistics(span, file) => match file {
1253                None => {
1254                    log::info!("Printed overall statistics");
1255                    return Ok(Some(CommandOutput::OverallStatistics(
1256                        self.overall_run_report.clone(),
1257                    )));
1258                }
1259                Some(path) => {
1260                    let mut file = std::fs::File::create(&path)
1261                        .map_err(|e| Error::IoError(path.clone().into(), e, span.clone()))?;
1262                    log::info!("Printed overall statistics to json file {path}");
1263
1264                    serde_json::to_writer(&mut file, &self.overall_run_report)
1265                        .expect("error serializing to json");
1266                }
1267            },
1268            ResolvedNCommand::Check(span, facts) => {
1269                self.check_facts(&span, &facts)?;
1270                log::info!("Checked fact {facts:?}.");
1271            }
1272            ResolvedNCommand::CoreAction(action) => match &action {
1273                ResolvedAction::Let(_, name, contents) => {
1274                    panic!("Globals should have been desugared away: {name} = {contents}")
1275                }
1276                _ => {
1277                    self.eval_actions(&ResolvedActions::new(vec![action.clone()]))?;
1278                }
1279            },
1280            ResolvedNCommand::Extract(span, expr, variants) => {
1281                let sort = expr.output_type();
1282
1283                let x = self.eval_resolved_expr(span.clone(), &expr)?;
1284                let n = self.eval_resolved_expr(span, &variants)?;
1285                let n: i64 = self.backend.base_values().unwrap(n);
1286
1287                let mut termdag = TermDag::default();
1288
1289                let extractor = Extractor::compute_costs_from_rootsorts(
1290                    Some(vec![sort]),
1291                    self,
1292                    TreeAdditiveCostModel::default(),
1293                );
1294                return if n == 0 {
1295                    if let Some((cost, term)) = extractor.extract_best(self, &mut termdag, x) {
1296                        // dont turn termdag into a string if we have messages disabled for performance reasons
1297                        if log_enabled!(Level::Info) {
1298                            log::info!("extracted with cost {cost}: {}", termdag.to_string(term));
1299                        }
1300                        Ok(Some(CommandOutput::ExtractBest(termdag, cost, term)))
1301                    } else {
1302                        Err(Error::ExtractError(
1303                            "Unable to find any valid extraction (likely due to subsume or delete)"
1304                                .to_string(),
1305                        ))
1306                    }
1307                } else {
1308                    if n < 0 {
1309                        panic!("Cannot extract negative number of variants");
1310                    }
1311                    let terms: Vec<TermId> = extractor
1312                        .extract_variants(self, &mut termdag, x, n as usize)
1313                        .iter()
1314                        .map(|e| e.1)
1315                        .collect();
1316                    if log_enabled!(Level::Info) {
1317                        let expr_str = expr.to_string();
1318                        log::info!("extracted {} variants for {expr_str}", terms.len());
1319                    }
1320                    Ok(Some(CommandOutput::ExtractVariants(termdag, terms)))
1321                };
1322            }
1323            ResolvedNCommand::Push(n) => {
1324                (0..n).for_each(|_| self.push());
1325                log::info!("Pushed {n} levels.")
1326            }
1327            ResolvedNCommand::Pop(span, n) => {
1328                for _ in 0..n {
1329                    self.pop().map_err(|err| {
1330                        if let Error::Pop(_) = err {
1331                            Error::Pop(span.clone())
1332                        } else {
1333                            err
1334                        }
1335                    })?;
1336                }
1337                log::info!("Popped {n} levels.")
1338            }
1339            ResolvedNCommand::PrintFunction(span, f, n, file, mode) => {
1340                let file = file
1341                    .map(|file| {
1342                        std::fs::File::create(&file)
1343                            .map_err(|e| Error::IoError(file.into(), e, span.clone()))
1344                    })
1345                    .transpose()?;
1346                return self.print_function(&f, n, file, mode).map_err(|e| match e {
1347                    Error::TypeError(TypeError::UnboundFunction(f, _)) => {
1348                        Error::TypeError(TypeError::UnboundFunction(f, span.clone()))
1349                    }
1350                    // This case is currently impossible
1351                    _ => e,
1352                });
1353            }
1354            ResolvedNCommand::PrintSize(span, f) => {
1355                let res = self.print_size(f.as_deref()).map_err(|e| match e {
1356                    Error::TypeError(TypeError::UnboundFunction(f, _)) => {
1357                        Error::TypeError(TypeError::UnboundFunction(f, span.clone()))
1358                    }
1359                    // This case is currently impossible
1360                    _ => e,
1361                })?;
1362                return Ok(Some(res));
1363            }
1364            ResolvedNCommand::Fail(span, c) => {
1365                let result = self.run_command(*c);
1366                if let Err(e) = result {
1367                    log::info!("Command failed as expected: {e}");
1368                } else {
1369                    return Err(Error::ExpectFail(span));
1370                }
1371            }
1372            ResolvedNCommand::Input {
1373                span: _,
1374                name,
1375                file,
1376            } => {
1377                self.input_file(&name, file)?;
1378            }
1379            ResolvedNCommand::Output { span, file, exprs } => {
1380                let mut filename = self.fact_directory.clone().unwrap_or_default();
1381                filename.push(file.as_str());
1382                // append to file
1383                let mut f = File::options()
1384                    .append(true)
1385                    .create(true)
1386                    .open(&filename)
1387                    .map_err(|e| Error::IoError(filename.clone(), e, span.clone()))?;
1388
1389                let extractor = Extractor::compute_costs_from_rootsorts(
1390                    None,
1391                    self,
1392                    TreeAdditiveCostModel::default(),
1393                );
1394                let mut termdag: TermDag = Default::default();
1395
1396                use std::io::Write;
1397                for expr in exprs {
1398                    let value = self.eval_resolved_expr(span.clone(), &expr)?;
1399                    let expr_type = expr.output_type();
1400
1401                    let term = extractor
1402                        .extract_best_with_sort(self, &mut termdag, value, expr_type)
1403                        .unwrap()
1404                        .1;
1405                    writeln!(f, "{}", termdag.to_string(term))
1406                        .map_err(|e| Error::IoError(filename.clone(), e, span.clone()))?;
1407                }
1408
1409                log::info!("Output to '{filename:?}'.")
1410            }
1411            ResolvedNCommand::UserDefined(_span, name, exprs) => {
1412                let command = self.commands.swap_remove(&name).unwrap_or_else(|| {
1413                    panic!("Unrecognized user-defined command: {name}");
1414                });
1415                let res = command.update(self, &exprs);
1416                self.commands.insert(name, command);
1417                return res;
1418            }
1419            ResolvedNCommand::ProveExists(span, resolved_call) => {
1420                let mut instrument = ProofInstrumentor { egraph: self };
1421                let (proof_store, proof_id) =
1422                    instrument
1423                        .prove_exists(&resolved_call)
1424                        .map_err(|error| Error::ProofError {
1425                            span: span.clone(),
1426                            error,
1427                        })?;
1428                return Ok(Some(CommandOutput::ProveExists {
1429                    proof_store,
1430                    proof_id,
1431                }));
1432            }
1433        };
1434
1435        Ok(None)
1436    }

1438    fn input_file(&mut self, func_name: &str, file: String) -> Result<(), Error> {
1439        let function_type = self
1440            .type_info
1441            .get_func_type(func_name)
1442            .unwrap_or_else(|| panic!("Unrecognized function name {func_name}"));
1443        let func = self.functions.get_mut(func_name).unwrap();
1444
1445        let mut filename = self.fact_directory.clone().unwrap_or_default();
1446        filename.push(file.as_str());
1447
1448        // check that the function uses supported types
1449
1450        for t in &func.schema.input {
1451            match t.name() {
1452                "i64" | "f64" | "String" => {}
1453                s => panic!("Unsupported type {s} for input"),
1454            }
1455        }
1456
1457        if function_type.subtype != FunctionSubtype::Constructor {
1458            match func.schema.output.name() {
1459                "i64" | "String" | "Unit" => {}
1460                s => panic!("Unsupported type {s} for input"),
1461            }
1462        }
1463
1464        log::info!("Opening file '{filename:?}'...");
1465        let mut f = File::open(filename).unwrap();
1466        let mut contents = String::new();
1467        f.read_to_string(&mut contents).unwrap();
1468
1469        // Can also do a row-major Vec<Value>
1470        let mut parsed_contents: Vec<Vec<Value>> = Vec::with_capacity(contents.lines().count());
1471
1472        let mut row_schema = func.schema.input.clone();
1473        if function_type.subtype == FunctionSubtype::Custom {
1474            row_schema.push(func.schema.output.clone());
1475        }
1476
1477        log::debug!("{row_schema:?}");
1478
1479        let unit_val = self.backend.base_values().get(());
1480
1481        for line in contents.lines() {
1482            let mut it = line.split('\t').map(|s| s.trim());
1483
1484            let mut row: Vec<Value> = Vec::with_capacity(row_schema.len());
1485
1486            for sort in row_schema.iter() {
1487                if let Some(raw) = it.next() {
1488                    let val = match sort.name() {
1489                        "i64" => {
1490                            if let Ok(i) = raw.parse::<i64>() {
1491                                self.backend.base_values().get(i)
1492                            } else {
1493                                return Err(Error::InputFileFormatError(file));
1494                            }
1495                        }
1496                        "f64" => {
1497                            if let Ok(f) = raw.parse::<f64>() {
1498                                self.backend
1499                                    .base_values()
1500                                    .get::<F>(core_relations::Boxed::new(f.into()))
1501                            } else {
1502                                return Err(Error::InputFileFormatError(file));
1503                            }
1504                        }
1505                        "String" => self.backend.base_values().get::<S>(raw.to_string().into()),
1506                        "Unit" => unit_val,
1507                        _ => panic!("Unreachable"),
1508                    };
1509                    row.push(val);
1510                } else {
1511                    break;
1512                }
1513            }
1514
1515            if row.is_empty() {
1516                continue;
1517            }
1518
1519            if row.len() != row_schema.len() || it.next().is_some() {
1520                return Err(Error::InputFileFormatError(file));
1521            }
1522
1523            parsed_contents.push(row);
1524        }
1525
1526        log::debug!("Successfully loaded file.");
1527
1528        let num_facts = parsed_contents.len();
1529
1530        let table_action = egglog_bridge::TableAction::new(&self.backend, func.backend_id);
1531
1532        if function_type.subtype != FunctionSubtype::Constructor {
1533            self.backend.with_execution_state(|es| {
1534                for row in parsed_contents.iter() {
1535                    table_action.insert(es, row.iter().copied());
1536                }
1537                Some(unit_val)
1538            });
1539        } else {
1540            self.backend.with_execution_state(|es| {
1541                for row in parsed_contents.iter() {
1542                    table_action.lookup(es, row);
1543                }
1544                Some(unit_val)
1545            });
1546        }
1547
1548        self.backend.flush_updates();
1549
1550        log::info!("Read {num_facts} facts into {func_name} from '{file}'.");
1551        Ok(())
1552    }

1560        &mut self,
1561        command: Command,
1562    ) -> Result<Vec<ResolvedNCommand>, Error> {
1563        let desugared = desugar_command(command, &mut self.parser, self.proof_state.proof_testing)?;
1564        if let Some(original_typechecking) = self.proof_state.original_typechecking.as_mut() {
1565            // Typecheck using the original egraph
1566            // TODO this is ugly- we don't need an entire e-graph just for type information.
1567            let typechecked = original_typechecking.typecheck_program(&desugared)?;
1568
1569            for command in &typechecked {
1570                if let Err(reason) =
1571                    command_supports_proof_encoding(&command.to_command(), &self.type_info)
1572                {

1574                    return Err(Error::UnsupportedProofCommand {
1575                        command: command_text,
1576                        reason,
1577                    });
1578                }
1579            }
1580
1581            Ok(proof_form(typechecked, &mut self.parser.symbol_gen))
1582        } else {
1583            let mut typechecked = self.typecheck_program(&desugared)?;
1584
1585            typechecked = remove_globals::remove_globals(typechecked, &mut self.parser.symbol_gen);
1586            for command in &typechecked {
1587                self.names.check_shadowing(command)?;
1588            }
1589            Ok(typechecked)
1590        }

1593    /// Desugars, typechecks, and removes globals from a single [`Command`].
1594    /// Leverages previous type information in the [`EGraph`] to do so, adding new type information.
1595    /// When will_run is true, adds to `desugared_commands_run_so_far`, which is used for proof checking.
1596    fn resolve_command(&mut self, command: Command) -> Result<ResolvedNCommands, Error> {
1597        let resolved_before_proofs = self.resolve_command_before_proofs(command)?;
1598
1599        // Add term encoding when it is enabled
1600        if self.proof_state.original_typechecking.is_none() {
1601            Ok(ResolvedNCommands {
1602                desugared: resolved_before_proofs,
1603                desugared_before_proofs: vec![],
1604            })
1605        } else {
1606            // Now remove globals for actual execution (but NOT from desugared_commands)
1607            let typechecked_no_globals = proof_global_remover::remove_globals(
1608                resolved_before_proofs.clone(),
1609                &mut self.parser.symbol_gen,
1610            );
1611            for command in &typechecked_no_globals {
1612                self.names.check_shadowing(command)?;
1613            }
1614
1615            let term_encoding_added =
1616                ProofInstrumentor::add_term_encoding(self, typechecked_no_globals);
1617            let mut new_typechecked = vec![];
1618            for new_cmd in term_encoding_added {
1619                let desugared =
1620                    desugar_command(new_cmd, &mut self.parser, self.proof_state.proof_testing)?;
1621                for cmd in &desugared {
1622                    log::debug!("Desugared term encoding: {}", cmd.to_command());
1623                }
1624
1625                // Now typecheck using self, adding term type information.
1626                let desugared_typechecked = self.typecheck_program(&desugared)?;
1627                // remove globals again, but this time allow primitive globals
1628                let desugared_typechecked = remove_globals::remove_globals(
1629                    desugared_typechecked,
1630                    &mut self.parser.symbol_gen,
1631                );
1632
1633                new_typechecked.extend(desugared_typechecked);
1634            }
1635            Ok(ResolvedNCommands {
1636                desugared: new_typechecked,
1637                desugared_before_proofs: resolved_before_proofs,
1638            })
1639        }
1640    }

1642    /// Run a program, returning the desugared outputs as well as the CommandOutputs.
1643    /// Can optionally not run the commands, just adding type information.
1644    fn process_program_internal(
1645        &mut self,
1646        program: Vec<Command>,
1647        run_commands: bool,
1648    ) -> Result<ResolvedNCommandsWithOutput, Error> {
1649        let mut outputs = Vec::new();
1650        let mut desugared_before_proofs = Vec::new();
1651        let mut desugared = Vec::new();
1652
1653        for before_expanded_command in program {
1654            // First do user-provided macro expansion for this command,
1655            // which may rely on type information from previous commands.
1656            let macro_expanded = self.command_macros.apply(
1657                before_expanded_command,
1658                &mut self.parser.symbol_gen,
1659                &self.type_info,
1660            )?;
1661
1662            for command in macro_expanded {
1663                // handle include specially- we keep them as-is for desugaring
1664                if let Command::Include(span, file) = &command {
1665                    let s = std::fs::read_to_string(file)
1666                        .unwrap_or_else(|_| panic!("{span} Failed to read file {file}"));
1667                    let included_program = self
1668                        .parser
1669                        .get_program_from_string(Some(file.clone()), &s)?;
1670                    // run program internal on these include commands
1671                    let resolved = self.process_program_internal(included_program, run_commands)?;
1672                    outputs.extend(resolved.outputs);
1673                    desugared.extend(resolved.resolved);
1674                    desugared_before_proofs.extend(resolved.resolved_before_proofs);
1675                } else {
1676                    let resolved = self.resolve_command(command)?;
1677                    if run_commands && self.are_proofs_enabled() {
1678                        self.proof_check_program
1679                            .extend(resolved.desugared_before_proofs.clone());
1680                    }
1681
1682                    desugared_before_proofs.extend(resolved.desugared_before_proofs);
1683                    desugared.extend(resolved.desugared.clone());
1684
1685                    for processed in resolved.desugared {
1686                        // even in desugar mode we still run push and pop
1687                        if run_commands
1688                            || matches!(
1689                                processed,
1690                                ResolvedNCommand::Push(_) | ResolvedNCommand::Pop(_, _)
1691                            )
1692                        {
1693                            let result = self.run_command(processed)?;
1694                            if let Some(output) = result {
1695                                outputs.push(output);
1696                            }
1697                        }
1698                    }
1699                }
1700            }
1701        }
1702
1703        Ok(ResolvedNCommandsWithOutput {
1704            outputs,
1705            resolved_before_proofs: desugared_before_proofs,
1706            resolved: desugared,
1707        })
1708    }

1884struct BackendRule<'a> {
1885    rb: egglog_bridge::RuleBuilder<'a>,
1886    entries: HashMap<core::ResolvedAtomTerm, QueryEntry>,
1887    functions: &'a IndexMap<String, Function>,
1888    type_info: &'a TypeInfo,
1889}
1890
1891impl<'a> BackendRule<'a> {
1892    fn new(
1893        rb: egglog_bridge::RuleBuilder<'a>,
1894        functions: &'a IndexMap<String, Function>,
1895        type_info: &'a TypeInfo,
1896    ) -> BackendRule<'a> {
1897        BackendRule {
1898            rb,
1899            functions,
1900            type_info,
1901            entries: Default::default(),
1902        }
1903    }
1904
1905    fn entry(&mut self, x: &core::ResolvedAtomTerm) -> QueryEntry {
1906        self.entries
1907            .entry(x.clone())
1908            .or_insert_with(|| match x {
1909                core::GenericAtomTerm::Var(_, v) => self
1910                    .rb
1911                    .new_var_named(v.sort.column_ty(self.rb.egraph()), &v.name),
1912                core::GenericAtomTerm::Literal(_, l) => literal_to_entry(self.rb.egraph(), l),
1913                core::GenericAtomTerm::Global(..) => {
1914                    panic!("Globals should have been desugared")
1915                }
1916            })
1917            .clone()
1918    }
1919
1920    fn func(&self, f: &typechecking::FuncType) -> egglog_bridge::FunctionId {
1921        self.functions[&f.name].backend_id
1922    }
1923
1924    fn prim(
1925        &mut self,
1926        prim: &core::SpecializedPrimitive,
1927        args: &[core::ResolvedAtomTerm],
1928    ) -> (ExternalFunctionId, Vec<QueryEntry>, ColumnTy) {
1929        let mut qe_args = self.args(args);
1930
1931        if prim.name() == "unstable-fn" {
1932            let core::ResolvedAtomTerm::Literal(_, Literal::String(ref name)) = args[0] else {
1933                panic!("expected string literal after `unstable-fn`")
1934            };
1935            let id = if let Some(f) = self.type_info.get_func_type(name) {
1936                ResolvedFunctionId::Lookup(egglog_bridge::TableAction::new(
1937                    self.rb.egraph(),
1938                    self.func(f),
1939                ))
1940            } else if let Some(possible) = self.type_info.get_prims(name) {
1941                let mut ps: Vec<_> = possible.iter().collect();
1942                ps.retain(|p| {
1943                    self.type_info
1944                        .get_sorts::<FunctionSort>()
1945                        .into_iter()
1946                        .any(|f| {
1947                            let types: Vec<_> = prim
1948                                .input()
1949                                .iter()
1950                                .skip(1)
1951                                .chain(f.inputs())
1952                                .chain([&f.output()])
1953                                .cloned()
1954                                .collect();
1955                            p.accept(&types, self.type_info)
1956                        })
1957                });
1958                assert!(ps.len() == 1, "options for {name}: {ps:?}");
1959                ResolvedFunctionId::Prim(ps.into_iter().next().unwrap().id)
1960            } else {
1961                panic!("no callable for {name}");
1962            };
1963            let partial_arcsorts = prim.input().iter().skip(1).cloned().collect();
1964
1965            qe_args[0] = self.rb.egraph().base_value_constant(ResolvedFunction {
1966                id,
1967                partial_arcsorts,
1968                name: name.clone(),
1969            });
1970        }
1971
1972        (
1973            prim.external_id(),
1974            qe_args,
1975            prim.output().column_ty(self.rb.egraph()),
1976        )
1977    }
1978
1979    fn args<'b>(
1980        &mut self,
1981        args: impl IntoIterator<Item = &'b core::ResolvedAtomTerm>,
1982    ) -> Vec<QueryEntry> {
1983        args.into_iter().map(|x| self.entry(x)).collect()
1984    }
1985
1986    fn query(&mut self, query: &core::Query<ResolvedCall, ResolvedVar>, include_subsumed: bool) {
1987        for atom in &query.atoms {
1988            match &atom.head {
1989                ResolvedCall::Func(f) => {
1990                    let f = self.func(f);
1991                    let args = self.args(&atom.args);
1992                    let is_subsumed = match include_subsumed {
1993                        true => None,
1994                        false => Some(false),
1995                    };
1996                    self.rb.query_table(f, &args, is_subsumed).unwrap();
1997                }
1998                ResolvedCall::Primitive(p) => {
1999                    let (p, args, ty) = self.prim(p, &atom.args);
2000                    self.rb.query_prim(p, &args, ty).unwrap()
2001                }
2002            }
2003        }
2004    }
2005
2006    fn actions(&mut self, actions: &core::ResolvedCoreActions) -> Result<(), Error> {
2007        for action in &actions.0 {
2008            match action {
2009                core::GenericCoreAction::Let(span, v, f, args) => {
2010                    let v = core::GenericAtomTerm::Var(span.clone(), v.clone());
2011                    let y = match f {
2012                        ResolvedCall::Func(f) => {
2013                            let name = f.name.clone();
2014                            let f = self.func(f);
2015                            let args = self.args(args);
2016                            let span = span.clone();
2017                            self.rb.lookup(f, &args, move || {
2018                                format!("{span}: lookup of function {name} failed")
2019                            })
2020                        }
2021                        ResolvedCall::Primitive(p) => {
2022                            let name = p.name().to_owned();
2023                            let (p, args, ty) = self.prim(p, args);
2024                            let span = span.clone();
2025                            self.rb.call_external_func(p, &args, ty, move || {
2026                                format!("{span}: call of primitive {name} failed")
2027                            })
2028                        }
2029                    };
2030                    self.entries.insert(v, y.into());
2031                }
2032                core::GenericCoreAction::LetAtomTerm(span, v, x) => {
2033                    let v = core::GenericAtomTerm::Var(span.clone(), v.clone());
2034                    let x = self.entry(x);
2035                    self.entries.insert(v, x);
2036                }
2037                core::GenericCoreAction::Set(_, f, xs, y) => match f {
2038                    ResolvedCall::Primitive(..) => panic!("runtime primitive set!"),
2039                    ResolvedCall::Func(f) => {
2040                        let f = self.func(f);
2041                        let args = self.args(xs.iter().chain([y]));
2042                        self.rb.set(f, &args)
2043                    }
2044                },
2045                core::GenericCoreAction::Change(span, change, f, args) => match f {
2046                    ResolvedCall::Primitive(..) => panic!("runtime primitive change!"),
2047                    ResolvedCall::Func(f) => {
2048                        let name = f.name.clone();
2049                        let can_subsume = self.functions[&f.name].can_subsume;
2050                        let f = self.func(f);
2051                        let args = self.args(args);
2052                        match change {
2053                            Change::Delete => self.rb.remove(f, &args),
2054                            Change::Subsume if can_subsume => self.rb.subsume(f, &args),
2055                            Change::Subsume => {
2056                                return Err(Error::SubsumeMergeError(name, span.clone()));
2057                            }
2058                        }
2059                    }
2060                },
2061                core::GenericCoreAction::Union(_, x, y) => {
2062                    let x = self.entry(x);
2063                    let y = self.entry(y);
2064                    self.rb.union(x, y)
2065                }
2066                core::GenericCoreAction::Panic(_, message) => self.rb.panic(message.clone()),
2067            }
2068        }
2069        Ok(())
2070    }
2071
2072    fn build(self) -> egglog_bridge::RuleId {
2073        self.rb.build()
2074    }
2075}

2077fn literal_to_entry(egraph: &egglog_bridge::EGraph, l: &Literal) -> QueryEntry {
2078    match l {
2079        Literal::Int(x) => egraph.base_value_constant::<i64>(*x),
2080        Literal::Float(x) => egraph.base_value_constant::<sort::F>(x.into()),
2081        Literal::String(x) => egraph.base_value_constant::<sort::S>(sort::S::new(x.clone())),
2082        Literal::Bool(x) => egraph.base_value_constant::<bool>(*x),
2083        Literal::Unit => egraph.base_value_constant::<()>(()),
2084    }
2085}

2087fn literal_to_value(egraph: &egglog_bridge::EGraph, l: &Literal) -> Value {
2088    match l {
2089        Literal::Int(x) => egraph.base_values().get::<i64>(*x),
2090        Literal::Float(x) => egraph.base_values().get::<sort::F>(x.into()),
2091        Literal::String(x) => egraph.base_values().get::<sort::S>(sort::S::new(x.clone())),
2092        Literal::Bool(x) => egraph.base_values().get::<bool>(*x),
2093        Literal::Unit => egraph.base_values().get::<()>(()),
2094    }
