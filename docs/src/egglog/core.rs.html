1//! This file implements the core IR of the language, which is called CoreRule.
2//! CoreRule uses a conjunctive query-like IR for the body (queries) and a
3//! SSA-like IR for the head (actions) based on the previous CoreAction form.
4//! Every construct has two forms: a standard (unresolved) form and a resolved form,
5//! which differs in whether the head is a symbol or a resolved call.
6//! Currently, CoreRule has several usages:
7//!   Typechecking is done over CoreRule format
8//!   Canonicalization is done over CoreRule format
9//!   ActionCompilers further compiles core actions to programs in a small VM
10//!   GJ compiler further compiler core queries to gj's CompiledQueries
11//!
12//! Most compiler-time optimizations are expected to be done over CoreRule format.

71#[derive(Debug, Clone)]
72pub enum ResolvedCall {
73    Func(FuncType),
74    Primitive(SpecializedPrimitive),
75}
76
77impl ResolvedCall {
78    pub fn output(&self) -> &ArcSort {
79        match self {
80            ResolvedCall::Func(func) => &func.output,
81            ResolvedCall::Primitive(prim) => prim.output(),
82        }
83    }
84
85    // Different from `from_resolution`, this function only considers function types and not primitives.
86    // As a result, it only requires input argument types, so types.len() == func.input.len(),
87    // while for `from_resolution`, types.len() == func.input.len() + 1 to account for the output type
88    pub fn from_resolution_func_types(
89        head: &str,
90        types: &[ArcSort],
91        typeinfo: &TypeInfo,
92    ) -> Option<ResolvedCall> {
93        if let Some(ty) = typeinfo.get_func_type(head) {
94            // As long as input types match, a result is returned.
95            let expected = ty.input.iter().map(|s| s.name());
96            let actual = types.iter().map(|s| s.name());
97            if expected.eq(actual) {
98                return Some(ResolvedCall::Func(ty.clone()));
99            }
100        }
101        None
102    }
103
104    pub fn from_resolution(head: &str, types: &[ArcSort], typeinfo: &TypeInfo) -> ResolvedCall {
105        let mut resolved_call = Vec::with_capacity(1);
106        if let Some(ty) = typeinfo.get_func_type(head) {
107            let expected = ty.input.iter().chain(once(&ty.output)).map(|s| s.name());
108            let actual = types.iter().map(|s| s.name());
109            if expected.eq(actual) {
110                resolved_call.push(ResolvedCall::Func(ty.clone()));
111            }
112        }
113
114        if let Some(primitives) = typeinfo.get_prims(head) {
115            for primitive in primitives {
116                if primitive.accept(types, typeinfo) {
117                    let (out, inp) = types.split_last().unwrap();
118                    resolved_call.push(ResolvedCall::Primitive(SpecializedPrimitive {
119                        primitive: primitive.clone(),
120                        input: inp.to_vec(),
121                        output: out.clone(),
122                    }));
123                }
124            }
125        }
126        assert!(
127            resolved_call.len() == 1,
128            "Ambiguous resolution for {:?}",
129            head,
130        );
131        resolved_call.pop().unwrap()
132    }
133}
134
135impl Display for ResolvedCall {
136    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
137        match self {
138            ResolvedCall::Func(func) => write!(f, "{}", func.name),
139            ResolvedCall::Primitive(prim) => write!(f, "{}", prim.name()),
140        }
141    }

168#[derive(Debug, Clone)]
169pub enum GenericAtomTerm<Leaf> {
170    Var(Span, Leaf),
171    Literal(Span, Literal),
172    Global(Span, Leaf),
173}
174
175// Ignores annotations for equality and hasing
176impl<Leaf> PartialEq for GenericAtomTerm<Leaf>
177where
178    Leaf: PartialEq,
179{
180    fn eq(&self, other: &Self) -> bool {
181        match (self, other) {
182            (GenericAtomTerm::Var(_, v1), GenericAtomTerm::Var(_, v2)) => v1 == v2,
183            (GenericAtomTerm::Literal(_, l1), GenericAtomTerm::Literal(_, l2)) => l1 == l2,
184            (GenericAtomTerm::Global(_, g1), GenericAtomTerm::Global(_, g2)) => g1 == g2,
185            _ => false,
186        }
187    }
188}
189
190impl<Leaf> Eq for GenericAtomTerm<Leaf> where Leaf: Eq {}

192impl<Leaf> Hash for GenericAtomTerm<Leaf>
193where
194    Leaf: Hash,
195{
196    fn hash<H: Hasher>(&self, state: &mut H) {
197        match self {
198            GenericAtomTerm::Var(_, v) => v.hash(state),
199            GenericAtomTerm::Literal(_, l) => l.hash(state),
200            GenericAtomTerm::Global(_, g) => g.hash(state),
201        }
202    }
203}

263impl<Head, Leaf> GenericAtom<Head, Leaf>
264where
265    Leaf: Clone + Eq + Hash,
266    Head: Clone,
267{
268    pub fn vars(&self) -> impl Iterator<Item = Leaf> + '_ {
269        self.args.iter().filter_map(|t| match t {
270            GenericAtomTerm::Var(_, v) => Some(v.clone()),
271            GenericAtomTerm::Literal(..) => None,
272            GenericAtomTerm::Global(..) => None,
273        })
274    }
275
276    fn subst(&mut self, subst: &HashMap<Leaf, GenericAtomTerm<Leaf>>) {
277        for arg in self.args.iter_mut() {
278            match arg {
279                GenericAtomTerm::Var(_, v) => {
280                    if let Some(at) = subst.get(v) {
281                        *arg = at.clone();
282                    }
283                }
284                GenericAtomTerm::Literal(..) => (),
285                GenericAtomTerm::Global(..) => (),
286            }
287        }
288    }
289}

296            self.args[0..n - 1]
297                .iter()
298                .map(|arg| arg.to_expr())
299                .collect(),
300        )
301    }

410#[derive(Clone, Debug, PartialEq, Eq, Hash)]
411pub enum GenericCoreAction<Head, Leaf> {
412    Let(Span, Leaf, Head, Vec<GenericAtomTerm<Leaf>>),
413    LetAtomTerm(Span, Leaf, GenericAtomTerm<Leaf>),
414    Set(
415        Span,
416        Head,
417        Vec<GenericAtomTerm<Leaf>>,
418        GenericAtomTerm<Leaf>,
419    ),
420    Change(Span, Change, Head, Vec<GenericAtomTerm<Leaf>>),
421    Union(Span, GenericAtomTerm<Leaf>, GenericAtomTerm<Leaf>),
422    Panic(Span, String),
423}

428pub struct GenericCoreActions<Head, Leaf>(pub(crate) Vec<GenericCoreAction<Head, Leaf>>);
429pub(crate) type ResolvedCoreActions = GenericCoreActions<ResolvedCall, ResolvedVar>;
430
431impl<Head, Leaf> Default for GenericCoreActions<Head, Leaf> {
432    fn default() -> Self {
433        Self(vec![])
434    }

437impl<Head, Leaf> GenericCoreActions<Head, Leaf>
438where
439    Leaf: Clone + Eq + Hash,
440{
441    pub(crate) fn subst(&mut self, subst: &HashMap<Leaf, GenericAtomTerm<Leaf>>) {
442        let actions = subst.iter().map(|(symbol, atom_term)| {
443            GenericCoreAction::LetAtomTerm(
444                atom_term.span().clone(),
445                symbol.clone(),
446                atom_term.clone(),
447            )
448        });
449        let existing_actions = std::mem::take(&mut self.0);
450        self.0 = actions.chain(existing_actions).collect();
451    }
452
453    fn new(actions: Vec<GenericCoreAction<Head, Leaf>>) -> GenericCoreActions<Head, Leaf> {
454        Self(actions)
455    }
456
457    pub(crate) fn get_free_vars(&self) -> HashSet<Leaf> {
458        let at_free_var = |at: &GenericAtomTerm<Leaf>| match at {
459            GenericAtomTerm::Var(_, v) => Some(v.clone()),
460            GenericAtomTerm::Literal(..) => None,
461            GenericAtomTerm::Global(..) => None,
462        };
463
464        let add_from_atom = |free_vars: &mut HashSet<Leaf>, at: &GenericAtomTerm<Leaf>| {
465            if let Some(v) = at_free_var(at) {
466                free_vars.insert(v);
467            }
468        };
469
470        let add_from_atoms = |free_vars: &mut HashSet<Leaf>, ats: &[GenericAtomTerm<Leaf>]| {
471            ats.iter().flat_map(&at_free_var).for_each(|v| {
472                free_vars.insert(v);
473            });
474        };
475
476        let mut free_vars = HashSet::default();
477        for action in self.0.iter().rev() {
478            match action {
479                GenericCoreAction::Let(_span, v, _, ats) => {
480                    add_from_atoms(&mut free_vars, ats);
481                    free_vars.remove(v);
482                }
483                GenericCoreAction::LetAtomTerm(_span, v, at) => {
484                    add_from_atom(&mut free_vars, at);
485                    free_vars.remove(v);
486                }
487                GenericCoreAction::Set(_span, _, ats, at) => {
488                    add_from_atoms(&mut free_vars, ats);
489                    add_from_atom(&mut free_vars, at);
490                }
491                GenericCoreAction::Change(_span, _change, _, ats) => {
492                    add_from_atoms(&mut free_vars, ats);
493                }
494                GenericCoreAction::Union(_span, at, at1) => {
495                    add_from_atom(&mut free_vars, at);
496                    add_from_atom(&mut free_vars, at1);
497                }
498                GenericCoreAction::Panic(_span, _) => {}
499            }
500        }
501        free_vars
502    }
503}

505pub(crate) trait GenericActionsExt<Head, Leaf> {
506    #[allow(clippy::type_complexity)]
507    fn to_core_actions<FG>(
508        &self,
509        typeinfo: &TypeInfo,
510        binding: &mut IndexSet<Leaf>,
511        fresh_gen: &mut FG,
512    ) -> Result<(GenericCoreActions<Head, Leaf>, MappedActions<Head, Leaf>), TypeError>
513    where
514        Head: Clone + Display + IsFunc,
515        Leaf: Clone + PartialEq + Eq + Display + Hash,
516        FG: FreshGen<Head, Leaf>;
517}
518
519impl<Head, Leaf> GenericActionsExt<Head, Leaf> for GenericActions<Head, Leaf>
520where
521    Head: Clone + Display + IsFunc,
522    Leaf: Clone + PartialEq + Eq + Display + Hash,
523{
524    #[allow(clippy::type_complexity)]
525    fn to_core_actions<FG>(
526        &self,
527        typeinfo: &TypeInfo,
528        binding: &mut IndexSet<Leaf>,
529        fresh_gen: &mut FG,
530    ) -> Result<(GenericCoreActions<Head, Leaf>, MappedActions<Head, Leaf>), TypeError>
531    where
532        Head: Clone + Display + IsFunc,
533        Leaf: Clone + PartialEq + Eq + Display + Hash,
534        FG: FreshGen<Head, Leaf>,
535    {
536        let mut norm_actions = vec![];
537        let mut mapped_actions: MappedActions<Head, Leaf> = GenericActions(vec![]);
538
539        // During the lowering, there are two important guaratees:
540        //   Every used variable should be bound.
541        //   Every introduced variable should be unbound before.
542        for action in self.0.iter() {
543            match action {
544                GenericAction::Let(span, var, expr) => {
545                    if binding.contains(var) {
546                        return Err(TypeError::AlreadyDefined(var.to_string(), span.clone()));
547                    }
548                    let mapped_expr =
549                        expr.to_core_actions(typeinfo, binding, fresh_gen, &mut norm_actions)?;
550                    norm_actions.push(GenericCoreAction::LetAtomTerm(
551                        span.clone(),
552                        var.clone(),
553                        mapped_expr.get_corresponding_var_or_lit(typeinfo),
554                    ));
555                    mapped_actions.0.push(GenericAction::Let(
556                        span.clone(),
557                        var.clone(),
558                        mapped_expr,
559                    ));
560                    binding.insert(var.clone());
561                }
562                GenericAction::Set(span, head, args, expr) => {
563                    let mut mapped_args = vec![];
564                    for arg in args {
565                        let mapped_arg =
566                            arg.to_core_actions(typeinfo, binding, fresh_gen, &mut norm_actions)?;
567                        mapped_args.push(mapped_arg);
568                    }
569                    let mapped_expr =
570                        expr.to_core_actions(typeinfo, binding, fresh_gen, &mut norm_actions)?;
571                    norm_actions.push(GenericCoreAction::Set(
572                        span.clone(),
573                        head.clone(),
574                        mapped_args
575                            .iter()
576                            .map(|e| e.get_corresponding_var_or_lit(typeinfo))
577                            .collect(),
578                        mapped_expr.get_corresponding_var_or_lit(typeinfo),
579                    ));
580                    let v = fresh_gen.fresh(head);
581                    mapped_actions.0.push(GenericAction::Set(
582                        span.clone(),
583                        CorrespondingVar::new(head.clone(), v),
584                        mapped_args,
585                        mapped_expr,
586                    ));
587                }
588                GenericAction::Change(span, change, head, args) => {
589                    let mut mapped_args = vec![];
590                    for arg in args {
591                        let mapped_arg =
592                            arg.to_core_actions(typeinfo, binding, fresh_gen, &mut norm_actions)?;
593                        mapped_args.push(mapped_arg);
594                    }
595                    norm_actions.push(GenericCoreAction::Change(
596                        span.clone(),
597                        *change,
598                        head.clone(),
599                        mapped_args
600                            .iter()
601                            .map(|e| e.get_corresponding_var_or_lit(typeinfo))
602                            .collect(),
603                    ));
604                    let v = fresh_gen.fresh(head);
605                    mapped_actions.0.push(GenericAction::Change(
606                        span.clone(),
607                        *change,
608                        CorrespondingVar::new(head.clone(), v),
609                        mapped_args,
610                    ));
611                }
612                GenericAction::Union(span, e1, e2) => {
613                    match (e1, e2) {
614                        (var @ GenericExpr::Var(..), GenericExpr::Call(_, f, args))
615                        | (GenericExpr::Call(_, f, args), var @ GenericExpr::Var(..))
616                            if f.is_constructor(typeinfo) =>
617                        {
618                            let head = f;
619                            let expr = var;
620                            let mut mapped_args = vec![];
621                            for arg in args {
622                                let mapped_arg = arg.to_core_actions(
623                                    typeinfo,
624                                    binding,
625                                    fresh_gen,
626                                    &mut norm_actions,
627                                )?;
628                                mapped_args.push(mapped_arg);
629                            }
630                            let mapped_expr = expr.to_core_actions(
631                                typeinfo,
632                                binding,
633                                fresh_gen,
634                                &mut norm_actions,
635                            )?;
636                            norm_actions.push(GenericCoreAction::Set(
637                                span.clone(),
638                                head.clone(),
639                                mapped_args
640                                    .iter()
641                                    .map(|e| e.get_corresponding_var_or_lit(typeinfo))
642                                    .collect(),
643                                mapped_expr.get_corresponding_var_or_lit(typeinfo),
644                            ));
645                            let v = fresh_gen.fresh(head);
646                            mapped_actions.0.push(GenericAction::Set(
647                                span.clone(),
648                                CorrespondingVar::new(head.clone(), v),
649                                mapped_args,
650                                mapped_expr,
651                            ));
652                        }
653                        _ => {
654                            let mapped_e1 = e1.to_core_actions(
655                                typeinfo,
656                                binding,
657                                fresh_gen,
658                                &mut norm_actions,
659                            )?;
660                            let mapped_e2 = e2.to_core_actions(
661                                typeinfo,
662                                binding,
663                                fresh_gen,
664                                &mut norm_actions,
665                            )?;
666                            norm_actions.push(GenericCoreAction::Union(
667                                span.clone(),
668                                mapped_e1.get_corresponding_var_or_lit(typeinfo),
669                                mapped_e2.get_corresponding_var_or_lit(typeinfo),
670                            ));
671                            mapped_actions.0.push(GenericAction::Union(
672                                span.clone(),
673                                mapped_e1,
674                                mapped_e2,
675                            ));
676                        }
677                    };
678                }
679                GenericAction::Panic(span, string) => {
680                    norm_actions.push(GenericCoreAction::Panic(span.clone(), string.clone()));
681                    mapped_actions
682                        .0
683                        .push(GenericAction::Panic(span.clone(), string.clone()));
684                }
685                GenericAction::Expr(span, expr) => {
686                    let mapped_expr =
687                        expr.to_core_actions(typeinfo, binding, fresh_gen, &mut norm_actions)?;
688                    mapped_actions
689                        .0
690                        .push(GenericAction::Expr(span.clone(), mapped_expr));
691                }
692            }
693        }
694        Ok((GenericCoreActions::new(norm_actions), mapped_actions))
695    }
696}

698pub(crate) trait GenericExprExt<Head, Leaf>
699where
700    Head: Clone + Display,
701    Leaf: Clone + PartialEq + Eq + Display + Hash,
702{
703    fn to_query(
704        &self,
705        typeinfo: &TypeInfo,
706        fresh_gen: &mut impl FreshGen<Head, Leaf>,
707    ) -> (
708        Vec<GenericAtom<HeadOrEq<Head>, Leaf>>,
709        MappedExpr<Head, Leaf>,
710    );
711
712    fn to_core_actions<FG: FreshGen<Head, Leaf>>(
713        &self,
714        typeinfo: &TypeInfo,
715        binding: &mut IndexSet<Leaf>,
716        fresh_gen: &mut FG,
717        out_actions: &mut Vec<GenericCoreAction<Head, Leaf>>,
718    ) -> Result<MappedExpr<Head, Leaf>, TypeError>;
719}
720
721impl<Head, Leaf> GenericExprExt<Head, Leaf> for GenericExpr<Head, Leaf>
722where
723    Head: Clone + Display,
724    Leaf: Clone + PartialEq + Eq + Display + Hash,
725{
726    fn to_query(
727        &self,
728        typeinfo: &TypeInfo,
729        fresh_gen: &mut impl FreshGen<Head, Leaf>,
730    ) -> (
731        Vec<GenericAtom<HeadOrEq<Head>, Leaf>>,
732        MappedExpr<Head, Leaf>,
733    )
734    where
735        Head: Clone + Display,
736        Leaf: Clone + PartialEq + Eq + Display + Hash,
737    {
738        match self {
739            GenericExpr::Lit(span, lit) => (vec![], GenericExpr::Lit(span.clone(), lit.clone())),
740            GenericExpr::Var(span, v) => (vec![], GenericExpr::Var(span.clone(), v.clone())),
741            GenericExpr::Call(span, f, children) => {
742                let fresh = fresh_gen.fresh(f);
743                let mut new_children = vec![];
744                let mut atoms = vec![];
745                let mut child_exprs = vec![];
746                for child in children {
747                    let (child_atoms, child_expr) = child.to_query(typeinfo, fresh_gen);
748                    let child_atomterm = child_expr.get_corresponding_var_or_lit(typeinfo);
749                    new_children.push(child_atomterm);
750                    atoms.extend(child_atoms);
751                    child_exprs.push(child_expr);
752                }
753                let args = {
754                    new_children.push(GenericAtomTerm::Var(span.clone(), fresh.clone()));
755                    new_children
756                };
757                atoms.push(GenericAtom {
758                    span: span.clone(),
759                    head: HeadOrEq::Head(f.clone()),
760                    args,
761                });
762                (
763                    atoms,
764                    GenericExpr::Call(
765                        span.clone(),
766                        CorrespondingVar::new(f.clone(), fresh),
767                        child_exprs,
768                    ),
769                )
770            }
771        }
772    }
773
774    fn to_core_actions<FG: FreshGen<Head, Leaf>>(
775        &self,
776        typeinfo: &TypeInfo,
777        binding: &mut IndexSet<Leaf>,
778        fresh_gen: &mut FG,
779        out_actions: &mut Vec<GenericCoreAction<Head, Leaf>>,
780    ) -> Result<MappedExpr<Head, Leaf>, TypeError> {
781        match self {
782            GenericExpr::Lit(span, lit) => Ok(GenericExpr::Lit(span.clone(), lit.clone())),
783            GenericExpr::Var(span, v) => {
784                let sym = v.to_string();
785                if binding.contains(v) || typeinfo.is_global(&sym) {
786                    Ok(GenericExpr::Var(span.clone(), v.clone()))
787                } else {
788                    Err(TypeError::Unbound(sym, span.clone()))
789                }
790            }
791            GenericExpr::Call(span, f, args) => {
792                let mut norm_args = vec![];
793                let mut mapped_args = vec![];
794                for arg in args {
795                    let mapped_arg =
796                        arg.to_core_actions(typeinfo, binding, fresh_gen, out_actions)?;
797                    norm_args.push(mapped_arg.get_corresponding_var_or_lit(typeinfo));
798                    mapped_args.push(mapped_arg);
799                }
800                let var = fresh_gen.fresh(f);
801                binding.insert(var.clone());
802                out_actions.push(GenericCoreAction::Let(
803                    span.clone(),
804                    var.clone(),
805                    f.clone(),
806                    norm_args,
807                ));
808                Ok(GenericExpr::Call(
809                    span.clone(),
810                    CorrespondingVar::new(f.clone(), var),
811                    mapped_args,
812                ))
813            }
814        }
815    }
816}

818/// A [`GenericCoreRule`] represents a generalization of lowered form of a rule.
819/// Unlike other `Generic`-prefixed types, [`GenericCoreRule`] takes two `Head`
820/// parameters instead of one. This is because the `Head` parameter of `body` and
821/// `head` can be different. In particular, early in the compilation pipeline,
822/// `body` can contain `Eq` atoms, which denotes equality constraints, so the `Head`
823/// for `body` needs to be a `HeadOrEq<Head>`, while `head` does not have equality
824/// constraints.
825#[derive(Debug, Clone)]
826pub struct GenericCoreRule<HeadQ, HeadA, Leaf> {
827    pub span: Span,
828    pub body: Query<HeadQ, Leaf>,
829    pub head: GenericCoreActions<HeadA, Leaf>,
830}
831
832pub(crate) type CoreRule = GenericCoreRule<StringOrEq, String, String>;
833pub(crate) type ResolvedCoreRule = GenericCoreRule<ResolvedCall, ResolvedCall, ResolvedVar>;
834
835impl<Head1, Head2, Leaf> GenericCoreRule<Head1, Head2, Leaf>
836where
837    Head1: Clone,
838    Head2: Clone,
839    Leaf: Clone + Eq + Hash,
840{
841    pub fn subst(&mut self, subst: &HashMap<Leaf, GenericAtomTerm<Leaf>>) {
842        for atom in &mut self.body.atoms {
843            atom.subst(subst);
844        }
845        self.head.subst(subst);
846    }
847}
848
849impl<Head, Leaf> GenericCoreRule<HeadOrEq<Head>, Head, Leaf>
850where
851    Leaf: Eq + Clone + Hash + Debug,
852    Head: Clone,
853{
854    /// Transformed a UnresolvedCoreRule into a CanonicalizedCoreRule.
855    /// In particular, it removes equality checks between variables and
856    /// other arguments, and turns equality checks between non-variable arguments
857    /// into a primitive equality check `value-eq`.
858    pub(crate) fn canonicalize(
859        self,
860        // Users need to pass in a substitute for equality constraints.
861        value_eq: impl Fn(&GenericAtomTerm<Leaf>, &GenericAtomTerm<Leaf>) -> Head,
862    ) -> GenericCoreRule<Head, Head, Leaf> {
863        let mut result_rule = self;
864        loop {
865            let mut to_subst = None;
866            for atom in result_rule.body.atoms.iter() {
867                if atom.head.is_eq() && atom.args[0] != atom.args[1] {
868                    match &atom.args[..] {
869                        [GenericAtomTerm::Var(_, x), y] | [y, GenericAtomTerm::Var(_, x)] => {
870                            to_subst = Some((x, y));
871                            break;
872                        }
873                        _ => (),
874                    }
875                }
876            }
877            if let Some((x, y)) = to_subst {
878                let subst = HashMap::from_iter([(x.clone(), y.clone())]);
879                result_rule.subst(&subst);
880            } else {
881                break;
882            }
883        }
884
885        let atoms = result_rule
886            .body
887            .atoms
888            .into_iter()
889            .filter_map(|atom| match atom.head {
890                HeadOrEq::Eq => {
891                    assert_eq!(atom.args.len(), 2);
892                    match (&atom.args[0], &atom.args[1]) {
893                        (GenericAtomTerm::Var(_, v1), GenericAtomTerm::Var(_, v2)) => {
894                            assert_eq!(v1, v2);
895                            None
896                        }
897                        (GenericAtomTerm::Var(..), _) | (_, GenericAtomTerm::Var(..)) => {
898                            panic!("equalities between variable and non-variable arguments should have been canonicalized")
899                        }
900                        (at1, at2) => {
901                            if at1 == at2 {
902                                None
903                            } else {
904                                Some(GenericAtom {
905                                    span: atom.span.clone(),
906                                    head: value_eq(&atom.args[0], &atom.args[1]),
907                                    args: vec![
908                                        atom.args[0].clone(),
909                                        atom.args[1].clone(),
910                                        GenericAtomTerm::Literal(atom.span.clone(), Literal::Unit),
911                                    ],
912                                })
913                            }
914                        },
915                    }
916                }
917                HeadOrEq::Head(symbol) => Some(GenericAtom {
918                    span: atom.span.clone(),
919                    head: symbol,
920                    args: atom.args,
921                }),
922            })
923            .collect();
924        GenericCoreRule {
925            span: result_rule.span,
926            body: Query { atoms },
927            head: result_rule.head,
928        }
929    }
930}

932pub(crate) trait GenericRuleExt<Head, Leaf> {
933    fn to_core_rule(
934        &self,
935        typeinfo: &TypeInfo,
936        fresh_gen: &mut impl FreshGen<Head, Leaf>,
937    ) -> Result<GenericCoreRule<HeadOrEq<Head>, Head, Leaf>, TypeError>
938    where
939        Head: Clone + Display + IsFunc,
940        Leaf: Clone + PartialEq + Eq + Display + Hash + Debug;
941}
942
943impl<Head, Leaf> GenericRuleExt<Head, Leaf> for GenericRule<Head, Leaf>
944where
945    Head: Clone + Display + IsFunc,
946    Leaf: Clone + PartialEq + Eq + Display + Hash + Debug,
947{
948    fn to_core_rule(
949        &self,
950        typeinfo: &TypeInfo,
951        fresh_gen: &mut impl FreshGen<Head, Leaf>,
952    ) -> Result<GenericCoreRule<HeadOrEq<Head>, Head, Leaf>, TypeError>
953    where
954        Head: Clone + Display + IsFunc,
955        Leaf: Clone + PartialEq + Eq + Display + Hash + Debug,
956    {
957        let (body, _correspondence) = Facts(self.body.clone()).to_query(typeinfo, fresh_gen);
958        let mut binding = body.get_vars();
959        let (head, _correspondence) =
960            self.head
961                .to_core_actions(typeinfo, &mut binding, fresh_gen)?;
962        Ok(GenericCoreRule {
963            span: self.span.clone(),
964            body,
965            head,
966        })
967    }
968}

970pub(crate) trait ResolvedRuleExt {
971    fn to_canonicalized_core_rule(
972        &self,
973        typeinfo: &TypeInfo,
974        fresh_gen: &mut SymbolGen,
975    ) -> Result<ResolvedCoreRule, TypeError>;
976}
977
978impl ResolvedRuleExt for ResolvedRule {
979    fn to_canonicalized_core_rule(
980        &self,
981        typeinfo: &TypeInfo,
982        fresh_gen: &mut SymbolGen,
983    ) -> Result<ResolvedCoreRule, TypeError> {
984        let value_eq = &typeinfo.get_prims("value-eq").unwrap()[0];
985        let value_eq = |at1: &ResolvedAtomTerm, at2: &ResolvedAtomTerm| {
986            ResolvedCall::Primitive(SpecializedPrimitive {
987                primitive: value_eq.clone(),
988                input: vec![at1.output(), at2.output()],
989                output: UnitSort.to_arcsort(),
990            })
991        };
992
993        let rule = self.to_core_rule(typeinfo, fresh_gen)?;
994
995        // The groundedness check happens before canonicalization, because canonicalization
996        // may turn ungrounded variables in a query to unbounded variables in actions (e.g.,
997        // `(rule ((= x y)) ((R x y)))`) but unboundedness is only checked during type checking.
998        grounded_check(&rule)?;
999
1000        let rule = rule.canonicalize(value_eq);
1001
1002        Ok(rule)
1003    }
