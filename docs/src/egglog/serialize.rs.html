6pub struct SerializeConfig {
7    // Maximumum number of functions to include in the serialized graph, any after this will be discarded
8    pub max_functions: Option<usize>,
9    // Maximum number of calls to include per function, any after this will be discarded
10    pub max_calls_per_function: Option<usize>,
11    // Whether to include temporary functions in the serialized graph
12    pub include_temporary_functions: bool,
13    // Root eclasses to include in the output
14    pub root_eclasses: Vec<(ArcSort, Value)>,

35/// A node in the serialized egraph.
36#[derive(PartialEq, Debug, Clone)]
37pub enum SerializedNode {
38    /// A user defined function call.
39    Function {
40        /// The name of the function.
41        name: String,
42        /// The offset of the index in the table.
43        /// This can be resolved to the output and input values with table.get_index(offset, true).
44        offset: usize,
45    },
46    /// A primitive value.
47    Primitive(Value),
48    /// A dummy node used to represent omitted nodes.
49    Dummy(Value),
50    /// A node that was split into multiple e-classes.
51    Split(Box<SerializedNode>),
52}
53
54impl SerializedNode {
55    /// Returns true if the node is a primitive value.
56    pub fn is_primitive(&self) -> bool {
57        match self {
58            SerializedNode::Primitive(_) => true,
59            SerializedNode::Split(node) => node.is_primitive(),
60            _ => false,
61        }
62    }
63}

65impl EGraph {
66    /// Serialize the egraph into a format that can be read by the egraph-serialize crate.
67    ///
68    /// There are multiple different semantically valid ways to do this. This is how this implementation does it:
69    ///
70    /// For node costs:
71    /// - Primitives: 1.0
72    /// - Function without costs: 1.0
73    /// - Function with costs: the cost
74    /// - Omitted nodes: infinite
75    ///
76    /// For node IDs:
77    /// - Functions: Function name + hash of input values
78    /// - Args which are eq sorts: Choose one ID from the e-class, distribute roughly evenly.
79    /// - Args and outputs values which are primitives: Sort name + hash of value
80    ///
81    /// For e-classes IDs:
82    /// - tag and value of canonicalized value
83    ///
84    /// This is to achieve the following properties:
85    /// - Equivalent primitive values will show up once in the e-graph.
86    /// - Functions which return primitive values will be added to the e-class of that value.
87    /// - Nodes will have consistant IDs throughout execution of e-graph (used for animating changes in the visualization)
88    /// - Edges in the visualization will be well distributed (used for animating changes in the visualization)
89    ///   (Note that this will be changed in `<https://github.com/egraphs-good/egglog/pull/158>` so that edges point to exact nodes instead of looking up the e-class)
90    pub fn serialize(&self, config: SerializeConfig) -> egraph_serialize::EGraph {
91        // First collect a list of all the calls we want to serialize
92        let all_calls: Vec<(
93            &Function,
94            Vec<Value>, // inputs
95            Value,      // output
96            bool,       // is subsumed
97            egraph_serialize::ClassId,
98            egraph_serialize::NodeId,
99        )> = self
100            .functions
101            .iter()
102            .filter(|(_, function)| !function.decl.ignore_viz)
103            .map(|(name, function)| {
104                let mut tuples = vec![];
105                self.backend.for_each_while(function.backend_id, |row| {
106                    if tuples.len() >= config.max_calls_per_function.unwrap_or(usize::MAX) {
107                        return false;
108                    }
109                    let (out, inps) = row.vals.split_last().unwrap();
110                    tuples.push((
111                        function,
112                        inps.to_vec(),
113                        *out,
114                        row.subsumed,
115                        self.value_to_class_id(&function.schema.output, *out),
116                        self.to_node_id(
117                            None,
118                            SerializedNode::Function {
119                                name: name.clone(),
120                                offset: tuples.len(),
121                            },
122                        ),
123                    ));
124                    true
125                });
126                tuples
127            })
128            // Filter out functions with no calls
129            .filter(|f| !f.is_empty())
130            .take(config.max_functions.unwrap_or(usize::MAX))
131            .flatten()
132            .collect();
133
134        // Then create a mapping from each canonical e-class ID to the set of node IDs in that e-class
135        // Note that this is only for e-classes, primitives have e-classes equal to their node ID
136        // This is for when we need to find what node ID to use for an edge to an e-class, we can rotate them evenly
137        // amoung all possible options.
138        let node_ids: NodeIDs = all_calls.iter().fold(
139            HashMap::default(),
140            |mut acc, (func, _input, _output, _subsumed, class_id, node_id)| {
141                if func.schema.output.is_eq_sort() {
142                    acc.entry(class_id.clone())
143                        .or_default()
144                        .push_back(node_id.clone());
145                }
146                acc
147            },
148        );
149
150        let mut serializer = Serializer {
151            node_ids,
152            result: egraph_serialize::EGraph::default(),
153        };
154
155        for (func, input, output, subsumed, class_id, node_id) in all_calls {
156            self.serialize_value(&mut serializer, &func.schema.output, output, &class_id);
157
158            assert_eq!(input.len(), func.schema.input.len());
159            let children: Vec<_> = input
160                .iter()
161                .zip(&func.schema.input)
162                .map(|(&v, sort)| {
163                    self.serialize_value(&mut serializer, sort, v, &self.value_to_class_id(sort, v))
164                })
165                .collect();
166            serializer.result.nodes.insert(
167                node_id,
168                egraph_serialize::Node {
169                    op: func.decl.name.to_string(),
170                    eclass: class_id.clone(),
171                    cost: NotNan::new(func.decl.cost.unwrap_or(1) as f64).unwrap(),
172                    children,
173                    subsumed,
174                },
175            );
176        }
177
178        serializer.result.root_eclasses = config
179            .root_eclasses
180            .iter()
181            .map(|(sort, v)| self.value_to_class_id(sort, *v))
182            .collect();
183
184        serializer.result
185    }

187    /// Gets the serialized class ID for a value.
188    pub fn value_to_class_id(&self, sort: &ArcSort, value: Value) -> egraph_serialize::ClassId {
189        // Canonicalize the value first so that we always use the canonical e-class ID
190        let value = self.backend.get_canon(value);
191        assert!(
192            !sort.name().to_string().contains('-'),
193            "Tag cannot contain '-' when serializing"
194        );
195        use numeric_id::NumericId;
196        format!("{}-{}", sort.name(), value.rep()).into()
197    }
198
199    /// Gets the value for a serialized class ID.
200    pub fn class_id_to_value(&self, eclass_id: &egraph_serialize::ClassId) -> Value {
201        let s = eclass_id.to_string();
202        let (_tag, bits) = s.split_once('-').unwrap();
203        Value::new_const(bits.parse().unwrap())
204    }
205
206    /// Gets the serialized node ID for the primitive, omitted, or function value.
207    pub fn to_node_id(
208        &self,
209        sort: Option<&ArcSort>,
210        node: SerializedNode,
211    ) -> egraph_serialize::NodeId {
212        match node {
213            SerializedNode::Function { name, offset } => {
214                assert!(sort.is_none());
215                format!("function-{}-{}", offset, name).into()
216            }
217            SerializedNode::Primitive(value) => {
218                format!("primitive-{}", self.value_to_class_id(sort.unwrap(), value)).into()
219            }
220            SerializedNode::Dummy(value) => {
221                format!("dummy-{}", self.value_to_class_id(sort.unwrap(), value)).into()
222            }
223            SerializedNode::Split(node) => format!("split-{}", self.to_node_id(sort, *node)).into(),
224        }
225    }
226
227    /// Gets the serialized node for the node ID.
228    pub fn from_node_id(&self, node_id: &egraph_serialize::NodeId) -> SerializedNode {
229        let node_id = node_id.to_string();
230        let (tag, rest) = node_id.split_once('-').unwrap();
231        match tag {
232            "function" => {
233                let (offset, name) = rest.split_once('-').unwrap();
234                SerializedNode::Function {
235                    name: name.into(),
236                    offset: offset.parse().unwrap(),
237                }
238            }
239            "primitive" => {
240                let class_id: egraph_serialize::ClassId = rest.into();
241                SerializedNode::Primitive(self.class_id_to_value(&class_id))
242            }
243            "dummy" => {
244                let class_id: egraph_serialize::ClassId = rest.into();
245                SerializedNode::Dummy(self.class_id_to_value(&class_id))
246            }
247            "split" => {
248                let (_offset, rest) = rest.split_once('-').unwrap();
249                let node_id: egraph_serialize::NodeId = rest.into();
250                SerializedNode::Split(Box::new(self.from_node_id(&node_id)))
251            }
252            _ => std::panic::panic_any(format!("Unknown node ID: {}-{}", tag, rest)),
253        }
254    }

256    /// Serialize the value and return the eclass and node ID
257    /// If this is a primitive value, we will add the node to the data, but if it is an eclass, we will not
258    /// When this is called on the output of a node, we only use the e-class to know which e-class its a part of
259    /// When this is called on an input of a node, we only use the node ID to know which node to point to.
260    fn serialize_value(
261        &self,
262        serializer: &mut Serializer,
263        sort: &ArcSort,
264        value: Value,
265        class_id: &egraph_serialize::ClassId,
266    ) -> egraph_serialize::NodeId {
267        let node_id = if sort.is_eq_sort() {
268            let node_ids = serializer
269                .node_ids
270                .entry(class_id.clone())
271                .or_insert_with(|| {
272                    // If we don't find node IDs for this class, it means that all nodes for it were omitted due to size constraints
273                    // In this case, add a dummy node in this class to represent the missing nodes
274                    let node_id = self.to_node_id(Some(sort), SerializedNode::Dummy(value));
275                    serializer.result.nodes.insert(
276                        node_id.clone(),
277                        egraph_serialize::Node {
278                            op: "[...]".to_string(),
279                            eclass: class_id.clone(),
280                            cost: NotNan::new(f64::INFINITY).unwrap(),
281                            children: vec![],
282                            subsumed: false,
283                        },
284                    );
285                    VecDeque::from(vec![node_id])
286                });
287            node_ids.rotate_left(1);
288            node_ids.front().unwrap().clone()
289        } else {
290            let node_id = self.to_node_id(Some(sort), SerializedNode::Primitive(value));
291            // Add node for value
292            {
293                // Children will be empty unless this is a container sort
294                let children: Vec<egraph_serialize::NodeId> = sort
295                    .inner_values(self.backend.container_values(), value)
296                    .into_iter()
297                    .map(|(s, v)| {
298                        self.serialize_value(serializer, &s, v, &self.value_to_class_id(&s, v))
299                    })
300                    .collect();
301                // If this is a container sort, use the name, otherwise use the value
302                let op = if sort.is_container_sort() {
303                    sort.serialized_name(value).to_string()
304                } else {
305                    let primitive_id = self
306                        .backend
307                        .base_values()
308                        .get_ty_by_id(sort.value_type().unwrap());
309                    let formatted_val = BaseValuePrinter {
310                        base: self.backend.base_values(),
311                        ty: primitive_id,
312                        val: value,
313                    };
314                    format!("{:?}", formatted_val)
315                };
316                serializer.result.nodes.insert(
317                    node_id.clone(),
318                    egraph_serialize::Node {
319                        op,
320                        eclass: class_id.clone(),
321                        cost: NotNan::new(1.0).unwrap(),
322                        children,
323                        subsumed: false,
324                    },
325                );
326            };
327            node_id
328        };
329        serializer.result.class_data.insert(
330            class_id.clone(),
331            egraph_serialize::ClassData {
332                typ: Some(sort.name().to_string()),
333            },
334        );
335        node_id
336    }
