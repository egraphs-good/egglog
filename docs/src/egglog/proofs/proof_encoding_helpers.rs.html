50/// Packages proof information for instrumenting actions.

56    Proof(String),                 // existing proof

273            let ast_sort = &self.proof_names().ast_sort;

276            "".to_string()

354    /// Header string for proof encoding, defining sorts and constructors.
355    /// Correspondings to [`RawProof`] in the Rust code.
356    pub(crate) fn proof_header(&mut self) -> String {
357        let mut to_ast_constructors = Vec::new();
358        // need to build a Ast{lit} for each lit sort in self
359        for sort_name in self.egraph.type_info.sorts.keys().clone() {
360            if !self
361                .proof_names()
362                .sort_to_ast_constructor
363                .contains_key(sort_name)
364            {
365                let ast_constructor = self
366                    .egraph
367                    .parser
368                    .symbol_gen
369                    .fresh(&format!("Ast{}", sort_name));
370                self.egraph
371                    .proof_state
372                    .proof_names
373                    .sort_to_ast_constructor
374                    .insert(sort_name.clone(), ast_constructor.clone());
375                to_ast_constructors.push(format!(
376                    "(constructor {ast_constructor} ({sort_name} ) {} :internal-hidden)",
377                    self.proof_names().ast_sort
378                ));
379            }
380        }
381        let to_ast_str = to_ast_constructors.join("\n");
382
383        let EncodingNames {
384            ref proof_list_sort,
385            ref ast_sort,
386            ref proof_datatype,
387            ref fiat_constructor,
388            ref rule_constructor,
389            ref merge_fn_constructor,
390            ref eq_trans_constructor,
391            ref eq_sym_constructor,
392            ref congr_constructor,
393            ref pcons,
394            ref pnil,
395            ..
396        } = *self.proof_names();
397
398        format!(
399            "
400(sort {proof_list_sort})
401(sort {ast_sort}) ;; wrap sorts in this for proofs
402(sort {proof_datatype})
403
404(constructor {pcons} ({proof_datatype} {proof_list_sort}) {proof_list_sort} :internal-hidden)
405(constructor {pnil} () {proof_list_sort} :internal-hidden)
406
407{to_ast_str}
408
409;; Fiat justification for globals and primitives, gives two terms t1 = t2 for the proposition being justified
410(constructor {fiat_constructor} ({ast_sort} {ast_sort}) {proof_datatype} :internal-hidden)
411;; name of rule, one proof per fact in the query, proposition being proven t1 = t2
412(constructor {rule_constructor} (String {proof_list_sort} {ast_sort} {ast_sort}) {proof_datatype} :internal-hidden)
413
414;; merge function justification- name of function and two proofs for the two terms being merged,
415;; and the proposition being justified t = t
416(constructor {merge_fn_constructor} (String {proof_datatype} {proof_datatype} {ast_sort}) {proof_datatype} :internal-hidden)
417
418;; transitivity of equality proofs
419(constructor {eq_trans_constructor} ({proof_datatype} {proof_datatype}) {proof_datatype} :internal-hidden)
420
421;; symmetry of equality proofs
422(constructor  {eq_sym_constructor} ({proof_datatype}) {proof_datatype} :internal-hidden)
423;; given a proof that t1 = f(..., ci, ...)
424;; and the child index i of ci in the term f(..., ci, ...)
425;; and a proof that ci = c2,
426;; produces a justification that t1 = f(..., c2, ...)
427(constructor  {congr_constructor} ({proof_datatype} i64 {proof_datatype}) {proof_datatype} :internal-hidden)
428                "
429        )
430    }

433/// Reads a file and checks that its commands support the proof encoding.
434pub fn file_supports_proofs(path: &Path) -> bool {
435    let contents = match std::fs::read_to_string(path) {
436        Ok(contents) => contents,
437        Err(_) => return false,
438    };
439
440    let canonical = match std::fs::canonicalize(path) {
441        Ok(canonical) => canonical,
442        Err(_) => return false,
443    };
444
445    let mut egraph = EGraph::default();
446    let filename = canonical.to_string_lossy().into_owned();
447    let desugared = match egraph.desugar_program(Some(filename.clone()), &contents) {
448        Ok(commands) => commands,
449        Err(_) => return false,
450    };
451
452    program_supports_proofs(&desugared, &egraph.type_info)
453}

455/// Reasons why a command doesn't support proof encoding
456#[derive(Debug, Clone, thiserror::Error)]
457pub enum ProofEncodingUnsupportedReason {
458    #[error("primitive operation lacks a validator function")]
459    PrimitiveWithoutValidator,
460    #[error(
461        "action contains a function lookup. Finding the output of a function is only supported in queries."
462    )]
463    FunctionLookupInAction,
464    #[error(
465        "sort has a presort (custom sort container implementation). Custom sorts are not supported by proof encoding."
466    )]
467    SortWithPresort,
468    #[error(
469        "sort has a :uf annotation. The :uf annotation is used internally by term encoding and cannot be specified manually in proof mode."
470    )]
471    SortWithUfAnnotation,
472    #[error("user-defined commands are not supported.")]
473    UserDefinedCommand,
474    #[error("input commands are not supported.")]
475    InputCommand,
476    #[error("missing merge function. All functions need to specify a :merge function.")]
477    NoMergeOnNonGlobalFunction,
478    #[error(
479        "let binding with a primitive in the body. For silly internal reasons, we don't support primitive bindings for proofs at the moment, sorry."
480    )]
481    LetBindingWithNonEqSort,
482}

484/// Checks whether a desugared program supports proof encoding.
485pub fn program_supports_proofs(commands: &[ResolvedCommand], type_info: &TypeInfo) -> bool {
486    for command in commands {
487        if command_supports_proof_encoding(command, type_info).is_err() {
488            return false;
489        }
490    }
491    true
492}

494/// Recursively check if all primitives in an expression have validators
495fn expr_primitives_have_validators(expr: &ResolvedExpr) -> bool {
496    use crate::ast::GenericExpr;
497    use crate::core::ResolvedCall;
498
499    let mut all_valid = true;
500    expr.walk(
501        &mut |e| {
502            if let GenericExpr::Call(_, ResolvedCall::Primitive(prim), _) = e {
503                if prim.validator().is_none() {
504                    all_valid = false;
505                }
506            }
507        },
508        &mut |_| {},
509    );
510    all_valid
511}

513/// Check if an action contains non-global function lookups in any of its expressions
514fn action_has_function_lookup(action: &ResolvedAction, type_info: &TypeInfo) -> bool {
515    let mut has_lookup = false;
516    action.clone().visit_exprs(&mut |expr| {
517        if type_info.expr_has_function_lookup(&expr).is_some() {
518            has_lookup = true;
519        }
520        expr
521    });
522    has_lookup
523}

525/// Checks whether a resolved command supports proof encoding.
526/// Returns Ok(()) if supported, or Err with the reason if not.
527pub(crate) fn command_supports_proof_encoding(
528    command: &ResolvedCommand,
529    type_info: &TypeInfo,
530) -> Result<(), ProofEncodingUnsupportedReason> {
531    // Check all expressions for primitives without validators
532    let mut all_primitives_have_validators = true;
533    command.clone().visit_exprs(&mut |expr| {
534        if !expr_primitives_have_validators(&expr) {
535            all_primitives_have_validators = false;
536        }
537        expr
538    });
539
540    if !all_primitives_have_validators {
541        return Err(ProofEncodingUnsupportedReason::PrimitiveWithoutValidator);
542    }
543
544    // Check actions (not queries) for function lookups
545    // Egglog supports lookups in actions at the global level, but not in proofs mode
546    // (global function calls are allowed - they get desugared to constructors)
547    let mut has_function_lookup_in_action = false;
548    command.clone().visit_actions(&mut |action| {
549        has_function_lookup_in_action |= action_has_function_lookup(&action, type_info);
550        action
551    });
552
553    if has_function_lookup_in_action {
554        return Err(ProofEncodingUnsupportedReason::FunctionLookupInAction);
555    }
556
557    // Now check command-specific constraints
558    match command {
559        GenericCommand::Sort {
560            presort_and_args: Some(_),
561            ..
562        } => Err(ProofEncodingUnsupportedReason::SortWithPresort),
563        GenericCommand::Sort { uf: Some(_), .. } => {
564            Err(ProofEncodingUnsupportedReason::SortWithUfAnnotation)
565        }
566        GenericCommand::UserDefined(..) => Err(ProofEncodingUnsupportedReason::UserDefinedCommand),
567        GenericCommand::Input { .. } => Err(ProofEncodingUnsupportedReason::InputCommand),
568        // Extract commands can't have non-global function lookups
569        // because instrument_action_expr doesn't support them
570        // (global function calls are fine - they get desugared to constructors)
571        GenericCommand::Extract(_, expr, variants) => {
572            if type_info.expr_has_function_lookup(expr).is_some()
573                || type_info.expr_has_function_lookup(variants).is_some()
574            {
575                Err(ProofEncodingUnsupportedReason::FunctionLookupInAction)
576            } else {
577                Ok(())
578            }
579        }
580        // no-merge on a non-global function
581        // To add support: https://github.com/egraphs-good/egglog/issues/774
582        GenericCommand::Function {
583            merge: None, name, ..
584        } => {
585            if type_info.is_global(name) {
586                Ok(())
587            } else {
588                Err(ProofEncodingUnsupportedReason::NoMergeOnNonGlobalFunction)
589            }
590        }
591        // let binding with non-eq sort not supported by proof_global_desugar
592        ResolvedCommand::Action(ResolvedAction::Let(_, _, expr)) => {
593            // let binding with non-eq sort not supported by proof_global_desugar
594            // we detect as setting something that is no-merge to a primitive not supported (global primitive binding)
595            if expr.output_type().is_eq_sort() {
596                Ok(())
597            } else {
598                Err(ProofEncodingUnsupportedReason::LetBindingWithNonEqSort)
599            }
600        }
601        // After global desugar it may look like this
602        ResolvedCommand::Action(ResolvedAction::Set(_span, head, _children, expr)) => {
603            if !type_info.is_global(head.name()) || expr.output_type().is_eq_sort() {
604                Ok(())
605            } else {
606                Err(ProofEncodingUnsupportedReason::LetBindingWithNonEqSort)
607            }
608        }
609        _ => Ok(()),
610    }
