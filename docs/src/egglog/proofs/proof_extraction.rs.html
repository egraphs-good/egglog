1use crate::ast::FunctionSubtype;
2use crate::extract::{Extractor, TreeAdditiveCostModel};
3use crate::proofs::proof_encoding::ProofInstrumentor;
4use crate::proofs::proof_format::{Justification, ProofId, ProofStore, proof_store_from_term};
5use crate::{ResolvedCall, TermDag};
6use thiserror::Error;
7
8#[derive(Debug, Error)]
9pub enum ProveExistsError {
10    #[error("prove-exists requires a constructor")]
11    RequiresConstructor,
12    #[error("prove-exists does not support primitives")]
13    PrimitivesUnsupported,
14    #[error("prove requires proofs mode")]
15    ProofsDisabled,
16    #[error("Could not find a proof due to query not matching (constructor {constructor}).")]
17    QueryDidNotMatch { constructor: String },
18}
19
20impl ProofInstrumentor<'_> {
21    /// Prove the existence of a constructor or fail if a proof cannot be found.
22    /// We use a constructor because inserting a value at the top level would give a trivial proof.
23    pub(crate) fn prove_exists(
24        &mut self,
25        call: &ResolvedCall,
26    ) -> Result<(ProofStore, ProofId), ProveExistsError> {
27        let func = match call {
28            ResolvedCall::Func(func) if func.subtype == FunctionSubtype::Constructor => func,
29            ResolvedCall::Func(_) => {
30                return Err(ProveExistsError::RequiresConstructor);
31            }
32            ResolvedCall::Primitive(_) => {
33                return Err(ProveExistsError::PrimitivesUnsupported);
34            }
35        };
36
37        let function = self
38            .egraph
39            .functions
40            .get(&func.name)
41            .unwrap_or_else(|| panic!("constructor {} is not declared", func.name));
42
43        if !self.egraph.proof_state.proofs_enabled {
44            return Err(ProveExistsError::ProofsDisabled);
45        }
46
47        let backend_id = function.backend_id;
48        let output_sort = function.schema.output.clone();
49
50        // Use the version that ignores unextractable flag since proof extraction
51        // needs to extract proofs from all terms including those marked unextractable
52        let extractor = Extractor::compute_costs_from_rootsorts_allow_unextractable(
53            None,
54            self.egraph,
55            TreeAdditiveCostModel::default(),
56        );
57
58        let mut termdag = TermDag::default();
59        let mut witness_value = None;
60
61        self.egraph.backend.for_each_while(backend_id, |row| {
62            let value = *row
63                .vals
64                .last()
65                .expect("constructor rows include their output value");
66            witness_value = Some(value);
67            false
68        });
69
70        let witness_value = witness_value.ok_or_else(|| ProveExistsError::QueryDidNotMatch {
71            constructor: func.name.clone(),
72        })?;
73
74        let proof_function_name = self.term_proof_name(output_sort.name());
75        let proof_value = self
76            .egraph
77            .lookup_function(&proof_function_name, &[witness_value])
78            .unwrap_or_else(|| panic!("no proof recorded for constructor {}", func.name));
79
80        let proof_sort = self
81            .egraph
82            .functions
83            .get(&proof_function_name)
84            .unwrap_or_else(|| {
85                panic!(
86                    "proof table {proof_function_name} for constructor {} was not declared",
87                    func.name
88                )
89            })
90            .schema
91            .output
92            .clone();
93
94        let (_, proof_term_id) = extractor
95            .extract_best_with_sort(self.egraph, &mut termdag, proof_value, proof_sort)
96            .unwrap_or_else(|| {
97                panic!("failed to extract proof term for constructor {}", func.name)
98            });
99
100        let (mut proof_store, proof_id) = proof_store_from_term(
101            &self.egraph.proof_state.proof_names,
102            termdag,
103            proof_term_id,
104            &self.egraph.desugared_commands,
105        );
106
107        // Remove globals from the proof
108        if let Result::Err(e) = proof_store.remove_globals(&self.egraph.desugared_commands) {
109            panic!("Failed to remove globals from proof: {e}");
110        }
111
112        // if the existence proof has a single premise, extract that premise proof
113        let proof = proof_store.get(proof_id);
114        let extra_rule_removed = match proof.justification() {
115            Justification::Rule { premise_proofs, .. } => match premise_proofs.as_slice() {
116                [premise_proof_id] => *premise_proof_id,
117                _ => proof_id,
118            },
119            _ => panic!("expected rule justification for existence proof"),
120        };
121
122        // Check the proof before simplification
123        if let Result::Err(e) =
124            proof_store.check_proof(extra_rule_removed, &self.egraph.desugared_commands)
125        {
126            panic!("Existence proof should be valid before simplification: {e}");
127        }
128
129        // simplify the proof
130        let simplified_proof = proof_store.simplify(extra_rule_removed);
131
132        // Check the proof after simplification
133        proof_store
134            .check_proof(simplified_proof, &self.egraph.desugared_commands)
135            .expect("simplified existence proof should still be valid");
136
137        Ok((proof_store, extra_rule_removed))
138    }
139}
