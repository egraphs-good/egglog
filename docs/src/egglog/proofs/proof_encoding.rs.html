52    /// Mark two things as equal, adding proof if proofs are enabled.
53    pub(crate) fn union(
54        &mut self,
55        type_name: &str,
56        lhs: &str,
57        rhs: &str,
58        justification: &Justification,
59    ) -> String {
60        let uf_name = self.uf_name(type_name);
61        let smaller = format!("(ordering-min {lhs} {rhs})");
62        let larger = format!("(ordering-max {lhs} {rhs})");
63        let set_proof = if self.egraph.proof_state.proofs_enabled {
64            let uf_proof_name = self.uf_proof_name(type_name);
65            let to_ast_constructor = self
66                .proof_names()
67                .sort_to_ast_constructor
68                .get(type_name)
69                .unwrap();
70            let rule_constructor = &self.proof_names().rule_constructor;
71            let fiat_constructor = &self.proof_names().fiat_constructor;
72            let proof = match justification {
73                Justification::Rule(rule_name, proof_list) => format!(
74                    "({rule_constructor} \"{rule_name}\" {proof_list} ({to_ast_constructor} {larger}) ({to_ast_constructor} {smaller}))"
75                ),
76                Justification::Fiat => format!(
77                    "({fiat_constructor} ({to_ast_constructor} {larger}) ({to_ast_constructor} {smaller}))"
78                ),
79                Justification::Merge(_func_name, _proof1, _proof2) => panic!(
80                    "Merge functions do not include union actions, so proof should not be by merge"
81                ),
82                Justification::Proof(existing_proof) => existing_proof.clone(),
83            };
84            format!("(set ({uf_proof_name} {larger} {smaller}) {proof})")
85        } else {
86            "".to_string()
87        };
88
89        format!(
90            "
91        ({uf_name} {larger} {smaller})
92         {set_proof}",
93        )
94    }

96    /// The parent table is the database representation of a union-find datastructure.
97    /// When one term has two parents, those parents are unioned in the merge action.
98    /// Also, we have a rule that maintains the invariant that each term points to its
99    /// canonical representative.
100    fn declare_sort(&mut self, sort_name: &str) -> Vec<Command> {
101        let pname = self.uf_name(sort_name);
102        let uf_function_name = self.uf_function_name(sort_name);
103        let fresh_sort = self.egraph.parser.symbol_gen.fresh("uf");
104        let fresh_name = self.egraph.parser.symbol_gen.fresh("uf_update");
105        let uf_function_index_name = self.egraph.parser.symbol_gen.fresh("uf_function_index");
106        let proof_tables = if self.egraph.proof_state.proofs_enabled {
107            let term_proof_name = self.term_proof_name(sort_name);
108            let proof_type = self.proof_names().proof_datatype.clone();
109            let uf_proof_name = self.uf_proof_name(sort_name);
110            format!(
111                "
112                (function {term_proof_name} ({sort_name}) {proof_type} :merge old :internal-hidden)
113                (function {uf_proof_name} ({sort_name} {sort_name}) {proof_type} :merge old :internal-hidden)
114                 "
115            )
116        } else {
117            "".to_string()
118        };
119
120        let (proof_query1, proof_action1, to_ast_constructor_code, proof_query2, proof_action2) =
121            if self.egraph.proof_state.proofs_enabled {
122                let p1_fresh = self.egraph.parser.symbol_gen.fresh("p1");
123                let p2_fresh = self.egraph.parser.symbol_gen.fresh("p2");
124                assert!(
125                    self.proof_names()
126                        .sort_to_ast_constructor
127                        .get(sort_name)
128                        .is_none()
129                );
130
131                let code = self.add_to_ast(sort_name);
132                let uf_proof_name = self.uf_proof_name(sort_name);
133                let trans_constructor = &self.proof_names().eq_trans_constructor;
134                let sym_constructor = &self.proof_names().eq_sym_constructor;
135
136                (
137                    format!(
138                        "(= {p1_fresh} ({uf_proof_name} a b))
139                         (= {p2_fresh} ({uf_proof_name} b c))"
140                    ),
141                    format!(
142                        "(set ({uf_proof_name} a c)
143                              ({trans_constructor} {p1_fresh} {p2_fresh}))"
144                    ),
145                    code,
146                    format!(
147                        "(= {p1_fresh} ({uf_proof_name} a b))
148                         (= {p2_fresh} ({uf_proof_name} a c))"
149                    ),
150                    format!(
151                        "(set ({uf_proof_name} b c)
152                          ({trans_constructor}
153                             ({sym_constructor} {p1_fresh})
154                             {p2_fresh}))"
155                    ),
156                )
157            } else {
158                (
159                    "".to_string(),
160                    "".to_string(),
161                    "".to_string(),
162                    "".to_string(),
163                    "".to_string(),
164                )
165            };
166
167        let path_compress_ruleset_name = self.proof_names().path_compress_ruleset_name.clone();
168        let single_parent_ruleset_name = self.proof_names().single_parent_ruleset_name.clone();
169        let uf_function_index_ruleset_name =
170            self.proof_names().uf_function_index_ruleset_name.clone();
171
172        self.parse_program(&format!(
173            "(sort {fresh_sort})
174             (constructor {pname} ({sort_name} {sort_name}) {fresh_sort} :internal-hidden)
175             (function {uf_function_name} ({sort_name}) {sort_name} :merge new :internal-hidden)
176             {to_ast_constructor_code}
177             {proof_tables}
178             ;; performs path compression, ensuring each term points to the representative
179             (rule (({pname} a b)
180                    ({pname} b c)
181                    (!= b c)
182                    {proof_query1})
183                  ((delete ({pname} a b))
184                   ({pname} a c)
185                   {proof_action1})
186                   :ruleset {path_compress_ruleset_name}
187                   :name \"{fresh_name}\")
188             ;; ensures each term has only one parent
189             (rule (({pname} a b)
190                    ({pname} a c)
191                    (!= b c)
192                    (= (ordering-max b c) b)
193                    {proof_query2})
194                  ((delete ({pname} a b))
195                   ({pname} b c)
196                    {proof_action2})
197                   :ruleset {single_parent_ruleset_name}
198                   :name \"singleparent{fresh_name}\")
199             ;; mirrors UF rows into a function-backed UF index for faster rebuild lookups
200             (rule (({pname} a b))
201                   ((set ({uf_function_name} a) b))
202                   :ruleset {uf_function_index_ruleset_name}
203                   :name \"{uf_function_index_name}\")
204                   ",
205        ))
206    }

208    /// Rules that execute deletion and subsumption based on the tables requesting the deletion/subsumption.
209    fn delete_and_subsume(&mut self, fdecl: &ResolvedFunctionDecl) -> String {
210        let child_names = fdecl
211            .schema
212            .input
213            .iter()
214            .enumerate()
215            .map(|(i, _)| format!("c{i}_"))
216            .collect::<Vec<_>>()
217            .join(" ");
218        let to_delete_name = self.delete_name(&fdecl.name);
219        let subsumed_name = self.subsumed_name(&fdecl.name);
220        let view_name = self.view_name(&fdecl.name);
221        let delete_subsume_ruleset = self.proof_names().delete_subsume_ruleset_name.clone();
222        let fresh_name = self.egraph.parser.symbol_gen.fresh("delete_rule");
223
224        // Subsume could use delete, except that `check` ignores subsumption.
225        format!(
226            "(rule (({to_delete_name} {child_names})
227                    ({view_name} {child_names} out))
228                   ((delete ({view_name} {child_names} out))
229                    (delete ({to_delete_name} {child_names})))
230                    :ruleset {delete_subsume_ruleset}
231                    :name \"{fresh_name}\")
232             (rule (({subsumed_name} {child_names})
233                    ({view_name} {child_names} out))
234                   ((subsume ({view_name} {child_names} out)))
235                    :ruleset {delete_subsume_ruleset}
236                    :name \"{fresh_name}_subsume\")"
237        )
238    }

240    /// Generate rules that run a merge function for a custom function.
241    /// One rule runs the merge function when two different values are present for the same children.
242    /// Another rule cleans up old values, necessary because the newly merged value may be equal to one of the old values.
243    fn handle_merge_fn(
244        &mut self,
245        fdecl: &ResolvedFunctionDecl,
246        child_names: &[String],
247        child_names_str: &str,
248        view_name: &str,
249        rebuilding_ruleset: &str,
250    ) -> String {
251        let name = &fdecl.name;
252
253        let merge_fn = &fdecl
254            .merge
255            .as_ref()
256            .unwrap_or_else(|| panic!("Proofs don't support :no-merge"));
257
258        let fresh_name = self.egraph.parser.symbol_gen.fresh("merge_rule");
259        let cleanup_name = self.egraph.parser.symbol_gen.fresh("merge_cleanup");
260
261        let p1_fresh = self.egraph.parser.symbol_gen.fresh("p1");
262        let p2_fresh = self.egraph.parser.symbol_gen.fresh("p2");
263        let view_proof_name = self.view_proof_name(&fdecl.name);
264        let rebuilding_cleanup_ruleset = self.proof_names().rebuilding_cleanup_ruleset_name.clone();
265        let proof_query = if self.egraph.proof_state.proofs_enabled {
266            format!(
267                "(= {p1_fresh} ({view_proof_name} {child_names_str} old))
268                     (= {p2_fresh} ({view_proof_name} {child_names_str} new))
269                    "
270            )
271        } else {
272            "".to_string()
273        };
274        let proof_var = self.fresh_var();
275        let mut merge_fn_code = vec![];
276        let merge_fn_var = self.instrument_action_expr(
277            merge_fn,
278            &mut merge_fn_code,
279            &Justification::Merge(name.clone(), p1_fresh.clone(), p2_fresh.clone()),
280        );
281        let merge_fn_code_str = merge_fn_code.join("\n");
282        let mut updated = child_names.to_vec();
283        updated.push(merge_fn_var.clone());
284        let term = format!("({name} {child_names_str} {merge_fn_var})");
285
286        let rule_proof = if self.egraph.proof_state.proofs_enabled {
287            let to_ast = self.fname_to_ast_name(name);
288            let merge_fn_constructor = self.proof_names().merge_fn_constructor.clone();
289            format!(
290                "(let {proof_var}
291                            ({merge_fn_constructor} \"{name}\"
292                                  {p1_fresh}
293                                  {p2_fresh}
294                                  ({to_ast} {term})))"
295            )
296        } else {
297            "".to_string()
298        };
299        let term_and_proof = self.update_view(name, &updated, &proof_var);
300        let cleanup_constructor = self.egraph.parser.symbol_gen.fresh("mergecleanup");
301        let fresh_sort = self.egraph.parser.symbol_gen.fresh("mergecleanupsort");
302        let output_sort = fdecl.schema.output.clone();
303
304        // The first runs the merge function adding a new row.
305        // The second deletes rows with old values for the old variable, while the third deletes rows with new values for the new variable.
306        format!(
307            "(sort {fresh_sort})
308                 (constructor {cleanup_constructor} ({output_sort} {output_sort}) {fresh_sort} :internal-hidden)
309                 (rule (({view_name} {child_names_str} old)
310                        ({view_name} {child_names_str} new)
311                        (!= old new)
312                        (= (ordering-max old new) new)
313                        {proof_query})
314                       (
315                        {merge_fn_code_str}
316                        {rule_proof}
317                        {term_and_proof}
318                        ({cleanup_constructor} {merge_fn_var} old)
319                        ({cleanup_constructor} {merge_fn_var} new)
320                       )
321                        :ruleset {rebuilding_ruleset}
322                        :name \"{fresh_name}\")
323                
324                 (rule (({cleanup_constructor} merged old)
325                        ({view_name} {child_names_str} merged)
326                        ({view_name} {child_names_str} old)
327                        (!= merged old))
328                       ((delete ({view_name} {child_names_str} old)))
329                        :ruleset {rebuilding_cleanup_ruleset}
330                        :name \"{cleanup_name}\")
331                ",
332        )
333    }

335    /// Generate a rule that handles congruence for constructors.
336    /// When two different values are present for the same children,
337    /// we union those two values together.
338    fn handle_congruence(
339        &mut self,
340        fdecl: &ResolvedFunctionDecl,
341        child_names: &[String],
342        rebuilding_ruleset: &str,
343    ) -> String {
344        // Congruence rule
345        let fresh_name = self.egraph.parser.symbol_gen.fresh("congruence_rule");
346        let mut child_names_new = child_names.to_vec();
347        child_names_new.push("new".to_string());
348        let mut child_names_old = child_names.to_vec();
349        child_names_old.push("old".to_string());
350        let (query1, prf1) = self.query_view_and_get_proof(&fdecl.name, &child_names_new);
351        let (query2, prf2) = self.query_view_and_get_proof(&fdecl.name, &child_names_old);
352        let sym = &self.proof_names().eq_sym_constructor;
353        let trans = &self.proof_names().eq_trans_constructor;
354
355        // Proof is by transitivity. A view proof gives a proof that
356        // representative r_1 = f(c_1,...,c_n).
357        // We also have a proof that other eclass representative r_2 = f(c_1,...,c_n), the same term.
358        // We want a proof that r1 = r2, which we get by transitivity.
359        let union_code = self.union(
360            &fdecl.schema.output,
361            "new",
362            "old",
363            &Justification::Proof(format!("({trans} {prf1} ({sym} {prf2}))",)),
364        );
365        format!(
366            "(rule ({query1}
367                        {query2}
368                        (!= old new)
369                        (= (ordering-max old new) new))
370                       ({union_code})
371                        :ruleset {rebuilding_ruleset}
372                        :name \"{fresh_name}\")"
373        )
374    }

376    /// Generate rules that handle merge functions or congruence.
377    /// For custom functions, we generate rules that run the merge function.
378    /// For constructors, we generate congruence rules.
379    fn handle_merge_or_congruence(&mut self, fdecl: &ResolvedFunctionDecl) -> String {
380        let child_names = fdecl
381            .schema
382            .input
383            .iter()
384            .enumerate()
385            .map(|(i, _)| format!("c{i}_"))
386            .collect::<Vec<_>>();
387        let child_names_str = child_names.join(" ");
388        let rebuilding_ruleset = self.proof_names().rebuilding_ruleset_name.clone();
389        let view_name = self.view_name(&fdecl.name);
390        if fdecl.subtype == FunctionSubtype::Custom {
391            self.handle_merge_fn(
392                fdecl,
393                &child_names,
394                &child_names_str,
395                &view_name,
396                &rebuilding_ruleset,
397            )
398        } else {
399            self.handle_congruence(fdecl, &child_names, &rebuilding_ruleset)
400        }
401    }

403    /// Each function/constructor gets a term table and a view table.
404    /// The term table stores underlying representative terms.
405    /// The view table stores child terms and their eclass.
406    /// The view table is mutated using delete, but we never delete from term tables.
407    /// We re-use the original name of the function for the term table.
408    fn term_and_view(&mut self, fdecl: &ResolvedFunctionDecl) -> Vec<Command> {
409        let schema = &fdecl.schema;
410        let out_type = schema.output.clone();
411
412        let name = &fdecl.name;
413        let view_name = self.view_name(&fdecl.name);
414        let in_sorts = ListDisplay(schema.input.clone(), " ");
415        let fresh_sort = self.egraph.parser.symbol_gen.fresh("view");
416        let delete_rule = self.delete_and_subsume(fdecl);
417        let to_delete_name = self.delete_name(&fdecl.name);
418        let subsumed_name = self.subsumed_name(&fdecl.name);
419        let term_sorts = format!(
420            "{in_sorts} {}",
421            if fdecl.subtype == FunctionSubtype::Constructor {
422                "".to_string()
423            } else {
424                schema.output.to_string()
425            }
426        );
427        let view_sorts = format!("{in_sorts} {out_type}");
428        let proof_constructors = self.proof_functions(fdecl, &view_sorts);
429
430        let view_sort = if fdecl.subtype == FunctionSubtype::Constructor {
431            schema.output.clone()
432        } else {
433            fresh_sort.clone()
434        };
435        let to_ast_view_sort = self.add_to_ast(&view_sort);
436
437        if self.egraph.proof_state.proofs_enabled {
438            self.egraph
439                .proof_state
440                .proof_names
441                .fn_to_term_sort
442                .insert(name.clone(), view_sort.clone());
443        }
444        let merge_rule = self.handle_merge_or_congruence(fdecl);
445        // the term table has child_sorts as inputs
446        // the view table has child_sorts + the leader term for the eclass
447        // Propagate cost, unextractable, hidden, and internal_let flags from the original function
448        let mut term_flags = String::new();
449        let mut view_flags = String::new();
450        if let Some(cost) = fdecl.cost {
451            term_flags.push_str(&format!(" :cost {cost}"));
452            view_flags.push_str(&format!(" :cost {cost}"));
453        }
454        if fdecl.unextractable {
455            view_flags.push_str(" :unextractable");
456        }
457        if fdecl.internal_hidden {
458            view_flags.push_str(" :internal-hidden");
459        }
460        if fdecl.internal_let {
461            view_flags.push_str(" :internal-let");
462        }
463        self.parse_program(&format!(
464            "
465            (sort {fresh_sort})
466            {to_ast_view_sort}
467            (constructor {name} ({term_sorts}) {view_sort}{term_flags} :internal-hidden :unextractable)
468            (constructor {view_name} ({view_sorts}) {fresh_sort} :term-constructor {name}{view_flags})
469            (constructor {to_delete_name} ({in_sorts}) {fresh_sort} :internal-hidden)
470            (constructor {subsumed_name} ({in_sorts}) {fresh_sort} :internal-hidden)
471            {proof_constructors}
472            {merge_rule}
473            {delete_rule}",
474        ))
475    }

477    fn proof_functions(&mut self, fdecl: &ResolvedFunctionDecl, view_sorts: &str) -> String {
478        if !self.egraph.proof_state.proofs_enabled {
479            return "".to_string();
480        }
481
482        let view_proof_name = self.view_proof_name(&fdecl.name);
483        let proof_type = self.proof_names().proof_datatype.clone();
484
485        // A view proof gives a proof that the representative term t_r equals the term in the view.
486        // Example: (AddView 2 3 t_r) proves the proposition that t_r = Add(2, 3) in that direction.
487        // This direction makes proof instrumentation use fewer symmetries, since the goal is to
488        // match the right-hand side of the equality to the concrete syntax
489        // of the original rule.
490        format!(
491            "
492            (function {view_proof_name} ({view_sorts}) {proof_type} :merge old :internal-hidden)
493            "
494        )
495    }

497    /// Rules that update the views when children change.
498    fn rebuilding_rules(&mut self, fdecl: &ResolvedFunctionDecl) -> Vec<Command> {
499        let types = fdecl.resolved_schema.view_types();
500
501        // Check if there are any eq-sort columns at all; if not, no rebuild rule needed.
502        if !types.iter().any(|t| t.is_eq_sort()) {
503            return vec![];
504        }
505
506        let view_name = self.view_name(&fdecl.name);
507        let child = |i: usize| format!("c{i}_");
508        let children_vec: Vec<String> = (0..types.len()).map(child).collect();
509        let children = format!("{}", ListDisplay(&children_vec, " "));
510
511        // For each eq-sort column, look up its leader via the UF table.
512        // For non-eq-sort columns, the leader is the same as the original.
513        let mut uf_queries = vec![];
514        let mut leader_vars: Vec<String> = vec![];
515        let mut bool_neq_exprs = vec![];
516        let mut uf_proof_vars: Vec<Option<String>> = vec![];
517
518        for (i, ty) in types.iter().enumerate() {
519            if ty.is_eq_sort() {
520                let leader_var = format!("c{i}_leader_");
521                let uf_function_name = self.uf_function_name(ty.name());
522                let ci = child(i);
523
524                if self.egraph.proof_state.proofs_enabled {
525                    let uf_proof = self.uf_proof_name(ty.name());
526                    let proof_var = self.fresh_var();
527                    uf_queries.push(format!(
528                        "(= {leader_var} ({uf_function_name} {ci}))
529                         (= {proof_var} ({uf_proof} {ci} {leader_var}))"
530                    ));
531                    uf_proof_vars.push(Some(proof_var));
532                } else {
533                    uf_queries.push(format!("(= {leader_var} ({uf_function_name} {ci}))"));
534                    uf_proof_vars.push(None);
535                }
536
537                bool_neq_exprs.push(format!("(bool-!= {ci} {leader_var})"));
538                leader_vars.push(leader_var);
539            } else {
540                leader_vars.push(child(i));
541                uf_proof_vars.push(None);
542            }
543        }
544
545        let uf_query_str = uf_queries.join("\n       ");
546        let or_expr = format!("(or {})", bool_neq_exprs.join("\n             "));
547        let filter_query = format!("(guard {or_expr})");
548
549        // Build the updated children: use leader_var for eq-sort columns, original for others.
550        let children_updated: Vec<String> = leader_vars.clone();
551
552        let fresh_name = self.egraph.parser.symbol_gen.fresh("rebuild_rule");
553        let (query_view, view_prf) = self.query_view_and_get_proof(&fdecl.name, &children_vec);
554
555        // Build proof code if proofs are enabled.
556        // We chain congruence proofs for each updated child and a transitivity proof
557        // for the representative (last column) update.
558        let (pf_code, pf_var) = if self.egraph.proof_state.proofs_enabled {
559            let eq_trans_constructor = self.proof_names().eq_trans_constructor.clone();
560            let congr_constructor = self.proof_names().congr_constructor.clone();
561            let sym_constructor = self.proof_names().eq_sym_constructor.clone();
562
563            // Start with the view proof and apply congruence for each eq-sort child
564            // (excluding the last column if this is a constructor, since that's the representative).
565            let mut current_proof = view_prf.clone();
566            let mut proof_code_parts = vec![];
567
568            for (i, ty) in types.iter().enumerate() {
569                if !ty.is_eq_sort() {
570                    continue;
571                }
572
573                let uf_prf = uf_proof_vars[i].as_ref().unwrap();
574
575                if fdecl.subtype == FunctionSubtype::Constructor && i == types.len() - 1 {
576                    // Updating the representative term (last column of constructor):
577                    // use transitivity with sym of the UF proof
578                    let new_proof = self.fresh_var();
579                    proof_code_parts.push(format!(
580                        "(let {new_proof}
581                           ({eq_trans_constructor}
582                              ({sym_constructor} {uf_prf})
583                              {current_proof}))",
584                    ));
585                    current_proof = new_proof;
586                } else {
587                    // Updating a child via congruence
588                    let new_proof = self.fresh_var();
589                    proof_code_parts.push(format!(
590                        "(let {new_proof}
591                              ({congr_constructor} {current_proof} {i}
592                                                   {uf_prf}))",
593                    ));
594                    current_proof = new_proof;
595                }
596            }
597
598            (proof_code_parts.join("\n"), current_proof)
599        } else {
600            ("".to_string(), "".to_string())
601        };
602
603        let updated_view = self.update_view(&fdecl.name, &children_updated, &pf_var);
604
605        // Make a single rule that updates the view when any child's leader differs.
606        let rule = format!(
607            "(rule ({query_view}
608                    {uf_query_str}
609                    {filter_query}
610                    )
611                 (
612                  {pf_code}
613                  {updated_view}
614                  (delete ({view_name} {children}))
615                 )
616                  :ruleset {} :name \"{fresh_name}\")",
617            self.proof_names().rebuilding_ruleset_name
618        );
619        self.parse_program(&rule)
620    }

622    /// Instrument fact replaces terms with looking up
623    /// canonical versions in the view.
624    /// It also needs to look up references to globals.
625    /// Adds the instrumented fact to `res` and returns a proof that the fact matched.
626    fn instrument_fact(&mut self, fact: &ResolvedFact, res: &mut Vec<String>) -> String {
627        match fact {
628            // In proof normal form, this is the only way that function calls apppear.
629            ResolvedFact::Eq(
630                _span,
631                ResolvedExpr::Call(
632                    _span2,
633                    head @ ResolvedCall::Func(FuncType {
634                        subtype: FunctionSubtype::Custom,
635                        ..
636                    }),
637                    args,
638                ),
639                // TODO this could actually be arbitrary pretty easily, it's just nested functions that are hard.
640                ResolvedExpr::Var(_span3, v),
641            ) => {
642                let mut new_args = vec![];
643                let mut arg_proofs = vec![];
644                for arg in args {
645                    let (var, proof) = self.instrument_fact_expr(arg, res);
646                    new_args.push(var);
647                    arg_proofs.push(proof);
648                }
649                new_args.push(v.to_string());
650
651                let view_name = self.view_name(head.name());
652                let args_str = ListDisplay(new_args, " ");
653                res.push(format!("({view_name} {args_str})",));
654
655                if self.egraph.proof_state.proofs_enabled {
656                    let view_proof_name = self.view_proof_name(head.name());
657                    let proof_var = self.fresh_var();
658                    res.push(format!("(= {proof_var} ({view_proof_name} {args_str}))"));
659                    let mut proof = proof_var;
660                    for (i, arg_proof) in arg_proofs.into_iter().enumerate() {
661                        let congr = &self.proof_names().congr_constructor;
662                        // add a congruence from the argument (representative) to the term
663                        proof = format!(
664                            "
665                            ({congr} {proof} {i} {arg_proof})
666                            "
667                        );
668                    }
669
670                    proof
671                } else {
672                    "".to_string()
673                }
674            }
675            ResolvedFact::Eq(_span, left_expr, right_expr) => {
676                let (v1, p1) = self.instrument_fact_expr(left_expr, res);
677                let (v2, p2) = self.instrument_fact_expr(right_expr, res);
678                res.push(format!("(= {v1} {v2})"));
679                let sym = &self.proof_names().eq_sym_constructor;
680                let trans = &self.proof_names().eq_trans_constructor;
681
682                format!("({trans} ({sym} {p1}) {p2})",)
683            }
684            ResolvedFact::Fact(generic_expr) => {
685                let (_, proof) = self.instrument_fact_expr(generic_expr, res);
686                proof
687            }
688        }
689    }

691    /// Instruments a fact expression to use the view tables.
692    /// Assumes there are no function lookups in the term.
693    /// Returns a variable representing the expression and a proof that the expression was matched.
694    /// Proves a ground equality t1 = t2 where t1 is the eclass representative and t2 matches `expr` syntactically.
695    fn instrument_fact_expr(
696        &mut self,
697        expr: &ResolvedExpr,
698        res: &mut Vec<String>,
699    ) -> (String, String) {
700        match expr {
701            ResolvedExpr::Lit(_, lit) => {
702                let fiat_constructor = &self.proof_names().fiat_constructor;
703                let lit_sort = literal_sort(lit);
704                let proof_code = if self.egraph.proof_state.proofs_enabled {
705                    let to_ast = self
706                        .proof_names()
707                        .sort_to_ast_constructor
708                        .get(lit_sort.name())
709                        .unwrap();
710                    format!("({fiat_constructor} ({to_ast} {lit}) ({to_ast} {lit}))")
711                } else {
712                    "".to_string()
713                };
714
715                (format!("{lit}"), proof_code)
716            }
717            ResolvedExpr::Var(_, resolved_var) => {
718                let var = &resolved_var.name;
719                (
720                    resolved_var.name.clone(),
721                    if !self.egraph.proof_state.proofs_enabled {
722                        "".to_string()
723                    } else if resolved_var.sort.is_eq_sort() {
724                        let term_proof_name = self.term_proof_name(resolved_var.sort.name());
725                        let fresh_proof = self.fresh_var();
726                        res.push(format!("(= {fresh_proof} ({term_proof_name} {var}))"));
727                        fresh_proof
728                    } else {
729                        let fiat_constructor = &self.proof_names().fiat_constructor;
730                        let lit_sort = resolved_var.sort.name();
731                        let to_ast = self
732                            .proof_names()
733                            .sort_to_ast_constructor
734                            .get(lit_sort)
735                            .unwrap();
736                        format!("({fiat_constructor} ({to_ast} {var}) ({to_ast} {var}))")
737                    },
738                )
739            }
740            ResolvedExpr::Call(_, resolved_call, args) => {
741                let mut new_args = vec![];
742                // Variables and constants don't need subproofs, but constructor calls do.
743                let mut arg_proofs: Vec<Option<String>> = vec![];
744                for arg in args {
745                    if matches!(arg, ResolvedExpr::Var(_, _) | ResolvedExpr::Lit(_, _)) {
746                        new_args.push(arg.to_string());
747                        arg_proofs.push(None);
748                    } else {
749                        let (arg_str, proof) = self.instrument_fact_expr(arg, res);
750                        new_args.push(arg_str);
751                        arg_proofs.push(Some(proof));
752                    }
753                }
754                match resolved_call {
755                    ResolvedCall::Func(func_type) => {
756                        assert!(
757                            func_type.subtype == FunctionSubtype::Constructor,
758                            "Only constructor function calls are allowed in fact expressions due to proof normal form. Got {func_type:?}",
759                        );
760
761                        let fv = self.fresh_var();
762                        let view_name = self.view_name(&func_type.name);
763                        let args_str = ListDisplay(new_args, " ");
764                        res.push(format!("({view_name} {args_str} {fv})",));
765
766                        let proof = if self.proofs_enabled() {
767                            let view_proof_var = self.fresh_var();
768                            let view_proof_name = self.view_proof_name(&func_type.name);
769                            res.push(format!(
770                                "(= {view_proof_var} ({view_proof_name} {args_str} {fv}))"
771                            ));
772                            let mut proof = view_proof_var;
773                            for (i, arg_proof) in arg_proofs.into_iter().enumerate() {
774                                if let Some(arg_proof) = arg_proof {
775                                    let congr = &self.proof_names().congr_constructor;
776                                    // add a congruence from the argument (representative) to the term
777                                    proof = format!(
778                                        "
779                            ({congr} {proof} {i} {arg_proof})
780                            "
781                                    );
782                                }
783                            }
784                            proof
785                        } else {
786                            "".to_string()
787                        };
788                        (fv, proof)
789                    }
790                    ResolvedCall::Primitive(specialized_primitive) => {
791                        if specialized_primitive.output().is_eq_sort() {
792                            panic!(
793                                "Term encoding does not support eq-sort primitive expressions in facts"
794                            );
795                        }
796                        let fv = self.fresh_var();
797                        res.push(format!(
798                            "(= {fv} ({} {}))",
799                            specialized_primitive.name(),
800                            ListDisplay(new_args, " ")
801                        ));
802
803                        let proof = if self.proofs_enabled() {
804                            let fiat_constructor = &self.proof_names().fiat_constructor;
805                            let to_ast = self
806                                .proof_names()
807                                .sort_to_ast_constructor
808                                .get(specialized_primitive.output().name())
809                                .unwrap();
810                            format!("({fiat_constructor} ({to_ast} {fv}) ({to_ast} {fv}))")
811                        } else {
812                            "".to_string()
813                        };
814
815                        (fv.clone(), proof)
816                    }
817                }
818            }
819        }
820    }

835    // Actions need to be instrumented to add to the view
836    // as well as to the terms tables.
837    fn instrument_action(
838        &mut self,
839        action: &ResolvedAction,
840        justification: &Justification,
841    ) -> Vec<String> {
842        let mut res = vec![];
843
844        match action {
845            ResolvedAction::Let(_span, v, generic_expr) => {
846                let v2 = self.instrument_action_expr(generic_expr, &mut res, justification);
847                res.push(format!("(let {} {})", v.name, v2));
848            }
849            ResolvedAction::Set(_span, h, generic_exprs, generic_expr) => {
850                let mut exprs = vec![];
851                for e in generic_exprs.iter().chain(std::iter::once(generic_expr)) {
852                    exprs.push(self.instrument_action_expr(e, &mut res, justification));
853                }
854
855                let ResolvedCall::Func(func_type) = h else {
856                    panic!(
857                        "Set action on non-function, should have been prevented by typechecking"
858                    );
859                };
860
861                let (add_code, _fv) = self.add_term_and_view(func_type, &exprs, justification);
862                res.extend(add_code);
863            }
864            ResolvedAction::Change(_span, change, h, generic_exprs) => {
865                if let ResolvedCall::Func(func_type) = h {
866                    let symbol = match change {
867                        Change::Delete => self.delete_name(&func_type.name),
868                        Change::Subsume => self.subsumed_name(&func_type.name),
869                    };
870                    let children = generic_exprs
871                        .iter()
872                        .map(|e| self.instrument_action_expr(e, &mut res, justification))
873                        .collect::<Vec<_>>();
874
875                    res.push(format!("({symbol} {})", ListDisplay(children, " ")));
876                } else {
877                    panic!(
878                        "Delete action on non-function, should have been prevented by typechecking"
879                    );
880                }
881            }
882            ResolvedAction::Union(_span, generic_expr, generic_expr1) => {
883                let v1 = self.instrument_action_expr(generic_expr, &mut res, justification);
884                let v2 = self.instrument_action_expr(generic_expr1, &mut res, justification);
885                let ot = generic_expr.output_type();
886                let type_name = ot.name();
887                let unioned = self.union(type_name, &v1, &v2, justification);
888                res.push(unioned);
889            }
890            ResolvedAction::Panic(..) => {
891                res.push(format!("{action}"));
892            }
893            ResolvedAction::Expr(_span, generic_expr) => {
894                self.instrument_action_expr(generic_expr, &mut res, justification);
895            }
896        }
897
898        res
899    }

921    /// Return some code adding to the view and term tables.
922    /// For constructors, `args` should not include the eclass of the resulting term (since it may not exist yet).
923    /// For custom functions, `args` should include all arguments (including the output for the function).
924    ///
925    /// Returns a vector of strings representing code to add and a variable for the created term.
926    /// We could return the term itself, but this might make the encoding blow up the code.
927    fn add_term_and_view(
928        &mut self,
929        func_type: &FuncType,
930        args: &[String],
931        justification: &Justification,
932    ) -> (Vec<String>, String) {
933        // A fresh variable for the new term.
934        let fv = self.fresh_var();
935        let mut res = vec![];
936        // TODO might be able to get rid of this intermediate variable in encoding
937        res.push(format!(
938            "(let {fv} ({} {}))",
939            func_type.name,
940            ListDisplay(args, " ")
941        ));
942
943        let args_with_fv = if func_type.subtype == FunctionSubtype::Constructor {
944            let mut a = args.to_vec();
945            a.push(fv.clone());
946            a
947        } else {
948            args.to_vec()
949        };
950
951        let (proof_str, view_proof_var) = if self.egraph.proof_state.proofs_enabled {
952            let to_ast = self.fname_to_ast_name(&func_type.name);
953            let rule_constructor = &self.proof_names().rule_constructor;
954            let fiat_constructor = &self.proof_names().fiat_constructor;
955
956            let proof = match justification {
957                Justification::Rule(rule_name, rule_proof) => {
958                    format!(
959                        "({rule_constructor} \"{rule_name}\" {rule_proof} ({to_ast} {fv}) ({to_ast} {fv}))",
960                    )
961                }
962                Justification::Fiat => {
963                    format!("({fiat_constructor} ({to_ast} {fv}) ({to_ast} {fv}))",)
964                }
965                Justification::Merge(fn_name, p1, p2) => {
966                    let merge_constructor = &self.proof_names().merge_fn_constructor;
967                    format!("({merge_constructor} \"{fn_name}\" {p1} {p2} ({to_ast} {fv}))",)
968                }
969                Justification::Proof(existing_proof) => existing_proof.clone(),
970            };
971
972            let proof_var = self.fresh_var();
973            // add a proof for the constructor if needed
974            let term_proof = if func_type.subtype == FunctionSubtype::Constructor {
975                let term_proof_constructor = self.term_proof_name(func_type.output.name());
976                format!("(set ({term_proof_constructor} {fv}) {proof_var})")
977            } else {
978                "".to_string()
979            };
980
981            (
982                format!(
983                    "(let {proof_var} {proof})
984                     {term_proof}"
985                ),
986                proof_var,
987            )
988        } else {
989            ("".to_string(), "".to_string())
990        };
991
992        res.push(proof_str);
993        res.push(self.update_view(&func_type.name, &args_with_fv, &view_proof_var));
994
995        // add to uf table to initialize eclass for constructors
996        if func_type.subtype == FunctionSubtype::Constructor {
997            res.push(self.union(
998                func_type.output.name(),
999                &fv,
1000                &fv,
1001                &Justification::Proof(view_proof_var),
1002            ));
1003        }
1004
1005        (res, fv)
1006    }

1032    // Add to view and term tables, returning a variable for the created term.
1033    fn instrument_action_expr(
1034        &mut self,
1035        expr: &ResolvedExpr,
1036        res: &mut Vec<String>,
1037        proof: &Justification,
1038    ) -> String {
1039        match expr {
1040            ResolvedExpr::Lit(_, lit) => format!("{lit}"),
1041            ResolvedExpr::Var(_, resolved_var) => resolved_var.name.clone(),
1042            ResolvedExpr::Call(_, resolved_call, args) => {
1043                let args = args
1044                    .iter()
1045                    .map(|arg| self.instrument_action_expr(arg, res, proof))
1046                    .collect::<Vec<_>>();
1047                match resolved_call {
1048                    ResolvedCall::Func(func_type) => {
1049                        if func_type.subtype == FunctionSubtype::Custom {
1050                            // Globals are desugared to no-arg functions (in non-proof mode)
1051                            // They're allowed, in proof mode they are constructors.
1052                            if self.egraph.type_info.is_global(&func_type.name) {
1053                                return format!("({} {})", func_type.name, ListDisplay(&args, " "));
1054                            }
1055                            panic!(
1056                                "Found a function lookup in actions, should have been prevented by typechecking"
1057                            );
1058                        }
1059                        let (add_code, fv) = self.add_term_and_view(func_type, &args, proof);
1060                        res.extend(add_code);
1061
1062                        fv
1063                    }
1064                    ResolvedCall::Primitive(specialized_primitive) => {
1065                        let fv = self.fresh_var();
1066                        res.push(format!(
1067                            "(let {} ({} {}))",
1068                            fv,
1069                            specialized_primitive.name(),
1070                            ListDisplay(args, " ")
1071                        ));
1072                        fv
1073                    }
1074                }
1075            }
1076        }
1077    }

1092    /// Instrument a rule to use term encoding. This involves using the view tables in facts,
1093    /// adding to term and view tables in actions.
1094    /// When proofs are enabled we query proof tables, then build a proof for the rule in the actions.
1095    /// Finally, each view update also updates the proof tables.
1096    fn instrument_rule(&mut self, rule: &ResolvedRule) -> Vec<Command> {
1097        let (facts, proof_str) = self.instrument_facts(&rule.body);
1098        let proof_var = self.fresh_var();
1099        let proof = Justification::Rule(rule.name.clone(), proof_var.clone());
1100        let proof_var_binding = if self.egraph.proof_state.proofs_enabled {
1101            format!(
1102                "(let {proof_var}
1103                          {proof_str})"
1104            )
1105        } else {
1106            "".to_string()
1107        };
1108
1109        let actions = self.instrument_actions(&rule.head.0, &proof);
1110        let name = &rule.name;
1111        let instrumented = format!(
1112            "(rule ({})
1113                   ({proof_var_binding}
1114                    {})
1115                    {}
1116                    :name \"{name}\")",
1117            ListDisplay(facts, " "),
1118            ListDisplay(actions, " "),
1119            if rule.ruleset.is_empty() {
1120                "".to_string()
1121            } else {
1122                format!(":ruleset {}", rule.ruleset)
1123            }
1124        );
1125        self.parse_program(&instrumented)
1126    }

1128    /// Any schedule should be sound as long as we saturate.
1129    fn rebuild(&mut self) -> Schedule {
1130        let path_compress_ruleset = self.proof_names().path_compress_ruleset_name.clone();
1131        let single_parent = self.proof_names().single_parent_ruleset_name.clone();
1132        let uf_function_index = self.proof_names().uf_function_index_ruleset_name.clone();
1133        let rebuilding_cleanup_ruleset = self.proof_names().rebuilding_cleanup_ruleset_name.clone();
1134        let rebuilding_ruleset = self.proof_names().rebuilding_ruleset_name.clone();
1135        let delete_ruleset = self.proof_names().delete_subsume_ruleset_name.clone();
1136        self.parse_schedule(format!(
1137            "(seq
1138              (saturate
1139                  {rebuilding_cleanup_ruleset}
1140                  (saturate {single_parent})
1141                  (saturate {path_compress_ruleset})
1142                  (saturate {uf_function_index})
1143                  {rebuilding_ruleset})
1144              {delete_ruleset})"
1145        ))
1146    }

1148    fn instrument_schedule(&mut self, schedule: &ResolvedSchedule) -> Schedule {
1149        match schedule {
1150            ResolvedSchedule::Run(span, config) => {
1151                let new_run = match config.until {
1152                    Some(ref facts) => {
1153                        let (instrumented, _proof) = self.instrument_facts(facts);
1154                        let instrumented_facts = self.parse_facts(&instrumented);
1155                        Schedule::Run(
1156                            span.clone(),
1157                            RunConfig {
1158                                ruleset: config.ruleset.clone(),
1159                                until: Some(instrumented_facts),
1160                            },
1161                        )
1162                    }
1163                    None => Schedule::Run(
1164                        span.clone(),
1165                        RunConfig {
1166                            ruleset: config.ruleset.clone(),
1167                            until: None,
1168                        },
1169                    ),
1170                };
1171                Schedule::Sequence(span.clone(), vec![new_run, self.rebuild()])
1172            }
1173            ResolvedSchedule::Sequence(span, schedules) => Schedule::Sequence(
1174                span.clone(),
1175                schedules
1176                    .iter()
1177                    .map(|s| self.instrument_schedule(s))
1178                    .collect(),
1179            ),
1180            ResolvedSchedule::Saturate(span, schedule) => {
1181                Schedule::Saturate(span.clone(), Box::new(self.instrument_schedule(schedule)))
1182            }
1183            GenericSchedule::Repeat(span, n, schedule) => Schedule::Repeat(
1184                span.clone(),
1185                *n,
1186                Box::new(self.instrument_schedule(schedule)),
1187            ),
1188        }
1189    }

1193        match &command {
1194            ResolvedNCommand::Sort {
1195                span,
1196                name,
1197                presort_and_args,
1198                unionable,
1199                ..
1200            } => {
1201                let uf_name = self.uf_name(name);
1202                let proof_func = if self.egraph.proof_state.proofs_enabled {
1203                    Some(self.term_proof_name(name))
1204                } else {
1205                    None
1206                };
1207                res.push(Command::Sort {
1208                    span: span.clone(),
1209                    name: name.clone(),
1210                    presort_and_args: presort_and_args.clone(),
1211                    uf: Some(uf_name),
1212                    proof_func,
1213                    unionable: *unionable,
1214                });
1215                res.extend(self.declare_sort(name));
1216            }
1217            ResolvedNCommand::Function(fdecl) => {
1218                res.extend(self.term_and_view(fdecl));
1219                res.extend(self.rebuilding_rules(fdecl));
1220            }
1221            ResolvedNCommand::NormRule { rule } => {
1222                res.extend(self.instrument_rule(rule));
1223            }
1224            ResolvedNCommand::CoreAction(action) => {
1225                let instrumented = self
1226                    .instrument_action(action, &Justification::Fiat)
1227                    .join("\n");
1228                res.extend(self.parse_program(&instrumented));
1229            }
1230            ResolvedNCommand::Check(span, facts) => {
1231                let (instrumented, _proof) = self.instrument_facts(facts);
1232                res.push(Command::Check(
1233                    span.clone(),
1234                    self.parse_facts(&instrumented),
1235                ));
1236            }
1237            ResolvedNCommand::RunSchedule(schedule) => {
1238                res.push(Command::RunSchedule(self.instrument_schedule(schedule)));
1239            }
1240            ResolvedNCommand::Fail(span, cmd) => {
1241                self.term_encode_command(cmd, res);
1242                let last = res.pop().unwrap();
1243                res.push(Command::Fail(span.clone(), Box::new(last)));
1244            }
1245            ResolvedNCommand::Extract(span, expr, variants) => {
1246                // Instrument the expressions to use view tables (like actions, not facts)
1247                let mut action_stmts = vec![];
1248                let instrumented_expr =
1249                    self.instrument_action_expr(expr, &mut action_stmts, &Justification::Fiat);
1250                let instrumented_variants =
1251                    self.instrument_action_expr(variants, &mut action_stmts, &Justification::Fiat);
1252
1253                // Add any action statements needed to set up the expressions
1254                for stmt in action_stmts {
1255                    res.extend(self.parse_program(&stmt));
1256                }
1257                // Rebuild before extract; we may have added new view rows that need canonicalization
1258                res.push(Command::RunSchedule(self.rebuild()));
1259                res.push(Command::Extract(
1260                    span.clone(),
1261                    self.parse_expr(&instrumented_expr),
1262                    self.parse_expr(&instrumented_variants),
1263                ));
1264            }
1265            ResolvedNCommand::PrintSize(span, name) => {
1266                // In proof mode, print the size of the view table for constructors
1267                let new_name = name.as_ref().map(|n| {
1268                    if self
1269                        .egraph
1270                        .type_info
1271                        .get_func_type(n)
1272                        .is_some_and(|f| f.subtype == FunctionSubtype::Constructor)
1273                    {
1274                        self.view_name(n)
1275                    } else {
1276                        n.clone()
1277                    }
1278                });
1279                res.push(Command::PrintSize(span.clone(), new_name));
1280            }
1281            ResolvedNCommand::Pop(..)
1282            | ResolvedNCommand::Push(..)
1283            | ResolvedNCommand::AddRuleset(..)
1284            | ResolvedNCommand::Output { .. }
1285            | ResolvedNCommand::Input { .. }
1286            | ResolvedNCommand::UnstableCombinedRuleset(..)
1287            | ResolvedNCommand::PrintOverallStatistics(..)
1288            | ResolvedNCommand::PrintFunction(..)
1289            | ResolvedNCommand::ProveExists(..) => {
1290                res.push(command.to_command().make_unresolved());
1291            }
1292            ResolvedNCommand::UserDefined(..) => {
1293                panic!("User defined commands unsupported in term encoding");
1294            }
1295        }
1296    }
1297
1298    pub(crate) fn add_term_encoding_helper(
1299        &mut self,
1300        program: Vec<ResolvedNCommand>,
1301    ) -> Vec<Command> {
1302        let mut res = vec![];
1303
1304        if !self.egraph.proof_state.term_header_added {
1305            res.extend(self.term_header());
1306            if self.egraph.proof_state.proofs_enabled {
1307                let proof_header = self.proof_header();
1308                res.extend(self.parse_program(&proof_header));
1309            }
1310            self.egraph.proof_state.term_header_added = true;
1311        }
1312
1313        for command in program {
1314            self.term_encode_command(&command, &mut res);
1315
1316            // run rebuilding after every command except a few
1317            if let ResolvedNCommand::Function(..)
1318            | ResolvedNCommand::NormRule { .. }
1319            | ResolvedNCommand::Sort { .. } = &command
1320            {
1321            } else {
1322                res.push(Command::RunSchedule(self.rebuild()));
1323            }
1324        }
1325
1326        res
1327    }
