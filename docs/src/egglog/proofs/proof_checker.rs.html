1//! Proof checking for egglog proofs.
2//! Given an egglog program and a proof for some proposition, check that the proof is valid.
3//! The main work is to check proofs for rules, ensuring that under the substitution
4//! the rule would have matched the propositions for each premise, and that the conclusion
5//! is contained in the rule's actions.
6//! For checking queries and actions, we evaluate the expressions under a substitution
7//! and produce a set of valid propositions with `process_actions`.
8//! The set of valid propositions includes reflexive equalities for all subterms.
9
10use crate::{
11    Term, TermDag, TermId,

23/// Result of processing actions: terms bound to variables and propositions
24#[derive(Debug, Clone)]
25pub(crate) struct ActionContext {
26    /// Terms bound to variables (from Let actions)
27    pub var_bindings: HashMap<String, TermId>,
28    /// Propositions (equalities) implied by the actions
29    pub propositions: HashSet<Proposition>,
30}

32/// Gathers all global CoreActions from a program.
33/// This extracts all actions that occur at the top level, filtering out NormRule and other commands.
34pub(crate) fn gather_global_actions(
35    prog: &[ResolvedNCommand],
36) -> impl Iterator<Item = &GenericAction<ResolvedCall, crate::ast::ResolvedVar>> {
37    prog.iter().filter_map(|cmd| {
38        if let GenericNCommand::CoreAction(action) = cmd {
39            Some(action)
40        } else {
41            None
42        }
43    })
44}

46/// Run a merge function and return the resulting term, as well as a set of propositions learned.
47pub(crate) fn run_merge(
48    term_dag: &mut TermDag,
49    func_name: &str,
50    prog: &[ResolvedNCommand],
51    old_term: TermId,
52    new_term: TermId,
53) -> Result<(TermId, HashSet<Proposition>), ProofCheckError> {
54    let mut subst = HashMap::default();
55    subst.insert("old".to_string(), old_term);
56    subst.insert("new".to_string(), new_term);
57    for cmd in prog {
58        if let GenericNCommand::Function(func_decl) = cmd {
59            if func_decl.name == func_name {
60                // run the merge function for this function using eval_expr
61                let expr = func_decl.merge.as_ref().ok_or_else(|| {
62                    ProofCheckError::from(ProofCheckErrorKind::FunctionNotFound {
63                        function_name: func_name.to_string(),
64                    })
65                })?;
66                return eval_expr_with_subst("merge_function", expr, term_dag, &subst);
67            }
68        }
69    }
70    Err(ProofCheckErrorKind::FunctionNotFound {
71        function_name: func_name.to_string(),
72    }
73    .into())
74}

76/// Given a sequence of actions, computes:
77/// 1. All the terms bound to variables (from Let actions)
78/// 2. All the propositions implied by the actions:
79///    - Reflexive equalities for all subterms
80///    - Ground equalities from union statements (bidirectional)
81///    - Reflexive equalities from set statements
82pub(crate) fn process_actions(
83    rule_name: &str,
84    mut bindings: HashMap<String, TermId>,
85    actions: &[&GenericAction<ResolvedCall, crate::ast::ResolvedVar>],
86    term_dag: &mut TermDag,
87) -> Result<ActionContext, ProofCheckError> {
88    let mut propositions = HashSet::default();
89
90    // Single pass: process all actions, accumulating bindings and propositions
91    for action in actions {
92        match action {
93            GenericAction::Let(_, var, expr) => {
94                // Evaluate the expression and collect propositions
95                let (term_id, new_props) =
96                    eval_expr_with_subst(rule_name, expr, term_dag, &bindings)?;
97                bindings.insert(var.name.clone(), term_id);
98                propositions.extend(new_props);
99            }
100            GenericAction::Union(_, lhs_expr, rhs_expr) => {
101                // Union creates ground equalities
102                let (lhs_term, lhs_props) =
103                    eval_expr_with_subst(rule_name, lhs_expr, term_dag, &bindings)?;
104                let (rhs_term, rhs_props) =
105                    eval_expr_with_subst(rule_name, rhs_expr, term_dag, &bindings)?;
106
107                // Collect propositions from evaluating both sides
108                propositions.extend(lhs_props);
109                propositions.extend(rhs_props);
110                // Store both directions of the equality
111                propositions.insert(Proposition::new(lhs_term, rhs_term));
112                propositions.insert(Proposition::new(rhs_term, lhs_term));
113            }
114            GenericAction::Set(_, func, args, rhs) => {
115                // Set creates reflexive equality for the resulting term
116                let mut all_args = args.to_vec();
117                all_args.push(rhs.clone());
118                let call_expr = ResolvedExpr::Call(crate::ast::Span::Panic, func.clone(), all_args);
119                let (_term, new_props) =
120                    eval_expr_with_subst(rule_name, &call_expr, term_dag, &bindings)?;
121                propositions.extend(new_props);
122            }
123            GenericAction::Expr(_, expr) => {
124                // Expr creates reflexive equality for its result
125                let (_, new_props) = eval_expr_with_subst(rule_name, expr, term_dag, &bindings)?;
126                propositions.extend(new_props);
127            }
128            GenericAction::Panic(_, _) => {
129                // Panics do not create propositions
130            }
131            GenericAction::Change(_, _, _, _) => {
132                // Changes do not create propositions
133            }
134        }
135    }
136
137    Ok(ActionContext {
138        var_bindings: bindings,
139        propositions,
140    })
141}

143/// Evaluate an expression under a substitution.
144/// Returns Ok((TermId, propositions)) if successful, where propositions include
145/// all reflexive equalities for the term and its subterms.
146/// Returns Err(()) if evaluation fails.
147fn eval_expr_with_subst(
148    rule_name: &str,
149    expr: &ResolvedExpr,
150    dag: &mut TermDag,
151    subst: &HashMap<String, TermId>,
152) -> Result<(TermId, HashSet<Proposition>), ProofCheckError> {
153    let mut propositions = HashSet::default();
154
155    let term_id = match expr {
156        ResolvedExpr::Lit(_, lit) => dag.lit(lit.clone()),
157        ResolvedExpr::Var(_, var) => subst.get(&var.name).copied().ok_or_else(|| {
158            ProofCheckError::from(ProofCheckErrorKind::UnboundVariable {
159                rule_name: rule_name.to_string(),
160                variable: var.name.clone(),
161                available: subst.keys().cloned().collect::<Vec<_>>().join(", "),
162            })
163        })?,
164        ResolvedExpr::Call(_, head, args) => match head {
165            ResolvedCall::Func(_func_type) => {
166                let mut arg_terms = Vec::new();
167                for arg in args {
168                    let (arg_term, arg_props) = eval_expr_with_subst(rule_name, arg, dag, subst)?;
169                    arg_terms.push(arg_term);
170                    propositions.extend(arg_props);
171                }
172                dag.app(head.name().to_string(), arg_terms)
173            }
174            ResolvedCall::Primitive(specialized_primitive) => {
175                // run validator, throwing error if it fails
176                let mut arg_terms = Vec::new();
177                for arg in args {
178                    let (arg_term, arg_props) = eval_expr_with_subst(rule_name, arg, dag, subst)?;
179                    arg_terms.push(arg_term);
180                    propositions.extend(arg_props);
181                }
182                // checked by earlier code showing this program supports proofs
183                let validator = specialized_primitive
184                    .validator()
185                    .expect("Expected primitive to have validator since proof mode is enabled");
186                validator(dag, &arg_terms).ok_or_else(|| {
187                    ProofCheckError::from(ProofCheckErrorKind::PrimitiveValidatorFailed {
188                        function_name: specialized_primitive.name().to_string(),
189                    })
190                })?
191            }
192        },
193    };
194
195    // Add reflexive equality for this term and all its subterms
196    add_subterm_reflexive_equalities(term_id, dag, &mut propositions);
197
198    Ok((term_id, propositions))
199}

201/// Add reflexive equalities for all subterms of a term.
202/// For example, if we have proved `(f (g 1)) = (f (g 1))`, then we have also proved
203// that `(g 1) = (g 1)` and `1 = 1`.
204fn add_subterm_reflexive_equalities(
205    term_id: TermId,
206    term_dag: &TermDag,
207    propositions: &mut HashSet<Proposition>,
208) {
209    // Add reflexive equality for this term
210    propositions.insert(Proposition::new(term_id, term_id));
211
212    // Recursively add for all children
213    if let Term::App(_, children) = term_dag.get(term_id) {
214        for &child_id in children {
215            add_subterm_reflexive_equalities(child_id, term_dag, propositions);
216        }
217    }
218}

220/// Gathers all global variables from a program and computes their values as terms
221/// without using globals (i.e., all global references are replaced with their definitions).
222///
223/// This expects the program to be in proof normalized form where globals appear as Let actions.
224pub(crate) fn gather_globals(
225    prog: &[ResolvedNCommand],
226    term_dag: &mut TermDag,
227) -> Result<HashMap<String, TermId>, ProofCheckError> {
228    let actions: Vec<_> = gather_global_actions(prog).collect();
229    let ctx = process_actions("global_action", HashMap::default(), &actions, term_dag)?;
230    Ok(ctx.var_bindings)
231}

252/// The kinds of errors that can occur during proof checking
253#[derive(Debug, Clone, Error)]
254pub enum ProofCheckErrorKind {
255    /// The proof claims terms are equal but they don't match the actual terms
256    #[error(
257        "Proof {proof_id} claims to prove {expected_lhs:?} = {expected_rhs:?}, but actually proves {actual_lhs:?} = {actual_rhs:?}"
258    )]
259    TermMismatch {
260        proof_id: ProofId,
261        expected_lhs: TermId,
262        expected_rhs: TermId,
263        actual_lhs: TermId,
264        actual_rhs: TermId,
265    },
266    /// Transitivity requires matching middle terms
267    #[error(
268        "Proof {proof_id}: transitivity requires matching middle terms, but left.rhs = {left_rhs:?} and right.lhs = {right_lhs:?}"
269    )]
270    TransitivityMismatch {
271        proof_id: ProofId,
272        left_rhs: TermId,
273        right_lhs: TermId,
274    },
275    /// Congruence proof: base rhs is not a function application
276    #[error("Proof {proof_id}: congruence error - base proof rhs is not a function application")]
277    CongruenceBaseNotApp { proof_id: ProofId },
278    /// Congruence proof: child index out of bounds
279    #[error(
280        "Proof {proof_id}: congruence error - child index {child_index} out of bounds for term with {num_children} children"
281    )]
282    CongruenceChildIndexOutOfBounds {
283        proof_id: ProofId,
284        child_index: usize,
285        num_children: usize,
286    },
287    /// Congruence proof: child proof lhs doesn't match base term child
288    #[error(
289        "Proof {proof_id}: congruence error - child proof lhs {child_lhs:?} doesn't match base term child {base_child:?} at index {child_index}"
290    )]
291    CongruenceChildMismatch {
292        proof_id: ProofId,
293        child_lhs: TermId,
294        base_child: TermId,
295        child_index: usize,
296    },
297    /// Congruence proof: result doesn't match expected
298    #[error(
299        "Proof {proof_id}: congruence error - proof rhs {proof_rhs:?} doesn't match expected {expected_rhs:?}"
300    )]
301    CongruenceResultMismatch {
302        proof_id: ProofId,
303        proof_rhs: TermId,
304        expected_rhs: TermId,
305    },
306    /// Congruence proof: lhs doesn't match base proof lhs
307    #[error("Proof {proof_id}: congruence error - proof lhs doesn't match base proof lhs")]
308    CongruenceLhsMismatch { proof_id: ProofId },
309    /// Rule application has wrong number of premises
310    #[error(
311        "Proof {proof_id}: rule '{rule_name}' expects {expected} premises, but proof has {actual}"
312    )]
313    RulePremiseCountMismatch {
314        proof_id: ProofId,
315        rule_name: String,
316        expected: usize,
317        actual: usize,
318    },
319    /// Variable not found in substitution during proof checking
320    #[error(
321        "Rule '{rule_name}': variable '{variable}' not found in substitution. Available: {available}"
322    )]
323    UnboundVariable {
324        rule_name: String,
325        variable: String,
326        available: String,
327    },
328    /// Function fact doesn't match the expected reflexive equality proposition
329    #[error(
330        "Rule '{rule_name}': function fact mismatch - expected reflexive equality for {expected}, got {actual_lhs} = {actual_rhs}"
331    )]
332    FunctionFactMismatch {
333        rule_name: String,
334        expected: String,
335        actual_lhs: String,
336        actual_rhs: String,
337    },
338    /// Equality fact doesn't match the proven proposition under substitution
339    #[error(
340        "Rule '{rule_name}': equality fact mismatch under substitution.\nFact: {fact}\nSubstituted: (= {substituted_lhs} {substituted_rhs})\nPremise proves: (= {proven_lhs} {proven_rhs})"
341    )]
342    EqualityFactMismatch {
343        rule_name: String,
344        fact: String,
345        substituted_lhs: String,
346        substituted_rhs: String,
347        proven_lhs: String,
348        proven_rhs: String,
349    },
350    /// Plain fact expression doesn't match proposition under substitution
351    #[error(
352        "Rule '{rule_name}': fact mismatch - {fact} under substitution {substitution} gives {actual}, expected {expected}"
353    )]
354    FactMismatch {
355        rule_name: String,
356        fact: String,
357        substitution: String,
358        actual: String,
359        expected: String,
360    },
361    /// Rule head actions don't produce the claimed equality
362    #[error(
363        "Rule '{rule_name}': rule head doesn't produce claimed equality.\nLHS: {claimed_lhs}\nRHS: {claimed_rhs}\nSubstitution: {substitution}"
364    )]
365    RuleHeadMismatch {
366        rule_name: String,
367        claimed_lhs: String,
368        claimed_rhs: String,
369        substitution: String,
370    },
371    /// Primitive operation validator failed
372    #[error("Primitive '{function_name}' validation failed")]
373    PrimitiveValidatorFailed { function_name: String },
374    /// Primitive has no validator for proof checking
375    #[error("Primitive '{function_name}' has no validator - cannot verify in proof")]
376    PrimitiveNoValidator { function_name: String },
377    /// Could not find the rule referenced in a proof
378    #[error("Could not find rule '{rule_name}'")]
379    RuleNotFound { rule_name: String },
380    /// Could not find the function referenced in a proof
381    #[error("Could not find function '{function_name}'")]
382    FunctionNotFound { function_name: String },
383    /// Fiat proof claims equality not established by globals
384    #[error(
385        "Proof {proof_id}: Fiat proof claims {lhs:?} = {rhs:?}, which is not established by globals"
386    )]
387    InvalidFiat {
388        proof_id: ProofId,
389        lhs: TermId,
390        rhs: TermId,
391    },
392    /// MergeFn proof: old and new proofs are for different functions
393    #[error(
394        "Proof {proof_id}: MergeFn error - old and new proofs should be for the same function, but got {old_func} and {new_func}"
395    )]
396    MergeFnFunctionMismatch {
397        proof_id: ProofId,
398        old_func: String,
399        new_func: String,
400    },
401    /// MergeFn proof: view term has no arguments
402    #[error("Proof {proof_id}: MergeFn error - {which} view term has no arguments")]
403    MergeFnEmptyArgs { proof_id: ProofId, which: String },
404    /// MergeFn proof: old and new view terms have different input arguments
405    #[error(
406        "Proof {proof_id}: MergeFn error - old and new view terms have different input arguments"
407    )]
408    MergeFnInputMismatch { proof_id: ProofId },
409    /// MergeFn proof: expected function application terms
410    #[error(
411        "Proof {proof_id}: MergeFn error - expected function application terms, got {old_term:?} and {new_term:?}"
412    )]
413    MergeFnNotApp {
414        proof_id: ProofId,
415        old_term: TermId,
416        new_term: TermId,
417    },
418    /// MergeFn proof: claimed equality not established by merge function
419    #[error(
420        "Proof {proof_id}: MergeFn error - proof claims {claimed_lhs} = {claimed_rhs}, which is not established by merge function"
421    )]
422    MergeFnResultMismatch {
423        proof_id: ProofId,
424        claimed_lhs: String,
425        claimed_rhs: String,
426    },
427    /// MergeFn proof: sub-proof is not reflexive
428    #[error(
429        "Proof {proof_id}: MergeFn error - {which} proof is not reflexive, lhs {lhs:?} != rhs {rhs:?}"
430    )]
431    MergeFnNotReflexive {
432        proof_id: ProofId,
433        which: String,
434        lhs: TermId,
435        rhs: TermId,
436    },
437}

439/// Context needed for proof checking
440pub(crate) struct ProofCheckContext {
441    /// Set of equalities established by global union/set actions
442    /// Each entry is a pair (lhs, rhs) that was unified
443    /// This includes reflexive equalities (term, term) for all globals
444    global_equalities: HashSet<Proposition>,
445    /// Map of global variable names to their TermIds
446    global_bindings: HashMap<String, TermId>,
447    /// Cache of already-checked proofs
448    checked_proofs: HashMap<ProofId, Proposition>,
449}
450
451impl ProofCheckContext {
452    /// Create a new proof check context by analyzing the program.
453    /// This gathers all equalities established by global actions (unions and sets).
454    fn new(prog: &[ResolvedNCommand], term_dag: &mut TermDag) -> Result<Self, ProofCheckError> {
455        // Use the new refactored functions
456        let actions: Vec<_> = gather_global_actions(prog).collect();
457        let action_ctx = process_actions("global_actions", HashMap::default(), &actions, term_dag)?;
458
459        Ok(ProofCheckContext {
460            global_equalities: action_ctx.propositions,
461            checked_proofs: HashMap::default(),
462            global_bindings: action_ctx.var_bindings,
463        })
464    }
465
466    fn in_globals(&self, lhs: TermId, rhs: TermId) -> bool {
467        self.global_equalities.contains(&Proposition::new(lhs, rhs))
468    }
469}

471/// Helper function to format a term with let bindings
472fn format_term(term_dag: &TermDag, term_id: TermId) -> String {
473    term_dag.to_string_with_let(&mut SymbolGen::new("".to_string()), term_id)
474}

476/// Helper function to format a substitution as a string
477fn format_substitution(term_dag: &TermDag, substitution: &HashMap<String, TermId>) -> String {
478    substitution
479        .iter()
480        .map(|(k, v)| format!("{} -> {}", k, format_term(term_dag, *v)))
481        .collect::<Vec<_>>()
482        .join(", ")
483}

485impl ProofStore {
486    /// Check that a proof is valid with respect to a typechecked program.
487    pub(crate) fn check_proof(
488        &mut self,
489        proof_id: ProofId,
490        program: &[ResolvedNCommand],
491    ) -> Result<Proposition, ProofCheckError> {
492        let mut ctx = ProofCheckContext::new(program, &mut self.term_dag)?;
493        self.check_proof_with_context(proof_id, program, &mut ctx)
494    }
495
496    /// Internal recursive proof checker with context
497    fn check_proof_with_context(
498        &mut self,
499        proof_id: ProofId,
500        program: &[ResolvedNCommand],
501        ctx: &mut ProofCheckContext,
502    ) -> Result<Proposition, ProofCheckError> {
503        // Check cache first
504        if let Some(prop) = ctx.checked_proofs.get(&proof_id) {
505            return Ok(prop.clone());
506        }
507
508        let proof = self.id_to_proof[proof_id].clone();
509        let result = match &proof.justification {
510            Justification::Fiat => {
511                // if the both terms are primitives and equal, accept
512                let term = self.term_dag.get(proof.lhs());
513                if (matches!(term, Term::Lit(_)) && proof.lhs() == proof.rhs())
514                    || ctx.in_globals(proof.lhs(), proof.rhs())
515                {
516                    Ok(Proposition::new(proof.lhs(), proof.rhs()))
517                } else {
518                    Err(ProofCheckErrorKind::InvalidFiat {
519                        proof_id,
520                        lhs: proof.lhs(),
521                        rhs: proof.rhs(),
522                    }
523                    .into())
524                }
525            }
526
527            Justification::Rule {
528                name,
529                premise_proofs,
530                substitution,
531            } => {
532                // Find the rule in the program
533                let rule = program
534                    .iter()
535                    .find_map(|cmd| match cmd {
536                        GenericNCommand::NormRule { rule } if &rule.name == name => Some(rule),
537                        _ => None,
538                    })
539                    .ok_or_else(|| {
540                        ProofCheckError::from(ProofCheckErrorKind::RuleNotFound {
541                            rule_name: name.clone(),
542                        })
543                    })?;
544
545                // Check premise count
546                if rule.body.len() != premise_proofs.len() {
547                    return Err(ProofCheckErrorKind::RulePremiseCountMismatch {
548                        proof_id,
549                        rule_name: name.clone(),
550                        expected: rule.body.len(),
551                        actual: premise_proofs.len(),
552                    }
553                    .into());
554                }
555
556                // Check each premise proof
557                let mut premise_propositions = Vec::new();
558                for &premise_id in premise_proofs {
559                    let prop = self.check_proof_with_context(premise_id, program, ctx)?;
560                    premise_propositions.push(prop);
561                }
562
563                // Verify that premises match the rule body under the substitution
564                for (fact, prop) in rule.body.iter().zip(premise_propositions.iter()) {
565                    self.check_fact_matches_proposition(fact, prop, substitution, name)?;
566                }
567
568                // Verify that the conclusion matches what the rule produces
569                self.check_rule_produces_equality(
570                    ctx,
571                    rule,
572                    substitution,
573                    proof.proposition(),
574                    name,
575                )?;
576
577                Ok(Proposition::new(proof.lhs(), proof.rhs()))
578            }
579
580            Justification::MergeFn {
581                function,
582                old_proof,
583                new_proof,
584            } => {
585                // Check both sub-proofs - they should be reflexive proofs
586                let old_prop = self.check_proof_with_context(*old_proof, program, ctx)?;
587                let new_prop = self.check_proof_with_context(*new_proof, program, ctx)?;
588
589                let (old_lhs, old_rhs) = (old_prop.lhs, old_prop.rhs);
590                let (new_lhs, new_rhs) = (new_prop.lhs, new_prop.rhs);
591
592                // MergeFn proofs expect reflexive equality proofs
593                if old_lhs != old_rhs {
594                    return Err(ProofCheckErrorKind::MergeFnNotReflexive {
595                        proof_id,
596                        which: "old".to_string(),
597                        lhs: old_lhs,
598                        rhs: old_rhs,
599                    }
600                    .into());
601                }
602                if new_lhs != new_rhs {
603                    return Err(ProofCheckErrorKind::MergeFnNotReflexive {
604                        proof_id,
605                        which: "new".to_string(),
606                        lhs: new_lhs,
607                        rhs: new_rhs,
608                    }
609                    .into());
610                }
611
612                let old_view_term = self.term_dag.get(old_rhs);
613                let new_view_term = self.term_dag.get(new_rhs);
614
615                let (old_term, new_term, view_head, input_args) =
616                    match (old_view_term.clone(), new_view_term.clone()) {
617                        (Term::App(old_head, old_args), Term::App(new_head, new_args)) => {
618                            // Verify both are views of the same function
619                            if old_head != new_head {
620                                return Err(ProofCheckErrorKind::MergeFnFunctionMismatch {
621                                    proof_id,
622                                    old_func: old_head.clone(),
623                                    new_func: new_head.clone(),
624                                }
625                                .into());
626                            }
627                            // The last argument is the output
628                            let old_output = *old_args.last().ok_or_else(|| {
629                                ProofCheckError::from(ProofCheckErrorKind::MergeFnEmptyArgs {
630                                    proof_id,
631                                    which: "old".to_string(),
632                                })
633                            })?;
634                            let new_output = *new_args.last().ok_or_else(|| {
635                                ProofCheckError::from(ProofCheckErrorKind::MergeFnEmptyArgs {
636                                    proof_id,
637                                    which: "new".to_string(),
638                                })
639                            })?;
640                            // Get the input arguments (all but the last)
641                            let inputs: Vec<TermId> = old_args[..old_args.len() - 1].to_vec();
642                            // inputs should match for old and new
643                            if inputs.len() != new_args.len() - 1
644                                || inputs
645                                    .iter()
646                                    .zip(new_args[..new_args.len() - 1].iter())
647                                    .any(|(a, b)| a != b)
648                            {
649                                return Err(
650                                    ProofCheckErrorKind::MergeFnInputMismatch { proof_id }.into()
651                                );
652                            }
653
654                            (old_output, new_output, old_head.clone(), inputs)
655                        }
656                        _ => {
657                            return Err(ProofCheckErrorKind::MergeFnNotApp {
658                                proof_id,
659                                old_term: old_rhs,
660                                new_term: new_rhs,
661                            }
662                            .into());
663                        }
664                    };
665
666                // Run the merge function to get the expected result
667                let (merged_term_child, mut merged_props) =
668                    run_merge(&mut self.term_dag, function, program, old_term, new_term)?;
669                // Add f(inputs..., merged_term) to merged_props
670                let mut merged_view_args = input_args.clone();
671                merged_view_args.push(merged_term_child);
672                let merged_term = self.term_dag.app(view_head, merged_view_args);
673                merged_props.insert(Proposition::new(merged_term, merged_term));
674                // Verify the proof's claimed equality is in the merged propositions
675                if !merged_props.contains(&Proposition::new(proof.lhs(), proof.rhs())) {
676                    return Err(ProofCheckErrorKind::MergeFnResultMismatch {
677                        proof_id,
678                        claimed_lhs: self
679                            .term_dag
680                            .to_string_with_let(&mut SymbolGen::new("".to_string()), proof.lhs()),
681                        claimed_rhs: self
682                            .term_dag
683                            .to_string_with_let(&mut SymbolGen::new("".to_string()), proof.rhs()),
684                    }
685                    .into());
686                }
687
688                Ok(Proposition::new(proof.lhs(), proof.rhs()))
689            }
690
691            Justification::Trans(left_id, right_id) => {
692                // Check both sub-proofs
693                let left_prop = self.check_proof_with_context(*left_id, program, ctx)?;
694                let right_prop = self.check_proof_with_context(*right_id, program, ctx)?;
695
696                let (left_lhs, left_rhs) = (left_prop.lhs, left_prop.rhs);
697                let (right_lhs, right_rhs) = (right_prop.lhs, right_prop.rhs);
698
699                // Check transitivity: left.rhs must equal right.lhs
700                if left_rhs != right_lhs {
701                    return Err(ProofCheckErrorKind::TransitivityMismatch {
702                        proof_id,
703                        left_rhs,
704                        right_lhs,
705                    }
706                    .into());
707                }
708
709                // Result should be left_lhs = right_rhs
710                if proof.lhs() != left_lhs || proof.rhs() != right_rhs {
711                    return Err(ProofCheckErrorKind::TermMismatch {
712                        proof_id,
713                        expected_lhs: proof.lhs(),
714                        expected_rhs: proof.rhs(),
715                        actual_lhs: left_lhs,
716                        actual_rhs: right_rhs,
717                    }
718                    .into());
719                }
720
721                Ok(Proposition::new(proof.lhs(), proof.rhs()))
722            }
723
724            Justification::Sym(inner_id) => {
725                // Check the inner proof
726                let inner_prop = self.check_proof_with_context(*inner_id, program, ctx)?;
727                let (inner_lhs, inner_rhs) = (inner_prop.lhs, inner_prop.rhs);
728
729                // Symmetry swaps lhs and rhs
730                if proof.lhs() != inner_rhs || proof.rhs() != inner_lhs {
731                    return Err(ProofCheckErrorKind::TermMismatch {
732                        proof_id,
733                        expected_lhs: proof.lhs(),
734                        expected_rhs: proof.rhs(),
735                        actual_lhs: inner_rhs,
736                        actual_rhs: inner_lhs,
737                    }
738                    .into());
739                }
740
741                Ok(Proposition::new(proof.lhs(), proof.rhs()))
742            }
743
744            Justification::Congr {
745                proof: base_id,
746                child_index,
747                child_proof: child_id,
748            } => {
749                // Check the base proof (proves t1 = f(..., ci, ...))
750                let base_prop = self.check_proof_with_context(*base_id, program, ctx)?;
751                let (base_lhs, base_rhs) = (base_prop.lhs, base_prop.rhs);
752
753                // Check the child proof (proves ci = c2)
754                let child_prop = self.check_proof_with_context(*child_id, program, ctx)?;
755                let (child_lhs, child_rhs) = (child_prop.lhs, child_prop.rhs);
756
757                // base_rhs should be an application f(...)
758                let (func_name, children) = match self.term_dag.get(base_rhs) {
759                    Term::App(f, cs) => (f.clone(), cs.clone()),
760                    _ => {
761                        return Err(ProofCheckErrorKind::CongruenceBaseNotApp { proof_id }.into());
762                    }
763                };
764
765                // Check child_index is valid
766                if *child_index >= children.len() {
767                    return Err(ProofCheckErrorKind::CongruenceChildIndexOutOfBounds {
768                        proof_id,
769                        child_index: *child_index,
770                        num_children: children.len(),
771                    }
772                    .into());
773                }
774
775                // Check that child_lhs matches the child at child_index
776                if children[*child_index] != child_lhs {
777                    return Err(ProofCheckErrorKind::CongruenceChildMismatch {
778                        proof_id,
779                        child_lhs,
780                        base_child: children[*child_index],
781                        child_index: *child_index,
782                    }
783                    .into());
784                }
785
786                // Construct the expected new term by replacing the child
787                let expected_rhs_children: Vec<TermId> = children
788                    .iter()
789                    .enumerate()
790                    .map(
791                        |(i, &child)| {
792                            if i == *child_index { child_rhs } else { child }
793                        },
794                    )
795                    .collect();
796
797                let expected_rhs_id = self.term_dag.app(func_name, expected_rhs_children);
798
799                // Verify proof.rhs() matches expected
800                if proof.rhs() != expected_rhs_id {
801                    return Err(ProofCheckErrorKind::CongruenceResultMismatch {
802                        proof_id,
803                        proof_rhs: proof.rhs(),
804                        expected_rhs: expected_rhs_id,
805                    }
806                    .into());
807                }
808
809                // Verify proof.lhs() matches base_lhs
810                if proof.lhs() != base_lhs {
811                    return Err(ProofCheckErrorKind::CongruenceLhsMismatch { proof_id }.into());
812                }
813
814                Ok(Proposition::new(proof.lhs(), proof.rhs()))
815            }
816        };
817
818        // Cache the result
819        if let Ok(ref prop) = result {
820            ctx.checked_proofs.insert(proof_id, prop.clone());
821        }
822
823        result
824    }

826    /// Check that a fact matches a proposition under a substitution  
827    fn check_fact_matches_proposition(
828        &mut self,
829        fact: &ResolvedFact,
830        prop: &Proposition,
831        substitution: &HashMap<String, TermId>,
832        rule_name: &str,
833    ) -> Result<(), ProofCheckError> {
834        let (lhs, rhs) = (prop.lhs, prop.rhs);
835        match fact {
836            // proof normal form for functions: (= (f args...) v)
837            // In the term representation, custom functions store output as last arg: f(args..., v)
838            ResolvedFact::Eq(
839                _,
840                ResolvedExpr::Call(
841                    _,
842                    ResolvedCall::Func(FuncType {
843                        subtype: FunctionSubtype::Custom,
844                        name,
845                        ..
846                    }),
847                    args,
848                ),
849                ResolvedExpr::Var(_, v),
850            ) => {
851                // Get the output variable's term
852                let var_term = substitution.get(&v.name).copied().ok_or_else(|| {
853                    ProofCheckErrorKind::UnboundVariable {
854                        rule_name: rule_name.to_string(),
855                        variable: v.name.clone(),
856                        available: substitution.keys().cloned().collect::<Vec<_>>().join(", "),
857                    }
858                })?;
859
860                // Evaluate all the input arguments
861                let mut arg_terms = Vec::new();
862                for arg in args {
863                    arg_terms.push(self.eval_expr_with_subst(rule_name, arg, substitution)?);
864                }
865                // Add the output variable as the last argument
866                arg_terms.push(var_term);
867
868                let expected_term_id = self.term_dag.app(name.clone(), arg_terms);
869
870                // The proposition should be a reflexive equality for this term
871                if lhs != expected_term_id || rhs != expected_term_id {
872                    return Err(ProofCheckErrorKind::FunctionFactMismatch {
873                        rule_name: rule_name.to_string(),
874                        expected: format_term(&self.term_dag, expected_term_id),
875                        actual_lhs: format_term(&self.term_dag, lhs),
876                        actual_rhs: format_term(&self.term_dag, rhs),
877                    }
878                    .into());
879                }
880
881                Ok(())
882            }
883            ResolvedFact::Eq(_, lhs_expr, rhs_expr) => {
884                let fact_lhs = self.eval_expr_with_subst(rule_name, lhs_expr, substitution)?;
885                let fact_rhs = self.eval_expr_with_subst(rule_name, rhs_expr, substitution)?;
886                if fact_lhs != lhs || fact_rhs != rhs {
887                    return Err(ProofCheckErrorKind::EqualityFactMismatch {
888                        rule_name: rule_name.to_string(),
889                        fact: format!("{fact}"),
890                        substituted_lhs: self.term_dag.to_string(fact_lhs),
891                        substituted_rhs: self.term_dag.to_string(fact_rhs),
892                        proven_lhs: format_term(&self.term_dag, lhs),
893                        proven_rhs: format_term(&self.term_dag, rhs),
894                    }
895                    .into());
896                }
897
898                Ok(())
899            }
900            // For a plain expr, the proof should have the form t1 = t2 where t2 matches the expr under substitution
901            ResolvedFact::Fact(expr) => {
902                let fact_term = self.eval_expr_with_subst(rule_name, expr, substitution)?;
903
904                if fact_term != rhs {
905                    return Err(ProofCheckErrorKind::FactMismatch {
906                        rule_name: rule_name.to_string(),
907                        fact: format!("{fact}"),
908                        substitution: format_substitution(&self.term_dag, substitution),
909                        actual: format_term(&self.term_dag, rhs),
910                        expected: format_term(&self.term_dag, fact_term),
911                    }
912                    .into());
913                }
914
915                Ok(())
916            }
917        }
918    }

920    /// Evaluate an expression with a variable substitution
921    fn eval_expr_with_subst(
922        &mut self,
923        rule_name: &str,
924        expr: &ResolvedExpr,
925        substitution: &HashMap<String, TermId>,
926    ) -> Result<TermId, ProofCheckError> {
927        match expr {
928            ResolvedExpr::Lit(_, lit) => Ok(self.term_dag.lit(lit.clone())),
929            ResolvedExpr::Var(_, var) => substitution.get(&var.name).copied().ok_or_else(|| {
930                ProofCheckError::from(ProofCheckErrorKind::UnboundVariable {
931                    rule_name: rule_name.to_string(),
932                    variable: var.name.clone(),
933                    available: substitution.keys().cloned().collect::<Vec<_>>().join(", "),
934                })
935            }),
936            ResolvedExpr::Call(_, head, args) => {
937                // Evaluate all arguments first
938                let mut arg_terms = Vec::new();
939                for arg in args {
940                    arg_terms.push(self.eval_expr_with_subst(rule_name, arg, substitution)?);
941                }
942
943                match head {
944                    ResolvedCall::Primitive(prim) => {
945                        // Use the validator to compute the primitive result
946                        if let Some(validator) = prim.validator() {
947                            let result =
948                                validator(&mut self.term_dag, &arg_terms).ok_or_else(|| {
949                                    ProofCheckErrorKind::PrimitiveValidatorFailed {
950                                        function_name: prim.name().to_string(),
951                                    }
952                                })?;
953                            Ok(result)
954                        } else {
955                            // No validator available - primitives without validators can't be checked in proofs
956                            Err(ProofCheckErrorKind::PrimitiveNoValidator {
957                                function_name: prim.name().to_string(),
958                            }
959                            .into())
960                        }
961                    }
962                    ResolvedCall::Func(func) => {
963                        match func.subtype {
964                            FunctionSubtype::Constructor => {
965                                Ok(self.term_dag.app(func.name.clone(), arg_terms))
966                            }
967                            FunctionSubtype::Custom => {
968                                // Custom functions should not appear in proof normal form!
969                                // They should be in the form (= (f args...) v) in the rule body
970                                panic!(
971                                    "Custom function {} should not appear in expression evaluation during proof checking. \
972                                    Functions should be in proof normal form: (= (function args...) output_var)",
973                                    func.name
974                                );
975                            }
976                        }
977                    }
978                }
979            }
980        }
981    }

983    /// Check that a rule produces the claimed equality
984    fn check_rule_produces_equality(
985        &mut self,
986        ctx: &ProofCheckContext,
987        rule: &crate::ast::GenericRule<ResolvedCall, crate::ast::ResolvedVar>,
988        substitution: &HashMap<String, TermId>,
989        claimed: &Proposition,
990        rule_name: &str,
991    ) -> Result<(), ProofCheckError> {
992        // Use process_actions to get propositions from the rule head
993        // Note: process_actions expects global variable bindings, but substitution
994        // has the same structure, so we can pass it directly
995        let action_refs: Vec<&GenericAction<ResolvedCall, crate::ast::ResolvedVar>> =
996            rule.head.0.iter().collect();
997        let mut bindings = ctx.global_bindings.clone();
998        bindings.extend(substitution.clone());
999        let action_ctx = process_actions(rule_name, bindings, &action_refs, &mut self.term_dag)?;
1000
1001        // Check if the claimed equality is in the propositions
1002        if action_ctx.propositions.contains(claimed) {
1003            return Ok(());
1004        }
1005
1006        Err(ProofCheckErrorKind::RuleHeadMismatch {
1007            rule_name: rule_name.to_string(),
1008            claimed_lhs: format_term(&self.term_dag, claimed.lhs()),
1009            claimed_rhs: format_term(&self.term_dag, claimed.rhs()),
1010            substitution: format_substitution(&self.term_dag, substitution),
1011        }
1012        .into())
1013    }
